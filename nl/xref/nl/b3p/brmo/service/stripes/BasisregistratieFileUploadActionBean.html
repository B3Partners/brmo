38  /**
39   * Action voor het handmatig uploaden van bestanden met basisregistratiegegevens via een formulier.
40   */
41  @StrictBinding
42  public class BasisregistratieFileUploadActionBean implements ActionBean {
43      private static final int ZIP_HEADER = 0x504b0304;
44  
45      private static final Log log = LogFactory.getLog(BasisregistratieFileUploadActionBean.class);
46  
47      private ActionBeanContext context;
48  
49      @Validate(required = true)
50      private String basisregistratie;
51  
52      @Validate(required = true)
53      private FileBean bestand;
54  
55      @DefaultHandler
56      @DontValidate
57      public Resolution form() {
58          return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
59      }
60  
61      public Resolution upload() throws BrmoException {
62          DataSource ds = ConfigUtil.getDataSourceStaging();
63          BrmoFramework brmo = new BrmoFramework(ds, null, null);
64          ZipInputStream zip = null;
65          ZipEntry entry = null;
66          int errors = 0;
67          int empty = 0;
68          int duplicate = 0;
69          final int ERROR_LIMIT = 5;
70          try {
71              final MutableLong theTotal = new MutableLong(0);
72              int extractedFiles = 0;
73              final ProgressUpdateListener totalAdder =
74                      new ProgressUpdateListener() {
75  
76                          @Override
77                          public void total(long total) {
78                              theTotal.setValue(theTotal.getValue() + total);
79                          }
80  
81                          @Override
82                          public void progress(long progress) {}
83  
84                          @Override
85                          public void exception(Throwable t) {}
86                      };
87  
88              // Check of bestand begint met ZIP header
89              InputStream in = new BufferedInputStream(bestand.getInputStream());
90              in.mark(4);
91              int header = new DataInputStream(in).readInt();
92              in.reset();
93  
94              if (header == ZIP_HEADER) {
95                  // Pak .xml bestanden in ZIP uit en verwerk deze
96  
97                  zip = new ZipInputStream(in);
98                  try {
99                      entry = zip.getNextEntry();
100                     while (entry != null) {
101                         if (!entry.getName().toLowerCase().endsWith(".xml")) {
102                             log.warn("Overslaan zip entry geen XML: " + entry.getName());
103                         } else {
104                             log.debug("Lezen XML bestand uit zip: " + entry.getName());
105                             try {
106                                 brmo.loadFromStream(
107                                         basisregistratie,
108                                         CloseShieldInputStream.wrap(zip),
109                                         bestand.getFileName() + "/" + entry.getName(),
110                                         totalAdder,
111                                         null);
112                             } catch (BrmoLeegBestandException e) {
113                                 log.info(
114                                         "Negeer ZIP entry "
115                                                 + entry.getName()
116                                                 + ": "
117                                                 + e.getMessage());
118                                 empty++;
119                             } catch (BrmoDuplicaatLaadprocesException e) {
120                                 log.info(
121                                         "Negeer ZIP entry "
122                                                 + entry.getName()
123                                                 + ": duplicaat laadproces");
124                                 duplicate++;
125                             } catch (BrmoException e) {
126                                 errors++;
127                                 log.error(
128                                         "BrmoException bij laden ZIP entry "
129                                                 + entry.getName()
130                                                 + (errors > ERROR_LIMIT
131                                                         ? "; afbreken"
132                                                         : "; doorgaan met volgende"),
133                                         e);
134                                 if (errors > ERROR_LIMIT) {
135                                     throw new BrmoException(
136                                             "Maximum aantal fouten ("
137                                                     + ERROR_LIMIT
138                                                     + ") bereikt bij inladen ZIP bestanden, inladen afgebroken. Controleer brmo-service.log",
139                                             e);
140                                 }
141                             }
142                             extractedFiles++;
143                         }
144                         entry = zip.getNextEntry();
145                     }
146                 } finally {
147                     zip.close();
148                 }
149             } else {
150                 // Geen ZIP, direct BR XML laden
151                 brmo.loadFromStream(basisregistratie, in, bestand.getFileName(), totalAdder, null);
152             }
153 
154             List<String> warnings = new ArrayList<>();
155             if (errors > 0) {
156                 warnings.add(
157                         "let op: "
158                                 + errors
159                                 + " bericht"
160                                 + (errors > 1 ? "en" : "")
161                                 + " niet ingeladen wegens fouten");
162             }
163             if (empty > 0) {
164                 warnings.add(
165                         "let op: "
166                                 + empty
167                                 + (empty > 1 ? " lege berichten" : " leeg bericht")
168                                 + " overgeslagen");
169             }
170             if (duplicate > 0) {
171                 warnings.add(
172                         "let op: "
173                                 + duplicate
174                                 + (duplicate > 1 ? " duplicate berichten" : " duplicaat bericht")
175                                 + " overgeslagen");
176             }
177 
178             String warning =
179                     warnings.isEmpty()
180                             ? ""
181                             : ", "
182                                     + String.join(", ", warnings.toArray(new String[] {}))
183                                     + ", zie log voor bestandsnamen uit ZIP en details!";
184 
185             getContext()
186                     .getMessages()
187                     .add(
188                             new SimpleMessage(
189                                     "Bestand "
190                                             + bestand.getFileName()
191                                             + " is ingelezen, "
192                                             + theTotal.getValue()
193                                             + " berichten"
194                                             + (extractedFiles > 0
195                                                     ? " (uit "
196                                                             + extractedFiles
197                                                             + " uitgepakte XML bestanden)"
198                                                     : "")
199                                             + warning));
200             log.info(
201                     String.format(
202                             "Stored %s data from file \"%s\" uploaded via form",
203                             basisregistratie, bestand.getFileName()));
204         } catch (Exception ex) {
205             String msg =
206                     "Fout bij inlezen bestand "
207                             + (zip != null && entry != null
208                                     ? entry.getName() + " uit ZIP bestand " + bestand.getFileName()
209                                     : bestand.getFileName());
210             log.error(msg, ex);
211 
212             getContext()
213                     .getMessages()
214                     .add(new SimpleMessage(msg + ": " + ExceptionUtils.getRootCauseMessage(ex)));
215         } finally {
216             try {
217                 bestand.delete();
218             } catch (IOException e) {
219                 // ignore
220             }
221             brmo.closeBrmoFramework();
222         }
223 
224         return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
225     }
