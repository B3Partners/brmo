143     @Override
144     public void execute(ProgressUpdateListener listener) {
145         this.l = listener;
146 
147         try {
148             l.updateStatus("Initialiseren...");
149             final Date startDate = new Date();
150             l.addLog(
151                     String.format(
152                             "Initialiseren GDS2 ophalen proces %d... %tc",
153                             config.getId(), startDate));
154             this.config.updateSamenvattingEnLogfile(
155                     String.format("Het GDS2 ophalen proces is gestart op %tc.", startDate));
156             this.config.setStatus(AutomatischProces.ProcessingStatus.PROCESSING);
157             this.config.setLastrun(startDate);
158             Stripersist.getEntityManager().merge(this.config);
159             Stripersist.getEntityManager().flush();
160 
161             BestandenlijstOpvragenRequest request = new BestandenlijstOpvragenRequest();
162             BestandenlijstOpvragenType verzoek = new BestandenlijstOpvragenType();
163             AfgifteSelectieCriteriaType criteria = new AfgifteSelectieCriteriaType();
164             request.setVerzoek(verzoek);
165             verzoek.setAfgifteSelectieCriteria(criteria);
166             criteria.setBestandKenmerken(new BestandKenmerkenType());
167 
168             String contractnummer =
169                     ClobElement.nullSafeGet(this.config.getConfig().get("gds2_contractnummer"));
170             if (contractnummer != null) {
171                 log.debug("GDS2 contractnummer is: " + contractnummer);
172                 criteria.getBestandKenmerken().setContractnummer(contractnummer);
173             }
174             String artikelnummer =
175                     ClobElement.nullSafeGet(this.config.getConfig().get("gds2_artikelnummer"));
176             if (artikelnummer != null) {
177                 log.debug("GDS2 artikelnummer is: " + artikelnummer);
178                 l.addLog("GDS2 artikelnummer is: " + artikelnummer);
179                 criteria.getBestandKenmerken().setArtikelnummer(artikelnummer);
180             }
181 
182             String alGerapporteerd =
183                     ClobElement.nullSafeGet(
184                             this.config.getConfig().get("gds2_al_gerapporteerde_afgiftes"));
185             if (!"true".equals(alGerapporteerd)) {
186                 criteria.setNogNietGerapporteerd(true);
187             }
188 
189             // datum tijd parsen/instellen
190             GregorianCalendar vanaf;
191             GregorianCalendar tot =
192                     getDatumTijd(ClobElement.nullSafeGet(this.config.getConfig().get("totdatum")));
193             String sVanaf = ClobElement.nullSafeGet(this.config.getConfig().get("vanafdatum"));
194             if (tot != null && ("-1".equals(sVanaf) | "-2".equals(sVanaf) | "-3".equals(sVanaf))) {
195                 vanaf =
196                         getDatumTijd(
197                                 ClobElement.nullSafeGet(this.config.getConfig().get("totdatum")),
198                                 Integer.parseInt(sVanaf));
199             } else {
200                 vanaf = getDatumTijd(sVanaf);
201             }
202 
203             List<AfgifteType> afgiftes = new ArrayList<>();
204             BestandenlijstOpvragenResponse response;
205             int hoogsteKlantAfgifteNummer = -1;
206 
207             Gds2AfgifteServiceV20170401 gds2 = initGDS2();
208             l.updateStatus("Uitvoeren SOAP request naar Kadaster...");
209 
210             boolean afgifteNummerGebaseerdOphalen = false;
211             if (vanaf == null && tot == null) {
212                 //  instellen laagste en hoogste afgiftenummer voor dit verzoek, mag alleen als er
213                 // geen periode is gedefinieerd
214                 KlantAfgiftenummerReeksType afgiftenummers = new KlantAfgiftenummerReeksType();
215                 String nr =
216                         ClobElement.nullSafeGet(
217                                 this.config.getConfig().get("klantafgiftenummer_vanaf"));
218                 if (StringUtils.isNotBlank(nr)) {
219                     BigInteger afgifteNrVanaf = new BigInteger(nr);
220                     afgiftenummers.setKlantAfgiftenummerVanaf(afgifteNrVanaf);
221                     log.info("Ophalen vanaf klant afgifte nummer: " + afgifteNrVanaf);
222                     l.addLog("Ophalen vanaf klant afgifte nummer: " + afgifteNrVanaf);
223 
224                     nr =
225                             ClobElement.nullSafeGet(
226                                     this.config.getConfig().get("klantafgiftenummer_totenmet"));
227                     if (StringUtils.isNotBlank(nr)) {
228                         BigInteger afgifteNrTM = new BigInteger(nr);
229                         afgiftenummers.setKlantAfgiftenummerTotmet(afgifteNrTM);
230                         log.info("Ophalen tot en met klant afgifte nummer: " + afgifteNrTM);
231                         l.addLog("Ophalen tot en met klant afgifte nummer: " + afgifteNrTM);
232                     }
233                     criteria.setKlantAfgiftenummerReeks(afgiftenummers);
234                     afgifteNummerGebaseerdOphalen = true;
235                 }
236             }
237 
238             if (afgifteNummerGebaseerdOphalen) {
239                 // <editor-fold> afgiftenummer-based ophalen
240                 l.addLog(
241                         "GDS2 verzoek voor afgiftenummers vanaf: "
242                                 + criteria.getKlantAfgiftenummerReeks()
243                                         .getKlantAfgiftenummerVanaf());
244                 l.addLog(
245                         "GDS2 verzoek voor afgiftenummers tot-en-met: "
246                                 + criteria.getKlantAfgiftenummerReeks()
247                                         .getKlantAfgiftenummerTotmet());
248                 response =
249                         retryBestandenLijstOpvragen(
250                                 gds2, request, BESTANDENLIJST_ATTEMPTS, BESTANDENLIJST_RETRY_WAIT);
251                 hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
252                 boolean hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
253                 int aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
254                 if (aantalInAntwoord > 0) {
255                     afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
256                 }
257                 l.addLog("Aantal in antwoord: " + aantalInAntwoord);
258                 l.addLog("Meer afgiftes beschikbaar: " + hasMore);
259 
260                 while (hasMore) {
261                     criteria.getKlantAfgiftenummerReeks()
262                             .setKlantAfgiftenummerVanaf(
263                                     // vanaf ophogen met aantal opgehaald
264                                     criteria.getKlantAfgiftenummerReeks()
265                                             .getKlantAfgiftenummerVanaf()
266                                             .add(BigInteger.valueOf(aantalInAntwoord)));
267                     l.addLog(
268                             "GDS2 verzoek voor afgiftenummers vanaf: "
269                                     + criteria.getKlantAfgiftenummerReeks()
270                                             .getKlantAfgiftenummerVanaf());
271                     l.addLog(
272                             "GDS2 verzoek voor afgiftenummers tot-en-met: "
273                                     + criteria.getKlantAfgiftenummerReeks()
274                                             .getKlantAfgiftenummerTotmet());
275                     response =
276                             retryBestandenLijstOpvragen(
277                                     gds2,
278                                     request,
279                                     BESTANDENLIJST_ATTEMPTS,
280                                     BESTANDENLIJST_RETRY_WAIT);
281                     hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
282                     aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
283                     hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
284                     l.addLog("Aantal in antwoord: " + aantalInAntwoord);
285                     l.addLog("Meer afgiftes beschikbaar: " + hasMore);
286 
287                     if (aantalInAntwoord > 0) {
288                         afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
289                     }
290                 }
291                 // </editor-fold> afgiftenummer-based ophalen
292             } else {
293                 // <editor-fold> time-based ophalen
294                 GregorianCalendar currentMoment = null;
295                 boolean parseblePeriod = false;
296                 int loopType = Calendar.DAY_OF_MONTH;
297                 int loopMax = 180;
298                 int loopNum = 0;
299                 boolean reducePeriod = false;
300                 boolean increasePeriod = false;
301 
302                 if (vanaf != null && tot != null && vanaf.before(tot)) {
303                     parseblePeriod = true;
304                     currentMoment = vanaf;
305                 }
306 
307                 boolean morePeriods2Process = false;
308                 do /* while morePeriods2Process is true */ {
309                     l.addLog("\n*** start periode ***");
310                     // zet periode in criteria indien gewenst
311                     if (parseblePeriod) {
312                         // check of de periodeduur verkleind moet worden
313                         if (reducePeriod) {
314                             switch (loopType) {
315                                 case Calendar.DAY_OF_MONTH:
316                                     currentMoment.add(loopType, -1);
317                                     loopType = Calendar.HOUR_OF_DAY;
318                                     l.addLog("* Verklein loop periode naar uur");
319                                     break;
320                                 case Calendar.HOUR_OF_DAY:
321                                     currentMoment.add(loopType, -1);
322                                     loopType = Calendar.MINUTE;
323                                     l.addLog("* Verklein loop periode naar minuut");
324                                     break;
325                                 case Calendar.MINUTE:
326                                 default:
327                                     /*
328                                      * Hier kom je alleen als binnen een minuut meer
329                                      * dan 2000 berichten zijn aangamaakt en het
330                                      * vinkje ook "al rapporteerde berichten
331                                      * ophalen" staat aan.
332                                      */
333                                     l.addLog("Niet alle gevraagde berichten zijn opgehaald");
334                             }
335                             reducePeriod = false;
336                         }
337 
338                         // check of de periodeduur vergroot moet worden
339                         if (increasePeriod) {
340                             switch (loopType) {
341                                 case Calendar.HOUR_OF_DAY:
342                                     loopType = Calendar.DAY_OF_MONTH;
343                                     l.addLog("* Vergroot loop periode naar dag");
344                                     break;
345                                 case Calendar.MINUTE:
346                                     loopType = Calendar.HOUR_OF_DAY;
347                                     l.addLog("* Vergroot loop periode naar uur");
348                                     break;
349                                 case Calendar.DAY_OF_MONTH:
350                                 default:
351                                     // not possible
352                             }
353                             increasePeriod = false;
354                         }
355 
356                         FilterDatumTijdType d = new FilterDatumTijdType();
357                         d.setDatumTijdVanaf(getXMLDatumTijd(currentMoment));
358                         l.addLog(String.format("Datum vanaf: %tc", currentMoment.getTime()));
359                         currentMoment.add(loopType, 1);
360                         d.setDatumTijdTotmet(getXMLDatumTijd(currentMoment));
361                         l.addLog(String.format("Datum tot: %tc", currentMoment.getTime()));
362                         criteria.setPeriode(d);
363 
364                         switch (loopType) {
365                             case Calendar.HOUR_OF_DAY:
366                                 // 0-23
367                                 if (currentMoment.get(loopType) == 0) {
368                                     increasePeriod = true;
369                                 }
370                                 break;
371                             case Calendar.MINUTE:
372                                 // 0-59
373                                 if (currentMoment.get(loopType) == 0) {
374                                     increasePeriod = true;
375                                 }
376                                 break;
377                             case Calendar.DAY_OF_MONTH:
378                             default:
379                                 // alleen dagen tellen, uur en minuut altijd helemaal
380                                 loopNum++;
381                         }
382 
383                         // bereken of einde van periode bereikt is
384                         morePeriods2Process = currentMoment.before(tot) && loopNum < loopMax;
385                     }
386 
387                     verzoek.setAfgifteSelectieCriteria(criteria);
388                     response =
389                             retryBestandenLijstOpvragen(
390                                     gds2,
391                                     request,
392                                     BESTANDENLIJST_ATTEMPTS,
393                                     BESTANDENLIJST_RETRY_WAIT);
394                     hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
395 
396                     int aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
397                     l.addLog("Aantal in antwoord: " + aantalInAntwoord);
398                     // opletten; in de xsd staat een default value van 'J' voor
399                     // meerAfgiftesbeschikbaar
400                     boolean hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
401                     l.addLog("Meer afgiftes beschikbaar: " + hasMore);
402 
403                     /*
404                      * Als "al gerapporteerde berichten" moeten worden opgehaald en
405                      * er zitten dan 2000 berichten in het antwoord dan heeft het
406                      * geen zin om meer keer de berichten op te halen, je krijgt
407                      * telkens dezelfde.
408                      */
409                     if (hasMore && "true".equals(alGerapporteerd)) {
410                         reducePeriod = true;
411                         morePeriods2Process = true;
412                         increasePeriod = false;
413                         // als er geen parsable periode is
414                         // (geen periode ingevuld en alGerapporteerd is true
415                         // dan moet morePeriods2Process false worden om een
416                         // eindloze while loop te voorkomen
417                         if (!parseblePeriod) {
418                             morePeriods2Process = false;
419                         } else {
420                             continue;
421                         }
422                     }
423 
424                     if (aantalInAntwoord > 0) {
425                         afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
426                     }
427 
428                     /*
429                      * Indicatie nog niet gerapporteerd: Met deze indicatie wordt
430                      * aangegeven of uitsluitend de nog niet gerapporteerde
431                      * bestanden moeten worden opgenomen in de lijst, of dat alle
432                      * beschikbare bestanden worden genoemd. Niet gerapporteerd
433                      * betekent in dit geval ‘niet eerder opgevraagd in deze
434                      * bestandenlijst’. Als deze indicator wordt gebruikt, dan
435                      * worden na terugmelding van de bestandenlijst de bijbehorende
436                      * bestanden gemarkeerd als zijnde ‘gerapporteerd’ in het
437                      * systeem van GDS.
438                      */
439                     int moreCount = 0;
440                     String dontGetMoreConfig =
441                             ClobElement.nullSafeGet(
442                                     this.config
443                                             .getConfig()
444                                             .get("gds2_niet_gerapporteerde_afgiftes_niet_ophalen"));
445 
446                     while (hasMore && !"true".equals(dontGetMoreConfig)) {
447                         l.updateStatus(
448                                 "Uitvoeren SOAP request naar Kadaster voor meer afgiftes..."
449                                         + moreCount++);
450                         criteria.setNogNietGerapporteerd(true);
451                         response =
452                                 retryBestandenLijstOpvragen(
453                                         gds2,
454                                         request,
455                                         BESTANDENLIJST_ATTEMPTS,
456                                         BESTANDENLIJST_RETRY_WAIT);
457                         hoogsteKlantAfgifteNummer =
458                                 response.getAntwoord().getKlantAfgiftenummerMax();
459 
460                         List<AfgifteType> afgifteLijst =
461                                 response.getAntwoord().getBestandenLijst().getAfgifte();
462                         for (AfgifteType t : afgiftes) {
463                             // lijst urls naar aparte logfile loggen
464                             if (t.getDigikoppelingExternalDatareferences() != null
465                                     && t.getDigikoppelingExternalDatareferences().getDataReference()
466                                             != null) {
467                                 for (DataReference dr :
468                                         t.getDigikoppelingExternalDatareferences()
469                                                 .getDataReference()) {
470                                     log.info(
471                                             "GDS2url te downloaden: "
472                                                     + dr.getTransport()
473                                                             .getLocation()
474                                                             .getSenderUrl()
475                                                             .getValue());
476                                     break;
477                                 }
478                             }
479                         }
480                         afgiftes.addAll(afgifteLijst);
481                         aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
482                         l.addLog("Aantal in antwoord: " + aantalInAntwoord);
483                         hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
484                         l.addLog("Nog meer afgiftes beschikbaar: " + hasMore);
485                     }
486 
487                 } while (morePeriods2Process);
488                 // </editor-fold> time-based ophalen
489             }
490             l.total(afgiftes.size());
491             l.addLog("Totaal aantal op te halen berichten: " + afgiftes.size());
492             l.addLog("Hoogste klant afgifte nummer: " + hoogsteKlantAfgifteNummer);
493 
494             this.config
495                     .getConfig()
496                     .put("hoogste_afgiftenummer", new ClobElement("" + hoogsteKlantAfgifteNummer));
497 
498             final String soort =
499                     this.config
500                             .getConfig()
501                             .getOrDefault("gds2_br_soort", new ClobElement("brk"))
502                             .getValue();
503             switch (soort) {
504                 case "brk2":
505                     verwerkAfgiftes(afgiftes, getCertificaatBaseURL(response.getAntwoord()), soort);
506                     break;
507                 case "brk":
508                     verwerkAfgiftes(afgiftes, getCertificaatBaseURL(response.getAntwoord()), soort);
509                     break;
510                 case "bag":
511                     // bag heeft geen contractnummer
512                     verwerkAfgiftes(afgiftes, getAnoniemBaseURL(response.getAntwoord()), soort);
513                     break;
514                 default:
515                     throw new BrmoException("Onbekende basisregistratie soort: " + soort);
516             }
517 
518         } catch (Exception e) {
519             log.error("Fout bij ophalen van GDS2 berichten", e);
520             this.config.setLastrun(new Date());
521             this.config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
522             this.config.setStatus(AutomatischProces.ProcessingStatus.ERROR);
523             String m = "Fout bij ophalen van GDS2 berichten: " + ExceptionUtils.getMessage(e);
524             if (e.getCause() != null) {
525                 m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(e);
526             }
527             l.addLog(m);
528             l.exception(e);
529         } finally {

933                                 sdf.format(new Date()),

961     private void verwerkAfgiftes(

963         int filterAlVerwerkt = 0;

975                 if (isAfgifteAlGeladen(a)) {

977                 } else if (soort.equalsIgnoreCase(BrmoFramework.BR_BAG)) {
978                     laadBagAfgifte(a, url);
979                     aantalGeladen++;
980                 } else {
981                     Bericht b = laadAfgifte(a, url, soort);
982                     if (b != null && doorsturenUrl != null) {
983                         geladenBerichtIds.add(b.getId());

986                 }

1003                 Stripersist.getEntityManager().merge(b);
1004                 Stripersist.getEntityManager().merge(this.config);
1005                 Stripersist.getEntityManager().flush();
1006                 Stripersist.getEntityManager().getTransaction().commit();
1007                 Stripersist.getEntityManager().clear();
1008                 aantalDoorgestuurd++;
1009                 l.progress(++progress);
1010             }
1011         }
1012 
1013         l.addLog("\n\n**** resultaat ****\n");
1014         l.addLog("Aantal afgiftes die al waren verwerkt: " + filterAlVerwerkt);
1015         l.addLog("\nAantal afgiftes geladen: " + aantalGeladen);
1016         l.addLog("\nAantal afgiftes doorgestuurd: " + aantalDoorgestuurd + "\n");
1017 
1018         l.addLog(
1019                 String.format(
1020                         "Het GDS2 ophalen proces met ID %d is afgerond op %tc",
1021                         config.getId(), Calendar.getInstance()));
1022         this.config.updateSamenvattingEnLogfile(
1023                 String.format(
1024                         "Het GDS2 ophalen proces, gestart op %tc, is afgerond op %tc. "
1025                                 + LOG_NEWLINE
1026                                 + "Aantal afgiftes die al waren verwerkt: %d"
1027                                 + LOG_NEWLINE
1028                                 + "Aantal afgiftes geladen: %d"
1029                                 + LOG_NEWLINE
1030                                 + "Aantal afgiftes doorgestuurd: %d"
1031                                 + LOG_NEWLINE
1032                                 + "Hoogste klantafgiftenummer: %s",
1033                         this.config.getLastrun(),
1034                         Calendar.getInstance(),
1035                         filterAlVerwerkt,
1036                         aantalGeladen,
1037                         aantalDoorgestuurd,
1038                         ClobElement.nullSafeGet(
1039                                 this.config.getConfig().get("hoogste_afgiftenummer"))));
1040 
1041         this.config.setStatus(AutomatischProces.ProcessingStatus.WAITING);
1042         this.config.setLastrun(new Date());
1043         Stripersist.getEntityManager().merge(this.config);
1044         // om geheugen problemen te vookomen bij runs met veel downloads en berichten
1045         // een flush/commit/clear forceren
1046         Stripersist.getEntityManager().flush();
1047         Stripersist.getEntityManager().getTransaction().commit();
1048         Stripersist.getEntityManager().clear();
1049     }
