7   package nl.b3p.brmo.schema.mapping;
8   
9   import java.text.ParseException;
10  import java.text.SimpleDateFormat;
11  import java.util.Date;
12  
13  public class SimpleDateFormatAttributeColumnMapping extends AttributeColumnMapping {
14      public static final String PATTERN_XML_DATE = "yyyy-MM-dd";
15      public static final String PATTERN_XML_TIMESTAMP = "yyyy-MM-dd'T'HH:mm:ss";
16  
17      private final SimpleDateFormat dateFormat;
18  
19      public SimpleDateFormatAttributeColumnMapping(
20              String name,
21              String type,
22              boolean notNull,
23              boolean primaryKey,
24              String dateFormatPattern) {
25          super(name, type, notNull, primaryKey);
26          dateFormat = new SimpleDateFormat(dateFormatPattern);
27      }
28  
29      public SimpleDateFormatAttributeColumnMapping(
30              String name, String type, boolean notNull, String dateFormatPattern) {
31          this(name, type, notNull, false, dateFormatPattern);
32      }
33  
34      public SimpleDateFormatAttributeColumnMapping(
35              String name, String type, String dateFormatPattern) {
36          this(name, type, true, false, dateFormatPattern);
37      }
38  
39      @Override
40      public Object toQueryParameter(Object value) throws ParseException {
41          if (value == null) {
42              return null;
43          } else {
44              Date date = dateFormat.parse(value.toString());
45              if ("timestamp".equals(getType())) {
46                  return new java.sql.Timestamp(date.getTime());
47              } else {
48                  return new java.sql.Date(date.getTime());
49              }
50          }
51      }
52  }
