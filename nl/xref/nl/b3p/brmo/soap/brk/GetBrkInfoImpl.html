18  @WebService(
19          targetNamespace = "http://brmo.b3p.nl/brk/1.0/soap-brk",
20          wsdlLocation = "WEB-INF/wsdl/brkinfo.wsdl")
21  @HandlerChain(file = "/handler-chain.xml")
22  public class GetBrkInfoImpl {
23      private static final Log LOG = LogFactory.getLog(GetBrkInfoImpl.class);
24      /** Web service operation */
25      @WebMethod(operationName = "getBrkInfo")
26      @RequestWrapper(className = "nl.b3p.brmo.soap.brk.getBrkInfo")
27      @ResponseWrapper(className = "nl.b3p.brmo.soap.brk.getBrkInfoResponse")
28      public BrkInfoResponse getBrkInfo(@WebParam(name = "request") BrkInfoRequest request)
29              throws BrkInfoException {
30  
31          Map<String, Object> searchContext = BrkInfo.createSearchContext(request);
32          if (searchContext.isEmpty()) {
33              throw new BrkInfoException("Request not valid", "minimaal één zoekterm vereist!");
34          }
35  
36          ArrayList<Long> ids = null;
37          try {
38              ids = BrkInfo.findKozIDs(searchContext);
39          } catch (Exception ex) {
40              LOG.error(ex);
41              throw new BrkInfoException("Database reported errors", ex.getLocalizedMessage());
42          }
43  
44          if (ids == null || ids.isEmpty()) {
45              throw new BrkInfoException("Database reported", "Geen resultaten");
46          }
47  
48          if (ids.size() > (Integer) searchContext.get(BrkInfo.MAXAANTALRESULTATEN)) {
49              throw new BrkInfoException(
50                      "Database reported",
51                      "Meer resultaten dan toegestaan: "
52                              + (Integer) searchContext.get(BrkInfo.MAXAANTALRESULTATEN)
53                              + ", pas uw voorwaarden aan!");
54          }
55  
56          BrkInfoResponse result = null;
57          try {
58              result = BrkInfo.createResponse(ids, searchContext);
59          } catch (Exception ex) {
60              LOG.error(ex);
61              throw new BrkInfoException("Database reported errors", ex.getLocalizedMessage());
62          }
63  
64          return result;
65      }
66  }
