276     private static StringBuilder createFullColumnsSQL() {
277         StringBuilder sql = new StringBuilder();
278         sql.append("    app_re.ka_sectie AS app_re_sectie,");
279         sql.append("    app_re.ka_perceelnummer  AS app_re_perceelnummer,");
280         sql.append("    app_re.ka_kad_gemeentecode AS app_re_kad_gemeentecode,");
281         sql.append("    app_re.ka_appartementsindex,");
282         sql.append("    kad_perceel.ka_sectie,");
283         sql.append("    kad_perceel.ka_perceelnummer,");
284         sql.append("    kad_perceel.ka_kad_gemeentecode,");
285         sql.append("    kad_perceel.ka_deelperceelnummer,");
286         sql.append("    kad_onrrnd_zk.dat_beg_geldh,");
287         sql.append("    kad_onrrnd_zk.datum_einde_geldh,");
288         sql.append("    kad_onrrnd_zk.kad_identif,");
289         sql.append("    kad_perceel.aand_soort_grootte,");
290         sql.append("    kad_onrrnd_zk.cu_aard_cultuur_onbebouwd,");
291         sql.append("    kad_onrrnd_zk.lr_bedrag,");
292         sql.append("    kad_perceel.begrenzing_perceel,");
293         sql.append("    kad_perceel.grootte_perceel,");
294         sql.append("    kad_onrrnd_zk.ks_koopjaar,");
295         sql.append("    kad_onrrnd_zk.ks_meer_onroerendgoed,");
296         sql.append("    kad_perceel.omschr_deelperceel ");
297         return sql;
298     }

300     private static StringBuilder createFromSQL() {
301         StringBuilder sql = new StringBuilder();
302         sql.append("    kad_onrrnd_zk ");
303         sql.append("LEFT OUTER JOIN ");
304         sql.append("    kad_perceel ");
305         sql.append("ON ");
306         sql.append("    ( ");
307         sql.append("        kad_onrrnd_zk.kad_identif = kad_perceel.sc_kad_identif) ");
308         sql.append("LEFT OUTER JOIN ");
309         sql.append("    app_re ");
310         sql.append("ON ");
311         sql.append("    ( ");
312         sql.append("        kad_onrrnd_zk.kad_identif = app_re.sc_kad_identif) ");
313         return sql;
314     }
315 
316     private static StringBuilder createWhereSQL() {
317         StringBuilder sql = new StringBuilder();
318         sql.append("    kad_onrrnd_zk.kad_identif = ? ");
319         return sql;
320     }

322     public static KadOnrndZkInfoResponse getRecord(Long id, Map<String, Object> searchContext)
323             throws Exception {
324 
325         DataSource ds = ConfigUtil.getDataSourceRsgb();
326         PreparedStatement stm = null;
327         Connection connRsgb = null;
328         ResultSet rs = null;
329         try {
330             connRsgb = ds.getConnection();
331             StringBuilder sql = new StringBuilder();
332             sql.append("SELECT ");
333             sql.append(createFullColumnsSQL());
334             sql.append("FROM ");
335             sql.append(createFromSQL());
336             sql.append("WHERE ");
337             sql.append(createWhereSQL());
338             LOG.trace(sql);
339             LOG.trace(id);
340             stm = connRsgb.prepareStatement(sql.toString());
341             stm.setLong(1, id);
342             rs = stm.executeQuery();
343 
344             KadOnrndZkInfoResponse koz = new KadOnrndZkInfoResponse();
345             if (rs.next()) {
346                 koz.setIdentificatie(Long.toString(rs.getLong("kad_identif")));
347                 koz.setAandSoortGrootte(rs.getString("aand_soort_grootte"));
348                 koz.setAardCultuurOnbebouwd(rs.getString("cu_aard_cultuur_onbebouwd"));
349                 koz.setBedrag(rs.getFloat("lr_bedrag"));
350                 koz.setDatumBeginGeldigheid(rs.getDate("dat_beg_geldh"));
351                 koz.setDatumEindeGeldigheid(rs.getDate("datum_einde_geldh"));
352                 koz.setKoopjaar(rs.getInt("ks_koopjaar"));
353                 // ks_meer_onroerendgoed is 'J' of 'N' string, geen boolean
354                 koz.setMeerOnroerendgoed(
355                         rs.getString("ks_meer_onroerendgoed").equalsIgnoreCase("J"));
356                 String type = "perceel";
357                 if (rs.getString("ka_appartementsindex") != null) {
358                     type = "appartement";
359                 }
360                 koz.setType(type);
361                 if (type.equalsIgnoreCase("appartement")) {
362                     koz.setGemeentecode(rs.getString("app_re_kad_gemeentecode"));
363                     koz.setPerceelnummer(rs.getString("app_re_perceelnummer"));
364                     koz.setSectie(rs.getString("app_re_sectie"));
365                     koz.setAppReVolgnummer(rs.getString("ka_appartementsindex"));
366                 } else {
367                     koz.setGemeentecode(rs.getString("ka_kad_gemeentecode"));
368                     koz.setPerceelnummer(rs.getString("ka_perceelnummer"));
369                     koz.setSectie(rs.getString("ka_sectie"));
370                     //                koz.setBegrenzingPerceel(rs.getString("begrenzing_perceel"));
371                     koz.setGroottePerceel(rs.getFloat("grootte_perceel"));
372                     koz.setOmschr_deelperceel(rs.getString("omschr_deelperceel"));
373                 }
374 
375                 koz.setRechten(RechtenResponse.getRechtenByKoz(id, searchContext));
376                 Boolean at = (Boolean) searchContext.get(BrkInfo.ADRESSENTOEVOEGEN);
377                 if (at != null && at) {
378                     koz.setAdressen(AdressenResponse.getAdressenByKoz(id));
379                 }
380                 //            koz.setRelaties(null);
381 
382             } else {
383                 return null;
384             }
385             return koz;
386 
387         } finally {
388             if (rs != null) {
389                 rs.close();
390             }
391             if (stm != null) {
392                 stm.close();
393             }
394             if (connRsgb != null) {
395                 connRsgb.close();
396             }
397         }
398     }
