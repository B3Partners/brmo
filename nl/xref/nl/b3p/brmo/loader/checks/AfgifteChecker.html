17  package nl.b3p.brmo.loader.checks;
18  
19  import nl.b3p.brmo.loader.StagingProxy;
20  import nl.b3p.brmo.loader.entity.Bericht;
21  import nl.b3p.brmo.loader.entity.LaadProces;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  
26  import java.io.File;
27  import java.io.FileInputStream;
28  import java.io.FileNotFoundException;
29  import java.io.IOException;
30  import java.io.InputStream;
31  import java.sql.SQLException;
32  import java.util.List;
33  import java.util.Map;
34  
35  /** @author Meine Toonen */
36  public class AfgifteChecker {
37      private static final Log log = LogFactory.getLog(AfgifteChecker.class);
38  
39      private List<Afgifte> afgiftes;
40      private StagingProxy staging;
41  
42      public void init(String input, StagingProxy staging) throws IOException {
43          try (FileInputStream fin = new FileInputStream(new File(input))) {
44              init(fin, staging);
45          }
46      }
47  
48      public void init(InputStream input, StagingProxy staging) throws IOException {
49          this.staging = staging;
50          // parse csv/excel
51          AfgiftelijstParser ap = new AfgiftelijstParser();
52          afgiftes = ap.parse(input);
53      }
54  
55      public void check() {
56          afgiftes.forEach(
57                  (afgifte) -> {
58                      check(afgifte);
59                  });
60      }
61  
62      /**
63       * afgifte controle.
64       *
65       * @param afgifte de te controleren afgifte
66       */
67      private void check(Afgifte afgifte) {
68          try {
69              LaadProces lp = staging.getLaadProcesByRestoredFilename(afgifte.getBestandsnaam());
70              // kijk of afgifte bestaat in staging
71              if (lp != null) {
72                  afgifte.setFoundInStaging(true);
73                  processFoundLaadprocess(afgifte, lp);
74              }
75          } catch (SQLException ex) {
76              log.error(
77                      "Error querying staging for laadproces for afgifte: " + afgifte.toString(), ex);
78          }
79      }
80      /**
81       * zoek de status van alle bijbehorende berichten op en leg die vast in de afgifte.
82       *
83       * @param afgifte de te controleren afgifte
84       * @param lp het laadproces waar de berichten bij zoeken
85       * @throws SQLException if any
86       */
87      private void processFoundLaadprocess(Afgifte afgifte, LaadProces lp) throws SQLException {
88          List<Bericht> berichten = staging.getBerichtByLaadProces(lp);
89          Map<Bericht.STATUS, Integer> counts = afgifte.getStatussen();
90          berichten.stream()
91                  .map(
92                          (bericht) -> {
93                              if (!counts.containsKey(bericht.getStatus())) {
94                                  counts.put(bericht.getStatus(), 0);
95                              }
96                              return bericht;
97                          })
98                  .forEachOrdered(
99                          (bericht) -> {
100                             counts.put(bericht.getStatus(), counts.get(bericht.getStatus()) + 1);
101                         });
102     }
103 
104     public File getResults(String input, String f) throws FileNotFoundException {
105         return getResults(input, new File(f));
106     }
107 
108     public File getResults(String input, File f) throws FileNotFoundException {
109         AfgiftelijstReport reporter = new AfgiftelijstReport();
110 
111         reporter.createReport(afgiftes, input, f);
112         return f;
113     }
114 }
