8   package nl.b3p.brmo.bgt.loader;
9   
10  import static org.junit.jupiter.api.Assertions.*;
11  
12  import org.junit.jupiter.api.Test;
13  
14  import java.time.Duration;
15  
16  class FormatTimeTest {
17      @Test
18      void seconds() {
19          assertEquals("5s", Utils.formatDuration(Duration.ofSeconds(5)));
20          assertEquals("12s", Utils.formatDuration(Duration.ofSeconds(12)));
21      }
22  
23      @Test
24      void minutes() {
25          assertEquals("5m  0s", Utils.formatDuration(Duration.ofMinutes(5)));
26          assertEquals("43m  0s", Utils.formatDuration(Duration.ofMinutes(43)));
27          assertEquals("43m 12s", Utils.formatDuration(Duration.ofMinutes(43).plusSeconds(12)));
28      }
29  
30      @Test
31      void hours() {
32          assertEquals("5h  0m  0s", Utils.formatDuration(Duration.ofHours(5)));
33          assertEquals("5h 32m  0s", Utils.formatDuration(Duration.ofHours(5).plusMinutes(32)));
34          assertEquals(
35                  "5h 32m  4s",
36                  Utils.formatDuration(Duration.ofHours(5).plusMinutes(32).plusSeconds(4)));
37      }
38  
39      @Test
40      void days() {
41          assertEquals("5d 0s", Utils.formatDuration(Duration.ofDays(5)));
42          assertEquals("5d 12s", Utils.formatDuration(Duration.ofDays(5).plusSeconds(12)));
43          assertEquals(
44                  "5d 2m 12s",
45                  Utils.formatDuration(Duration.ofDays(5).plusMinutes(2).plusSeconds(12)));
46          assertEquals(
47                  "5d 3h  2m 12s",
48                  Utils.formatDuration(
49                          Duration.ofDays(5).plusHours(3).plusMinutes(2).plusSeconds(12)));
50      }
51  }
