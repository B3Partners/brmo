61      /**

64       * @return een test data {@code Object[]} met daarin {@code {"bestandNaam", aantalBerichten,

67      public static Collection<Object[]> testdata() {
68          return Arrays.asList(
69                  new Object[][] {

71                      {"/GH-280/b1.xml", 1, 1, "2016-03-08T16:00:09.000023"},

76      @BeforeEach
77      @Override
78      public void setUp() throws Exception {
79          BasicDataSource dsStaging = new BasicDataSource();
80          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
81          dsStaging.setUsername(params.getProperty("staging.user"));
82          dsStaging.setPassword(params.getProperty("staging.passwd"));
83          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
84          staging = new DatabaseDataSourceConnection(dsStaging);
85  
86          BasicDataSource dsRsgb = new BasicDataSource();
87          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
88          dsRsgb.setUsername(params.getProperty("rsgb.user"));
89          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
90          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
91          rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
92  
93          brmo = new BrmoFramework(dsStaging, dsRsgb, null);
94          brmo.setOrderBerichten(true);
95  
96          if (this.isOracle) {
97              staging =
98                      new DatabaseConnection(
99                              OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
100                             params.getProperty("staging.user").toUpperCase());
101             staging.getConfig()
102                     .setProperty(
103                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
104                             new Oracle10DataTypeFactory());
105             staging.getConfig()
106                     .setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
107 
108             rsgb =
109                     new DatabaseConnection(
110                             OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
111                             params.getProperty("rsgb.user").toUpperCase());
112             rsgb.getConfig()
113                     .setProperty(
114                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
115                             new Oracle10DataTypeFactory());
116             rsgb.getConfig()
117                     .setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
118         } else if (this.isPostgis) {
119             staging.getConfig()
120                     .setProperty(
121                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
122                             new PostgresqlDataTypeFactory());
123             rsgb.getConfig()
124                     .setProperty(
125                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
126                             new PostgresqlDataTypeFactory());
127         }
128 
129         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
130         fxdb.setCaseSensitiveTableNames(false);
131         // lege bericht en laadproces tabellen
132         IDataSet stagingDataSet =
133                 fxdb.build(
134                         new FileInputStream(
135                                 new File(
136                                         BagBerichtIntegrationTest.class
137                                                 .getResource("/staging-empty-flat.xml")
138                                                 .toURI())));
139 
140         sequential.lock();
141 
142         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
143         assumeTrue(
144                 0L == brmo.getCountBerichten(BR_BAG, "STAGING_OK"), "Er zijn STAGING_OK berichten");
145         assumeTrue(
146                 0L == brmo.getCountLaadProcessen(BR_BAG, "STAGING_OK"),
147                 "Er zijn STAGING_OK laadprocessen");
148     }

150     @AfterEach
151     public void cleanup() throws Exception {
152         brmo.closeBrmoFramework();
153 
154         CleanUtil.cleanSTAGING(staging, false);
155         staging.close();
156 
157         CleanUtil.cleanRSGB_BAG(rsgb, true);
158         rsgb.close();
159 
160         try {
161             sequential.unlock();
162         } catch (IllegalMonitorStateException e) {
163             // in geval van niet waar gemaakte assumptions
164             LOG.debug(
165                     "unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
166         }
167     }

169     @Test
170     public void loadInStaging() throws Exception {
171         Iterator<Object[]> i = testdata().iterator();
172         Object[] data = i.next();
173         String bestandNaam = (String) data[0];
174         int aantalBerichten = (int) data[1];
175         int aantalProcessen = (int) data[2];
176         String timestamp = (String) data[3];
177 
178         brmo.loadFromFile(
179                 BR_BAG, BagBerichtIntegrationTest.class.getResource(bestandNaam).getFile(), null);
180         assertEquals(
181                 aantalBerichten,
182                 brmo.getCountBerichten(BR_BAG, "STAGING_OK"),
183                 "Aantal berichten is niet als verwacht");
184         assertEquals(
185                 aantalProcessen,
186                 brmo.getCountLaadProcessen(BR_BAG, "STAGING_OK"),
187                 "Aantal laadprocessen is niet als verwacht");
188 
189         ITable bericht = staging.createDataSet().getTable("bericht");
190         assertEquals(
191                 LocalDateTime.parse(timestamp)
192                         .format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss.S")),
193                 bericht.getValue(0, "datum").toString(),
194                 "Datum komt niet overeen");
195     }

197     /**
198      * laad de berichten in de staging db, verwerk vervolgens naar rsgb.
199      *
200      * @throws Exception if any
201      */
202     @Test
203     public void loadAll() throws Exception {
204         int _aantalBerichten = 0;
205         int _aantalProcessen = 0;
206 
207         Iterator<Object[]> i = testdata().iterator();
208         LOG.debug("Aantal te laden testdata berichten: " + testdata().size());
209         while (i.hasNext()) {
210             Object[] data = i.next();
211             // LOG.debug("bestandNaam:     " + data[0]);
212             // LOG.debug("aantalBerichten: " + data[1]);
213             // LOG.debug("aantalProcessen: " + data[2]);
214             // LOG.debug("timestamp:       " + data[3]);
215             brmo.loadFromFile(
216                     BR_BAG,
217                     BagBerichtIntegrationTest.class.getResource(data[0].toString()).getFile(),
218                     null);
219             _aantalBerichten += (int) data[1];
220             _aantalProcessen += (int) data[2];
221         }
222 
223         assertEquals(
224                 _aantalBerichten,
225                 brmo.getCountBerichten(BR_BAG, "STAGING_OK"),
226                 "Aantal berichten is niet als verwacht");
227         assertEquals(
228                 _aantalProcessen,
229                 brmo.getCountLaadProcessen(BR_BAG, "STAGING_OK"),
230                 "Aantal laadprocessen is niet als verwacht");
231 
232         ITable bericht = staging.createDataSet().getTable("bericht");
233         assertEquals(2, bericht.getRowCount(), "Het aantal bericht");
234         Date datum_b1 = (Date) bericht.getValue(0, "datum");
235         Date datum_b2 = (Date) bericht.getValue(1, "datum");
236 
237         Thread t = brmo.toRsgb();
238         t.join();
239 
240         ITable pand = rsgb.createDataSet().getTable("pand");
241         assertEquals(1, pand.getRowCount(), "Het aantal panden klopt niet");
242         assertEquals(
243                 "0613100000136918", pand.getValue(0, "identif"), "Pand identificatie klopt niet");
244 
245         ITable pand_archief = rsgb.createDataSet().getTable("pand_archief");
246         assertEquals(1, pand_archief.getRowCount(), "Het aantal archief panden");
247         assertEquals(
248                 pand_archief.getValue(0, "datum_einde_geldh"),
249                 pand_archief.getValue(0, "dat_beg_geldh"),
250                 "Datums komen niet overeen");
251 
252         ITable brondocument = rsgb.createDataSet().getTable("brondocument");
253         assertEquals(1, brondocument.getRowCount(), "Het aantal brondocumenten klopt niet");
254         assertEquals(
255                 "VA1071182",
256                 brondocument.getValue(0, "identificatie"),
257                 "brondocumentnummer klopt niet");
258 
259         assertEquals(
260                 0,
261                 brmo.getCountBerichten(BR_BAG, "STAGING_OK"),
262                 "Aantal verwerkte berichten is niet als verwacht");
263 
264         ITable bericht2 = staging.createDataSet().getTable("bericht");
265         assertEquals(
266                 datum_b1,
267                 bericht2.getValue(0, "datum"),
268                 "Datum voorafgaan en na afloop van transformatie komt niet overeen");
269         assertEquals(
270                 datum_b2,
271                 bericht2.getValue(1, "datum"),
272                 "Datum voorafgaan en na afloop van transformatie komt niet overeen");
273     }
