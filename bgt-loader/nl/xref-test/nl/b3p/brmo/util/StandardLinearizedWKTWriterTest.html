7   package nl.b3p.brmo.util;
8   
9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  
11  import org.junit.jupiter.api.Test;
12  import org.locationtech.jts.geom.Coordinate;
13  import org.locationtech.jts.geom.GeometryFactory;
14  
15  class StandardLinearizedWKTWriterTest {
16      StandardLinearizedWKTWriter writer = new StandardLinearizedWKTWriter();
17  
18      @Test
19      void write() {
20          GeometryFactory gf = new GeometryFactory();
21          assertEquals("POINT (0 0)", writer.write(gf.createPoint(new Coordinate(0, 0))));
22          assertEquals(
23                  "LINESTRING (0 0, 1 1, 0 0)",
24                  writer.write(
25                          gf.createLinearRing(
26                                  new Coordinate[] {
27                                      new Coordinate(0, 0),
28                                      new Coordinate(1, 1),
29                                      new Coordinate(0, 0),
30                                  })));
31      }
32  }
