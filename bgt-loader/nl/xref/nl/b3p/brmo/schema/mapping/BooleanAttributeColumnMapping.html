7   package nl.b3p.brmo.schema.mapping;
8   
9   public class BooleanAttributeColumnMapping extends AttributeColumnMapping {
10      public BooleanAttributeColumnMapping(String name, boolean notNull) {
11          super(name, "boolean", notNull, false);
12      }
13  
14      public BooleanAttributeColumnMapping(String name) {
15          this(name, true);
16      }
17  
18      @Override
19      public Object toQueryParameter(Object value) {
20          if (value == null) {
21              return null;
22          } else {
23              return Boolean.parseBoolean(value.toString());
24          }
25      }
26  }
