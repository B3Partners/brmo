56      private static void setDefaultDatabaseProperties() {
57          DatabaseOptions defaultDbOptions = new DatabaseOptions();
58          Stream.of(new String[][] {
59                  {"db.connectionString", defaultDbOptions.getConnectionString()},
60                  {"db.user", defaultDbOptions.getUser()},
61                  {"db.password", defaultDbOptions.getPassword()},
62          }).forEach(property -> {
63              String value = System.getProperty(property[0]);
64              if (value == null) {
65                  System.setProperty(property[0], property[1]);
66              }
67          });
68      }
69  
70      protected static DatabaseOptions getTestDatabaseOptions() {
71          // Get database connection options from test profile, but set defaults if no Maven profile specified
72          // With the defaults also set as properties these can be used for enabling/disabling tests for certain databases
73          setDefaultDatabaseProperties();
74          DatabaseOptions dbOptions = new DatabaseOptions();
75          dbOptions.setConnectionString(System.getProperty("db.connectionString"));
76          dbOptions.setUser(System.getProperty("db.user"));
77          dbOptions.setPassword(System.getProperty("db.password"));
78          return dbOptions;
79      }
80  
81      private static IDatabaseTester getDatabaseTester(DatabaseOptions dbOptions, String schema) throws ClassNotFoundException {
82          BGTDatabase db = new BGTDatabase(dbOptions);
83          return new JdbcDatabaseTester(db.getDialect().getDriverClass(), dbOptions.getConnectionString(), dbOptions.getUser(), dbOptions.getPassword(), schema);
84      }
85  
86      public static String getSchema(BGTDatabase db, DatabaseOptions dbOptions) {
87          if (db.getDialect() instanceof OracleDialect) {
88              // If we don't set the schema to the user the DatabaseMetaData.getTables() call without a schema filter will take 5 minutes
89              return dbOptions.getUser().toUpperCase();
90          }
91          return null;
92      }
93  
94      @BeforeEach
95      void setUp(TestInfo info) throws Exception {
96          dbOptions = getTestDatabaseOptions();
97          db = new BGTDatabase(dbOptions);
98          dbTest = getDatabaseTester(dbOptions, getSchema(db, dbOptions));
99          dbTestConnection = dbTest.getConnection();
100 
101         IDataTypeFactory dtf;
102         if (db.getDialect() instanceof PostGISDialect) {
103             dtf = new PostgresqlDataTypeFactory();
104             datasetFileSuffix = "_postgresql.xml";
105         } else if (db.getDialect() instanceof OracleDialect) {
106             dtf = new Oracle10DataTypeFactory();
107             datasetFileSuffix = "_oracle.xml";
108         } else {
109             throw new IllegalArgumentException();
110         }
111         dbTestConnection.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, dtf);
112         dbTestConnection.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
113 
114         if (info.getTestMethod().isPresent()) {
115             if(info.getTestMethod().get().getAnnotation(SkipDropTables.class) == null) {
116                 try {
117                     dropTables(dbTestConnection.getConnection(), getSchema(db, dbOptions), db.getDialect() instanceof OracleDialect);
118                 } catch(Exception e) {
119                     LOG.error("Exception dropping tables before test", e);
120                 }
121             }
122         }
123     }
124 
125     @AfterAll
126     static void dropTablesAfterAllTests() throws SQLException, ClassNotFoundException {
127         DatabaseOptions dbOptions = getTestDatabaseOptions();
128         try (BGTDatabase db = new BGTDatabase(dbOptions)) {
129             dropTables(db.getConnection(), getSchema(db, dbOptions), db.getDialect() instanceof OracleDialect);
130         }
131     }

133     protected void assertDataSetEquals(String tables, String dataSetXmlFiles) throws DatabaseUnitException, SQLException, IOException {
134         IDataSet actualDataSet = dbTestConnection.createDataSet(tables.split(","));
135         if (System.getProperty("db.writeActualDataSet") != null) {
136             write(actualDataSet);
137         }
138         List<IDataSet> dataSets = new ArrayList<>();
139         for(String dataSetXmlFile: dataSetXmlFiles.split(",")) {
140             dataSets.add(new XmlDataSet(BGTTestFiles.getTestInputStream("expected/" + dataSetXmlFile + datasetFileSuffix)));
141         }
142         IDataSet expectedDataSet = new CompositeDataSet(dataSets.toArray(new IDataSet[]{}));
143         Assertion.assertEquals(expectedDataSet, actualDataSet);
144     }
145 
146     protected void assertMinRowCounts(Object[][] featureTypesMinRowCounts) throws SQLException {
147         for(Object[] featureType: featureTypesMinRowCounts) {
148             String table = (String) featureType[0];
149             int minRowCount = (int) featureType[1];
150             int actualRowCount = dbTestConnection.getRowCount(table);
151             assertTrue(actualRowCount >= minRowCount,
152                     String.format("Expected a minimum row count of %d for table %s, actual %d",
153                             minRowCount,
154                             table,
155                             actualRowCount)
156             );
157         }
158     }

160     public static void write(IDataSet dataSet) throws DataSetException, IOException {
161         XmlDataSet.write(dataSet, System.out);
162     }

169     private static void dropTables(Connection connection, String schema, boolean isOracle) throws SQLException {
170         for (String tableName: Stream.concat(
171                 BGTSchema.getInstance().getAllObjectTypes().map(objectType -> BGTSchemaMapper.getInstance().getTableNameForObjectType(objectType, "")),
172                 Stream.of(BGTSchemaMapper.METADATA_TABLE_NAME)).collect(Collectors.toList())) {
173             if (isOracle) {
174                 tableName = tableName.toUpperCase();
175             }
176             if(new DefaultMetadataHandler().tableExists(connection.getMetaData(), schema, tableName)) {
177                 try {
178                     LOG.trace("Drop table: " + tableName);
179                     new LoggingQueryRunner().update(connection,"drop table " + tableName);
180                 } catch (SQLException se) {
181                     LOG.warn("Exception dropping table " + tableName + ": " + se.getLocalizedMessage());
182                 }
183             }
184         }
185         if (isOracle) {
186             new LoggingQueryRunner().update(connection,"delete from user_sdo_geom_metadata");
187         }
188     }
