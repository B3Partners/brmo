31  public class LoadCommandIntegrationTest extends CommandLineTestBase {
32  
33      @ParameterizedTest(name="[{index}] {0}")
34      @MethodSource
35      void load(String input, String tables, String dataSetFile) throws DatabaseUnitException, SQLException, IOException {
36          String file = getTestFile(input).getAbsolutePath();
37          run("load", "--include-history", file);
38          assertDataSetEquals(tables, dataSetFile);
39      }
40  
41      private static Stream<Arguments> load() {
42          return Stream.of(
43                  Arguments.of("bgt_wijk_curve.gml", "wijk", "bgt_wijk_curve"),
44                  Arguments.of("bgt_openbareruimtelabel.gml", "openbareruimtelabel", "bgt_openbareruimtelabel")
45          );
46      }
47  
48      @Test
49      void loadZipFromFile() throws SQLException {
50          String file = getTestFile("extract.zip").getAbsolutePath();
51          run("load", "--include-history", "--feature-types=all,plaatsbepalingspunt", file);
52          assertExtractZipRowCounts(true);
53      }
54  
55      @Test
56      void loadZipFromFileWithoutHistory() throws SQLException {
57          String file = getTestFile("extract.zip").getAbsolutePath();
58          run("load", "--feature-types=all,plaatsbepalingspunt", file);
59          assertExtractZipRowCounts(false);
60      }
61  
62      @Test
63      @EnabledIfSystemProperty(named = "db.connectionString", matches = ".*postgresql.*")
64      void postgreSQLNoCopy() throws SQLException {
65          String file = getTestFile("extract.zip").getAbsolutePath();
66          run("load", "--no-pg-copy", "--include-history", "--feature-types=all,plaatsbepalingspunt", file);
67          assertExtractZipRowCounts(true);
68      }
69  
70      @Test
71      void loadZipNoMultithreading() throws SQLException {
72          String file = getTestFile("extract.zip").getAbsolutePath();
73          run("load", "--no-multithreading", "--include-history", "--feature-types=all,plaatsbepalingspunt", file);
74          assertExtractZipRowCounts(true);
75      }
76  
77      @Test
78      void loadZipFromHttp() throws SQLException, IOException {
79          mockWebServer.enqueue(new MockResponse()
80                  .setBody(new Buffer().readFrom(getTestInputStream("extract.zip")))
81                  .setResponseCode(200)
82          );
83          String url = mockWebServer.url("/file.zip").toString();
84          // MockWebServer does not support partial content
85          run("load", "--no-http-zip-random-access", "--include-history", "--feature-types=all,plaatsbepalingspunt", url);
86          assertExtractZipRowCounts(true);
87      }
88  
89      @Test
90      @DisabledOnOs(OS.WINDOWS) // Does not have NL overheid SSL root certificate
91      void loadZipUsingHttpRandomAccessPDOKOnlineTest() throws SQLException {
92          String url = "https://api.pdok.nl/lv/bgt/download/v1_0/delta/predefined/bgt-citygml-nl-delta.zip";
93          Object[][] featureTypes = new Object[][] {
94                  {"openbareruimte", 9526},
95                  {"waterinrichtingselement", 64146},
96          };
97          String featureTypesString = Stream.of(featureTypes).map(e -> (String)e[0]).collect(Collectors.joining(","));
98          run("load", "--http-zip-random-access", "--debug-http-seeks", "--include-history", "--feature-types=" + featureTypesString, url);
99          assertMinRowCounts(featureTypes);
100     }
101 
102     private void assertExtractZipRowCounts(boolean withHistory) throws SQLException {
103         // Contents are compared in DownloadCommandMockIntegrationTest, here only compare row counts
104         for(BGTTestFiles.BGTRowCount tableRowCount: extractRowCounts) {
105             Integer expectedRowCount = withHistory ? tableRowCount.rowCountWithHistory : tableRowCount.rowCountWithoutHistory;
106             Integer actualRowCount = dbTestConnection.getRowCount(tableRowCount.table);
107             assertEquals(expectedRowCount, actualRowCount, "Expected row count for table %s" + tableRowCount.table);
108         }
109     }
