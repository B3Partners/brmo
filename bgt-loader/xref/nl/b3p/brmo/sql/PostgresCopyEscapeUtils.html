7   package nl.b3p.brmo.sql;
8   
9   import org.apache.commons.text.StringEscapeUtils;
10  import org.apache.commons.text.translate.CharSequenceTranslator;
11  
12  import java.io.IOException;
13  import java.io.Writer;
14  
15  public class PostgresCopyEscapeUtils {
16      /**
17       * Escapes column values for use in the text-based PostgreSQL COPY format with the default column and row delimiters,
18       * see https://www.postgresql.org/docs/current/sql-copy.html
19       */
20      public static final CharSequenceTranslator ESCAPE = new CharSequenceTranslator() {
21          @Override
22          public int translate(CharSequence charSequence, int start, Writer writer) throws IOException {
23              for(int i = start; i < charSequence.length(); i++) {
24                  char c = charSequence.charAt(i);
25                  String subst = null;
26                  if (c == '\\') {
27                      subst = "\\\\";
28                  } else if (c == '\t') {
29                      subst = "\\t";
30                  } else if (c == '\n') {
31                      subst = "\\n";
32                  }
33                  if (subst != null) {
34                      if (i > start) {
35                          writer.append(charSequence, start, i);
36                      }
37                      writer.append(subst);
38                      return i - start + 1;
39                  }
40              }
41              writer.append(charSequence, start, charSequence.length());
42              return charSequence.length() - start;
43          }
44      };
45  
46      public static StringEscapeUtils.Builder builder() {
47          return StringEscapeUtils.builder(ESCAPE);
48      }
49  }
