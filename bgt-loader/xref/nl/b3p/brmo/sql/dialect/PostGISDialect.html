7   package nl.b3p.brmo.sql.dialect;
8   
9   import nl.b3p.brmo.util.StandardLinearizedWKTWriter;
10  import org.geotools.geometry.jts.WKTWriter2;
11  import org.locationtech.jts.geom.Geometry;
12  import org.postgresql.util.PGobject;
13  
14  import java.sql.Connection;
15  import java.sql.SQLException;
16  
17  public class PostGISDialect implements SQLDialect {
18      private final StandardLinearizedWKTWriter wktWriter = new StandardLinearizedWKTWriter();
19      private final WKTWriter2 wktWriter2 = new WKTWriter2();
20  
21      @Override
22      public String getDriverClass() {
23          return "org.postgresql.Driver";
24      }
25  
26      @Override
27      public Object getGeometryParameter(Connection c, Geometry geometry, boolean linearizeCurves) throws SQLException {
28          if(geometry == null) {
29              return null;
30          } else {
31              String ewkt = getEWkt(geometry, linearizeCurves);
32              PGobject object = new PGobject();
33              object.setType("geometry");
34              object.setValue(ewkt);
35              return object;
36          }
37      }
38  
39      public String getEWkt(Geometry geometry, boolean linearizeCurves) {
40          String wkt = linearizeCurves ? wktWriter.write(geometry) : wktWriter2.write(geometry);
41          return "SRID=" + geometry.getSRID() + ";" + wkt;
42      }
43  
44      @Override
45      public String getCreateGeometryIndexSQL(String tableName, String geometryColumn, String type) {
46          return String.format("create index idx_%s_%s on %s using gist (%s)",
47              tableName, geometryColumn, tableName, geometryColumn);
48      }
49  
50      @Override
51      public int getDefaultOptimalBatchSize() {
52          return 2500;
53      }
54  }
