7   package nl.b3p.brmo.sql.dialect;
8   
9   import java.sql.Connection;
10  import java.sql.SQLException;
11  import java.util.regex.Matcher;
12  import java.util.regex.Pattern;
13  import nl.b3p.brmo.util.StandardLinearizedWKTWriter;
14  import org.geotools.geometry.jts.WKTWriter2;
15  import org.locationtech.jts.geom.Geometry;
16  import org.postgresql.util.PGobject;
17  
18  public class PostGISDialect implements SQLDialect {
19    private final StandardLinearizedWKTWriter wktWriter = new StandardLinearizedWKTWriter();
20    private final WKTWriter2 wktWriter2 = new WKTWriter2();
21  
22    @Override
23    public String getDriverClass() {
24      return "org.postgresql.Driver";
25    }
26  
27    @Override
28    public String getType(String type) {
29      Matcher sequence = Pattern.compile("sequence\\((.+)\\)").matcher(type);
30      if (sequence.matches()) {
31        return "integer default nextval('" + sequence.group(1) + "')";
32      }
33      return type;
34    }
35  
36    @Override
37    public Object getGeometryParameter(Connection c, Geometry geometry, boolean linearizeCurves)
38        throws SQLException {
39      if (geometry == null) {
40        return null;
41      } else {
42        String ewkt = getEWkt(geometry, linearizeCurves);
43        PGobject object = new PGobject();
44        object.setType("geometry");
45        object.setValue(ewkt);
46        return object;
47      }
48    }
49  
50    public String getEWkt(Geometry geometry, boolean linearizeCurves) {
51      String wkt = linearizeCurves ? wktWriter.write(geometry) : wktWriter2.write(geometry);
52      return "SRID=" + geometry.getSRID() + ";" + wkt;
53    }
54  
55    @Override
56    public String getCreateGeometryIndexSQL(String tableName, String geometryColumn, String type) {
57      return String.format(
58          "create index idx_%s_%s on %s using gist (%s)",
59          tableName, geometryColumn, tableName, geometryColumn);
60    }
61  
62    @Override
63    public int getDefaultOptimalBatchSize() {
64      return 2500;
65    }
66  }
