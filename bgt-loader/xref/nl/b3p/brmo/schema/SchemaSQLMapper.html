26  public abstract class SchemaSQLMapper {
27      private final Schema schema;
28  
29      protected final Map<String,String> objectTypeNameToTableName = new HashMap<>();
30  
31      private static final Set<String> reservedWords = Stream.of(new String[]{
32              "function"
33      }).collect(Collectors.toSet());
34  
35      public SchemaSQLMapper(Schema schema) {
36          this.schema = schema;
37          schema.getAllObjectTypes()
38                  .map(ObjectType::getName)
39                  .forEach(name -> objectTypeNameToTableName.put(name, name.toLowerCase()));
40      }
41  
42      public Schema getSchema() {
43          return schema;
44      }
45  
46      public abstract String getMetadataTableName();
47  
48      public String getTableNameForObjectType(ObjectType objectType, String tablePrefix) {
49          return tablePrefix + objectTypeNameToTableName.get(objectType.getName());
50      }
51  
52      public String getColumnNameForObjectType(ObjectType objectType, String attributeName) {
53          attributeName = attributeName.toLowerCase();
54          if (reservedWords.contains(attributeName)) {
55              attributeName = attributeName + "_";
56          }
57          String tableNameLower = getTableNameForObjectType(objectType, null).toLowerCase();
58          String attributeNameLower = attributeName.toLowerCase();
59          int i = attributeNameLower.indexOf(tableNameLower);
60          if (i != -1) {
61              attributeName = new StringBuilder(attributeName).replace(i, i + tableNameLower.length(), "").toString();
62          }
63          return attributeName.replaceAll("\\-", "_");
64      }

66      public void printSchema(SQLDialect dialect, String tablePrefix, Predicate<ObjectType> objectTypeFilter) {
67          // Sort object type names by table names
68          SortedMap<String, ObjectType> tableNamesObjectTypes = new TreeMap<>(getSchema().getAllObjectTypes()
69                  .filter(objectTypeFilter == null ? objectType -> true : objectTypeFilter)
70                  .collect(Collectors.toMap(objectType -> getTableNameForObjectType(objectType, ""), objectType -> objectType)));
71  
72          String createTable = tableNamesObjectTypes.values().stream()
73                  .flatMap(objectType -> getCreateTableStatements(objectType, dialect, tablePrefix))
74                  .collect(Collectors.joining("; \n\n"));
75          System.out.println(createTable + ";\n\n");
76  
77          String geometryMetadata = tableNamesObjectTypes.values().stream()
78                  .flatMap(objectType -> getCreateGeometryMetadataStatements(objectType, dialect, tablePrefix))
79                  .filter(sql -> sql.length() > 0)
80                  .collect(Collectors.joining(";\n"));
81          if (geometryMetadata.length() > 0) {
82              System.out.printf("-- %s\n\n", getBundleString("schema.geometry_metadata"));
83              System.out.println(geometryMetadata + ";\n");
84          }
85  
86          System.out.printf("-- %s\n\n", getBundleString("schema.loader_metadata"));
87          System.out.println(getCreateMetadataTableStatements(dialect, tablePrefix).stream().collect(Collectors.joining(";\n")) + ";\n");
88  
89          System.out.printf("-- %s %s\n\n", getBundleString("schema.primary_keys"), getBundleString("schema.after_initial_load"));
90          String primaryKeys = tableNamesObjectTypes.values().stream()
91                  .flatMap(objectType -> getCreatePrimaryKeyStatements(objectType, dialect, tablePrefix))
92                  .collect(Collectors.joining("; \n"));
93          System.out.println(primaryKeys + ";\n");
94  
95          System.out.printf("-- %s %s\n\n", getBundleString("schema.geometry_indexes"), getBundleString("schema.after_initial_load"));
96          String geometryIndexes = tableNamesObjectTypes.values().stream()
97                  .flatMap(objectType -> getCreateGeometryIndexStatements(objectType, dialect, tablePrefix))
98                  .collect(Collectors.joining(";\n"));
99          System.out.println(geometryIndexes + ";\n");
100     }

102     public Stream<String> getCreateTableStatements(ObjectType objectType, SQLDialect dialect, String tablePrefix) {
103         List<String> statements = new ArrayList<>();
104         String tableName = getTableNameForObjectType(objectType, tablePrefix);
105         if (dialect.supportsDropTableIfExists()) {
106             statements.add("drop table if exists " + tableName);
107         }
108         String columns = objectType.getDirectAttributes().stream()
109                 .map(column -> String.format("  %s %s%s",
110                         getColumnNameForObjectType(objectType, column.getName()),
111                         dialect.getType(column.getType()),
112                         column.isNotNull() ? " not null" : ""))
113                 .collect(Collectors.joining(",\n"));
114         statements.add(String.format("create table %s (\n%s\n)", tableName, columns));
115         statements.addAll(objectType.getOneToManyAttributeObjectTypes().stream()
116                 .flatMap(oneToManyObjectType -> getCreateTableStatements(oneToManyObjectType, dialect, tablePrefix))
117                 .collect(Collectors.toList()));
118         return statements.stream();
119     }

121     public Stream<String> getCreatePrimaryKeyStatements(ObjectType objectType, SQLDialect dialect, String tablePrefix) {
122         return getCreatePrimaryKeyStatements(objectType, dialect, tablePrefix, true);
123     }

125     public Stream<String> getCreatePrimaryKeyStatements(ObjectType objectType, SQLDialect dialect, String tablePrefix, boolean includeOneToMany) {
126         String tableName = getTableNameForObjectType(objectType, tablePrefix);
127         String columns = objectType.getDirectAttributes().stream()
128                 .filter(AttributeColumnMapping::isPrimaryKey)
129                 .map(column -> getColumnNameForObjectType(objectType, column.getName()))
130                 .collect(Collectors.joining(", "));
131         String sql = String.format("alter table %s add constraint %s_pkey primary key(%s)",
132                 tableName, tableName, columns);
133         return Stream.concat(
134                 Stream.of(sql),
135                 includeOneToMany
136                         ? objectType.getOneToManyAttributeObjectTypes().stream().flatMap(oneToManyObjectType -> getCreatePrimaryKeyStatements(oneToManyObjectType, dialect, tablePrefix))
137                         : Stream.empty()
138         );
139     }

141     public Stream<String> getCreateGeometryMetadataStatements(ObjectType objectType, SQLDialect dialect, String tablePrefix) {
142         return Stream.concat(
143                 objectType.getGeometryAttributes().stream().map(column -> dialect.getCreateGeometryMetadataSQL(
144                         getTableNameForObjectType(objectType, tablePrefix),
145                         getColumnNameForObjectType(objectType, column.getName()),
146                         column.getType())),
147                 objectType.getOneToManyAttributeObjectTypes().stream()
148                         .flatMap(oneToManyObjectType -> getCreateGeometryMetadataStatements(oneToManyObjectType, dialect, tablePrefix))
149         );
150     }

157         return Stream.concat(

160                         getColumnNameForObjectType(objectType, column.getName()),
161                         column.getType())),
162                 includeOneToMany

164                         : Stream.empty()
165         );
166     }

168     public List<String> getCreateMetadataTableStatements(SQLDialect dialect, String tablePrefix) {
169         List<String> statements = new ArrayList<>();
170         final String tableName = getMetadataTableName();
171         if (dialect.supportsDropTableIfExists()) {
172             statements.add("drop table if exists " + tableName);
173         }
174         String sql = String.format("create table %s(\n  naam %s,\n  waarde %s,\n  primary key(naam)\n)",
175                 tableName,
176                 dialect.getType("varchar(255)"),
177                 dialect.getType("text"));
178         statements.add(sql);
179 
180         return statements;
181     }

183     /**
184      * Use for aligning fixed width output.
185      */
186     public int getMaxTableLength() {
187         return objectTypeNameToTableName.values().stream().map(String::length).reduce(0, Integer::max);
188     }
