59  @Command(name = "bgt-loader", mixinStandardHelpOptions = true, versionProvider = BGTLoaderMain.class,
60          resourceBundle = Utils.BUNDLE_NAME, subcommands = {DownloadCommand.class})
61  public class BGTLoaderMain implements IVersionProvider {
62      private static Log log;
63  
64      /**
65       * init logging.
66       *
67       * @param standAlone set to {@code false} when using in a preconfigured environment, eg. calling methods from a servlet,
68       *                   use {@code true} for commandline usage.
69       */
70      public static void configureLogging(boolean standAlone) {
71          if (standAlone) {
72              PropertyConfigurator.configure(BGTLoaderMain.class.getResourceAsStream("/bgt-loader-cli-log4.properties"));
73              log = LogFactory.getLog(BGTLoaderMain.class);
74              try {
75                  Logging.ALL.setLoggerFactory("org.geotools.util.logging.Log4JLoggerFactory");
76              } catch (ClassNotFoundException ignored) {
77              }
78          } else {
79              log = LogFactory.getLog(BGTLoaderMain.class);
80          }
81      }
82  
83      public static void main(String[] args) {
84          configureLogging(true);
85          CommandLine cmd = new CommandLine(new BGTLoaderMain())
86                  .setUsageHelpAutoWidth(true);
87          System.exit(cmd.execute(args));
88      }

90      @Command(name = "schema", sortOptions = false)
91      public int schema(
92              @Option(names="--dialect", paramLabel="<dialect>", defaultValue = "postgis") BGTDatabase.SQLDialectEnum dialectEnum,
93              @Mixin FeatureTypeSelectionOptions featureTypeSelectionOptions,
94              @Option(names="--table-prefix", defaultValue = "", hidden = true) String tablePrefix,
95              @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp) throws SQLException {
96          SQLDialect dialect = BGTDatabase.createDialect(dialectEnum);
97          // For schema generation include plaatsbepalingspunt with 'all' and 'bgt'
98          if (featureTypeSelectionOptions.featureTypes.contains("all") || featureTypeSelectionOptions.featureTypes.contains("bgt")) {
99              featureTypeSelectionOptions.getFeatureTypes().add(DeltaCustomDownloadRequest.FeaturetypesEnum.PLAATSBEPALINGSPUNT.getValue());
100         }
101         Set<String> tableNames = featureTypeSelectionOptions.getFeatureTypesList().stream()
102                 .map(DeltaCustomDownloadRequest.FeaturetypesEnum::getValue)
103                 .collect(Collectors.toSet());
104         BGTSchemaMapper bgtSchemaMapper = BGTSchemaMapper.getInstance();
105         bgtSchemaMapper.printSchema(dialect, tablePrefix, objectType ->
106                 tableNames.contains(bgtSchemaMapper.getTableNameForObjectType(objectType, ""))
107         );
108         return ExitCode.OK;
109     }

111     @Command(name = "load", sortOptions = false)
112     public int load(
113             @Mixin DatabaseOptions dbOptions,
114             @Mixin LoadOptions loadOptions,
115             @Mixin FeatureTypeSelectionOptions featureTypeSelectionOptions,
116             @Parameters(paramLabel = "<file>") String file,
117             @Mixin CLIOptions cliOptions,
118             @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp) throws Exception {
119 
120         log.info(getUserAgent());
121 
122         try(BGTDatabase db = new BGTDatabase(dbOptions)) {
123             BGTObjectTableWriter writer = db.createObjectTableWriter(loadOptions, dbOptions);
124 
125             ProgressReporter progressReporter = cliOptions.isConsoleProgressEnabled()
126                     ? new ConsoleProgressReporter()
127                     : new ProgressReporter();
128             writer.setProgressUpdater(progressReporter);
129 
130             if (loadOptions.createSchema) {
131                 db.createMetadataTable(loadOptions);
132             }
133 
134             if (file.endsWith(".zip") && (file.startsWith("http://") || file.startsWith("https://"))) {
135                 loadZipFromURI(new URI(file), writer, featureTypeSelectionOptions, loadOptions, true);
136             } else if (file.endsWith(".zip")) {
137                 loadZip(new File(file), writer, featureTypeSelectionOptions);
138             } else if (file.matches(".*\\.[xg]ml")) {
139                 loadXml(new File(file), writer);
140             } else {
141                 log.error(getMessageFormattedString("load.invalid_extension", file));
142                 return ExitCode.USAGE;
143             }
144 
145             if (writer.getProgress() == null) {
146                 log.error(getBundleString("error.no_feature_types"));
147                 return ExitCode.SOFTWARE;
148             }
149             db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
150             // Set feature types list from options, not MutatieInhoud (if input has it)...
151             // FIXME if downloaded initial extract has less object types, update will fail -- should set to only encountered
152             // feature types
153             db.setFeatureTypesEnumMetadata(featureTypeSelectionOptions.getFeatureTypesList());
154             db.setMetadataValue(Metadata.INCLUDE_HISTORY, loadOptions.includeHistory + "");
155             db.setMetadataValue(Metadata.LINEARIZE_CURVES, loadOptions.linearizeCurves + "");
156             db.setMetadataValue(Metadata.TABLE_PREFIX, loadOptions.tablePrefix);
157             BGTObjectTableWriter.BGTProgress progress = writer.getProgress();
158             if (progress.getMutatieInhoud() != null) {
159                 db.setMetadataForMutaties(progress.getMutatieInhoud());
160                 db.setMetadataValue(Metadata.GEOM_FILTER, progress.getMutatieInhoud().getGebied());
161 
162                 log.info(getMessageFormattedString("load.mutatie",
163                         progress.getMutatieInhoud().getMutatieType(),
164                         progress.getMutatieInhoud().getLeveringsId()
165                 ));
166             }
167             progressReporter.reportTotalSummary();
168             db.getConnection().commit();
169         }
170 
171         return ExitCode.OK;
172     }

174     private static boolean isBGTZipEntrySelected(String entryName, FeatureTypeSelectionOptions featureTypeSelectionOptions, boolean logSkipAsInfo) {
175         Set<DeltaCustomDownloadRequest.FeaturetypesEnum> featureTypes = featureTypeSelectionOptions.getFeatureTypesList();
176         Pattern p = Pattern.compile("bgt_(.+).[xg]ml");
177 
178         Matcher m = p.matcher(entryName);
179         if (!m.matches()) {
180             log.warn(getMessageFormattedString("load.skip_entry", entryName));
181             return false;
182         }
183         String tableName = m.group(1);
184         try {
185             DeltaCustomDownloadRequest.FeaturetypesEnum featureType = DeltaCustomDownloadRequest.FeaturetypesEnum.fromValue(tableName);
186             if (!featureTypes.contains(featureType)) {
187                 String msg = getMessageFormattedString("load.skip_unselected", tableName);
188                 if (logSkipAsInfo) {
189                     log.info(msg);
190                 } else {
191                     log.debug(msg);
192                 }
193                 return false;
194             } else {
195                 return true;
196             }
197         } catch (IllegalArgumentException e) {
198             log.warn(getMessageFormattedString("load.skip_unknown_feature_type", entryName));
199             return false;
200         }
201     }

203     public void loadZipFromURI(URI uri, BGTObjectTableWriter writer, FeatureTypeSelectionOptions featureTypeSelectionOptions, LoadOptions loadOptions, boolean showSelected) throws Exception {
204         log.info(getMessageFormattedString("download.downloading_from", uri));
205         if (loadOptions.isHttpZipRandomAccess()) {
206             loadZipFromURIUsingRandomAccess(uri, writer, featureTypeSelectionOptions, showSelected, loadOptions.isDebugHttpSeeks());
207         } else {
208             loadZipFromURIUsingStreaming(uri, writer, featureTypeSelectionOptions);
209         }
210     }

212     public void loadZipFromURIUsingRandomAccess(URI uri, BGTObjectTableWriter writer, FeatureTypeSelectionOptions featureTypeSelectionOptions, boolean showSelected, boolean debugHttpSeeks) throws Exception {
213         Instant start = Instant.now();
214 
215         // NOTE: it can happen that not all entries from a ZIP are read because of https://issues.apache.org/jira/browse/COMPRESS-584
216         // This happened with https://api.pdok.nl/lv/bgt/download/v1_0/cache/2/ebe787b3-e113-4331-ab96-edd1e9bf5aa7/bgt-citygml-nl-nopbp.zip
217 
218         try(
219                 HttpSeekableByteChannel channel = new HttpSeekableByteChannel(uri).withDebug(debugHttpSeeks);
220                 CountingSeekableByteChannel loggingChannel = new CountingSeekableByteChannel(channel);
221                 org.apache.commons.compress.archivers.zip.ZipFile zipFile = new org.apache.commons.compress.archivers.zip.ZipFile(
222                         loggingChannel,
223                         uri.toString(),
224                         "UTF8",
225                         false,
226                         true
227                 )
228         ) {
229             if (debugHttpSeeks) {
230                 System.out.println();
231             }
232             int count = 0;
233             long uncompressed = 0;
234             for (Iterator<ZipArchiveEntry> it = zipFile.getEntries().asIterator(); it.hasNext(); ) {
235                 ZipArchiveEntry entry = it.next();
236                 count++;
237                 uncompressed += entry.getSize();
238             }
239             log.info(getMessageFormattedString("download.zip.read",
240                     formatTimeSince(start),
241                     count,
242                     byteCountToDisplaySize(channel.size()),
243                     byteCountToDisplaySize(uncompressed)));
244             if (debugHttpSeeks) {
245                 log.info(getMessageFormattedString("download.zip.debug-http-seeks.entries",
246                         loggingChannel.getNonConsecutiveIops(),
247                         channel.getHttpRequestCount(),
248                         channel.getBytesRead()));
249             }
250             loggingChannel.setLoggingEnabled(false);
251 
252             ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
253             List<ZipArchiveEntry> selected = new ArrayList<>();
254             zipFile.getEntries().asIterator().forEachRemaining(entry -> {
255                 if (isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, false)) {
256                     selected.add(entry);
257                 }
258             });
259 
260             if (selected.size() > 1) {
261                 // Only report total percentage when more than one entry
262                 long totalSize = selected.stream().map(ZipArchiveEntry::getSize).reduce(0L, Long::sum);
263                 Long totalCompressedSize = selected.stream().map(ZipArchiveEntry::getCompressedSize).reduce(0L, Long::sum);
264                 progressReporter.setTotalBytes(totalSize);
265                 if (showSelected) {
266                     log.info(getMessageFormattedString("download.zip.selected",
267                             selected.size(),
268                             byteCountToDisplaySize(totalCompressedSize),
269                             byteCountToDisplaySize(totalSize)));
270                 }
271             }
272             Long[] previousEntriesBytesRead = new Long[]{0L};
273             progressReporter.setTotalBytesReadFunction(() -> previousEntriesBytesRead[0] + writer.getProgress().getBytesRead());
274 
275             for (ZipArchiveEntry entry: selected) {
276                 progressReporter.startNewFile(entry.getName(), entry.getSize());
277                 writer.write(zipFile.getInputStream(entry));
278             }
279             if (debugHttpSeeks) {
280                 log.info(getMessageFormattedString("download.zip.debug-http-seeks.totals",
281                         loggingChannel.getNonConsecutiveIops(),
282                         channel.getHttpRequestCount(),
283                         channel.getBytesRead(),
284                         byteCountToDisplaySize(channel.getBytesRead())));
285             }
286         }
287     }

289     public void loadZipFromURIUsingStreaming(URI downloadURI, BGTObjectTableWriter writer, FeatureTypeSelectionOptions featureTypeSelectionOptions) throws Exception {
290         ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
291 
292         try (InputStream input = new ResumingBGTDownloadInputStream(downloadURI, writer)) {
293             CountingInputStream countingInputStream = new CountingInputStream(input);
294             progressReporter.setTotalBytesReadFunction(countingInputStream::getByteCount);
295 
296             try (ZipInputStream zis = new ZipInputStream(countingInputStream)) {
297                 ZipEntry entry = zis.getNextEntry();
298                 while (entry != null) {
299                     if (isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, true)) {
300                         progressReporter.startNewFile(entry.getName(), null);
301                         writer.write(CloseShieldInputStream.wrap(zis));
302                     }
303                     entry = zis.getNextEntry();
304                 }
305             }
306         }
307     }

309     public void loadZip(File file, BGTObjectTableWriter writer, FeatureTypeSelectionOptions featureTypeSelectionOptions) throws Exception {
310         try(ZipFile zipFile = new ZipFile(file)) {
311             List<ZipEntry> entries = zipFile.stream()
312                     .filter(entry -> isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, false))
313                     .collect(Collectors.toList());
314 
315             ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
316             if (entries.size() > 1) {
317                 // Only report total percentage when more than one entry
318                 Long totalSize = entries.stream().map(ZipEntry::getSize).reduce(0L, Long::sum);
319                 progressReporter.setTotalBytes(totalSize);
320             }
321             Long[] previousEntriesBytesRead = new Long[]{0L};
322             progressReporter.setTotalBytesReadFunction(() -> previousEntriesBytesRead[0] + writer.getProgress().getBytesRead());
323 
324             for (ZipEntry entry: entries) {
325                 try (InputStream in = zipFile.getInputStream(entry)) {
326                     // getSize() will not return -1 because ZipFile uses random access to read the ZIP central directory
327                     loadInputStream(entry.getName(), in, entry.getSize(), writer);
328                     previousEntriesBytesRead[0] += entry.getSize();
329                 }
330             }
331             progressReporter.reportTotalSummary();
332         }
333     }

335     public void loadXml(File file, BGTObjectTableWriter writer) throws Exception {
336         try(FileInputStream in = new FileInputStream(file)) {
337             loadInputStream(file.getName(), in, file.length(), writer);
338         }
339     }

341     public void loadInputStream(String name, InputStream input, long size, BGTObjectTableWriter writer) throws Exception {
342         ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
343         progressReporter.startNewFile(name, size);
344         writer.write(input);
345     }
