7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import picocli.CommandLine.Option;
10  
11  public class LoadOptions {
12    @Option(
13        names = {"--no-create-schema"},
14        negatable = true)
15    boolean createSchema = true;
16  
17    @Option(
18        names = {"--drop-if-exists"},
19        negatable = true)
20    boolean dropIfExists;
21  
22    @Option(names = {"--include-history"})
23    boolean includeHistory = false;
24  
25    @Option(names = "--table-prefix", defaultValue = "", hidden = true)
26    String tablePrefix = "";
27  
28    @Option(names = "--max-objects", paramLabel = "<number>", hidden = true)
29    Integer maxObjects;
30  
31    @Option(names = {"--linearize-curves"})
32    boolean linearizeCurves = false;
33  
34    @Option(
35        names = {"--no-multithreading"},
36        negatable = true,
37        hidden = true)
38    boolean multithreading = true;
39  
40    @Option(names = "--no-http-zip-random-access", negatable = true, hidden = true)
41    boolean httpZipRandomAccess = true;
42  
43    @Option(names = "--debug-http-seeks", hidden = true)
44    boolean debugHttpSeeks = false;
45  
46    public boolean isCreateSchema() {
47      return createSchema;
48    }
49  
50    public void setCreateSchema(boolean createSchema) {
51      this.createSchema = createSchema;
52    }
53  
54    public boolean isDropIfExists() {
55      return dropIfExists;
56    }
57  
58    public void setDropIfExists(boolean dropIfExists) {
59      this.dropIfExists = dropIfExists;
60    }
61  
62    public boolean isIncludeHistory() {
63      return includeHistory;
64    }
65  
66    public void setIncludeHistory(boolean includeHistory) {
67      this.includeHistory = includeHistory;
68    }
69  
70    public String getTablePrefix() {
71      return tablePrefix;
72    }
73  
74    public void setTablePrefix(String tablePrefix) {
75      this.tablePrefix = tablePrefix;
76    }
77  
78    public Integer getMaxObjects() {
79      return maxObjects;
80    }
81  
82    public void setMaxObjects(Integer maxObjects) {
83      this.maxObjects = maxObjects;
84    }
85  
86    public boolean isLinearizeCurves() {
87      return linearizeCurves;
88    }
89  
90    public void setLinearizeCurves(boolean linearizeCurves) {
91      this.linearizeCurves = linearizeCurves;
92    }
93  
94    public boolean isMultithreading() {
95      return multithreading;
96    }
97  
98    public void setMultithreading(boolean multithreading) {
99      this.multithreading = multithreading;
100   }
101 
102   public boolean isHttpZipRandomAccess() {
103     return httpZipRandomAccess;
104   }
105 
106   public void setHttpZipRandomAccess(boolean httpZipRandomAccess) {
107     this.httpZipRandomAccess = httpZipRandomAccess;
108   }
109 
110   public boolean isDebugHttpSeeks() {
111     return debugHttpSeeks;
112   }
113 
114   public void setDebugHttpSeeks(boolean debugHttpSeeks) {
115     this.debugHttpSeeks = debugHttpSeeks;
116   }
