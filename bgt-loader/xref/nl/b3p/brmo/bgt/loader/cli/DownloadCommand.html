53    public static ApiClient getApiClient(URI baseUri) {
54      ApiClient client = new ApiClient();
55      if (baseUri != null) {
56        client.updateBaseUri(baseUri.toString());
57      }
58      client.setRequestInterceptor(builder -> builder.headers("User-Agent", getUserAgent()));
59      return client;
60    }

62    private BGTObjectTableWriter createWriter(
63        BGTDatabase db, DatabaseOptions dbOptions, LoadOptions loadOptions, CLIOptions cliOptions)
64        throws SQLException {
65      BGTObjectTableWriter writer = db.createObjectTableWriter(loadOptions, dbOptions);
66      ProgressReporter progressReporter;
67      if (cliOptions.isConsoleProgressEnabled()) {
68        progressReporter = new ConsoleProgressReporter();
69      } else {
70        progressReporter = new ProgressReporter();
71      }
72      writer.setProgressUpdater(progressReporter);
73      return writer;
74    }

76    @Command(name = "initial", sortOptions = false)
77    public int initial(
78        @Mixin DatabaseOptions dbOptions,
79        @Mixin LoadOptions loadOptions,
80        @Mixin ExtractSelectionOptions extractSelectionOptions,
81        @Option(names = "--no-geo-filter") boolean noGeoFilter,
82        @Option(names = "--download-service", hidden = true) URI downloadServiceURI,
83        @Mixin CLIOptions cliOptions,
84        @Option(
85                names = {"-h", "--help"},
86                usageHelp = true)
87            boolean showHelp)
88        throws Exception {
89  
90      if (extractSelectionOptions.getGeoFilterWkt() == null && !noGeoFilter) {
91        log.error(getBundleString("download.no_geo_filter"));
92        return ExitCode.USAGE;
93      }
94  
95      log.info(getUserAgent());
96  
97      try (BGTDatabase db = this.getBGTdatabase(dbOptions)) {
98        if (loadOptions.createSchema) {
99          db.createMetadataTable(loadOptions);
100       } else {
101         log.info(getBundleString("download.connect_db"));
102       }
103 
104       db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
105       db.setMetadataValue(Metadata.BRMOVERSIE, getBrmoVersion());
106       db.setFeatureTypesEnumMetadata(extractSelectionOptions.getFeatureTypesList());
107       db.setMetadataValue(Metadata.INCLUDE_HISTORY, loadOptions.includeHistory + "");
108       db.setMetadataValue(Metadata.LINEARIZE_CURVES, loadOptions.linearizeCurves + "");
109       db.setMetadataValue(Metadata.TABLE_PREFIX, loadOptions.tablePrefix);
110 
111       Instant start = null;
112       URI uri;
113       if (noGeoFilter) {
114         uri = new URI(PREDEFINED_FULL_DELTA_URI);
115       } else {
116         // Close connection while waiting for extract
117         db.close();
118         start = Instant.now(); // Record total time waiting for extract
119         uri =
120             getCustomDownloadURI(
121                 downloadServiceURI,
122                 extractSelectionOptions,
123                 new CustomDownloadProgressReporter(cliOptions.isConsoleProgressEnabled()));
124       }
125       BGTObjectTableWriter writer = createWriter(db, dbOptions, loadOptions, cliOptions);
126       loadZipFromURI(uri, db, writer, extractSelectionOptions, loadOptions, noGeoFilter, start);
127 
128       db.setMetadataValue(Metadata.DELTA_TIME_TO, null);
129       // Do not set geom filter from MutatieInhoud, a custom download without geo filter will
130       // have gebied
131       // "POLYGON ((-100000 200000, 412000 200000, 412000 712000, -100000 712000, -100000
132       // 200000))"
133       db.setMetadataValue(Metadata.GEOM_FILTER, extractSelectionOptions.getGeoFilterWkt());
134 
135       db.getConnection().commit();
136       return ExitCode.OK;
137     }
138   }

156   private static URI getCustomDownloadURI(
157       URI downloadServiceURI,
158       ExtractSelectionOptions extractSelectionOptions,
159       Consumer<CustomDownloadProgress> progressConsumer)
160       throws ApiException, InterruptedException {
161     try {
162       log.info(getBundleString("download.create"));
163       ApiClient client = getApiClient(downloadServiceURI);
164       return DownloadApiUtils.getCustomDownloadURL(
165           client, null, extractSelectionOptions, progressConsumer);
166     } catch (ApiException apiException) {
167       printApiException(apiException);
168       throw apiException;
169     }
170   }

172   @Command(name = "update", sortOptions = false)
173   public int update(
174       @Mixin DatabaseOptions dbOptions,
175       @Mixin CLIOptions cliOptions,
176       @Option(names = "--download-service", hidden = true) URI downloadServiceURI,
177       @Option(names = "--no-http-zip-random-access", negatable = true, hidden = true)
178           boolean noHttpZipRandomAccess,
179       @Option(
180               names = {"-h", "--help"},
181               usageHelp = true)
182           boolean showHelp)
183       throws Exception {
184     log.info(getUserAgent());
185 
186     ApiClient client = getApiClient(downloadServiceURI);
187 
188     log.info(getBundleString("download.connect_db"));
189     try (BGTDatabase db = getBGTdatabase(dbOptions)) {
190       String deltaId = db.getMetadata(Metadata.DELTA_ID);
191       OffsetDateTime deltaIdTimeTo = null;
192       String s = db.getMetadata(Metadata.DELTA_TIME_TO);
193       if (s != null && s.length() > 0) {
194         deltaIdTimeTo = OffsetDateTime.parse(s);
195       }
196       if (deltaId == null) {
197         log.error(getBundleString("download.no_delta_id"));
198         return ExitCode.SOFTWARE;
199       }
200       ExtractSelectionOptions extractSelectionOptions = new ExtractSelectionOptions();
201       extractSelectionOptions.setGeoFilterWkt(db.getMetadata(Metadata.GEOM_FILTER));
202       if (extractSelectionOptions.getGeoFilterWkt() != null
203           && extractSelectionOptions.getGeoFilterWkt().length() == 0) {
204         extractSelectionOptions.setGeoFilterWkt(null);
205       }
206       extractSelectionOptions.setFeatureTypes(
207           Arrays.asList(db.getMetadata(Metadata.FEATURE_TYPES).split(",")));
208       LoadOptions loadOptions = new LoadOptions();
209       loadOptions.setHttpZipRandomAccess(!noHttpZipRandomAccess);
210       loadOptions.includeHistory = Boolean.parseBoolean(db.getMetadata(Metadata.INCLUDE_HISTORY));
211       loadOptions.linearizeCurves = Boolean.parseBoolean(db.getMetadata(Metadata.LINEARIZE_CURVES));
212       loadOptions.tablePrefix =
213           Objects.requireNonNullElse(db.getMetadata(Metadata.TABLE_PREFIX), "");
214 
215       log.info(
216           getMessageFormattedString("download.current_delta_id", deltaId)
217               + ", "
218               + (deltaIdTimeTo != null
219                   ? getMessageFormattedString(
220                       "download.current_delta_time",
221                       DateTimeFormatter.ISO_INSTANT.format(deltaIdTimeTo))
222                   : getBundleString("download.current_delta_time_unknown")));
223 
224       try {
225         Instant start = Instant.now();
226         log.info(getBundleString("download.loading_deltas"));
227         // Note that the afterDeltaId parameter is useless, because the response does not
228         // distinguish between
229         // "'afterDeltaId' is the latest" and "'afterDeltaId' not found or older than 31
230         // days"
231         GetDeltasResponse response = new DeltaApi(client).getDeltas(null, 1, 100);
232 
233         // Verify no links to other page, as we expect at most 31 delta's
234         if (response.getLinks() != null && !response.getLinks().isEmpty()) {
235           throw new IllegalStateException("Did not expect links in GetDeltas response");
236         }
237 
238         int i;
239         for (i = 0; i < response.getDeltas().size(); i++) {
240           Delta d = response.getDeltas().get(i);
241           if (deltaId.equals(d.getId())) {
242             break;
243           }
244         }
245         if (i == response.getDeltas().size()) {
246           // TODO automatically do initial load depending on option
247           log.error(getBundleString("download.current_delta_not_found"));
248           return ExitCode.SOFTWARE;
249         }
250 
251         List<Delta> deltas = response.getDeltas().subList(i + 1, response.getDeltas().size());
252         if (deltas.isEmpty()) {
253           log.info(getBundleString("download.uptodate"));
254           return ExitCode.OK;
255         }
256 
257         Delta latestDelta = deltas.get(deltas.size() - 1);
258         log.info(
259             getMessageFormattedString(
260                 "download.updates_available",
261                 deltas.size(),
262                 latestDelta.getId(),
263                 latestDelta.getTimeWindow().getTo()));
264 
265         BGTObjectTableWriter writer = createWriter(db, dbOptions, loadOptions, cliOptions);
266 
267         int deltaCount = 1;
268         for (Delta delta : deltas) {
269           log.info(
270               getMessageFormattedString(
271                   "download.creating_download", deltaCount++, deltas.size(), delta.getId()));
272           URI uri =
273               DownloadApiUtils.getCustomDownloadURL(
274                   client,
275                   delta,
276                   extractSelectionOptions,
277                   new CustomDownloadProgressReporter(cliOptions.isConsoleProgressEnabled()));
278           // TODO: BGTObjectTableWriter does setAutocommit(false) and commit() after each
279           // stream for a feature type
280           // is written, maybe use one transaction for all feature types?
281           loadZipFromURI(uri, db, writer, extractSelectionOptions, loadOptions, false, start);
282           db.setMetadataValue(Metadata.DELTA_TIME_TO, delta.getTimeWindow().getTo().toString());
283           db.getConnection().commit();
284         }
285 
286         db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
287         db.getConnection().commit();
288         return ExitCode.OK;
289       } catch (ApiException apiException) {
290         printApiException(apiException);
291         throw apiException;
292       }
293     }
294   }

296   private static void printApiException(ApiException apiException) {
297     log.error(
298         String.format(
299             "API status code: %d, body: %s\n",
300             apiException.getCode(), apiException.getResponseBody()));
301   }

303   private static void loadZipFromURI(
304       URI uri,
305       BGTDatabase db,
306       BGTObjectTableWriter writer,
307       ExtractSelectionOptions extractSelectionOptions,
308       LoadOptions loadOptions,
309       boolean showSelected,
310       Instant start)
311       throws Exception {
312     Instant loadStart = Instant.now();
313     new BGTLoaderMain()
314         .loadZipFromURI(uri, writer, extractSelectionOptions, loadOptions, showSelected);
315     db.setMetadataForMutaties(writer.getProgress().getMutatieInhoud());
316     log.info(
317         getMessageFormattedString(
318                 "download.complete",
319                 getBundleString(
320                     "download.mutatietype."
321                         + writer.getProgress().getMutatieInhoud().getMutatieType()),
322                 writer.getProgress().getMutatieInhoud().getLeveringsId(),
323                 formatTimeSince(loadStart))
324             + (start == null
325                 ? ""
326                 : " "
327                     + getMessageFormattedString(
328                         "download.complete_total", formatTimeSince(start))));
329   }
