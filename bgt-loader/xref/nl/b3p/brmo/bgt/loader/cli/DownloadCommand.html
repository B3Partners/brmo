52      public static ApiClient getApiClient(URI baseUri) {
53          ApiClient client = new ApiClient();
54          if (baseUri != null) {
55              client.updateBaseUri(baseUri.toString());
56          }
57          client.setRequestInterceptor(builder -> builder.headers("User-Agent", getUserAgent()));
58          return client;
59      }

61      private BGTObjectTableWriter createWriter(BGTDatabase db, DatabaseOptions dbOptions, LoadOptions loadOptions, CLIOptions cliOptions) throws SQLException {
62          BGTObjectTableWriter writer = db.createObjectTableWriter(loadOptions, dbOptions);
63          ProgressReporter progressReporter;
64          if (cliOptions.isConsoleProgressEnabled()) {
65              progressReporter = new ConsoleProgressReporter();
66          } else {
67              progressReporter = new ProgressReporter();
68          }
69          writer.setProgressUpdater(progressReporter);
70          return writer;
71      }

73      @Command(name="initial", sortOptions = false)
74      public int initial(
75              @Mixin DatabaseOptions dbOptions,
76              @Mixin LoadOptions loadOptions,
77              @Mixin ExtractSelectionOptions extractSelectionOptions,
78              @Option(names="--no-geo-filter") boolean noGeoFilter,
79              @Option(names="--download-service", hidden = true) URI downloadServiceURI,
80              @Mixin CLIOptions cliOptions,
81              @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp
82      ) throws Exception {
83  
84          if (extractSelectionOptions.getGeoFilterWkt() == null && !noGeoFilter) {
85              System.err.println(getBundleString("download.no_geo_filter"));
86              return ExitCode.USAGE;
87          }
88  
89          log.info(getUserAgent());
90  
91          try(BGTDatabase db = this.getBGTdatabase(dbOptions)) {
92              if (loadOptions.createSchema) {
93                  db.createMetadataTable(loadOptions);
94              } else {
95                  log.info(getBundleString("download.connect_db"));
96              }
97  
98              db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
99              db.setMetadataValue(Metadata.BRMOVERSIE, getBrmoVersion());
100             db.setFeatureTypesEnumMetadata(extractSelectionOptions.getFeatureTypesList());
101             db.setMetadataValue(Metadata.INCLUDE_HISTORY, loadOptions.includeHistory + "");
102             db.setMetadataValue(Metadata.LINEARIZE_CURVES, loadOptions.linearizeCurves + "");
103             db.setMetadataValue(Metadata.TABLE_PREFIX, loadOptions.tablePrefix);
104 
105             Instant start = null;
106             URI uri;
107             if (noGeoFilter) {
108                 uri = new URI(PREDEFINED_FULL_DELTA_URI);
109             } else {
110                 // Close connection while waiting for extract
111                 db.close();
112                 start = Instant.now(); // Record total time waiting for extract
113                 uri = getCustomDownloadURI(downloadServiceURI, extractSelectionOptions, new CustomDownloadProgressReporter(cliOptions.isConsoleProgressEnabled()));
114             }
115             BGTObjectTableWriter writer = createWriter(db, dbOptions, loadOptions, cliOptions);
116             loadZipFromURI(uri, db, writer, extractSelectionOptions, loadOptions, noGeoFilter, start);
117 
118             db.setMetadataValue(Metadata.DELTA_TIME_TO, null);
119             // Do not set geom filter from MutatieInhoud, a custom download without geo filter will have gebied
120             // "POLYGON ((-100000 200000, 412000 200000, 412000 712000, -100000 712000, -100000 200000))"
121             db.setMetadataValue(Metadata.GEOM_FILTER, extractSelectionOptions.getGeoFilterWkt());
122 
123             db.getConnection().commit();
124             return ExitCode.OK;
125         }
126     }

143     private static URI getCustomDownloadURI(URI downloadServiceURI, ExtractSelectionOptions extractSelectionOptions, Consumer<CustomDownloadProgress> progressConsumer) throws ApiException, InterruptedException {
144         try {
145             log.info(getBundleString("download.create"));
146             ApiClient client = getApiClient(downloadServiceURI);
147             return DownloadApiUtils.getCustomDownloadURL(client, null, extractSelectionOptions, progressConsumer);
148         } catch(ApiException apiException) {
149             printApiException(apiException);
150             throw apiException;
151         }
152     }

154     @Command(name="update", sortOptions = false)
155     public int update(
156             @Mixin DatabaseOptions dbOptions,
157             @Mixin CLIOptions cliOptions,
158             @Option(names="--download-service", hidden = true) URI downloadServiceURI,
159             @Option(names="--no-http-zip-random-access", negatable = true, hidden = true) boolean noHttpZipRandomAccess,
160             @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp
161     ) throws Exception {
162         log.info(getUserAgent());
163 
164         ApiClient client = getApiClient(downloadServiceURI);
165 
166         log.info(getBundleString("download.connect_db"));
167         try(BGTDatabase db = getBGTdatabase(dbOptions)) {
168             String deltaId = db.getMetadata(Metadata.DELTA_ID);
169             OffsetDateTime deltaIdTimeTo = null;
170             String s = db.getMetadata(Metadata.DELTA_TIME_TO);
171             if (s != null && s.length() > 0) {
172                 deltaIdTimeTo = OffsetDateTime.parse(s);
173             }
174             if (deltaId == null) {
175                 System.err.println(getBundleString("download.no_delta_id"));
176                 return ExitCode.SOFTWARE;
177             }
178             ExtractSelectionOptions extractSelectionOptions = new ExtractSelectionOptions();
179             extractSelectionOptions.setGeoFilterWkt(db.getMetadata(Metadata.GEOM_FILTER));
180             if (extractSelectionOptions.getGeoFilterWkt() != null && extractSelectionOptions.getGeoFilterWkt().length() == 0) {
181                 extractSelectionOptions.setGeoFilterWkt(null);
182             }
183             extractSelectionOptions.setFeatureTypes(Arrays.asList(db.getMetadata(Metadata.FEATURE_TYPES).split(",")));
184             LoadOptions loadOptions = new LoadOptions();
185             loadOptions.setHttpZipRandomAccess(!noHttpZipRandomAccess);
186             loadOptions.includeHistory = Boolean.parseBoolean(db.getMetadata(Metadata.INCLUDE_HISTORY));
187             loadOptions.linearizeCurves = Boolean.parseBoolean(db.getMetadata(Metadata.LINEARIZE_CURVES));
188             loadOptions.tablePrefix = db.getMetadata(Metadata.TABLE_PREFIX);
189 
190             log.info(getMessageFormattedString("download.current_delta_id", deltaId) + ", " +
191                     (deltaIdTimeTo != null
192                             ? getMessageFormattedString("download.current_delta_time", DateTimeFormatter.ISO_INSTANT.format(deltaIdTimeTo))
193                             : getBundleString("download.current_delta_time_unknown")));
194 
195             try {
196                 Instant start = Instant.now();
197                 log.info(getBundleString("download.loading_deltas"));
198                 // Note that the afterDeltaId parameter is useless, because the response does not distinguish between
199                 // "'afterDeltaId' is the latest" and "'afterDeltaId' not found or older than 31 days"
200                 GetDeltasResponse response = new DeltaApi(client).getDeltas(null, 1, 100);
201 
202                 // Verify no links to other page, as we expect at most 31 delta's
203                 if (response.getLinks() != null && !response.getLinks().isEmpty()) {
204                     throw new IllegalStateException("Did not expect links in GetDeltas response");
205                 }
206 
207                 int i;
208                 for (i = 0; i < response.getDeltas().size(); i++) {
209                     Delta d = response.getDeltas().get(i);
210                     if (deltaId.equals(d.getId())) {
211                         break;
212                     }
213                 }
214                 if (i == response.getDeltas().size()) {
215                     // TODO automatically do initial load depending on option
216                     System.err.println(getBundleString("download.current_delta_not_found"));
217                     return ExitCode.SOFTWARE;
218                 }
219 
220                 List<Delta> deltas = response.getDeltas().subList(i + 1, response.getDeltas().size());
221                 if (deltas.isEmpty()) {
222                     log.info(getBundleString("download.uptodate"));
223                     return ExitCode.OK;
224                 }
225 
226                 Delta latestDelta = deltas.get(deltas.size() - 1);
227                 log.info(getMessageFormattedString("download.updates_available", deltas.size(), latestDelta.getId(), latestDelta.getTimeWindow().getTo()));
228 
229                 BGTObjectTableWriter writer = createWriter(db, dbOptions, loadOptions, cliOptions);
230 
231                 int deltaCount = 1;
232                 for (Delta delta: deltas) {
233                     log.info(getMessageFormattedString("download.creating_download", deltaCount++, deltas.size(), delta.getId()));
234                     URI uri = DownloadApiUtils.getCustomDownloadURL(client, delta, extractSelectionOptions, new CustomDownloadProgressReporter(cliOptions.isConsoleProgressEnabled()));
235                     // TODO: BGTObjectTableWriter does setAutocommit(false) and commit() after each stream for a feature type
236                     // is written, maybe use one transaction for all feature types?
237                     loadZipFromURI(uri, db, writer, extractSelectionOptions, loadOptions, false, start);
238                     db.setMetadataValue(Metadata.DELTA_TIME_TO, delta.getTimeWindow().getTo().toString());
239                     db.getConnection().commit();
240                 }
241 
242                 db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
243                 db.getConnection().commit();
244                 return ExitCode.OK;
245             } catch(ApiException apiException) {
246                 printApiException(apiException);
247                 throw apiException;
248             }
249         }
250     }

252     private static void printApiException(ApiException apiException) {
253         log.error(String.format("API status code: %d, body: %s\n", apiException.getCode(), apiException.getResponseBody()));
254     }

256     private static void loadZipFromURI(URI uri, BGTDatabase db, BGTObjectTableWriter writer, ExtractSelectionOptions extractSelectionOptions, LoadOptions loadOptions, boolean showSelected, Instant start) throws Exception {
257         Instant loadStart = Instant.now();
258         new BGTLoaderMain().loadZipFromURI(uri, writer, extractSelectionOptions, loadOptions, showSelected);
259         db.setMetadataForMutaties(writer.getProgress().getMutatieInhoud());
260         log.info(getMessageFormattedString("download.complete",
261                 getBundleString("download.mutatietype." + writer.getProgress().getMutatieInhoud().getMutatieType()),
262                 writer.getProgress().getMutatieInhoud().getLeveringsId(),
263                 formatTimeSince(loadStart)) +
264                 (start == null ? "" : " " + getMessageFormattedString("download.complete_total", formatTimeSince(start)))
265         );
266     }
