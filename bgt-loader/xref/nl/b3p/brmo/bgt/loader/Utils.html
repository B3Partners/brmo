7   package nl.b3p.brmo.bgt.loader;

9   import java.text.MessageFormat;
10  import java.time.Duration;
11  import java.time.Instant;
12  import java.util.ResourceBundle;
13  
14  public class Utils {
15  
16    public static String formatTimeSince(Instant start) {
17      return formatDuration(Duration.between(start, Instant.now()));
18    }
19  
20    public static String formatDuration(Duration d) {
21      String days = d.toDaysPart() > 0 ? d.toDaysPart() + "d " : "";
22      if (d.toHoursPart() == 0 && d.toMinutesPart() == 0) {
23        return days + d.toSecondsPart() + "s";
24      } else if (d.toHoursPart() == 0) {
25        return String.format("%s%dm %2ds", days, d.toMinutesPart(), d.toSecondsPart());
26      }
27      return String.format(
28          "%s%dh %2dm %2ds", days, d.toHoursPart(), d.toMinutesPart(), d.toSecondsPart());
29    }
30  
31    public static final String BUNDLE_NAME = "BGTLoader";
32  
33    private static final ResourceBundle bundle = ResourceBundle.getBundle(BUNDLE_NAME);
34  
35    public static ResourceBundle getBundle() {
36      return bundle;
37    }
38  
39    public static String getBundleString(String key) {
40      return bundle.getString(key);
41    }
42  
43    public static String getMessageFormattedString(String key, Object... args) {
44      ResourceBundle bundle = getBundle();
45      return new MessageFormat(bundle.getString(key), bundle.getLocale()).format(args);
46    }
47  
48    public static String getLoaderVersion() {
49      return getBundleString("app.version");
50    }
51  
52    public static String getBrmoVersion() {
53      return getBundleString("brmo.version");
54    }
55  
56    public static String getUserAgent() {
57      return String.format(
58          "%s, %s (%s)/%s, %s/%s",
59          getBundleString("app.user-agent"),
60          System.getProperty("os.name"),
61          System.getProperty("os.arch"),
62          System.getProperty("os.version"),
63          System.getProperty("java.vm.name"),
64          System.getProperty("java.vm.version"));
65    }
66  }
