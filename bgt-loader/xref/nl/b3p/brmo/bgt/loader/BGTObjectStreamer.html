114   private SMInputCursor initCursor(SMInputCursor cursor) throws XMLStreamException {
115     QName root = cursor.advance().getQName();
116 
117     if (root.equals(CITYGML_CITY_MODEL)) {
118       isMutaties = false;
119     } else if (root.equals(BGT_MUTATIES)) {
120       isMutaties = true;
121     } else {
122       throw new IllegalArgumentException("XML root element moet CityModel of bgtMutaties zijn");
123     }
124 
125     // Note: when using StaxMate childElementCursor() or similar, only an local name string
126     // element does not work
127     // when using Aalto. Always use a QName parameter!
128 
129     if (isMutaties) {
130       cursor = cursor.childElementCursor(MUTATIE_BERICHT).advance().childElementCursor().advance();
131       label:
132       do {
133         switch (cursor.getLocalName()) {
134           case "dataset":
135             assert cursor.collectDescendantText().equals("bgt");
136             break;
137           case "inhoud":
138             this.mutatieInhoud = parseInhoud(cursor);
139             break;
140           case "mutatieGroep":
141             // cursor positioned correctly
142             hasMutatieGroep = true;
143             break label;
144           default:
145             throw new IllegalStateException(
146                 "Verwacht mutatieGroep element, gevonden " + cursor.getQName());
147         }
148       } while (cursor.getNext() != null);
149 
150     } else {
151       cursor = cursor.childElementCursor(CITYGML_CITY_OBJECT_MEMBER).advance();
152     }
153 
154     return cursor;
155   }

157   private static MutatieInhoud parseInhoud(SMInputCursor inhoudCursor) throws XMLStreamException {
158     SMInputCursor cursor = inhoudCursor.childElementCursor();
159     MutatieInhoud mutatieInhoud = new MutatieInhoud();
160     while (cursor.getNext() != null) {
161       switch (cursor.getLocalName()) {
162         case "mutatieType":
163           mutatieInhoud.mutatieType = cursor.collectDescendantText().trim();
164           break;
165         case "gebied":
166           mutatieInhoud.gebied = cursor.collectDescendantText().trim();
167           break;
168         case "leveringsId":
169           mutatieInhoud.leveringsId = cursor.collectDescendantText().trim();
170           break;
171         case "objectTypen":
172           SMInputCursor c = cursor.childElementCursor();
173           while (c.getNext() != null) {
174             mutatieInhoud.objectTypen.add(c.collectDescendantText().trim());
175           }
176           break;
177       }
178     }
179     return mutatieInhoud;
180   }

186   protected SMInputFactory buildSMInputFactory() {
187     // Using alternative StAX parsers explicitly:
188     // final XMLInputFactory stax = new WstxInputFactory(); // Woodstox
189     // final XMLInputFactory stax = new com.fasterxml.aalto.stax.InputFactoryImpl(); // Aalto
190     final XMLInputFactory xmlInputFactory =
191         XMLInputFactory.newFactory(); // JRE Default, depends on JAR's present or
192     // javax.xml.stream.XMLInputFactory property, can be SJSXP
193     log.trace("StAX XMLInputFactory: " + xmlInputFactory.getClass().getName());
194 
195     xmlInputFactory.setProperty(XMLInputFactory.IS_COALESCING, Boolean.TRUE); // Coalesce characters
196     xmlInputFactory.setProperty(
197         XMLInputFactory.SUPPORT_DTD,
198         Boolean.FALSE); // No XML entity expansions or external entities
199 
200     return new SMInputFactory(xmlInputFactory);
201   }

203   @Override
204   public Iterator<BGTObject> iterator() {
205     return new Iterator<>() {
206       SMEvent event = cursor.getCurrEvent();
207 
208       /**
209        * A parse action may return multiple objects. Buffer the future objects to be returned in a
210        * next() call.
211        */
212       private final Queue<BGTObject> buffer = new LinkedList<>();
213 
214       @Override
215       public boolean hasNext() {
216         if (isMutaties && !hasMutatieGroep) {
217           return false;
218         }
219         if (!buffer.isEmpty()) {
220           return true;
221         }
222         if (event != null) {
223           return true;
224         }
225         try {
226           event = cursor.getNext();
227           return event != null;
228         } catch (XMLStreamException e) {
229           throw new RuntimeException(e);
230         }
231       }
232 
233       @Override
234       public BGTObject next() {
235         try {
236           if (!buffer.isEmpty()) {
237             return buffer.remove();
238           }
239           if (isMutaties && !hasMutatieGroep) {
240             throw new IllegalStateException("No items");
241           }
242           if (event == null) {
243             if (!hasNext()) {
244               throw new IllegalStateException("No more items");
245             }
246           }
247           // Make sure cityObjectMembers.getNext() is called with next Iterator.getNext()
248           // call
249           event = null;
250 
251           Map<String, Object> attributes = new HashMap<>();
252 
253           SMInputCursor cityObjectMemberChild;
254 
255           String gmlIdPreviousVersion = null;
256           BGTObject.MutatieStatus mutatieStatus = BGTObject.MutatieStatus.WORDT;
257 
258           if (isMutaties) {
259             // Assume single child per <mutatieGroep>, cursor is currently at
260             // <mutatieGroep>
261 
262             SMInputCursor mutatie = cursor.childElementCursor().advance();
263             String mutatieNaam = mutatie.getLocalName();
264             if (mutatieNaam.equals("wijziging") || mutatieNaam.equals("toevoeging")) {
265               mutatie = mutatie.childElementCursor().advance();
266             } else {
267               throw new IllegalStateException(
268                   "Ongeldig mutatieGroep child element: " + mutatie.getQName());
269             }
270 
271             if (mutatieNaam.equals("wijziging")) {
272               assert mutatie.getLocalName().equals("was");
273               gmlIdPreviousVersion = fixUUID(mutatie.getAttrValue("id"));
274 
275               if (mutatie.getNext() == null) {
276                 // To support deletes, we would need to parse the <cityObjectMember>
277                 // child element name
278                 // at least, but deletes do not occur for bgt
279                 throw new IllegalStateException(
280                     "Mutaties 'was' zonder 'wordt' worden niet ondersteund voor BGT");
281               }
282               assert mutatie.getLocalName().equals("wordt");
283               mutatieStatus = BGTObject.MutatieStatus.WAS_WORDT;
284             } else if (mutatieNaam.equals("toevoeging")) {
285               assert mutatie.getLocalName().equals("wordt");
286               // mutatieStatus is already IMGeoObject.MutatieStatus.WORDT
287             } else {
288               throw new IllegalStateException(
289                   "Ongeldig mutatie element: " + mutatie.getLocalName());
290             }
291 
292             // Move cursor from <wordt> to <cityObjectMember> child
293             cityObjectMemberChild =
294                 mutatie
295                     .childElementCursor(BGT_OBJECT)
296                     .advance()
297                     .childElementCursor(CITYGML_CITY_OBJECT_MEMBER)
298                     .advance()
299                     .childElementCursor()
300                     .advance();
301           } else {
302             cityObjectMemberChild = cursor.childElementCursor().advance();
303           }
304 
305           final String name = cityObjectMemberChild.getLocalName();
306           final BGTObjectType bgtObjectType = BGTSchema.getInstance().getObjectTypeByName(name);
307           if (bgtObjectType == null) {
308             throw new IllegalArgumentException("Onbekend object type: " + name);
309           }
310           final Location location = cityObjectMemberChild.getCursorLocation();
311           attributes.put("gmlId", fixUUID(cityObjectMemberChild.getAttrValue(NS_GML, "id")));
312 
313           SMInputCursor attributesCursor = cityObjectMemberChild.childElementCursor();
314 
315           // Only a single collection attribute is supported (no matrix of multiple
316           // collection attributes)
317           // An IMGeoObject will be returned for each collection item
318           Collection<Map<String, Object>> collectionAttribute = null;
319 
320           while (attributesCursor.getNext() != null) {
321             String attributeName = attributesCursor.getLocalName();
322             Object parsedAttribute = parseIMGeoAttribute(attributesCursor);
323 
324             if (parsedAttribute instanceof BGTObject) {
325               List<BGTObject> oneToMany = (List<BGTObject>) attributes.get(attributeName);
326               if (oneToMany == null) {
327                 oneToMany = new ArrayList<>();
328                 attributes.put(attributeName, oneToMany);
329               }
330               oneToMany.add((BGTObject) parsedAttribute);
331             } else if (parsedAttribute instanceof Collection) {
332               if (collectionAttribute != null) {
333                 throw new IllegalStateException("Only a single collection attribute is supported");
334               }
335               collectionAttribute = (Collection<Map<String, Object>>) parsedAttribute;
336             } else {
337               attributes.put(attributeName, parsedAttribute);
338             }
339           }
340 
341           if (collectionAttribute != null && !collectionAttribute.isEmpty()) {
342             int index = 0;
343             for (Map<String, Object> collectionItem : collectionAttribute) {
344               Map<String, Object> attributesForCollectionItem = new HashMap<>(attributes);
345               attributesForCollectionItem.putAll(collectionItem);
346               attributesForCollectionItem.put(BGTSchema.INDEX, index++);
347               buffer.add(new BGTObject(bgtObjectType, attributesForCollectionItem, location));
348             }
349             return buffer.remove();
350           } else {
351             return new BGTObject(
352                 bgtObjectType, attributes, location, mutatieStatus, gmlIdPreviousVersion);
353           }
354         } catch (Exception e) {
355           throw new RuntimeException(e);
356         }
357       }
358     };
359   }

361   private Object parseIMGeoAttribute(SMInputCursor attribute)
362       throws XMLStreamException, FactoryException, IOException {
363     if ("true".equals(attribute.getAttrValue("http://www.w3.org/2001/XMLSchema-instance", "nil"))) {
364       return null;
365     }
366 
367     if (attribute.hasLocalName("identificatie")) {
368       /*
369       <imgeo:identificatie>
370           <imgeo:NEN3610ID>
371               <imgeo:namespace>NL.IMGeo</imgeo:namespace>
372               <imgeo:lokaalID>G1234.abcdef01234567890abcdef012345678</imgeo:lokaalID>
373           </imgeo:NEN3610ID>
374       </imgeo:identificatie>
375        */
376       // Just get the UUID
377       String id =
378           attribute.descendantElementCursor(LOKAAL_ID).advance().collectDescendantText().trim();
379       // UUID is allowed to have '-' characters, normalize it to UUID without
380       return id.replaceAll("-", "");
381     }
382 
383     if (isGeometryElement(attribute.getLocalName())) {
384       // Position underlying XmlStreamReader at GML element
385       attribute.childElementCursor().advance();
386 
387       Geometry geom = geometryReader.readGeometry();
388       geom.setSRID(SRID);
389       return geom;
390     }
391 
392     if (attribute.hasLocalName("nummeraanduidingreeks")) {
393       Location location = attribute.getCursorLocation();
394       Map<String, Object> nummeraanduidingreeks = parseNummeraanduidingreeks(attribute);
395       return new BGTObject(
396           BGTSchema.getInstance().getObjectTypeByName("nummeraanduidingreeks"),
397           nummeraanduidingreeks,
398           location);
399     }
400 
401     if (attribute.hasLocalName("openbareRuimteNaam")) {
402       return parseOpenbareRuimteNaam(attribute);
403     }
404 
405     // String attribute value as default
406     return attribute.collectDescendantText().trim();
407   }

409   private static boolean isGeometryElement(String localName) {
410     return localName.startsWith("geometrie") || localName.startsWith("kruinlijn");
411   }

413   private static final QName CAPITALIZED_NUMMERAANDUIDINGREEKS =
414       new QName(NS_IMGEO, "Nummeraanduidingreeks");
415   private static final QName LOWERCASE_NUMMERAANDUIDINGREEKS =
416       new QName(NS_IMGEO, "nummeraanduidingreeks");
417   private static final QName LABEL = new QName(NS_IMGEO, "Label");
418   private static final QName LABELPOSITIE = new QName(NS_IMGEO, "Labelpositie");
419 
420   private Map<String, Object> parseNummeraanduidingreeks(SMInputCursor cursor)
421       throws XMLStreamException, FactoryException, IOException {
422     Map<String, Object> label = new HashMap<>();
423 
424     cursor =
425         cursor.childElementCursor(CAPITALIZED_NUMMERAANDUIDINGREEKS).advance().childElementCursor();
426     while (cursor.getNext() != null) {
427       if (cursor.getQName().equals(LOWERCASE_NUMMERAANDUIDINGREEKS)) {
428         SMInputCursor labelChilds = cursor.childElementCursor(LABEL).advance().childElementCursor();
429         while (labelChilds.getNext() != null) {
430           if (labelChilds.hasLocalName("tekst")) {
431             label.put("tekst", labelChilds.collectDescendantText().trim());
432           } else if (labelChilds.hasLocalName("positie")) {
433             SMInputCursor labelPositie = labelChilds.childElementCursor(LABELPOSITIE).advance();
434 
435             SMInputCursor labelPositieChilds = labelPositie.childElementCursor();
436             while (labelPositieChilds.getNext() != null) {
437               if (labelPositieChilds.hasLocalName("plaatsingspunt")) {
438                 // Positioneer op <gml:Point/>
439                 labelPositieChilds.childElementCursor().advance();
440                 labelPositieChilds
441                     .getStreamReader()
442                     .require(XMLStreamConstants.START_ELEMENT, NS_GML, "Point");
443                 Geometry geom = geometryReader.readGeometry();
444                 geom.setSRID(SRID);
445                 label.put("plaatsingspunt", geom);
446               } else if (labelPositieChilds.hasLocalName("hoek")) {
447                 String hoek = labelPositieChilds.collectDescendantText().trim();
448                 label.put("hoek", Double.parseDouble(hoek));
449               }
450             }
451           }
452         }
453       } else {
454         label.put(cursor.getLocalName(), cursor.collectDescendantText().trim());
455       }
456     }
457 
458     return label;
459   }

461   private List<Map<String, Object>> parseOpenbareRuimteNaam(SMInputCursor cursor)
462       throws XMLStreamException, FactoryException, IOException {
463 
464     cursor = cursor.childElementCursor(LABEL).advance().childElementCursor();
465 
466     List<Map<String, Object>> posities = new ArrayList<>();
467     String tekst = null;
468 
469     while (cursor.getNext() != null) {
470       if (cursor.getLocalName().equals("tekst")) {
471         tekst = cursor.collectDescendantText().trim();
472       } else if (cursor.getLocalName().equals("positie")) {
473         Map<String, Object> positie = new HashMap<>();
474         positie.put("tekst", tekst);
475         posities.add(positie);
476         SMInputCursor positieCursor = cursor.childElementCursor().advance().childElementCursor();
477         while (positieCursor.getNext() != null) {
478           if (positieCursor.getLocalName().equals("plaatsingspunt")) {
479             positieCursor.childElementCursor().advance();
480             Geometry geom = geometryReader.readGeometry();
481             geom.setSRID(SRID);
482             positie.put("plaatsingspunt", geom);
483           } else {
484             positie.put(positieCursor.getLocalName(), positieCursor.collectDescendantText().trim());
485           }
486         }
487       }
488     }
489 
490     return posities;
491   }
