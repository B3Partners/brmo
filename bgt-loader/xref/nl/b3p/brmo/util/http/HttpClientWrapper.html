8   package nl.b3p.brmo.util.http;
9   
10  import java.io.IOException;
11  import java.net.URI;
12  
13  /**
14   * A wrapper around a HTTP client with only the functionality needed by this package.
15   *
16   * @param <REQUEST> An implementation defined HTTP request type.
17   * @param <RESPONSE> An implementation defined HTTP response type.
18   * @author Matthijs Laan
19   */
20  public interface HttpClientWrapper<REQUEST, RESPONSE> {
21    /**
22     * Perform a synchronous HTTP GET request.
23     *
24     * @param uri The URI.
25     * @param requestHeaders The request headers to send. The supplied String instances must alternate
26     *     as header names and header values. Only single request header values are expected.
27     * @return A wrapper with the response.
28     * @throws IOException If an I/O error occurs.
29     * @throws InterruptedException if any
30     */
31    HttpResponseWrapper request(URI uri, String... requestHeaders)
32        throws IOException, InterruptedException;
33  
34    /**
35     * Override to modify the request before it is sent, such as adding a {@code User-Agent} header.
36     *
37     * @param request The implementation defined HTTP request type.
38     */
39    default void beforeRequest(REQUEST request) {}
40  
41    /**
42     * Wrap the HTTP response so the status code, response headers en the response body can be read.
43     *
44     * @param response The implementation defined HTTP response type.
45     * @return wrapped response
46     */
47    HttpResponseWrapper wrapResponse(RESPONSE response);
48  }
