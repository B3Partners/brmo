8   package nl.b3p.brmo.util.http;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.net.URI;
13  import java.net.http.HttpRequest;
14  import java.net.http.HttpResponse;
15  
16  /**
17   * A wrapper around a HTTP client with only the functionality needed by this package.
18   * @param <REQUEST> An implementation defined HTTP request type.
19   * @param <RESPONSE> An implementation defined HTTP response type.
20   *
21   * @author Matthijs Laan
22   */
23  public interface HttpClientWrapper<REQUEST,RESPONSE> {
24      /**
25       * Perform a synchronous HTTP GET request.
26       * @param uri The URI.
27       * @param requestHeaders The request headers to send. The supplied String instances must alternate as header names
28       *                       and header values. Only single request header values are expected.
29       * @return A wrapper with the response.
30       * @throws IOException If an I/O error occurs.
31       * @throws InterruptedException
32       */
33      HttpResponseWrapper request(URI uri, String... requestHeaders) throws IOException, InterruptedException;
34  
35      /**
36       * Override to modify the request before it is sent, such as adding a {@code User-Agent} header.
37       * @param request The implementation defined HTTP request type.
38       */
39      default void beforeRequest(REQUEST request) {
40      }
41  
42      /**
43       * Wrap the HTTP response so the status code, response headers en the response body can be read.
44       * @param response The implementation defined HTTP response type.
45       * @return
46       */
47      HttpResponseWrapper wrapResponse(RESPONSE response);
48  }
