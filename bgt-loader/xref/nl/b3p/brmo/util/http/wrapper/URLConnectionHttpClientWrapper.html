8   package nl.b3p.brmo.util.http.wrapper;
9   
10  import nl.b3p.brmo.util.http.HttpClientWrapper;
11  import nl.b3p.brmo.util.http.HttpResponseWrapper;
12  
13  import java.io.IOException;
14  import java.io.InputStream;
15  import java.net.HttpURLConnection;
16  import java.net.URI;
17  import java.net.URLConnection;
18  
19  /**
20   * Wraps a standard Java HttpURLConnection or HttpsURLConnection.
21   *
22   * @author Matthijs Laan
23   */
24  public class URLConnectionHttpClientWrapper implements HttpClientWrapper<HttpURLConnection,HttpURLConnection> {
25      @Override
26      public HttpResponseWrapper request(URI uri, String... requestHeaders) throws IOException {
27          URLConnection connection = uri.toURL().openConnection();
28          if (!(connection instanceof HttpURLConnection)) {
29              throw new IllegalArgumentException("Expected HttpURLConnection instance for URI " + uri);
30          }
31          HttpURLConnection httpURLConnection = (HttpURLConnection) connection;
32          httpURLConnection.setInstanceFollowRedirects(true);
33          if (requestHeaders.length % 2 != 0) {
34              throw new IllegalArgumentException();
35          }
36          for(int i = 0; i < requestHeaders.length; i += 2) {
37              httpURLConnection.setRequestProperty(requestHeaders[i], requestHeaders[i+1]);
38          }
39  
40          beforeRequest(httpURLConnection);
41          httpURLConnection.connect();
42  
43          return wrapResponse(httpURLConnection);
44      }
45  
46      /**
47       * Called before doing a request, override to modify or log it.
48       * @param httpURLConnection The HttpURLConnection that will be used, can be modified
49       */
50      public void beforeRequest(HttpURLConnection httpURLConnection) {
51      }
52  
53      /**
54       * Wraps a HttpURLConnection after HttpURLConnection.connect() has been called, override to modify or log it.
55       * @param httpURLConnection The URL connection to be wrapped
56       * @return The wrapped response
57       */
58      @Override
59      public HttpResponseWrapper wrapResponse(HttpURLConnection httpURLConnection) {
60          return new HttpResponseWrapper() {
61              @Override
62              public int getStatusCode() throws IOException {
63                  return httpURLConnection.getResponseCode();
64              }
65  
66              @Override
67              public String getHeader(String header) {
68                  return httpURLConnection.getHeaderField(header);
69              }
70  
71              @Override
72              public InputStream getResponseBody() throws IOException {
73                  return httpURLConnection.getInputStream();
74              }
75          };
76      }
77  }
