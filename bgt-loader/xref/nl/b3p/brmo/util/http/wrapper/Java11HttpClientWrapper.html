8   package nl.b3p.brmo.util.http.wrapper;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.net.URI;
13  import java.net.http.HttpClient;
14  import java.net.http.HttpRequest;
15  import java.net.http.HttpResponse;
16  import nl.b3p.brmo.util.http.HttpClientWrapper;
17  import nl.b3p.brmo.util.http.HttpResponseWrapper;
18  
19  /**
20   * Wraps the Java 11 {@link java.net.http.HttpClient}. Follows redirects by default but sets the
21   * HTTP version to 1.1 instead of HTTP/2 if the {@code httpclientwrapper.java11.http1_1} system
22   * property is set.
23   *
24   * @author Matthijs Laan
25   */
26  public class Java11HttpClientWrapper
27      implements HttpClientWrapper<HttpRequest.Builder, HttpResponse<InputStream>> {
28    private final HttpClient httpClient;
29  
30    public Java11HttpClientWrapper() {
31      this(HttpClient.newBuilder().followRedirects(HttpClient.Redirect.NORMAL));
32    }
33  
34    public Java11HttpClientWrapper(HttpClient.Builder builder) {
35      // Check system property if we must disable HTTP/2 if a server gives problems (such as
36      // https://npmjs.org/package/lite-server)
37      if (System.getProperty("httpclientwrapper.java11.http1_1") != null) {
38        builder.version(HttpClient.Version.HTTP_1_1);
39      }
40      this.httpClient = builder.build();
41    }
42  
43    @Override
44    public HttpResponseWrapper request(URI uri, String... requestHeaders)
45        throws IOException, InterruptedException {
46      HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(uri);
47      if (requestHeaders.length > 0) {
48        requestBuilder.headers(requestHeaders);
49      }
50      beforeRequest(requestBuilder);
51      HttpRequest request = requestBuilder.build();
52      HttpResponse<InputStream> response =
53          httpClient.send(request, HttpResponse.BodyHandlers.ofInputStream());
54      return wrapResponse(response);
55    }
56  
57    /**
58     * Called before doing a request, override to modify or log it.
59     *
60     * @param requestBuilder The request that will be done, can be modified
61     */
62    @Override
63    public void beforeRequest(HttpRequest.Builder requestBuilder) {}
64  
65    /**
66     * Wraps a HttpResponse, override to modify or log it.
67     *
68     * @param response The received response to be wrapped
69     * @return The wrapped response
70     */
71    @Override
72    public HttpResponseWrapper wrapResponse(HttpResponse<InputStream> response) {
73      return new HttpResponseWrapper() {
74        @Override
75        public int getStatusCode() {
76          return response.statusCode();
77        }
78  
79        @Override
80        public String getHeader(String header) {
81          return response.headers().firstValue(header).orElse(null);
82        }
83  
84        @Override
85        public InputStream getResponseBody() {
86          return response.body();
87        }
88      };
89    }
90  }
