8   package nl.b3p.brmo.util.http;
9   
10  import nl.b3p.brmo.util.http.wrapper.Java11HttpClientWrapper;
11  import nl.b3p.brmo.util.http.wrapper.URLConnectionHttpClientWrapper;
12  
13  /**
14   * Provides default wrapper implementations for Java HTTP clients.
15   *
16   * <p>If the {@code java.specification.version} system property is 11 or higher, returns a {@link
17   * Java11HttpClientWrapper} unless the system property {@code httpclientwrapper} is set to {@code
18   * urlconnection}. This client tries to upgrade the connection HTTP/2 by default, unless the {@code
19   * httpclientwrapper.java11.http1_1} system property is set.
20   *
21   * <p>When the Java version is lower than 11, a {@code URLConnectionHttpClientWrapper} is returned.
22   *
23   * @author Matthijs Laan
24   */
25  public class HttpClientWrappers {
26    public static HttpClientWrapper getDefault() {
27      if ("urlconnection".equals(System.getProperty("httpclientwrapper"))) {
28        return new URLConnectionHttpClientWrapper();
29      }
30      float f = Float.parseFloat(System.getProperty(("java.specification.version")));
31      if (f >= 11.0f) {
32        return new Java11HttpClientWrapper();
33      } else {
34        return new URLConnectionHttpClientWrapper();
35      }
36    }
37  }
