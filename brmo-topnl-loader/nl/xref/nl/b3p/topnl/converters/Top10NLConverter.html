76      @Override
77      public List<TopNLEntity> convert(List jaxbObject)
78              throws IOException, SAXException, ParserConfigurationException, TransformerException {
79          List<TopNLEntity> entities = null;
80          if (jaxbObject instanceof ArrayList) {
81              entities = convertFeatureCollection(jaxbObject);
82          } else {
83              entities = new ArrayList<>();
84              entities.add(convertObject(jaxbObject));
85          }
86          return entities;
87      }

89      @Override
90      public List<TopNLEntity> convertFeatureCollection(List jaxbFeatureCollection)
91              throws IOException, SAXException, ParserConfigurationException, TransformerException {
92          List<TopNLEntity> entities = new ArrayList<>();
93  
94          for (Object fm : jaxbFeatureCollection) {
95              FeatureMemberType featureMember = (FeatureMemberType) fm;
96              JAXBElement jaxbObject = featureMember.getTop10NlObject();
97              Object obj = jaxbObject.getValue();
98              TopNLEntity entity = convertObject(obj);
99              entities.add(entity);
100         }
101 
102         return entities;
103     }

105     @Override
106     public TopNLEntity convertObject(Object featureMember)
107             throws IOException, SAXException, ParserConfigurationException, TransformerException,
108                     IllegalArgumentException {
109         TopNLEntity entity = null;
110 
111         if (featureMember instanceof FeatureMemberType) {
112             FeatureMemberType fm = (FeatureMemberType) featureMember;
113             JAXBElement jaxbObject = fm.getTop10NlObject();
114             return convertObject(jaxbObject.getValue());
115         } else if (featureMember instanceof HoogteType) {
116             entity = convertHoogte(featureMember);
117         } else if (featureMember instanceof FunctioneelGebiedType) {
118             entity = convertFunctioneelGebied(featureMember);
119         } else if (featureMember instanceof GebouwType) {
120             entity = convertGebouw(featureMember);
121         } else if (featureMember instanceof GeografischGebiedType) {
122             entity = convertGeografischGebied(featureMember);
123         } else if (featureMember instanceof InrichtingselementType) {
124             entity = convertInrichtingselement(featureMember);
125         } else if (featureMember instanceof PlaatsType) {
126             entity = convertPlaats(featureMember);
127         } else if (featureMember instanceof RegistratiefGebiedType) {
128             entity = convertRegistratiefGebied(featureMember);
129         } else if (featureMember instanceof ReliefType) {
130             entity = convertRelief(featureMember);
131         } else if (featureMember instanceof SpoorbaandeelType) {
132             entity = convertSpoorbaandeel(featureMember);
133         } else if (featureMember instanceof TerreinType) {
134             entity = convertTerrein(featureMember);
135         } else if (featureMember instanceof WaterdeelType) {
136             entity = convertWaterdeel(featureMember);
137         } else if (featureMember instanceof WegdeelType) {
138             entity = convertWegdeel(featureMember);
139         } else if (featureMember instanceof PlanTopografieType) {
140             entity = convertPlanTopografie(featureMember);
141         } else {
142             throw new IllegalArgumentException("Type not recognized: " + featureMember.getClass());
143         }
144 
145         log.debug(
146                 "Converted type "
147                         + entity.getClass()
148                         + ", identificatie: "
149                         + entity.getIdentificatie());
150         entity.setTopnltype(TopNLType.TOP10NL.getType());
151         return entity;
152     }

154     @Override
155     public Hoogte convertHoogte(Object jaxbObject)
156             throws IOException, SAXException, ParserConfigurationException, TransformerException {
157         HoogteType h = (HoogteType) jaxbObject;
158 
159         Hoogte hoogte = new Hoogte();
160         convertTop10NlObjectType(h, hoogte);
161 
162         hoogte.setTypeHoogte(h.getTypeHoogte().getValue());
163 
164         hoogte.setHoogte(h.getHoogte());
165         hoogte.setObjectEindTijd(
166                 h.getObjectEindTijd() != null ? h.getObjectEindTijd().getTime() : null);
167 
168         Element el = h.getGeometrie();
169         Geometry geom = gc.convertGeometry(el);
170         hoogte.setGeometrie(geom);
171         hoogte.setReferentieVlak(
172                 h.getReferentievlak() != null ? h.getReferentievlak().getValue() : null);
173         return hoogte;
174     }

176     public String convertIdentificatie(Identificatie identificatie) {
177         String idString =
178                 identificatie.getNEN3610ID().getNamespace().trim()
179                         + "."
180                         + identificatie.getNEN3610ID().getLokaalID();
181         return idString;
182     }

184     private void convertTop10NlObjectType(Top10NlObjectType type, TopNLEntity entity) {
185         entity.setIdentificatie(convertIdentificatie(type.getIdentificatie()));
186         entity.setBrontype(type.getBrontype().getValue());
187         entity.setBronactualiteit(type.getBronactualiteit().getTime());
188         entity.setBronbeschrijving(type.getBronbeschrijving());
189         entity.setBronnauwkeurigheid(type.getBronnauwkeurigheid());
190         entity.setObjectBeginTijd(type.getObjectBeginTijd().getTime());
191         entity.setVisualisatieCode(
192                 type.getVisualisatieCode() != null ? type.getVisualisatieCode().longValue() : null);
193     }

210     @Override
211     public Gebouw convertGebouw(Object jaxbObject)
212             throws IOException, SAXException, ParserConfigurationException, TransformerException {
213         GebouwType g = (GebouwType) jaxbObject;
214 
215         Gebouw gb = new Gebouw();
216         convertTop10NlObjectType(g, gb);
217         String types = "";
218         for (CodeType typeGebouwT10Type : g.getTypeGebouw()) {
219             if (types.length() != 0) {
220                 types += ",";
221             }
222             types += typeGebouwT10Type.getValue();
223         }
224         gb.setTypeGebouw(types);
225         gb.setFysiekVoorkomen(
226                 g.getFysiekVoorkomen() != null ? g.getFysiekVoorkomen().getValue() : null);
227         gb.setHoogteklasse(g.getHoogteklasse() != null ? g.getHoogteklasse().getValue() : null);
228         gb.setHoogte(g.getHoogte());
229         gb.setNaam(String.join(",", g.getNaam()));
230         gb.setSoortnaam(g.getSoortnaam());
231         gb.setStatus(g.getStatus() != null ? g.getStatus().getValue() : null);
232         gb.setGeometrie(gc.convertGeometry(g.getGeometrie()));
233         return gb;
234     }

311     @Override
312     public Relief convertRelief(Object jaxbObject)
313             throws IOException, SAXException, ParserConfigurationException, TransformerException {
314         ReliefType r = (ReliefType) jaxbObject;
315 
316         Relief rg = new Relief();
317         convertTop10NlObjectType(r, rg);
318 
319         if (r.getLijnGeometrie() != null) {
320             rg.setGeometrie((LineString) gc.convertGeometry(r.getLijnGeometrie()));
321         }
322 
323         if (r.getTaludGeometrie() != null) {
324             BRTHogeEnLageZijdeType hl = r.getTaludGeometrie().getBRTHogeEnLageZijde();
325             Element[] els = hl.getHogeZijde();
326             for (Element el : els) {
327                 LineString g = (LineString) gc.convertGeometry(el);
328                 if (el.getLocalName().equals("lageZijde")) {
329                     rg.setTaludLageZijde(g);
330                 } else if (el.getLocalName().equals("hogeZijde")) {
331                     rg.setTaludHogeZijde(g);
332                 }
333             }
334         }
335 
336         rg.setHoogteklasse(r.getHoogteklasse().getValue());
337         rg.setTypeRelief(r.getTypeRelief() != null ? r.getTypeRelief().getValue() : null);
338         rg.setHoogteniveau(r.getHoogteniveau().longValue());
339 
340         return rg;
341     }

343     @Override
344     public Spoorbaandeel convertSpoorbaandeel(Object jaxbObject)
345             throws IOException, SAXException, ParserConfigurationException, TransformerException,
346                     ClassCastException {
347         SpoorbaandeelType r = (SpoorbaandeelType) jaxbObject;
348 
349         Spoorbaandeel rg = new Spoorbaandeel();
350         convertTop10NlObjectType(r, rg);
351 
352         final Geometry g = gc.convertGeometry(r.getGeometrie());
353         if (g instanceof LineString) {
354             rg.setGeometrie((LineString) g);
355         } else {
356             rg.setPuntGeometrie((Point) g);
357         }
358 
359         String fysieks = "";
360         for (CodeType fk : r.getFysiekVoorkomen()) {
361             if (fysieks.length() != 0) {
362                 fysieks += ",";
363             }
364             fysieks += fk.getValue();
365         }
366         rg.setFysiekVoorkomen(fysieks);
367 
368         rg.setTypeSpoorbaan(r.getTypeSpoorbaan() != null ? r.getTypeSpoorbaan().getValue() : null);
369         rg.setAantalSporen(r.getAantalSporen() != null ? r.getAantalSporen().getValue() : null);
370         rg.setStatus(r.getStatus().getValue());
371         rg.setHoogteniveau(r.getHoogteniveau().longValue());
372         rg.setTypeInfrastructuur(
373                 r.getTypeInfrastructuur() != null ? r.getTypeInfrastructuur().getValue() : null);
374         rg.setSpoorbreedte(r.getSpoorbreedte() != null ? r.getSpoorbreedte().getValue() : null);
375         rg.setVervoerfunctie(
376                 r.getVervoerfunctie() != null ? r.getVervoerfunctie().getValue() : null);
377         rg.setElektrificatie(
378                 r.getElektrificatie() != null
379                         ? r.getElektrificatie().equals(BRTJaNeeWaardeType.JA)
380                         : null);
381         return rg;
382     }

448     @Override
449     public Wegdeel convertWegdeel(Object jaxbObject)
450             throws IOException, SAXException, ParserConfigurationException, TransformerException {
451         WegdeelType r = (WegdeelType) jaxbObject;
452 
453         Wegdeel rg = new Wegdeel();
454         convertTop10NlObjectType(r, rg);
455 
456         String fysieks = "";
457         for (CodeType fk : r.getFysiekVoorkomen()) {
458             if (fysieks.length() != 0) {
459                 fysieks += ",";
460             }
461             fysieks += fk.getValue();
462         }
463         rg.setFysiekVoorkomen(fysieks);
464 
465         String typeweg = "";
466         for (CodeType codeType : r.getTypeWeg()) {
467             if (typeweg.length() != 0) {
468                 typeweg += ",";
469             }
470             typeweg += codeType.getValue();
471         }
472 
473         rg.setTypeWeg(typeweg);
474 
475         String hoofdverkeersgebruikt = "";
476         for (CodeType codeType : r.getHoofdverkeersgebruik()) {
477             if (hoofdverkeersgebruikt.length() != 0) {
478                 hoofdverkeersgebruikt += ",";
479             }
480             hoofdverkeersgebruikt += codeType.getValue();
481         }
482 
483         rg.setHoofdverkeersgebruik(hoofdverkeersgebruikt);
484 
485         rg.setNaam(String.join(",", r.getNaam()));
486 
487         Element[] els = r.getHartGeometrie();
488         for (Element el : els) {
489             String localname = el.getLocalName();
490             Geometry g = gc.convertGeometry(el);
491             if (localname.equals("hoofdGeometrie")) {
492                 rg.setGeometrie(g);
493             } else if (localname.equals("hartGeometrie")) {
494                 rg.setHartGeometrie(g);
495             }
496         }
497         rg.setTypeInfrastructuur(
498                 r.getTypeInfrastructuur() != null ? r.getTypeInfrastructuur().getValue() : null);
499         rg.setVerhardingstype(r.getVerhardingstype().getValue());
500         rg.setVerhardingsbreedteklasse(
501                 r.getVerhardingsbreedteklasse() != null
502                         ? r.getVerhardingsbreedteklasse().getValue()
503                         : null);
504         rg.setGescheidenRijbaan(r.getGescheidenRijbaan() == BRTJaNeeWaardeType.JA);
505         rg.setaWegnummer(String.join(",", r.getAWegnummer()));
506         rg.setnWegnummer(String.join(",", r.getNWegnummer()));
507         rg.seteWegnummer(String.join(",", r.getEWegnummer()));
508         rg.setsWegnummer(String.join(",", r.getSWegnummer()));
509         rg.setAfritnummer(r.getAfritnummer());
510         rg.setAfritnaam(String.join(",", r.getAfritnaam()));
511         rg.setKnooppuntnaam(String.join(",", r.getKnooppuntnaam()));
512         rg.setBrugnaam(String.join(",", r.getBrugnaam()));
513         rg.setTunnelnaam(String.join(",", r.getTunnelnaam()));
514         rg.setHoogteniveau(r.getHoogteniveau().longValue());
515         rg.setStatus(r.getStatus().getValue());
516         return rg;
517     }
