57    public Database(DataSource ds) throws SQLException {
58      this.dataSource = ds;
59      if (ds != null) {
60        gjc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(dataSource.getConnection());
61        run = new QueryRunner(dataSource, gjc.isPmdKnownBroken());
62      }
63    }

65    public void save(TopNLEntity entity) throws ParseException, SQLException {
66  
67      try {
68  
69        if (entity instanceof Hoogte) {
70          saveHoogte(entity);
71        } else if (entity instanceof FunctioneelGebied) {
72          saveFunctioneelGebied(entity);
73        } else if (entity instanceof Gebouw) {
74          saveGebouw(entity);
75        } else if (entity instanceof GeografischGebied) {
76          saveGeografischGebied(entity);
77        } else if (entity instanceof Inrichtingselement) {
78          saveInrichtingselement(entity);
79        } else if (entity instanceof Plaats) {
80          savePlaats(entity);
81        } else if (entity instanceof RegistratiefGebied) {
82          saveRegistratiefGebied(entity);
83        } else if (entity instanceof Relief) {
84          saveRelief(entity);
85        } else if (entity instanceof Spoorbaandeel) {
86          saveSpoorbaandeel(entity);
87        } else if (entity instanceof Terrein) {
88          saveTerrein(entity);
89        } else if (entity instanceof Waterdeel) {
90          saveWaterdeel(entity);
91        } else if (entity instanceof Wegdeel) {
92          saveWegdeel(entity);
93        } else if (entity instanceof PlanTopografie) {
94          savePlanTopografie(entity);
95        } else {
96          throw new IllegalArgumentException("Type of entity not (yet) implemented.");
97        }
98  
99      } catch (SQLException e) {
100       log.error("Error inserting entity: ", e);
101       throw e;
102     }
103   }

105   private Gebouw saveGebouw(TopNLEntity entity) throws SQLException, ParseException {
106     Gebouw h = (Gebouw) entity;
107 
108     ResultSetHandler<Gebouw> handler =
109         new BeanHandler(
110             Gebouw.class, new BasicRowProcessor(new DbUtilsGeometryColumnConverter(gjc)));
111     Object nativeGeom = gjc.convertToNativeGeometryObject(h.getGeometrie());
112     Object[] args =
113         getVarargs(
114             entity,
115             h.getTypeGebouw(),
116             h.getStatus(),
117             h.getFysiekVoorkomen(),
118             h.getHoogteklasse(),
119             h.getHoogte(),
120             h.getSoortnaam(),
121             h.getNaam(),
122             h.getNaamFries(),
123             nativeGeom);
124     Gebouw inserted =
125         run.insert(
126             "INSERT INTO "
127                 + h.getTopnltype()
128                 + ".gebouw ("
129                 + getTopNLEntityColumns()
130                 + ",typegebouw,status,fysiekvoorkomen,hoogteklasse,hoogte,soortnaam,naam,naamfries,geometrie) VALUES ("
131                 + getTopNLEntityReplacementChars()
132                 + ",?,?,?,?,?,?,?,?,?)",
133             handler,
134             args);
135 
136     return inserted;
137   }

533   private Object[] getVarargs(TopNLEntity entity, Object... specificArgs) {
534     Object[] genericArgs = {
535       entity.getIdentificatie(),
536       entity.getTopnltype(),
537       entity.getBrontype(),
538       entity.getBronactualiteit() != null
539           ? new java.sql.Date(entity.getBronactualiteit().getTime())
540           : null,
541       entity.getBronbeschrijving(),
542       entity.getBronnauwkeurigheid(),
543       entity.getObjectBeginTijd() != null
544           ? new java.sql.Date(entity.getObjectBeginTijd().getTime())
545           : null,
546       entity.getObjectEindTijd() != null
547           ? new java.sql.Date(entity.getObjectEindTijd().getTime())
548           : null,
549       entity.getVisualisatieCode()
550     };
551     int numGeneric = genericArgs.length;
552     Object[] completeArgs = new Object[numGeneric + specificArgs.length];
553     System.arraycopy(genericArgs, 0, completeArgs, 0, numGeneric);
554 
555     for (int i = 0; i < specificArgs.length; i++) {
556       Object specificArg = specificArgs[i];
557       completeArgs[numGeneric + i] = specificArg;
558     }
559 
560     return completeArgs;
561   }
