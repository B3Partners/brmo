17  package nl.b3p.topnl;
18  
19  import java.io.IOException;
20  import java.io.InputStreamReader;
21  import java.io.Reader;
22  import java.sql.Connection;
23  import java.sql.SQLException;
24  import javax.sql.DataSource;
25  import nl.b3p.jdbc.util.converter.GeometryJdbcConverter;
26  import nl.b3p.jdbc.util.converter.GeometryJdbcConverterFactory;
27  import org.apache.commons.dbutils.QueryRunner;
28  import org.apache.commons.logging.Log;
29  import org.apache.commons.logging.LogFactory;
30  import org.hsqldb.jdbc.JDBCDataSource;
31  import org.junit.jupiter.api.AfterEach;
32  import org.junit.jupiter.api.BeforeEach;
33  import org.junit.jupiter.api.TestInfo;
34  
35  /**
36   * @author Meine Toonen meinetoonen@b3partners.nl
37   * @author mprins
38   */
39  public class TestUtil {
40  
41    private static final Log LOG = LogFactory.getLog(TestUtil.class);
42    protected DataSource datasource;
43    protected boolean useDB = false;
44  
45    protected QueryRunner run;
46  
47    /** Log de naam van de test als deze begint. */
48    @BeforeEach
49    public void startTest(TestInfo testInfo) {
50      LOG.info("==== Start test methode: " + testInfo.getDisplayName());
51    }
52  
53    @BeforeEach
54    public void setUpClass(TestInfo testInfo) throws SQLException, IOException {
55      if (useDB) {
56        JDBCDataSource ds = new JDBCDataSource();
57        String testname = testInfo.getDisplayName().replace(':', '-').replace(' ', '_');
58        ds.setUrl(
59            "jdbc:hsqldb:file:./target/unittest-hsqldb_"
60                + testname
61                + "_"
62                + System.currentTimeMillis()
63                + "/db;shutdown=true");
64        datasource = ds;
65        initDB("initdb250nl.sql");
66        initDB("initdb100nl.sql");
67        initDB("initdb50nl.sql");
68        initDB("initdb10nl.sql");
69        GeometryJdbcConverter gjc =
70            GeometryJdbcConverterFactory.getGeometryJdbcConverter(datasource.getConnection());
71        run = new QueryRunner(datasource, gjc.isPmdKnownBroken());
72      }
73    }
74  
75    /** Log de naam van de test als deze eindigt. */
76    @AfterEach
77    public void endTest(TestInfo testInfo) {
78      LOG.info("==== Einde test methode: " + testInfo.getDisplayName());
79    }
80  
81    private void initDB(String file) throws IOException {
82      try {
83        Reader f = new InputStreamReader(TestUtil.class.getResourceAsStream(file));
84        executeScript(f);
85      } catch (SQLException sqle) {
86        LOG.error("Error initializing testdb:", sqle);
87      }
88    }
89  
90    public void executeScript(Reader f) throws IOException, SQLException {
91  
92      try (Connection conn = datasource.getConnection()) {
93        ScriptRunner sr = new ScriptRunner(conn, true, true);
94        sr.runScript(f);
95        conn.commit();
96      }
97    }
98  }
