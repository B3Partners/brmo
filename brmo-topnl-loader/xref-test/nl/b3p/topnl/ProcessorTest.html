57    /** Test of parse method, of class Processor. */
58    @Test
59    public void testParse() throws JAXBException, IOException {
60      URL in = ProcessorTest.class.getResource("top250nl_Hoogte.xml");
61      List jaxb = instance.parse(in);
62      assertNotNull(jaxb);
63      assertEquals(1, jaxb.size());
64      assertTrue(jaxb.get(0) instanceof nl.b3p.topnl.top250nl.FeatureMemberType);
65    }

67    /** Test of save method, of class Processor. */
68    @Test
69    public void testSave250()
70        throws JAXBException,
71            IOException,
72            SAXException,
73            TransformerException,
74            ParserConfigurationException,
75            SQLException,
76            ParseException {
77      URL in = ProcessorTest.class.getResource("top250nl_Hoogte.xml");
78      TopNLType type = TopNLType.TOP250NL;
79      List jaxb = instance.parse(in);
80      List<TopNLEntity> hoogte = instance.convert(jaxb, type);
81      instance.save(hoogte.get(0), type);
82  
83      ResultSetHandler<Hoogte> handler =
84          new BeanHandler<>(
85              Hoogte.class,
86              new BasicRowProcessor(
87                  new DbUtilsGeometryColumnConverter(
88                      GeometryJdbcConverterFactory.getGeometryJdbcConverter(
89                          datasource.getConnection()))));
90  
91      Hoogte real =
92          run.query(
93              "SELECT * FROM top250nl.Hoogte WHERE identificatie=?",
94              handler,
95              "NL.TOP250NL.16R09-0000084246");
96      assertNotNull(real);
97    }
