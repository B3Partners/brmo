109     @Before
110     @Override
111     public void setUp() throws Exception {
112         assumeTrue("Het bestand met staging testdata zou moeten bestaan.", AfgifteNummerScannerIntegrationTest.class.getResource(sBestandsNaam) != null);
113 
114         staging = new DatabaseDataSourceConnection(dsStaging);
115 
116         if (this.isMsSQL) {
117             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
118         } else if (this.isOracle) {
119             dsStaging.getConnection().setAutoCommit(true);
120             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), DBPROPS.getProperty("staging.username").toUpperCase());
121             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
122             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
123         } else if (this.isPostgis) {
124             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
125         } else {
126             fail("Geen ondersteunde database aangegegeven");
127         }
128 
129         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
130         fxdb.setCaseSensitiveTableNames(false);
131         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(AfgifteNummerScannerIntegrationTest.class.getResource(sBestandsNaam).toURI())));
132 
133         sequential.lock();
134         if (this.isMsSQL) {
135             // SET IDENTITY_INSERT op ON
136             InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
137         } else {
138             DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
139         }
140 
141         brmo = new BrmoFramework(dsStaging, null);
142 
143         scanner = new AfgifteNummerScanner(new AfgifteNummerScannerProces());
144         FieldUtils.writeField(scanner, "listener", new ProgressUpdateListener() {
145             @Override
146             public void total(long total) {
147             }
148 
149             @Override
150             public void progress(long progress) {
151             }
152 
153             @Override
154             public void exception(Throwable t) {
155             }
156 
157             @Override
158             public void updateStatus(String status) {
159             }
160 
161             @Override
162             public void addLog(String log) {
163             }
164         }, true);
165 
166         assumeTrue("Er zijn anders dan verwacht aantal laadprocessen", lAantalLaadProcessen == brmo.getCountLaadProcessen(null, null, null, null));
167 
168         if (!this.isMsSQL) {
169             // mssql gebruikt identity kolom
170             try {
171                 ITable lp = staging.createQueryTable("laadproces", "select max(id) as maxid from laadproces");
172                 Number maxLP = (Number) lp.getValue(0, "maxid");
173                 SequenceUtil.updateSequence("laadproces_id_seq", maxLP.longValue() + 9999999, dsStaging);
174             } catch (SQLException s) {
175                 LOG.error("Bijwerken laadproces sequnce is mislukt", s);
176             }
177         }
178     }

180     @After
181     public void tearDown() throws Exception {
182         if (brmo != null) {
183             brmo.closeBrmoFramework();
184         }
185         if (staging != null) {
186             CleanUtil.cleanSTAGING(staging, true);
187             staging.close();
188         }
189         try {
190             sequential.unlock();
191         } catch (IllegalMonitorStateException e) {
192             // in geval van niet waar gemaakte assumptions
193             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
194         }
195     }

227         List<Map<String, Object>> afgiftenummers = scanner.getOntbrekendeAfgiftenummers(sContractNummer, "contractafgiftenummer");
228 
229         Iterator<Map<String, Object>> records = afgiftenummers.iterator();
230         Map<String, Object> rec;
231         long inserted = 0;

234

240         }
241

243     }
