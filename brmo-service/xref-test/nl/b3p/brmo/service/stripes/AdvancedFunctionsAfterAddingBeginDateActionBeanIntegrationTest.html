184         }

187     @Test
188     public void testReplayBRKVerwijderBerichtenNaUpdateIngangsdatumRecht() throws Exception {
189         
190         updatesBean.populateUpdateProcesses();
191         updatesBean.setUpdateProcessName("Bijwerken ingangsdatum_recht zakelijk recht");
192         updatesBean.update();
193         Date now = new Date();
194         long maxWait = 20000;
195         while(!updatesBean.isComplete() && (new Date().getTime() - now.getTime() < maxWait)){
196             Thread.sleep(1000);
197             LOG.info("waiting...");
198         }
199         LOG.info("Finished waiting");
200         
201         final IDataSet rds = rsgb.createDataSet();
202         ITable zak_recht = rsgb.createDataSet().getTable("zak_recht");
203         for (int i = 0; i < zak_recht.getRowCount(); i++) {
204             assertNotNull("Ingangsdatum recht is niet gevuld", zak_recht.getValue(i, "ingangsdatum_recht"));
205             assertNotNull("fk_3avr_aand recht is niet gevuld", zak_recht.getValue(i, "fk_3avr_aand"));
206         }
207 
208         // deze gegevens komen uit de test data set
209         assertEquals("Er is een spook record in de kad_onrrnd_zk tabel", 1, rds.getTable("kad_onrrnd_zk").getRowCount());
210         assertEquals("De perceel tabel is leeg", 0, rds.getTable("kad_perceel").getRowCount());
211         assertEquals("De kad_onrrnd_zk_archief komt een record te kort", aantalBerichten - (1 + 1), rds.getTable("kad_onrrnd_zk_archief").getRowCount());
212         assertTrue("Er is minstens een perceel in kad_perceel_archief", 0 < rds.getTable("kad_perceel_archief").getRowCount());
213 
214         bean.replayBRKVerwijderBerichten(BrmoFramework.BR_BRK, Bericht.STATUS.RSGB_OK.toString());
215 
216         assertEquals("Er zijn geen spook records in de kad_onrrnd_zk tabel", 0, rds.getTable("kad_onrrnd_zk").getRowCount());
217         assertTrue("De kad_onrrnd_zk_archief tabel is niet leeg", 0 < rds.getTable("kad_onrrnd_zk_archief").getRowCount());
218         assertEquals("Er zit voor ieder bericht met perceel een record in kad_perceel_archief",
219                 aantalBerichten - 1,
220                 rds.getTable("kad_perceel_archief").getRowCount()
221         );
222 
223         assertEquals(
224                 "Alle berichten hebben status RSGB_OK",
225                 aantalBerichten,
226                 brmo.getCountBerichten(null,null,"brk","RSGB_OK")
227         );
