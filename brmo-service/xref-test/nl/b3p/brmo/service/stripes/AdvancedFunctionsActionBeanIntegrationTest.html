109     private void loadData(String sBestandsNaam, long aantalBerichten, long aantalProcessen, String rBestandsNaam)
110             throws Exception {
111         Assumptions.assumeTrue(AdvancedFunctionsActionBeanIntegrationTest.class.getResource(sBestandsNaam) != null,
112                 "Het bestand met staging testdata zou moeten bestaan.");
113         Assumptions.assumeTrue(AdvancedFunctionsActionBeanIntegrationTest.class.getResource(rBestandsNaam) != null,
114                 "Het bestand met rsgb testdata zou moeten bestaan.");
115         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
116         fxdb.setCaseSensitiveTableNames(false);
117         IDataSet stagingDataSet = fxdb.build(new FileInputStream(
118                 new File(AdvancedFunctionsActionBeanIntegrationTest.class.getResource(sBestandsNaam).toURI())));
119         IDataSet rsgbDataSet = fxdb.build(new FileInputStream(
120                 new File(AdvancedFunctionsActionBeanIntegrationTest.class.getResource(rBestandsNaam).toURI())));
121 
122         if (isMsSQL) {
123             // SET IDENTITY_INSERT op ON
124             InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
125             InsertIdentityOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
126         } else {
127             DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
128             DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
129         }
130         Assumptions.assumeTrue(aantalBerichten == brmo.getCountBerichten(null, null, "brk", "RSGB_OK"),
131                 "Er zijn x RSGB_OK berichten");
132         Assumptions.assumeTrue(aantalProcessen == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"),
133                 "Er zijn x STAGING_OK laadprocessen");
134     }

136     @AfterEach
137     public void cleanup() throws Exception {
138         if (brmo != null) {
139             // in geval van niet waar gemaakte assumptions
140             brmo.closeBrmoFramework();
141         }
142         if (rsgb != null) {
143             CleanUtil.cleanRSGB_BRK(rsgb, true);
144             rsgb.close();
145         }
146         if (staging != null) {
147             CleanUtil.cleanSTAGING(staging, true);
148             staging.close();
149         }
150 
151         try {
152             sequential.unlock();
153         } catch (IllegalMonitorStateException e) {
154             // in geval van niet waar gemaakte assumptions
155             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
156         }
157     }

159     @ParameterizedTest(name = "testReplayBRKVerwijderBerichten {index}: verwerken bestand: ''{0}''")
160     @MethodSource("argumentsProvider")
161     public void testReplayBRKVerwijderBerichten(String sBestandsNaam, long aantalBerichten, long aantalProcessen,
162                                                 String rBestandsNaam) throws Exception {
163         loadData(sBestandsNaam, aantalBerichten, aantalProcessen, rBestandsNaam);
164         final IDataSet rds = rsgb.createDataSet();
165 
166         assertEquals(1, rds.getTable("kad_onrrnd_zk").getRowCount(),
167                 "Er is een spook record in de kad_onrrnd_zk tabel");
168         assertEquals(0, rds.getTable("kad_perceel").getRowCount(), "De perceel tabel is leeg");
169         assertEquals(aantalBerichten - (1 + 1), rds.getTable("kad_onrrnd_zk_archief").getRowCount(),
170                 "De kad_onrrnd_zk_archief komt een record te kort");
171         assertTrue(0 < rds.getTable("kad_perceel_archief").getRowCount(),
172                 "Er is minstens een perceel in kad_perceel_archief");
173 
174         bean.replayBRKVerwijderBerichten(BrmoFramework.BR_BRK, Bericht.STATUS.RSGB_OK.toString());
175 
176         assertEquals(0, rds.getTable("kad_onrrnd_zk").getRowCount(),
177                 "Er zijn geen spook records in de kad_onrrnd_zk tabel");
178         assertTrue(0 < rds.getTable("kad_onrrnd_zk_archief").getRowCount(),
179                 "De kad_onrrnd_zk_archief tabel is niet leeg");
180         assertEquals(
181                 aantalBerichten - 1,
182                 rds.getTable("kad_perceel_archief").getRowCount(),
183                 "Er zit voor ieder bericht met perceel in record in kad_perceel_archief");
184 
185         ITable bericht = staging.createDataSet().getTable("bericht");
186         assertEquals(aantalBerichten, bericht.getRowCount(), "Alle berichten hebben status RSGB_OK");
187     }

189     @DisplayName("Cleanup")
190     @ParameterizedTest(name = "testFillbestandsNaamHersteld {index}: verwerken bestand: ''{0}''")
191     @MethodSource("argumentsProvider")
192     public void testFillbestandsNaamHersteld(String sBestandsNaam, long aantalBerichten, long aantalProcessen,
193                                              String rBestandsNaam) throws Exception {
194         loadData(sBestandsNaam, aantalBerichten, aantalProcessen, rBestandsNaam);
195         bean.fillbestandsNaamHersteld(BrmoFramework.BR_BRK, "0");
196         ITable laadproces = staging.createDataSet().getTable("laadproces");
197 
198         for (int i = 1; i < laadproces.getRowCount(); i++) {
199             if (laadproces.getValue(i, "bestand_naam").toString().equals("stand")) {
200                 assertNull(laadproces.getValue(i, "bestand_naam_hersteld"),
201                         "'bestand_naam_hersteld' moet leeg zijn voor stand");
202             } else if (laadproces.getValue(i, "bestand_naam").toString().equals("verwijder")) {
203                 assertNotNull(laadproces.getValue(i, "bestand_naam_hersteld"),
204                         "'bestand_naam_hersteld' mag niet leeg zijn");
205                 assertEquals(
206                         "bestandsnaam kon niet worden hersteld", laadproces.getValue(i, "bestand_naam_hersteld"),
207                         "inhoud mag niet herstelbaar zijn");
208             } else {
209                 assertNotNull(laadproces.getValue(i, "bestand_naam_hersteld"),
210                         "'bestand_naam_hersteld' mag niet leeg zijn");
211                 assertNotEquals(
212                         "bestandsnaam kon niet worden hersteld", laadproces.getValue(i, "bestand_naam_hersteld"),
213                         "inhoud moet herstelbaar zijn");
214             }
215         }
