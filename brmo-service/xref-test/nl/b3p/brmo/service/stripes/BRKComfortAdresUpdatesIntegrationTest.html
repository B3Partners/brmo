72      @BeforeEach
73      public void setUp() throws Exception {
74          // mock de context van de bean
75          ServletContext sctx = mock(ServletContext.class);
76          ActionBeanContext actx = mock(ActionBeanContext.class);
77          bean = spy(UpdatesActionBean.class);
78          when(bean.getContext()).thenReturn(actx);
79          when(actx.getServletContext()).thenReturn(sctx);
80  
81          rsgb = new DatabaseConnection(dsRsgb.getConnection());
82          staging = new DatabaseConnection(dsStaging.getConnection());
83  
84          if (isMsSQL) {
85              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
86              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
87          } else if (isOracle) {
88              rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
89                      DBPROPS.getProperty("rsgb.username").toUpperCase());
90              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
91              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
92              staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
93                      DBPROPS.getProperty("staging.username").toUpperCase());
94              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
95              staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
96          } else if (isPostgis) {
97              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
98              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
99          } else {
100             fail("Geen ondersteunde database aangegegeven");
101         }
102         staging.getConfig().setProperty(DatabaseConfig.FEATURE_ALLOW_EMPTY_FIELDS, true);
103         rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_ALLOW_EMPTY_FIELDS, true);
104         brmo = new BrmoFramework(dsStaging, dsRsgb);
105         brmo.setOrderBerichten(true);
106         sequential.lock();
107     }

109     private void loadData(String sStagingBestand, String sRsgbBestand) throws Exception {
110         Assumptions.assumeTrue(BRKComfortAdresUpdatesIntegrationTest.class.getResource(sStagingBestand) != null,
111                 "Het bestand met staging testdata zou moeten bestaan.");
112         Assumptions.assumeTrue(BRKComfortAdresUpdatesIntegrationTest.class.getResource(sRsgbBestand) != null,
113                 "Het bestand met rsgb testdata zou moeten bestaan.");
114         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
115         fxdb.setCaseSensitiveTableNames(false);
116         IDataSet stagingDataSet = fxdb.build(new FileInputStream(
117                 new File(BRKComfortAdresUpdatesIntegrationTest.class.getResource(sStagingBestand).toURI())));
118         IDataSet rsgbDataSet = fxdb.build(new FileInputStream(
119                 new File(BRKComfortAdresUpdatesIntegrationTest.class.getResource(sRsgbBestand).toURI())));
120 
121         if (isMsSQL) {
122             // SET IDENTITY_INSERT op ON
123             InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
124             InsertIdentityOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
125         } else {
126             DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
127             DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
128         }
129 
130         Assumptions.assumeTrue(
131                 1 == brmo.getCountBerichten(null, null, "brk", "RSGB_OK"),
132                 "Er zijn x RSGB_OK berichten"
133         );
134         Assumptions.assumeTrue(
135                 1 == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"),
136                 "Er zijn x STAGING_OK laadprocessen"
137         );
138     }

140     @AfterEach
141     public void cleanup() throws Exception {
142         if (brmo != null) {
143             // in geval van niet waar gemaakte assumptions
144             brmo.closeBrmoFramework();
145             brmo = null;
146         }
147         if (rsgb != null) {
148             CleanUtil.cleanRSGB_BRK(rsgb, true);
149             rsgb.close();
150             rsgb = null;
151         }
152         if (staging != null) {
153             CleanUtil.cleanSTAGING(staging, true);
154             staging.close();
155             staging = null;
156         }
157 
158         try {
159             sequential.unlock();
160         } catch (IllegalMonitorStateException e) {
161             // in geval van niet waar gemaakte assumptions
162             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
163         }
164     }

166     /**
167      * Test of de snelle update de kolom `adres_binnenland` bijwerkt met comfort
168      * adres op basis van woonlocatie, eventueel wordt ook postbus adres gevuld
169      * uit postlocatie.
170      *
171      * @throws Exception if any
172      */
173     @ParameterizedTest(name =  "Snelle Update {index}: verwerken bestanden: ''{0}'' en ''{1}''")
174     @MethodSource("argumentsProvider")
175     public void testSnelleUpdate(String sStagingBestand, String sRsgbBestand) throws Exception {
176         loadData(sStagingBestand, sRsgbBestand);
177 
178         final IDataSet rds = rsgb.createDataSet();
179         ITable subject = rds.getTable("subject");
180         for (int i = 0; i < subject.getRowCount(); i++) {
181             assertNull(subject.getValue(i, "adres_binnenland"), "'adres_binnenland' is niet 'null'");
182 
183             if (sStagingBestand.contains("postbus")) {
184                 assertNull(subject.getValue(i, "pa_postadres_postcode"), "'pa_postadres_postcode' is 'null'");
185                 assertNull(subject.getValue(i, "pa_postbus__of_antwoordnummer"),
186                         "'pa_postbus__of_antwoordnummer' is niet 'null'");
187                 assertNull(subject.getValue(i, "pa_postadrestype"), "'pa_postadrestype'  is niet 'null'");
188             }
189         }
190 
191         bean.populateUpdateProcesses();
192         bean.setUpdateProcessName("Bijwerken subject adres comfort data");
193         bean.update();
194 
195         assertEquals(1, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"),
196                 "niet alle berichten zijn 'RSGB_OK'");
197 
198         // subject tabel opnieuw openen
199         subject = rds.getTable("subject");
200         for (int i = 0; i < subject.getRowCount(); i++) {
201             LOG.debug("Rij: " + i + " adres: " + subject.getValue(i, "adres_binnenland"));
202             assertNotNull(subject.getValue(i, "adres_binnenland"), "'adres_binnenland' is 'null'");
203             assertFalse(subject.getValue(i, "adres_binnenland").toString().isEmpty(),
204                     "'adres_binnenland' is leeg");
205 
206             if (sStagingBestand.contains("postbus")) {
207                 assertNotNull(subject.getValue(i, "pa_postadres_postcode"),
208                         "'pa_postadres_postcode' is 'null'");
209                 assertNotNull(subject.getValue(i, "pa_postbus__of_antwoordnummer"),
210                         "'pa_postbus__of_antwoordnummer' is 'null'");
211                 assertEquals("P", subject.getValue(i, "pa_postadrestype"), "waarde moet 'P' zijn");
212             }
213         }
