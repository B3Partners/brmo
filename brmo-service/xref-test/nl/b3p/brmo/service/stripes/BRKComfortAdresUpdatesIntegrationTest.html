103       staging
104           .getConfig()

116   private void loadData(String sStagingBestand, String sRsgbBestand) throws Exception {
117     assumeTrue(
118         BRKComfortAdresUpdatesIntegrationTest.class.getResource(sStagingBestand) != null,
119         "Het bestand met staging testdata zou moeten bestaan.");
120     assumeTrue(
121         BRKComfortAdresUpdatesIntegrationTest.class.getResource(sRsgbBestand) != null,
122         "Het bestand met rsgb testdata zou moeten bestaan.");
123     FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
124     fxdb.setCaseSensitiveTableNames(false);
125     IDataSet stagingDataSet =
126         fxdb.build(
127             new FileInputStream(
128                 new File(
129                     BRKComfortAdresUpdatesIntegrationTest.class
130                         .getResource(sStagingBestand)
131                         .toURI())));
132     IDataSet rsgbDataSet =
133         fxdb.build(
134             new FileInputStream(
135                 new File(
136                     BRKComfortAdresUpdatesIntegrationTest.class
137                         .getResource(sRsgbBestand)
138                         .toURI())));
139 
140     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
141     DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
142 
143     assumeTrue(1 == brmo.getCountBerichten("brk", "RSGB_OK"), "Er zijn x RSGB_OK berichten");
144     assumeTrue(
145         1 == brmo.getCountLaadProcessen("brk", "STAGING_OK"), "Er zijn x STAGING_OK laadprocessen");
146   }

148   @AfterEach
149   public void cleanup() throws Exception {
150     if (brmo != null) {
151       // in geval van niet waar gemaakte assumptions
152       brmo.closeBrmoFramework();
153       brmo = null;
154     }
155     bean = null;
156     if (rsgb != null) {
157       CleanUtil.cleanRSGB_BRK(rsgb, true);
158       rsgb.close();
159       rsgb = null;
160     }
161     if (staging != null) {
162       CleanUtil.cleanSTAGING(staging, true);
163       staging.close();
164       staging = null;
165     }
166     try {
167       sequential.unlock();
168     } catch (IllegalMonitorStateException e) {
169       // in geval van niet waar gemaakte assumptions
170       LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
171     }
172   }

174   /**
175    * Test of de snelle update de kolom `adres_binnenland` bijwerkt met comfort adres op basis van
176    * woonlocatie, eventueel wordt ook postbus adres gevuld uit postlocatie.
177    *
178    * @throws Exception if any
179    */
180   @ParameterizedTest(name = "Snelle Update {index}: verwerken bestanden: ''{0}'' en ''{1}''")
181   @MethodSource("argumentsProvider")
182   public void testSnelleUpdate(String sStagingBestand, String sRsgbBestand) throws Exception {
183     loadData(sStagingBestand, sRsgbBestand);
184 
185     final IDataSet rds = rsgb.createDataSet();
186     ITable subject = rds.getTable("subject");
187     for (int i = 0; i < subject.getRowCount(); i++) {
188       assertNull(subject.getValue(i, "adres_binnenland"), "'adres_binnenland' is niet 'null'");
189 
190       if (sStagingBestand.contains("postbus")) {
191         assertNull(
192             subject.getValue(i, "pa_postadres_postcode"), "'pa_postadres_postcode' is 'null'");
193         assertNull(
194             subject.getValue(i, "pa_postbus__of_antwoordnummer"),
195             "'pa_postbus__of_antwoordnummer' is niet 'null'");
196         assertNull(subject.getValue(i, "pa_postadrestype"), "'pa_postadrestype'  is niet 'null'");
197       }
198     }
199 
200     bean.populateUpdateProcesses();
201     bean.setUpdateProcessName("Bijwerken subject adres comfort data");
202     bean.update();
203 
204     assertEquals(1, brmo.getCountBerichten("brk", "RSGB_OK"), "niet alle berichten zijn 'RSGB_OK'");
205 
206     // subject tabel opnieuw openen
207     subject = rds.getTable("subject");
208     for (int i = 0; i < subject.getRowCount(); i++) {
209       LOG.debug("Rij: " + i + " adres: " + subject.getValue(i, "adres_binnenland"));
210       assertNotNull(subject.getValue(i, "adres_binnenland"), "'adres_binnenland' is 'null'");
211       assertFalse(
212           subject.getValue(i, "adres_binnenland").toString().isEmpty(),
213           "'adres_binnenland' is leeg");
214 
215       if (sStagingBestand.contains("postbus")) {
216         assertNotNull(
217             subject.getValue(i, "pa_postadres_postcode"), "'pa_postadres_postcode' is 'null'");
218         assertNotNull(
219             subject.getValue(i, "pa_postbus__of_antwoordnummer"),
220             "'pa_postbus__of_antwoordnummer' is 'null'");
221         assertEquals("P", subject.getValue(i, "pa_postadrestype"), "waarde moet 'P' zijn");
222       }
223     }
