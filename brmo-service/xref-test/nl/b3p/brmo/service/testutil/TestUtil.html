4   package nl.b3p.brmo.service.testutil;
5   
6   import java.io.IOException;
7   import java.util.Properties;
8   import javax.naming.Context;
9   import javax.naming.InitialContext;
10  import javax.naming.NamingException;
11  import org.apache.commons.dbcp.BasicDataSource;
12  import org.apache.commons.logging.Log;
13  import org.apache.commons.logging.LogFactory;
14  import org.junit.After;
15  import static org.junit.Assume.assumeNotNull;
16  import org.junit.Before;
17  import org.junit.BeforeClass;
18  import org.junit.Rule;
19  import org.junit.rules.TestName;
20  
21  /**
22   *
23   * @author mprins
24   */
25  public abstract class TestUtil {
26  
27      private static boolean haveSetupJNDI = false;
28  
29      private static final Log LOG = LogFactory.getLog(TestUtil.class);
30      /**
31       * properties uit {@code <DB smaak>.properties} en
32       * {@code local.<DB smaak>.properties}.
33       *
34       * @see #loadDBprop()
35       */
36      protected final Properties DBPROPS = new Properties();
37  
38      /**
39       * {@code true} als we met een Oracle database bezig zijn.
40       */
41      protected boolean isOracle;
42  
43      /**
44       * {@code true} als we met een MS SQL Server database bezig zijn.
45       */
46      protected boolean isMsSQL;
47  
48      /**
49       * {@code true} als we met een Postgis database bezig zijn.
50       */
51      protected boolean isPostgis;
52  
53      protected BasicDataSource dsStaging;
54      protected BasicDataSource dsRsgb;
55      protected BasicDataSource dsRsgbBgt;
56  
57      /**
58       * logging rule.
59       */
60      @Rule
61      public TestName name = new TestName();
62  
63      /**
64       * test of de database properties zijn aangegeven, zo niet dan skippen we
65       * alle tests in deze test.
66       */
67      @BeforeClass
68      public static void checkDatabaseIsProvided() {
69          assumeNotNull("Verwacht database omgeving te zijn aangegeven.", System.getProperty("database.properties.file"));
70      }
71  
72      /**
73       * subklassen dienen zelf een setup te hebben; vanwege de overerving gaat
74       * deze methode af na de {@code @Before} methoden van de superklasse, bijv.
75       * {@link #loadDBprop()}.
76       *
77       * @throws Exception if any
78       */
79      @Before
80      abstract public void setUp() throws Exception;
81  
82      /**
83       * initialize database props using the environment provided file.
84       *
85       * @throws java.io.IOException if loading the property file fails
86       */
87      @Before
88      public void loadDBprop() throws IOException {
89          // the `database.properties.file`  is set in the pom.xml or using the commandline
90          DBPROPS.load(TestUtil.class.getClassLoader()
91                  .getResourceAsStream(System.getProperty("database.properties.file")));
92          try {
93              // see if a local version exists and use that to override
94              DBPROPS.load(TestUtil.class.getClassLoader()
95                      .getResourceAsStream("local." + System.getProperty("database.properties.file")));
96          } catch (IOException | NullPointerException e) {
97              // ignore this
98          }
99  
100         isOracle = "oracle".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
101         isMsSQL = "jtds-sqlserver".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
102         isPostgis = "postgis".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
103 
104         try {
105             Class driverClass = Class.forName(DBPROPS.getProperty("jdbc.driverClassName"));
106         } catch (ClassNotFoundException ex) {
107             LOG.error("Database driver niet gevonden.", ex);
108         }
109 
110         dsStaging = new BasicDataSource();
111         dsStaging.setUrl(DBPROPS.getProperty("staging.url"));
112         dsStaging.setUsername(DBPROPS.getProperty("staging.username"));
113         dsStaging.setPassword(DBPROPS.getProperty("staging.password"));
114         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
115 
116         dsRsgb = new BasicDataSource();
117         dsRsgb.setUrl(DBPROPS.getProperty("rsgb.url"));
118         dsRsgb.setUsername(DBPROPS.getProperty("rsgb.username"));
119         dsRsgb.setPassword(DBPROPS.getProperty("rsgb.password"));
120         dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
121 
122         dsRsgbBgt = new BasicDataSource();
123         dsRsgbBgt.setUrl(DBPROPS.getProperty("rsgb.url"));
124         dsRsgbBgt.setUsername(DBPROPS.getProperty("rsgb.username"));
125         dsRsgbBgt.setPassword(DBPROPS.getProperty("rsgb.password"));
126         dsRsgbBgt.setAccessToUnderlyingConnectionAllowed(true);
127 
128         setupJNDI();
129     }
130 
131     /**
132      * Log de naam van de test als deze begint.
133      */
134     @Before
135     public void startTest() {
136         LOG.info("==== Start test methode: " + name.getMethodName());
137     }
138 
139     /**
140      * Log de naam van de test als deze eindigt.
141      */
142     @After
143     public void endTest() {
144         LOG.info("==== Einde test methode: " + name.getMethodName());
145     }
146 
147     /**
148      * setup jndi voor testcases.
149      */
150     protected void setupJNDI() {
151         if (!haveSetupJNDI) {
152             try {
153                 System.setProperty(Context.INITIAL_CONTEXT_FACTORY, "org.apache.naming.java.javaURLContextFactory");
154                 System.setProperty(Context.URL_PKG_PREFIXES, "org.apache.naming");
155                 InitialContext ic = new InitialContext();
156                 ic.createSubcontext("java:");
157                 ic.createSubcontext("java:comp");
158                 ic.createSubcontext("java:comp/env");
159                 ic.createSubcontext("java:comp/env/jdbc");
160                 ic.createSubcontext("java:comp/env/jdbc/brmo");
161                 ic.bind("java:comp/env/jdbc/brmo/rsgb", dsRsgb);
162                 ic.bind("java:comp/env/jdbc/brmo/staging", dsStaging);
163                 ic.bind("java:comp/env/jdbc/brmo/staging", dsRsgbBgt);
164                 haveSetupJNDI = true;
165             } catch (NamingException ex) {
166                 LOG.warn("Opzetten van datasource jndi is mislukt", ex);
167             }
168         }
169     }
170 }
