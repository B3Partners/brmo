36   * @author Mark Prins
37   */
38  public class DownloadLogIntegrationTest extends WebTestUtil {
39  
40    /**
41     * Test login/downloadlog/logout sequentie.
42     *
43     * @throws IOException mag niet optreden
44     * @throws URISyntaxException mag niet optreden
45     */
46    @Test
47    public void testLoginDownloadLogsLogoutPage() throws Exception {
48      // login
49      HttpResponse response = client.execute(new HttpGet(BASE_TEST_URL));
50      EntityUtils.consume(response.getEntity());
51  
52      HttpUriRequest login =
53          RequestBuilder.post()
54              .setUri(new URI(BASE_TEST_URL + "j_security_check"))
55              .addParameter("j_username", "brmo")
56              .addParameter("j_password", "brmo")
57              .build();
58      response = client.execute(login);
59      EntityUtils.consume(response.getEntity());
60      assertThat(
61          "Response status is OK.",
62          response.getStatusLine().getStatusCode(),
63          equalTo(HttpStatus.SC_OK));
64  
65      // downloadlog
66      response = client.execute(new HttpGet(BASE_TEST_URL + "downloadlog.jsp"));
67      assertThat(
68          "Response status is OK.",
69          response.getStatusLine().getStatusCode(),
70          equalTo(HttpStatus.SC_OK));
71      String body = EntityUtils.toString(response.getEntity());
72      assertNotNull(body, "Response body mag niet null zijn.");
73      // System.out.println("\n\n=============== BODY:=================\n" + body + "\n\n");
74      // test eerste regel uit de logfile
75      assertTrue(
76          body.contains(
77              "Quartz scheduler 'BRMOgeplandeTaken' initialized from an externally provided properties instance."));
78  
79      // logout
80      response = client.execute(new HttpGet(BASE_TEST_URL + "logout.jsp"));
81      body = EntityUtils.toString(response.getEntity());
82      assertThat(
83          "Response status is OK.",
84          response.getStatusLine().getStatusCode(),
85          equalTo(HttpStatus.SC_OK));
86      assertNotNull(body, "Response body mag niet null zijn.");
87      assertTrue(body.contains("Opnieuw inloggen"), "Response moet 'Opnieuw inloggen' tekst hebben.");
88    }

92      // void implementatie
93    }
94  }
