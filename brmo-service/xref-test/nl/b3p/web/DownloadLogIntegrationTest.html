28  import org.apache.http.util.EntityUtils;
29  import static org.hamcrest.CoreMatchers.equalTo;
30  import static org.junit.Assert.assertNotNull;
31  import static org.junit.Assert.assertThat;
32  import static org.junit.Assert.assertTrue;
33  import org.junit.Test;
34  
35  /**
36   *
37   * @author Mark Prins
38   */
39  public class DownloadLogIntegrationTest extends WebTestUtil {
40  
41      /**
42       * onze test response.
43       */
44      private HttpResponse response;
45  
46      /**
47       * Test login/downloadlog/logout sequentie.
48       *
49       * @throws IOException mag niet optreden
50       * @throws URISyntaxException mag niet optreden
51       */
52      @Test
53      public void testLoginDownloadLogsLogoutPage() throws IOException, URISyntaxException {
54          // login
55          response = client.execute(new HttpGet(BASE_TEST_URL));
56          EntityUtils.consume(response.getEntity());
57  
58          HttpUriRequest login = RequestBuilder.post()
59                  .setUri(new URI(BASE_TEST_URL + "j_security_check"))
60                  .addParameter("j_username", "brmo")
61                  .addParameter("j_password", "brmo")
62                  .build();
63          response = client.execute(login);
64          EntityUtils.consume(response.getEntity());
65          assertThat("Response status is OK.", response.getStatusLine().getStatusCode(),
66                  equalTo(HttpStatus.SC_OK));
67  
68          // downloadlog
69          response = client.execute(new HttpGet(BASE_TEST_URL + "downloadlog.jsp"));
70          assertThat("Response status is OK.", response.getStatusLine().getStatusCode(),
71                  equalTo(HttpStatus.SC_OK));
72          String body = EntityUtils.toString(response.getEntity());
73          assertNotNull("Response body mag niet null zijn.", body);
74          // System.out.println("\n\n=============== BODY:=================\n" + body + "\n\n");
75          // test eerste regel uit de logfile
76          assertTrue(body.contains("Quartz scheduler 'BRMOgeplandeTaken' initialized from an externally provided properties instance."));
77  
78          // logout
79          response = client.execute(new HttpGet(BASE_TEST_URL + "logout.jsp"));
80          body = EntityUtils.toString(response.getEntity());
81          assertThat("Response status is OK.", response.getStatusLine().getStatusCode(),
82                  equalTo(HttpStatus.SC_OK));
83          assertNotNull("Response body mag niet null zijn.", body);
84          assertTrue("Response moet 'Opnieuw inloggen' tekst hebben.", body.contains("Opnieuw inloggen"));
85      }
86  
87      @Override
88      public void setUp() throws Exception {
89          // void implementatie
90      }
91  }
