4   package nl.b3p.brmo.service.jobs;
5   
6   import java.io.PrintWriter;
7   import java.io.StringWriter;
8   import java.util.Date;
9   import javax.persistence.EntityManager;
10  import nl.b3p.brmo.loader.util.BrmoException;
11  import nl.b3p.brmo.persistence.staging.AutomatischProces;
12  import nl.b3p.brmo.service.scanner.AbstractExecutableProces;
13  import org.apache.commons.logging.Log;
14  import org.apache.commons.logging.LogFactory;
15  import org.quartz.JobExecutionContext;
16  import org.quartz.JobExecutionException;
17  import org.quartz.Job;
18  import org.stripesstuff.stripersist.Stripersist;
19  
20  /**
21   * Start door quartz beheerde automatische processen.
22   *
23   * @author mprins
24   */
25  public class AutomatischProcesJob implements Job {
26  
27      private static final Log log = LogFactory.getLog(AutomatischProcesJob.class);
28  
29      @Override
30      public void execute(JobExecutionContext context) throws JobExecutionException {
31          Long id = null;
32          AutomatischProces p = null;
33          try {
34              id = context.getJobDetail().getJobDataMap().getLongValue("id");
35              log.debug("Poging gepland automatisch proces met id: " + id + " te starten.");
36              Stripersist.requestInit();
37              p = Stripersist.getEntityManager().find(AutomatischProces.class, id);
38              if (p != null) {
39                  p.addLogLine(String.format("Geplande taak gestart op %tc.", new Date()));
40                  AbstractExecutableProces.getProces(p).execute();
41                  p.addLogLine(String.format("Geplande taak afgerond op %tc.", new Date()));
42                  p.setLastrun(new Date());
43                  Stripersist.getEntityManager().merge(p);
44                  Stripersist.getEntityManager().flush();
45                  log.debug("Gepland automatisch proces met " + id + " is afgerond.");
46              } else {
47                  log.warn("Automatisch proces met id:" + id + " bestaat niet.");
48              }
49          } catch (ClassCastException cce) {
50              log.warn("Geen geldige ID gevonden in de jobdata map.", cce);
51          } catch (BrmoException ex) {
52              if (p != null) {
53                  StringWriter sw = new StringWriter();
54                  ex.printStackTrace(new PrintWriter(sw));
55                  p.setStatus(AutomatischProces.ProcessingStatus.ERROR);
56                  p.updateSamenvattingEnLogfile(String.format("Fout tijdens starten of uitvoeren van automatische proces met id %s: %s ", id, sw.toString()));
57                  Stripersist.getEntityManager().merge(p);
58              }
59              log.error("Fout tijdens starten of uitvoeren van automatische proces met id: " + id, ex);
60          } finally {
61              if (Stripersist.getEntityManager().getTransaction().isActive()) {
62                  Stripersist.getEntityManager().getTransaction().commit();
63              }
64          }
65      }
66  }
