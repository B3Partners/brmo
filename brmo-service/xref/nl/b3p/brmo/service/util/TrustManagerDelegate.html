4   package nl.b3p.brmo.service.util;
5   
6   import javax.net.ssl.X509TrustManager;
7   import java.security.cert.CertificateException;
8   import java.security.cert.X509Certificate;
9   
10  /**
11   *
12   * @author mprins
13   */
14  public class TrustManagerDelegate implements X509TrustManager {
15  
16      private final X509TrustManager mainTrustManager;
17      private final X509TrustManager fallbackTrustManager;
18  
19      public TrustManagerDelegate(X509TrustManager mainTrustManager, X509TrustManager fallbackTrustManager) {
20          this.mainTrustManager = mainTrustManager;
21          this.fallbackTrustManager = fallbackTrustManager;
22      }
23  
24      @Override
25      public void checkClientTrusted(final X509Certificate[] x509Certificates, final String authType) throws CertificateException {
26          try {
27              mainTrustManager.checkClientTrusted(x509Certificates, authType);
28          } catch (CertificateException ignored) {
29              this.fallbackTrustManager.checkClientTrusted(x509Certificates, authType);
30          }
31      }
32  
33      @Override
34      public void checkServerTrusted(final X509Certificate[] x509Certificates, final String authType) throws CertificateException {
35          try {
36              mainTrustManager.checkServerTrusted(x509Certificates, authType);
37          } catch (CertificateException ignored) {
38              this.fallbackTrustManager.checkServerTrusted(x509Certificates, authType);
39          }
40      }
41  
42      @Override
43      public X509Certificate[] getAcceptedIssuers() {
44          return this.fallbackTrustManager.getAcceptedIssuers();
45      }
46  }
