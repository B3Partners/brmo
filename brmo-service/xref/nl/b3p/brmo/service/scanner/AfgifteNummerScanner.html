60      /**
61       * Zoek contractnummers op van de GDS2 jobs.
62       *
63       * @return lijst met contractnummers, de lijst kan leeg zijn
64       */
65      public static List<String> contractnummers() {
66          try {
67              final DataSource ds = ConfigUtil.getDataSourceStaging();
68              try (final Connection conn = ds.getConnection()) {
69                  final GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(conn);
70                  String sql = "select distinct cast(value as varchar(15)) from automatisch_proces_config where config_key = 'gds2_contractnummer'";
71                  List<String> contractnummers = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(conn, sql, new ColumnListHandler<>());
72                  // omdat sql server geen distinct op een 'text' kolom kan doen halen we alles op en doen we distinct + sorteren aan de java kant
73                  //contractnummers = new ArrayList<>(new HashSet(contractnummers));
74                  contractnummers.sort(String::compareToIgnoreCase);
75                  DbUtils.closeQuietly(conn);
76                  return Collections.unmodifiableList(contractnummers);
77              }
78          } catch (BrmoException | SQLException | ClassCastException | UnsupportedOperationException | IllegalArgumentException ex) {
79              LOG.error("Ophalen contractnummers is mislukt.", ex);
80              return Collections.EMPTY_LIST;
81          }
82      }

117     @Override
118     public void execute(ProgressUpdateListener listener) {
119         this.listener = listener;
120         config.setStatus(PROCESSING);
121         config.setLastrun(new Date());
122         Stripersist.getEntityManager().merge(config);
123         Stripersist.getEntityManager().flush();
124 
125         String msg = String.format("De afgiftenummer scanner met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
126         LOG.info(msg);
127         listener.updateStatus(msg);
128         listener.addLog(msg);
129 
130         String contractnummer = this.config.getContracNummer();
131         String afgiftenummertype = this.config.getAfgifteNummerType();
132 
133         try {
134 
135             if (afgiftenummertype == null) {
136                 msg = "Geen afgiftenummertype opgegeven voor opzoeken van ontbrekende afgiftenummers.";
137                 LOG.warn(msg);
138                 listener.updateStatus(msg);
139                 listener.addLog(msg);
140                 afgiftenummertype = "";
141             }
142 
143             if (contractnummer == null) {
144                 msg = "Geen contractnummer opgegeven voor opzoeken van ontbrekende afgiftenummers.";
145                 LOG.error(msg);
146                 listener.updateStatus(msg);
147                 listener.addLog(msg);
148             }
149 
150             List<Map<String, Object>> afgiftenummers = getOntbrekendeAfgiftenummers(contractnummer, afgiftenummertype);
151 
152             if (!afgiftenummers.isEmpty()) {
153 
154                 Iterator<Map<String, Object>> records = afgiftenummers.iterator();
155                 Map<String, Object> rec;
156 
157                 msg = "Ontbrekende " + afgiftenummertype + "s voor contractnummer: " + contractnummer;
158                 LOG.info(msg);
159                 listener.addLog(msg);
160 
161                 while (records.hasNext()) {
162                     rec = records.next();
163 
164                     if (rec.get("eerst_ontbrekend").equals(rec.get("laatst_ontbrekend"))) {
165                         msg = "    " + afgiftenummertype + " " + rec.get("eerst_ontbrekend") + " ontbreekt (tussen laadproces.id " + rec.get("laatste_aanwezige_id") + " en " + rec.get("eerst_opvolgende_id") + ")";
166                     } else {
167                         msg = "    Meerdere " + afgiftenummertype + "s ontbreken; eerst ontbrekende " + afgiftenummertype + " " + rec.get("eerst_ontbrekend") + ", laatst ontbrekende " + afgiftenummertype + " " + rec.get("laatst_ontbrekend") + " (tussen laadproces.id " + rec.get("laatste_aanwezige_id") + " en " + rec.get("eerst_opvolgende_id") + ")";
168                     }
169                     LOG.info(msg);
170                     listener.addLog(msg);
171                 }
172             }
173 
174             msg = "Klaar met bepalen ontbrekende " + afgiftenummertype + "s.";
175             LOG.info(msg);
176             listener.addLog(msg);
177             listener.updateStatus(msg);
178 
179             config.setStatus(WAITING);
180             config.setLastrun(new Date());
181         } catch (BrmoException | SQLException e) {
182             config.setStatus(ERROR);
183             LOG.error("Fout tijdens scannen voor onbrekende afgiftenummers van contract nummer: " + contractnummer, e);
184             listener.exception(e);
185         } finally {
186             Stripersist.getEntityManager().merge(config);
187         }
188     }

190     /**
191      * @param contractnummer    GDS2 contractnummer
192      * @param afgiftenummertype naam van de kolom in de laadproces tabel,
193      *                          "contractafgiftenummer" of "klantafgiftenummer", niet {code null}
194      * @return lijst met ranges van onbrekende records
195      * {@code ["eerst_ontbrekend_nr", laatst_ontbrekend_nr, "laatste_aanwezige_id", "eerst_opvolgende_id"]}
196      * @throws BrmoException if any
197      * @throws SQLException  if any
198      */
199     /*package private tbv. unit test*/ List<Map<String, Object>> getOntbrekendeAfgiftenummers(final String contractnummer, final String afgiftenummertype) throws BrmoException, SQLException {
200         if (contractnummer == null) {
201             throw new BrmoException("Contractnummer voor bepalen van ontbrekende afgiftenummers ontbreekt.");
202         }
203 
204         final DataSource ds = ConfigUtil.getDataSourceStaging();
205         try (final Connection conn = ds.getConnection()) {
206             ;
207             final GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(conn);
208             final String sql;
209             switch (afgiftenummertype) {
210                 case "contractafgiftenummer":
211                     sql = "SELECT"
212                             + "    laadproces.contractafgiftenummer + 1 AS eerst_ontbrekend,"
213                             + "    MIN(fr.contractafgiftenummer) - 1    AS laatst_ontbrekend,"
214                             + "    MIN(laadproces.id)                   AS laatste_aanwezige_id,"
215                             + "    MIN(fr.id)                           AS eerst_opvolgende_id"
216                             + " FROM"
217                             + "    laadproces"
218                             + " LEFT JOIN laadproces r ON"
219                             + "    laadproces.contractafgiftenummer = r.contractafgiftenummer - 1"
220                             + "    AND r.contractnummer = ? "
221                             + " LEFT JOIN laadproces fr ON"
222                             + "    laadproces.contractafgiftenummer < fr.contractafgiftenummer "
223                             + "    AND fr.contractnummer = ? "
224                             + " WHERE"
225                             + "    r.contractafgiftenummer IS NULL"
226                             + "    AND fr.contractafgiftenummer IS NOT NULL"
227                             + "    AND laadproces.contractnummer = ? "
228                             + " GROUP BY"
229                             + "    laadproces.contractafgiftenummer,"
230                             + "    r.contractafgiftenummer";
231                     break;
232                 case "klantafgiftenummer":
233                 default:
234                     sql = "SELECT"
235                             + "    laadproces.klantafgiftenummer + 1 AS eerst_ontbrekend,"
236                             + "    MIN(fr.klantafgiftenummer) - 1    AS laatst_ontbrekend,"
237                             + "    MIN(laadproces.id)                AS laatste_aanwezige_id,"
238                             + "    MIN(fr.id)                        AS eerst_opvolgende_id"
239                             + " FROM"
240                             + "    laadproces"
241                             + " LEFT JOIN laadproces r ON"
242                             + "    laadproces.klantafgiftenummer = r.klantafgiftenummer - 1"
243                             + "    AND r.contractnummer = ? "
244                             + " LEFT JOIN laadproces fr ON"
245                             + "    laadproces.klantafgiftenummer < fr.klantafgiftenummer"
246                             + "    AND fr.contractnummer = ? "
247                             + " WHERE"
248                             + "    r.klantafgiftenummer IS NULL"
249                             + "    AND fr.klantafgiftenummer IS NOT NULL"
250                             + "    AND laadproces.contractnummer = ? "
251                             + " GROUP BY"
252                             + "    laadproces.klantafgiftenummer,"
253                             + "    r.klantafgiftenummer";
254                     break;
255 
256             }
257 
258             List<Map<String, Object>> afgiftenummers = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(conn, sql, new MapListHandler(), contractnummer, contractnummer, contractnummer);
259 
260             this.ontbrekendeNummersGevonden = !afgiftenummers.isEmpty();
261             LOG.debug("Ontbrekende " + (afgiftenummertype == "" ? "klantafgiftenummer" : afgiftenummertype) + "s voor contractnummer " + contractnummer + ": " + afgiftenummers);
262 
263             DbUtils.closeQuietly(conn);
264             return afgiftenummers;
265         }
266     }
