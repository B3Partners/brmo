4   package nl.b3p.brmo.service.scanner;
5   
6   import java.util.Calendar;
7   import java.util.Date;
8   import javax.sql.DataSource;
9   import nl.b3p.brmo.loader.BrmoFramework;
10  import nl.b3p.brmo.loader.util.BrmoException;
11  import nl.b3p.brmo.persistence.staging.AutomatischProces;
12  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.ERROR;
13  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.PROCESSING;
14  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.WAITING;
15  import nl.b3p.brmo.persistence.staging.BerichtTransformatieProces;
16  import nl.b3p.brmo.service.util.ConfigUtil;
17  import org.apache.commons.lang3.exception.ExceptionUtils;
18  import org.apache.commons.logging.Log;
19  import org.apache.commons.logging.LogFactory;
20  import org.stripesstuff.stripersist.Stripersist;
21  
22  /**
23   * Transformatie van berichten in een automatisch proces gebruikmakend van
24   * BrmoFramework.
25   *
26   * @see nl.b3p.brmo.loader.BrmoFramework
27   *
28   * @author mprins
29   */
30  public class BerichtTransformatieUitvoeren extends AbstractExecutableProces {
31  
32      private static final Log log = LogFactory.getLog(BerichtTransformatieUitvoeren.class);
33  
34      private final BerichtTransformatieProces config;
35  
36      private ProgressUpdateListener l;
37  
38      private boolean transformErrorOccured = false;
39  
40      public BerichtTransformatieUitvoeren(BerichtTransformatieProces config) {
41          this.config = config;
42      }
43  
44  
45      @Override
46      public void execute() throws BrmoException {
47          this.execute(new ProgressUpdateListener() {
48              @Override
49              public void total(long total) {
50              }
51  
52              @Override
53              public void progress(long progress) {
54              }
55  
56              @Override
57              public void exception(Throwable t) {
58                  // afvangen van de fout conditie zodat de status van het proces goed kan worden gezet
59                  transformErrorOccured = true;
60                  log.error(t);
61              }
62  
63              @Override
64              public void updateStatus(String status) {
65              }
66  
67              @Override
68              public void addLog(String log) {
69              }
70          });
71      }
72  
73      @Override
74      public void execute(ProgressUpdateListener listener) {
75          this.l = listener;
76          l.updateStatus("Initialiseren...");
77          l.addLog(String.format("Initialiseren... %tc", new Date()));
78          this.config.setStatus(PROCESSING);
79          this.config.setLastrun(new Date());
80          Stripersist.getEntityManager().merge(config);
81          Stripersist.getEntityManager().flush();
82  
83          final StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
84          String oldLog = config.getLogfile();
85          if (oldLog != null) {
86              if (oldLog.length() > OLD_LOG_LENGTH) {
87                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
88              } else {
89                  sb.append(oldLog);
90              }
91          }
92  
93          String msg = String.format("Het bericht transformatie proces met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
94          log.info(msg);
95          l.addLog(msg);
96          sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
97  
98          BrmoFramework brmo = null;
99          try {
100             DataSource dataSourceStaging = ConfigUtil.getDataSourceStaging();
101             DataSource dataSourceRsgb = ConfigUtil.getDataSourceRsgb();
102             brmo = new BrmoFramework(dataSourceStaging, dataSourceRsgb);
103             brmo.setEnablePipeline(true);
104             brmo.setOrderBerichten(true);
105             brmo.setTransformPipelineCapacity(100);
106 
107             Thread t = brmo.toRsgb(new nl.b3p.brmo.loader.ProgressUpdateListener() {
108                 @Override
109                 public void total(long total) {
110                     sb.append("Totaal te transformeren: ")
111                             .append(total).append(AutomatischProces.LOG_NEWLINE);
112                     l.total(total);
113                 }
114 
115                 @Override
116                 public void progress(long progress) {
117                     l.progress(progress);
118                 }
119 
120                 @Override
121                 public void exception(Throwable t) {
122                     sb.append("Fout tijdens transformeren: ")
123                             .append(t.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
124                     l.exception(t);
125                 }
126             });
127             t.join();
128 
129             if (transformErrorOccured) {
130                 this.config.setStatus(ERROR);
131                 msg = "Handmatige transformatie vanuit de berichten pagina is noodzakelijk.";
132                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
133                 log.warn(msg);
134                 msg = String.format("Bericht transformatie proces met ID %d is niet succesvol afgerond op %tc.", config.getId(), Calendar.getInstance());
135                 log.error(msg);
136             } else {
137                 this.config.setStatus(WAITING);
138                 msg = String.format("Bericht transformatie proces met ID %d is succesvol afgerond op %tc.", config.getId(), Calendar.getInstance());
139                 log.info(msg);
140             }
141             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
142             l.addLog(msg);
143             this.config.setSamenvatting(msg);
144             this.config.setLogfile(sb.toString());
145         } catch (BrmoException | InterruptedException t) {
146             log.error("Fout bij transformeren berichten naar RSGB", t);
147             String m = "Fout bij transformeren berichten naar RSGB: " + ExceptionUtils.getMessage(t);
148             if (t.getCause() != null) {
149                 m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(t);
150             }
151             this.config.setLogfile(sb.toString());
152             this.config.updateSamenvattingEnLogfile(m);
153             this.config.setStatus(ERROR);
154         } finally {
155             if (brmo != null) {
156                 brmo.closeBrmoFramework();
157             }
158             Stripersist.getEntityManager().merge(this.config);
159             Stripersist.getEntityManager().flush();
160         }
161     }
162 }
