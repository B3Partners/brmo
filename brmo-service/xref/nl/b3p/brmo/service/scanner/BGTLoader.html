30  public class BGTLoader extends AbstractExecutableProces {
31    private static final Log LOG = LogFactory.getLog(BGTLoader.class);
32    private final BGTLoaderProces config;
33    @Transient private ProgressUpdateListener listener;
34  
35    public BGTLoader(BGTLoaderProces config) {
36      this.config = config;
37    }
38  
39    /**
40     * Voert deze taak eenmalig uit.
41     *
42     * @throws BrmoException als er een fout optreed in het uitvoeren van het proces.
43     */
44    @Override
45    public void execute() throws BrmoException {
46      this.execute(
47          new ProgressUpdateListener() {
48            @Override
49            public void total(long total) {}
50  
51            @Override
52            public void progress(long progress) {}
53  
54            @Override
55            public void exception(Throwable t) {
56              LOG.error(t);
57            }
58  
59            @Override
60            public void updateStatus(String status) {}
61  
62            @Override
63            public void addLog(String log) {
64              LOG.info(log);
65            }
66          });
67    }
68  
69    /**
70     * Voert de taak uit en rapporteert de voortgang.
71     *
72     * @param listener voortgangs listener
73     */
74    @Override
75    public void execute(ProgressUpdateListener listener) {
76      this.listener = listener;
77      BGTLoaderMain.configureLogging(false);
78  
79      if (config.getStatus().equals(WAITING) || config.getStatus().equals(ERROR)) {
80        if (config.getStatus().equals(ERROR)) {
81          listener.addLog("Vorige run is met ERROR status afgerond, opnieuw proberen");
82        }
83  
84        int exitCode = -1;
85        BGTDatabase bgtDatabase;
86        try (Connection rsgbbgtConnection = ConfigUtil.getDataSourceRsgbBgt().getConnection()) {
87  
88          DatabaseOptions databaseOptions = new DatabaseOptions();
89          // Niet nodig (rsgbbgtConnection wordt gebruikt), maar voor de duidelijkheid
90          databaseOptions.setConnectionString(rsgbbgtConnection.getMetaData().getURL());
91          Connection connection = rsgbbgtConnection;
92          if (databaseOptions.getConnectionString().startsWith("jdbc:postgresql:")) {
93            // Voor gebruik van pgCopy is unwrappen van de connectie nodig.
94            // Ook al doet PostGISCopyInsertBatch zelf ook een unwrap, de
95            // PGConnectionUnwrapper kan ook Tomcat JNDI
96            // connection pool unwrapping aan welke niet met een normale Connection.unwrap()
97            // werkt.
98            connection = (Connection) PGConnectionUnwrapper.unwrap(rsgbbgtConnection);
99            databaseOptions.setUsePgCopy(true);
100         }
101         bgtDatabase =
102             new BGTDatabase(databaseOptions, connection) {
103               /** connectie niet sluiten; dat doen we later als we helemaal klaar zijn */
104               @Override
105               public void close() {
106                 LOG.debug("Had de BGT database kunnen sluiten... maar niet gedaan.");
107               }
108             };
109 
110         LoadOptions loadOptions = new LoadOptions();
111         loadOptions.setIncludeHistory(
112             ("true".equals(ClobElement.nullSafeGet(config.getConfig().get("include-history")))));
113         loadOptions.setCreateSchema(
114             ("true".equals(ClobElement.nullSafeGet(config.getConfig().get("create-schema")))));
115         loadOptions.setLinearizeCurves(
116             "true".equals(ClobElement.nullSafeGet(config.getConfig().get("linearize-curves"))));
117 
118         ExtractSelectionOptions extractSelectionOptions = new ExtractSelectionOptions();
119         extractSelectionOptions.setGeoFilterWkt(
120             ClobElement.nullSafeGet(config.getConfig().get("geo-filter")));
121         boolean noGeoFilter =
122             StringUtils.isEmpty(ClobElement.nullSafeGet(config.getConfig().get("geo-filter")));
123         String fTypes = ClobElement.nullSafeGet(config.getConfig().get("feature-types"));
124         extractSelectionOptions.setFeatureTypes(Arrays.asList(fTypes.split(",")));
125         DownloadCommand downloadCommand = new DownloadCommand();
126         downloadCommand.setBGTDatabase(bgtDatabase);
127 
128         listener.updateStatus(PROCESSING.toString());
129         config.setStatus(PROCESSING);
130         Stripersist.getEntityManager().merge(config);
131         Stripersist.getEntityManager().flush();
132 
133         String initialLoadDeltaId;
134         try {
135           initialLoadDeltaId =
136               bgtDatabase.getMetadata(BGTSchemaMapper.Metadata.INITIAL_LOAD_DELTA_ID);
137         } catch (SQLException e) {
138           // Metadata table may not exist
139           initialLoadDeltaId = null;
140         }
141 
142         if (initialLoadDeltaId == null) {
143           listener.updateStatus("Ophalen BGT stand...");
144           listener.addLog("Ophalen BGT stand");
145           exitCode =
146               downloadCommand.initial(
147                   databaseOptions,
148                   loadOptions,
149                   extractSelectionOptions,
150                   noGeoFilter,
151                   null,
152                   new CLIOptions(),
153                   false);
154           // exitCode 2 = USAGE / config fout
155           listener.updateStatus("Einde ophalen BGT stand");
156           listener.addLog("Einde ophalen BGT stand");
157         } else {
158           listener.updateStatus("Ophalen BGT mutaties...");
159           listener.addLog("Ophalen BGT mutaties");
160           exitCode = downloadCommand.update(databaseOptions, new CLIOptions(), null, false, false);
161           listener.updateStatus("Einde ophalen BGT mutaties");
162           listener.addLog("Einde ophalen BGT mutaties");
163         }
164 
165       } catch (ClassNotFoundException e) {
166         LOG.error(e.getLocalizedMessage());
167         listener.exception(e);
168       } catch (BrmoException | SQLException e) {
169         LOG.error("Fout tijdens benaderen BGT database", e);
170         listener.exception(e);
171       } catch (Exception e) {
172         LOG.error("Fout tijdens laden BGT", e);
173         listener.exception(e);
174       } finally {
175         if (exitCode == 0) {
176           config.setStatus(WAITING);
177           config.setLastrun(new Date());
178           listener.updateStatus(WAITING.toString());
179         } else {
180           config.setStatus(ERROR);
181           config.setLastrun(new Date());
182           listener.updateStatus(ERROR.toString());
183         }
184         listener.addLog("BGT laden afgerond");
185         config.setLastrun(new Date());
186         Stripersist.getEntityManager().merge(config);
187         Stripersist.getEntityManager().flush();
188       }
189     }
190   }
