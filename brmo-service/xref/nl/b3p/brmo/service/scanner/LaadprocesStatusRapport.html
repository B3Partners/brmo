5

7   import nl.b3p.brmo.loader.entity.LaadProces;
8   import nl.b3p.brmo.loader.util.BrmoException;
9   import nl.b3p.brmo.persistence.staging.AutomatischProces;
10  import nl.b3p.brmo.persistence.staging.LaadprocesStatusRapportProces;
11  import nl.b3p.brmo.service.util.ConfigUtil;
12  import org.apache.commons.logging.Log;
13  import org.apache.commons.logging.LogFactory;
14  import org.stripesstuff.stripersist.Stripersist;
15  
16  import javax.persistence.EntityManager;
17  import javax.persistence.Transient;
18  import java.util.Calendar;
19  import java.util.Date;
20  
21  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.PROCESSING;
22  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.WAITING;
23  
24  /**
25   *
26   * @author mprins
27   */
28  public class LaadprocesStatusRapport extends AbstractExecutableProces {
29  
30      private static final Log LOG = LogFactory.getLog(MailRapportage.class);
31      private LaadprocesStatusRapportProces config;
32      @Transient
33      private ProgressUpdateListener listener;
34  
35      public LaadprocesStatusRapport(LaadprocesStatusRapportProces config) {
36          this.config = config;
37      }
38  
39      @Override
40      public void execute() throws BrmoException {
41          this.execute(new ProgressUpdateListener() {
42              @Override
43              public void total(long total) {
44              }
45  
46              @Override
47              public void progress(long progress) {
48              }
49  
50              @Override
51              public void exception(Throwable t) {
52                  LOG.error(t);
53              }
54  
55              @Override
56              public void updateStatus(String status) {
57              }
58  
59              @Override
60              public void addLog(String log) {
61              }
62          });
63      }
64  
65      @Override
66      public void execute(ProgressUpdateListener listener) {
67          EntityManager em = Stripersist.getEntityManager();
68  
69          this.listener = listener;
70          config.setStatus(PROCESSING);
71          config.setLastrun(new Date());
72          em.merge(config);
73          em.flush();
74  
75          StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
76          String oldLog = config.getLogfile();
77          if (oldLog != null) {
78              if (oldLog.length() > OLD_LOG_LENGTH) {
79                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
80              } else {
81                  sb.append(oldLog).append(AutomatischProces.LOG_NEWLINE);
82              }
83          }
84  
85          config.setStatus(PROCESSING);
86          String msg = String.format("De Laadproces Status Rapportage met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
87          LOG.info(msg);
88          listener.updateStatus(msg);
89          listener.addLog(msg);
90          sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
91          this.active = true;
92  
93          LOG.info("Ophalen Laadproces Status informatie.");
94          BrmoFramework brmo = null;
95          long aantal = -1L;
96          StringBuilder samenvatting = new StringBuilder("Laadproces Status overzicht");
97          samenvatting.append(AutomatischProces.LOG_NEWLINE);
98          try {
99              brmo = new BrmoFramework(ConfigUtil.getDataSourceStaging(), null);
100             for (LaadProces.STATUS status : LaadProces.STATUS.values()) {
101                 aantal = brmo.getCountLaadProcessen(null, null, null, status.name());
102                 msg = String.format("Aantal laadprocessen met status %s: %s.", status.name(), aantal);
103                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
104                 samenvatting.append(msg).append(AutomatischProces.LOG_NEWLINE);
105                 LOG.info(msg);
106                 listener.updateStatus(msg);
107                 listener.addLog(msg);
108             }
109         } catch (BrmoException ex) {
110             LOG.error(ex.getLocalizedMessage());
111             listener.exception(ex);
112             sb.append(ex.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
113         } finally {
114             if (brmo != null) {
115                 brmo.closeBrmoFramework();
116             }
117 
118         }
119         msg = String.format("De Laadproces Status Rapportage met ID %d is afgerond op %tc.", config.getId(), Calendar.getInstance());
120         LOG.info(msg);
121         listener.updateStatus(msg);
122         listener.addLog(msg);
123         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
124         config.setSamenvatting(samenvatting.toString());
125         config.setStatus(WAITING);
126         config.setLastrun(new Date());
127         config.setLogfile(sb.toString());
128     }
129 }
