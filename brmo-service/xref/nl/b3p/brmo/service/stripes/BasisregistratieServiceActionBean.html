59      private static final int ZIP_HEADER = 0x504b0304;
60  
61      @DefaultHandler
62      public Resolution read() {
63          HttpServletRequest req = getContext().getRequest();
64  
65          if(!req.getMethod().equals("POST")) {
66              return new DirectResponseResolution(405, "Only POST allowed\n");
67          }
68  
69          if(req.getContentLength() > ConfigUtil.MAX_UPLOAD_SIZE) {
70              return new DirectResponseResolution(413, String.format("Maximum upload size of %s exceeded\n", FileUtils.byteCountToDisplaySize(ConfigUtil.MAX_UPLOAD_SIZE)));
71          }
72  
73          BrmoFramework brmo = null;
74          try {
75              DataSource ds = ConfigUtil.getDataSourceStaging();
76              brmo = new BrmoFramework(ds, null);
77  
78              // Check of bestand begint met ZIP header
79              InputStream in = new BufferedInputStream(req.getInputStream());
80              in.mark(4);
81              int header = new DataInputStream(in).readInt();
82              in.reset();
83  
84              String unzippedFile = null;
85  
86              if(header == ZIP_HEADER) {
87                  // Pak eerste .xml bestand in ZIP uit en verwerk deze
88                  // Gebruik niet bestandsnaam van ZIP entries, altijd "MUTBX01.xml" oid (iig niet uniek)
89                  ZipInputStream zip = new ZipInputStream(in);
90                  ZipEntry entry = zip.getNextEntry();
91                  while(entry != null && !entry.getName().toLowerCase().endsWith(".xml")) {
92                      log.warn("Overslaan zip entry geen XML: " + entry.getName());
93                      entry = zip.getNextEntry();
94                  }
95                  if(entry == null) {
96                      throw new BrmoException("Geen geschikt XML bestand gevonden in zip bestand ");
97                  }
98                  log.debug("Lezen XML bestand uit zip: " + entry.getName());
99                  unzippedFile = entry.getName();
100                 brmo.loadFromStream(basisregistratie, zip, getUniqueFilename(brmo), (Long) null);
101 
102             } else {
103                 // Geen ZIP, direct BR XML laden
104                 brmo.loadFromStream(basisregistratie, in, getUniqueFilename(brmo), (Long) null);
105             }
106 
107             log.info(String.format("Stored %s data received via service endpoint", basisregistratie));
108             return new DirectResponseResolution(200, "Data successfully received" + (header == ZIP_HEADER ? ", unzipped \"" + unzippedFile + "\"": "") + " and stored\n");
109             } catch(Exception e) {
110             log.error("Fout bij ontvangen basisregistratiegegevens via directe POST", e);
111 
112             return new DirectResponseResolution(500, String.format("Error receiving data: %s\n", ExceptionUtils.getRootCauseMessage(e)));
113         } finally {
114             if(brmo != null) {
115                 brmo.closeBrmoFramework();
116             }
117         }
118     }

120     private final SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
121 
122     private String getUniqueFilename(BrmoFramework brmo) throws Exception {
123         // Unieke bestandsnaam nodig
124         String name;
125         boolean wait = false;
126         do {
127             name = "Upload op " + sdf.format(new Date());
128             if(!wait) {
129                 wait = true;
130             } else {
131                 Thread.sleep(10);
132             }
133             // Check of in dezelfde milliseconde gePOST laadproces al bestaat...
134         } while(brmo.getLaadProcesIdByFileName(name) != null);
135 
136         return name;
137     }
