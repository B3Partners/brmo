182   @WaitPage(path = JSP_PROGRESS, delay = 1000, refresh = 1000)
183   public Resolution update() {
184     UpdateProcess process = null;
185 
186     for (UpdateProcess p : updateProcesses) {
187       if (p.getName().equals(updateProcessName)) {
188         process = p;
189         break;
190       }
191     }
192     if (process == null) {
193       getContext().getMessages().add(new SimpleMessage("Ongeldig proces"));
194       return new ForwardResolution(JSP);
195     }
196 
197     start = new Date();
198     log.info("Start update process: " + process.getName());
199 
200     // Get berichten
201     BrmoFramework brmo = null;
202     try {
203       DataSource dataSourceStaging = ConfigUtil.getDataSourceStaging();
204       DataSource dataSourceRsgb = ConfigUtil.getDataSourceRsgb();
205       DataSource dataSourceRsgbBrk = ConfigUtil.getDataSourceRsgbBrk();
206       brmo = new BrmoFramework(dataSourceStaging, dataSourceRsgb, dataSourceRsgbBrk);
207 
208       boolean enableTransformPipeline =
209           "true".equals(getContext().getServletContext().getInitParameter("pipelining.enabled"));
210       brmo.setEnablePipeline(enableTransformPipeline);
211       if (enableTransformPipeline) {
212         String capacityString =
213             getContext().getServletContext().getInitParameter("pipelining.capacity");
214         if (capacityString != null) {
215           brmo.setTransformPipelineCapacity(Integer.parseInt(capacityString));
216         }
217       }
218       String batchString = getContext().getServletContext().getInitParameter("batch.capacity");
219       if (batchString != null) {
220         brmo.setBatchCapacity(Integer.parseInt(batchString));
221       }
222       String errorState = getContext().getServletContext().getInitParameter("error.state");
223       if (errorState != null) {
224         brmo.setErrorState(errorState);
225       }
226 
227       Thread t = brmo.toRsgb(process, this);
228       t.join();
229 
230       if (this.exceptionStacktrace == null) {
231         getContext()
232             .getMessages()
233             .add(new SimpleMessage("Transformatie afgerond, controleer berichtenstatus."));
234       }
235     } catch (Throwable t) {
236       log.error("Fout bij updaten", t);
237       String m = "Fout bij updaten: " + ExceptionUtils.getMessage(t);
238       if (t.getCause() != null) {
239         m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(t);
240       }
241       getContext().getMessages().add(new SimpleMessage(m));
242     } finally {
243       if (brmo != null) {
244         brmo.closeBrmoFramework();
245       }
246       complete = true;
247     }
248     return new ForwardResolution(JSP_PROGRESS);
249   }
