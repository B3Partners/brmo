26  /**
27   * Action voor het handmatig uploaden van bestanden met basisregistratiegegevens
28   * via een formulier.
29   */
30  @StrictBinding
31  public class BasisregistratieFileUploadActionBean implements ActionBean {
32      private static final int ZIP_HEADER = 0x504b0304;
33  
34      private static final Log log = LogFactory.getLog(BasisregistratieFileUploadActionBean.class);
35  
36      private ActionBeanContext context;
37  
38      @Validate(required=true)
39      private String basisregistratie;
40  
41      @Validate(required=true)
42      private FileBean bestand;
43  
44      @DefaultHandler
45      @DontValidate
46      public Resolution form() {
47          return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
48      }
49  
50      public Resolution upload() throws BrmoException {
51          DataSource ds = ConfigUtil.getDataSourceStaging();
52          BrmoFramework brmo = null;
53          brmo = new BrmoFramework(ds, null);
54          ZipInputStream zip = null;
55          ZipEntry entry = null;
56          int errors = 0;
57          int empty = 0;
58          int duplicate = 0;
59          final int ERROR_LIMIT = 5;
60          try {
61              final MutableLong theTotal = new MutableLong(0);
62              int extractedFiles = 0;
63              final ProgressUpdateListener totalAdder = new ProgressUpdateListener() {
64  
65                  @Override
66                  public void total(long total) {
67                      theTotal.setValue(theTotal.getValue() + total);
68                  }
69  
70                  @Override
71                  public void progress(long progress) {
72                  }
73  
74                  @Override
75                  public void exception(Throwable t) {
76                  }
77              };
78              // Check of bestand begint met ZIP header
79  
80              InputStream in = new BufferedInputStream(bestand.getInputStream());
81              in.mark(4);
82              int header = new DataInputStream(in).readInt();
83              in.reset();
84  
85              if(header == ZIP_HEADER) {
86                  // Pak .xml bestanden in ZIP uit en verwerk deze
87  
88                  zip = new ZipInputStream(in);
89                  try {
90                      entry = zip.getNextEntry();
91                      while(entry != null) {
92                          if(!entry.getName().toLowerCase().endsWith(".xml")) {
93                              log.warn("Overslaan zip entry geen XML: " + entry.getName());
94                          } else {
95                              log.debug("Lezen XML bestand uit zip: " + entry.getName());
96                              try {
97                                  brmo.loadFromStream(basisregistratie, new CloseShieldInputStream(zip), bestand.getFileName() + "/" + entry.getName(), totalAdder, null);
98                              } catch(BrmoLeegBestandException e) {
99                                  log.info("Negeer ZIP entry " + entry.getName() + ": " + e.getMessage());
100                                 empty++;
101                             } catch(BrmoDuplicaatLaadprocesException e) {
102                                 log.info("Negeer ZIP entry " + entry.getName() + ": duplicaat laadproces");
103                                 duplicate++;
104                             } catch(BrmoException e) {
105                                 errors++;
106                                 log.error("BrmoException bij laden ZIP entry " + entry.getName() + (errors > ERROR_LIMIT ? "; afbreken" : "; doorgaan met volgende"), e);
107                                 if(errors > ERROR_LIMIT) {
108                                     throw new BrmoException("Maximum aantal fouten (" + ERROR_LIMIT + ") bereikt bij inladen ZIP bestanden, inladen afgebroken. Controleer brmo-service.log", e);
109                                 }
110                             }
111                             extractedFiles++;
112                         }
113                         entry = zip.getNextEntry();
114                     }
115                 } finally {
116                     zip.close();
117                 }
118             } else {
119                 // Geen ZIP, direct BR XML laden
120                 brmo.loadFromStream(basisregistratie, in, bestand.getFileName(), totalAdder, null);
121             }
122 
123             List warnings = new ArrayList();
124             if(errors > 0) {
125                 warnings.add("let op: " + errors + " bericht" + (errors > 1 ? "en" : "") + " niet ingeladen wegens fouten");
126             }
127             if(empty > 0) {
128                 warnings.add("let op: " + empty + (empty > 1 ? " lege berichten" : " leeg bericht") + " overgeslagen");
129             }
130             if(duplicate > 0) {
131                 warnings.add("let op: " + duplicate + (duplicate > 1 ? " duplicate berichten" : " duplicaat bericht") + " overgeslagen");
132             }
133 
134             String warning = warnings.isEmpty() ? "" : ", " + String.join(", ", (String[])warnings.toArray(new String[] {})) + ", zie log voor bestandsnamen uit ZIP en details!";
135 
136             getContext().getMessages().add(new SimpleMessage("Bestand " + bestand.getFileName() + " is ingelezen, " + theTotal.getValue() + " berichten" + (extractedFiles > 0 ? " (uit " + extractedFiles + " uitgepakte XML bestanden)" : "") + warning));
137             log.info(String.format("Stored %s data from file \"%s\" uploaded via form", basisregistratie, bestand.getFileName()));
138         } catch(Exception ex) {
139             String msg = "Fout bij inlezen bestand " + (zip != null && entry != null ? entry.getName() + " uit ZIP bestand " + bestand.getFileName() : bestand.getFileName());
140             log.error(msg, ex);
141 
142             getContext().getMessages().add(new SimpleMessage(msg + ": " + ExceptionUtils.getRootCauseMessage(ex)));
143         } finally {
144             try {
145                 bestand.delete();
146             } catch(IOException e) {
147                 // ignore
148             }
149             brmo.closeBrmoFramework();
150         }
151 
152         return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
153     }
