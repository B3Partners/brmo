74          @Validate(field = "cronExpressie", on = "save")
75      })

103     @DontValidate
104     public Resolution delete() {
105         if (proces != null) {
106 
107             Stripersist.getEntityManager().createQuery("update LaadProces set automatischProces = null where automatischProces = :this")
108                     .setParameter("this", proces)
109                     .executeUpdate();
110 
111             Stripersist.getEntityManager().remove(proces);
112             load();
113             Stripersist.getEntityManager().getTransaction().commit();
114             deleteScheduledJob(proces);
115             proces = null;
116             getContext().getMessages().add(new SimpleMessage("Proces is verwijderd"));
117         }
118         return new ForwardResolution(JSP);
119     }
120 
121     @DontValidate
122     public Resolution add() {
123         return new ForwardResolution(JSP).addParameter("type", type);
124     }
125 
126     public Resolution save() {
127         if (proces == null) {
128             proces = getProces(type);
129         }
130 
131         proces.getConfig().clear();
132         proces.getConfig().putAll(config);
133 
134         if(proces instanceof BerichtDoorstuurProces) {
135             String id = ClobElement.nullSafeGet(config.get("gds2_ophaalproces_id"));
136             GDS2OphaalProces p = Stripersist.getEntityManager().find(GDS2OphaalProces.class, Long.parseLong(id));
137             if(p != null) {
138                 String label = ClobElement.nullSafeGet(p.getConfig().get("label"));
139                 proces.getConfig().put("label", new ClobElement("Doorsturen " + label + " afgiftes"));
140             }
141         }
142 
143         Stripersist.getEntityManager().persist(proces);
144         load();
145         Stripersist.getEntityManager().getTransaction().commit();
146         getContext().getMessages().add(new SimpleMessage("Proces is opgeslagen"));
147 
148         try {
149             this.updateJobSchedule(proces);
150         } catch (SchedulerException ex) {
151             getContext().getMessages().add(
152                     new SimpleError("Er is een fout opgetreden tijdens inplannen van de taak. {2}",
153                             ex.getMessage()));
154         }
155 
156         return new ForwardResolution(JSP).addParameter("proces", proces.getId());
157     }

159     /**

162      * @param type the type of {@code AutomatischProces} to create.
163      * @return an instance of the specified type

167         switch (type) {

174             case GDS2OphaalProces:
175                 return new GDS2OphaalProces();
176             case BerichtTransformatieProces:

178             case BerichtDoorstuurProces:
179                 return new BerichtDoorstuurProces();
180             case WebMirrorBAGScannerProces:
181                 return new WebMirrorBAGScannerProces();
182             case BGTLightScannerProces:
183                 return new BGTLightScannerProces();
184             case BGTLightOphaalProces:
185                 return new BGTLightOphaalProces();
186             case LaadprocesTransformatieProces:
187                 return new LaadprocesTransformatieProces();
188             case MaterializedViewRefresh:
189                 return new MaterializedViewRefresh();
190             case BerichtstatusRapportProces:
191                 return new BerichtstatusRapportProces();
192             case LaadprocesStatusRapportProces:
193                 return new LaadprocesStatusRapportProces();

207     /**
208      * Vervang de proces job door een aangepaste job of plaats een nieuwe job mocht
209      * deze nog niet bestaan. Als de cron expressie {@code null} is wordt de job
210      * verwijderd uit de scheduler.
211      *
212      * @param p bij te werken proces
213      * @throws SchedulerException
214      */
215     private void updateJobSchedule(AutomatischProces p) throws SchedulerException {
216         log.debug("Update scheduled job:" + p.getId());
217 
218         StdSchedulerFactory factory = (StdSchedulerFactory) getContext().getServletContext()
219                 .getAttribute(QUARTZ_FACTORY_KEY);
220         Scheduler scheduler = factory.getScheduler(SCHEDULER_NAME);
221         JobKey key = new JobKey(GeplandeTakenInit.JOBKEY_PREFIX + p.getId());
222         log.debug("Jobkey voor id " + p.getId() + " gevonden? " + scheduler.checkExists(key));
223 
224         scheduler.deleteJob(key);
225         if (p.getCronExpressie() != null) {
226             GeplandeTakenInit.addJobDetails(scheduler, p);
227         }
228     }
229 
230     /**
231      * verwijder een proces job uit de scheduler door update met een {@code null} cron
232      * schedule.
233      *
234      * @param p te verwijderen proces
235      */
236     private void deleteScheduledJob(AutomatischProces p) {
237         try {
238             p.setCronExpressie(null);
239             updateJobSchedule(p);
240         } catch (SchedulerException se) {
241             log.warn("Ingeplande taak uit de planner halen is mislukt", se);
242         }
243     }

245     /**
246      * validatie van de cron expressie van een proces voorafgaand aan save.
247      */
248     @ValidationMethod(on = "save")
249     public void validateCronExpressie() {
250         if (proces != null) {
251             String expr = proces.getCronExpressie();
252             try {
253                 if (expr != null) {
254                     CronExpression cron = new CronExpression(expr);
255                 }
256             } catch (ParseException ex) {
257                 Stripersist.getEntityManager().refresh(proces);
258                 load();
259                 getContext().getValidationErrors().add("cronExpressie",
260                         new SimpleError("{0} {2} is ongeldig, (melding: {4}, mogelijk nabij positie {3})",
261                                 expr, ex.getErrorOffset(), ex.getLocalizedMessage()));
262             }
263         }
264     }
