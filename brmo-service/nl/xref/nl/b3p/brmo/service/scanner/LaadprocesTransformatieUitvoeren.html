32  /**
33   * Transformeert laadprocessn, zoals bijvoorbeeld TopNL.
34   *
35   * @author mprins
36   */
37  public class LaadprocesTransformatieUitvoeren extends AbstractExecutableProces {
38  
39      private static final Log LOG = LogFactory.getLog(LaadprocesTransformatieUitvoeren.class);
40  
41      private final LaadprocesTransformatieProces config;
42  
43      @Transient private ProgressUpdateListener listener;
44  
45      private boolean transformErrorOccured = false;
46  
47      public LaadprocesTransformatieUitvoeren(LaadprocesTransformatieProces config) {
48          this.config = config;
49      }
50  
51      @Override
52      public void execute() throws BrmoException {
53          this.execute(
54                  new ProgressUpdateListener() {
55                      @Override
56                      public void total(long total) {}
57  
58                      @Override
59                      public void progress(long progress) {}
60  
61                      @Override
62                      public void exception(Throwable t) {
63                          // afvangen van de fout conditie zodat de status van het proces goed kan
64                          // worden gezet
65                          transformErrorOccured = true;
66                          LOG.error(t);
67                      }
68  
69                      @Override
70                      public void updateStatus(String status) {}
71  
72                      @Override
73                      public void addLog(String log) {}
74                  });
75      }
76  
77      @Override
78      public void execute(final ProgressUpdateListener listener) {
79          this.listener = listener;
80          listener.updateStatus("Initialiseren...");
81          listener.addLog(String.format("Initialiseren... %tc", new Date()));
82          this.config.setStatus(PROCESSING);
83          this.config.setLastrun(new Date());
84          Stripersist.getEntityManager().merge(config);
85          Stripersist.getEntityManager().flush();
86  
87          final StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
88          String oldLog = config.getLogfile();
89          if (oldLog != null) {
90              if (oldLog.length() > OLD_LOG_LENGTH) {
91                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
92              } else {
93                  sb.append(oldLog);
94              }
95          }
96  
97          String msg =
98                  String.format(
99                          "Het laadproces transformatie proces met ID %d is gestart op %tc.",
100                         config.getId(), Calendar.getInstance());
101         LOG.info(msg);
102         listener.addLog(msg);
103         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
104 
105         BrmoFramework brmo = null;
106         try {
107             DataSource dataSourceStaging = ConfigUtil.getDataSourceStaging();
108             DataSource dataSourceTopNL = null;
109             if (LaadprocesTransformatieProces.LaadprocesSoorten.BR_TOPNL
110                     .getSoort()
111                     .equalsIgnoreCase(this.config.getSoort())) {
112                 dataSourceTopNL = ConfigUtil.getDataSourceTopNL();
113             }
114             brmo = new BrmoFramework(dataSourceStaging, null, null, dataSourceTopNL, null);
115             long[] lpIds;
116 
117             if (LaadprocesTransformatieProces.LaadprocesSoorten.BR_TOPNL
118                     .getSoort()
119                     .equalsIgnoreCase(this.config.getSoort())) {
120                 List<Long> ids = new ArrayList<>();
121                 for (String type : TopNLDirectoryScanner.subdirectoryNames) {
122                     ids.addAll(
123                             Arrays.asList(
124                                     brmo.getLaadProcessenIds(
125                                             "bestand_datum", "ASC", type, "STAGING_OK")));
126                 }
127                 lpIds = ArrayUtils.toPrimitive(ids.toArray(new Long[ids.size()]));
128             } else {
129                 lpIds =
130                         ArrayUtils.toPrimitive(
131                                 brmo.getLaadProcessenIds(
132                                         "bestand_datum", "ASC", config.getSoort(), "STAGING_OK"));
133             }
134             brmo.setOrderBerichten(!config.alsStandTransformeren());
135             if (lpIds == null || lpIds.length == 0) {
136                 msg =
137                         "Er zijn geen laadprocessen van soort "
138                                 + config.getSoort()
139                                 + " en status: 'STAGING_OK' om te transformeren.";
140                 LOG.info(msg);
141                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
142                 listener.addLog(msg);
143             } else {
144                 listener.updateStatus("Transformeren...");
145                 Thread t =
146                         brmo.toRsgb(
147                                 RsgbProxy.BerichtSelectMode.BY_LAADPROCES,
148                                 lpIds,
149                                 new nl.b3p.brmo.loader.ProgressUpdateListener() {
150                                     @Override
151                                     public void total(long total) {
152                                         sb.append("Totaal te transformeren: ")
153                                                 .append(total)
154                                                 .append(AutomatischProces.LOG_NEWLINE);
155                                         listener.total(total);
156                                     }
157 
158                                     @Override
159                                     public void progress(long progress) {
160                                         listener.progress(progress);
161                                     }
162 
163                                     @Override
164                                     public void exception(Throwable t) {
165                                         sb.append("Fout tijdens laadproces transformeren: ")
166                                                 .append(t.getLocalizedMessage())
167                                                 .append(AutomatischProces.LOG_NEWLINE);
168                                         listener.exception(t);
169                                     }
170                                 });
171                 t.join();
172             }
173 
174             if (transformErrorOccured) {
175                 this.config.setStatus(ERROR);
176                 msg = "Handmatige transformatie vanuit de laadprocessen pagina is noodzakelijk.";
177                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
178                 LOG.warn(msg);
179                 msg =
180                         String.format(
181                                 "Laadproces transformatie proces met ID %d is niet succesvol afgerond op %tc.",
182                                 config.getId(), Calendar.getInstance());
183                 LOG.error(msg);
184                 listener.updateStatus("Er is een fout opgetreden.");
185             } else {
186                 this.config.setStatus(WAITING);
187                 msg =
188                         String.format(
189                                 "Laadproces transformatie proces met ID %d is succesvol afgerond op %tc.",
190                                 config.getId(), Calendar.getInstance());
191                 LOG.info(msg);
192                 listener.updateStatus("Transformatie afgerond.");
193             }
194 
195             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
196             this.listener.addLog(msg);
197             this.config.setSamenvatting(msg);
198             this.config.setLogfile(sb.toString());
199         } catch (BrmoException | InterruptedException e) {
200             LOG.error("Fout bij transformeren laadprocessen naar RSGB", e);
201             String m =
202                     "Fout bij transformeren laadprocessen naar RSGB: "
203                             + ExceptionUtils.getMessage(e);
204             if (e.getCause() != null) {
205                 m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(e);
206             }
207             this.config.setLogfile(sb.toString());
208             this.config.updateSamenvattingEnLogfile(m);
209             this.config.setStatus(ERROR);
210         } finally {
211             if (brmo != null) {
212                 brmo.closeBrmoFramework();
213             }
214             Stripersist.getEntityManager().merge(this.config);
215             Stripersist.getEntityManager().flush();
216             Stripersist.getEntityManager().getTransaction().commit();
217             Stripersist.getEntityManager().clear();
218         }
219     }
