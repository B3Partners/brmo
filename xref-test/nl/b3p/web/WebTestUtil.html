15  /**
16   * Integratie test utility klasse om online integratie tests te bouwen welke
17   * tegen een brmo-service kunnen worden uitgevoerd.
18   *
19   * @author Mark Prins
20   */
21  public abstract class WebTestUtil extends TestUtil {
22  
23      private static final Log LOG = LogFactory.getLog(WebTestUtil.class);
24  
25      /**
26       * the server root url. {@value}
27       */
28      public static final String BASE_TEST_URL = "http://localhost:9091/brmo-service/";
29  
30  
31      /**
32       * our test client.
33       */
34      protected static CloseableHttpClient client;
35  
36  
37      /**
38       * initialize http client.
39       */
40      @BeforeAll
41      public static void setUpClass() {
42          client = HttpClients.custom()
43                  .useSystemProperties()
44                  .setUserAgent("brmo integration test")
45                  .setRedirectStrategy(new LaxRedirectStrategy())
46                  .setDefaultCookieStore(new BasicCookieStore())
47                  .build();
48      }
49  
50      /**
51       * close http client connections.
52       *
53       * @throws IOException if any occurs closing the http connection
54       */
55      @AfterAll
56      public static void tearDownClass() throws IOException {
57          client.close();
58      }
59  
60  }
