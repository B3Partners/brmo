17  package nl.b3p.web.jsp;
18  
19  import java.io.File;
20  import java.util.List;
21  import org.junit.Test;
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertFalse;
24  import static org.junit.Assert.assertNotNull;
25  
26  /**
27   * testcases voor {@link LogfileUtil}.
28   *
29   * @author Mark Prins
30   */
31  public class LogfileUtilTest {
32  
33      @Test
34      public void testGetLogFile() {
35          String l = LogfileUtil.getLogfile();
36          assertNotNull(l);
37          assertEquals(getTestFileName(), l);
38      }
39  
40      @Test
41      public void getLogfileList() {
42          List<String> l = LogfileUtil.getLogfileList();
43          assertNotNull(l);
44          assertFalse(l.isEmpty());
45          assertEquals(getTestFileName(), l.get(0));
46      }
47      
48      private String getTestFileName() {
49          String s = System.getProperty("java.io.tmpdir");
50          if (s.endsWith(File.separator)) {
51              s = s.substring(0, s.length() - 1);
52          }
53          return s + File.separator + "LogfileUtilTest.log";
54      }
55  }
