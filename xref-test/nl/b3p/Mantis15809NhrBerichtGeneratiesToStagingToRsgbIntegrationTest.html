36   * bevat een testcase voor mantis15809 waarbij berichten van dezelfde inschrijving op verschillende tijdstippen
37   * worden geladen.
38   * <br>Draaien met:
39   * {@code mvn -Dit.test=Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest -Dtest.onlyITs=true verify -Ppostgresql -pl :brmo-loader > /tmp/postgresql.log}
40   * voor bijvoorbeeld PostgreSQL of
41   * {@code mvn -Dit.test=Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest -Dtest.onlyITs=true verify -Pmssql  -pl :brmo-loader > /tmp/mssql.log}
42   * voor bijvoorbeeld MS SQL of
43   * {@code mvn -Dit.test=Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest -Dtest.onlyITs=true verify -Poracle  -pl :brmo-loader > /tmp/oracle.log}

54      private static final String BESTANDTYPE = "nhr";
55      private final Lock sequential = new ReentrantLock(true);

61      @BeforeEach
62      @Override
63      public void setUp() throws Exception {
64          BasicDataSource dsStaging = new BasicDataSource();
65          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
66          dsStaging.setUsername(params.getProperty("staging.user"));
67          dsStaging.setPassword(params.getProperty("staging.passwd"));
68          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
69          dsStaging.setConnectionProperties(params.getProperty("staging.options", ""));
70  
71          BasicDataSource dsRsgb = new BasicDataSource();
72          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
73          dsRsgb.setUsername(params.getProperty("rsgb.user"));
74          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
75          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
76          dsRsgb.setConnectionProperties(params.getProperty("rsgb.options", ""));
77  
78          staging = new DatabaseDataSourceConnection(dsStaging);
79          rsgb = new DatabaseDataSourceConnection(dsRsgb);
80  
81          if (this.isMsSQL) {
82              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
83              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
84          } else if (this.isOracle) {
85              staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
86                      params.getProperty("staging.user").toUpperCase());
87              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
88              staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
89  
90              rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
91                      params.getProperty("rsgb.user").toUpperCase());
92              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
93              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
94          } else if (this.isPostgis) {
95              // we hebben alleen nog postgres over
96              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
97              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
98          }
99  
100         brmo = new BrmoFramework(dsStaging, dsRsgb);
101         brmo.setOrderBerichten(true);
102 
103         sequential.lock();
104 
105         // staging en rsgb tabellen leeg maken voorafgaan aan de test
106         CleanUtil.cleanSTAGING(staging, false);
107         CleanUtil.cleanRSGB_NHR(rsgb);
108 
109         Assumptions.assumeTrue(0L == brmo.getCountBerichten(null, null, "nhr", "STAGING_OK"),
110                 "Er zijn geen STAGING_OK berichten");
111         Assumptions.assumeTrue(0L == brmo.getCountLaadProcessen(null, null, "nhr", "STAGING_OK"),
112                 "Er zijn geen STAGING_OK laadprocessen");
113     }

133     @Test
134     public void test() throws Exception {
135         // te laden berichten + kvknummer + vestg nummer + sbicode + aantallen
136         Object[][] testdataSet = new Object[][]{
137                 // {"filename", aantalBerichten, aantalProcessen, hoofdVestgID, aantalVestg_activiteit,
138                 // kvkNummer, sbiCodes (eerste  sbiCode is hoofdactiviteit) },
139                 {"/mantis15809-nhr-bericht-generaties/78597161_20200903.anon.xml", 3, 1, "nhr.comVestg.000006873758", 4,
140                         78597161, new String[]{"46738", "47789", "47918", "4329"}},
141                 {"/mantis15809-nhr-bericht-generaties/78597161_20201029.anon.xml", 3, 1, "nhr.comVestg.000006873758", 4,
142                         78597161, new String[]{"46738", "47789", "47918", "4329"}},
143                 {"/mantis15809-nhr-bericht-generaties/78597161_20210127.anon.xml", 3, 1, "nhr.comVestg.000006873758", 2,
144                         78597161, new String[]{"4329", "4339"}}
145         };
146 
147         String bestandNaam;
148         long aantalBerichten = 0;
149         long aantalProcessen = 0;
150         String vestgID;
151         long aantalVestg_activiteit;
152         long kvkNummer;
153         String[] sbiCodes = new String[]{};
154         int verwerktNHr = 0;
155 
156         // berichten laden en transformeren
157         for (Object[] testdata : testdataSet) {
158             bestandNaam = (String) testdata[0];
159             aantalBerichten += Long.valueOf((Integer) testdata[1]);
160             aantalProcessen += Long.valueOf((Integer) testdata[2]);
161             vestgID = (String) testdata[3];
162             aantalVestg_activiteit = Long.valueOf((Integer) testdata[4]);
163             kvkNummer = Long.valueOf((Integer) testdata[5]);
164             sbiCodes = (String[]) testdata[6];
165 
166             testNhrXMLToStagingToRsgb(bestandNaam, aantalBerichten, aantalProcessen,
167                     vestgID, aantalVestg_activiteit, kvkNummer, sbiCodes, ++verwerktNHr);
168         }
169 
170         // dubbele controle van aantal brmo berichten/laadprocessen als alles geladen/getransformeerd is
171         assertEquals(aantalBerichten, brmo.listBerichten().size(), "Het aantal berichten is niet als verwacht.");
172         assertEquals(aantalProcessen, brmo.listLaadProcessen().size(), "Het aantal processen is niet als verwacht.");
173 
174         // dubbele controle van vestg_activiteit als alles geladen/getransformeerd is
175         ITable vestg_activiteit = rsgb.createDataSet().getTable("vestg_activiteit");
176         assertEquals(sbiCodes.length, vestg_activiteit.getRowCount(),
177                 "Het aantal 'vestg_activiteit' records klopt niet");
178         vestg_activiteit = rsgb.createQueryTable("vestg_activiteit",
179                 "select * from vestg_activiteit where indicatie_hoofdactiviteit=1");
180         assertEquals(sbiCodes[0], vestg_activiteit.getValue(0, "fk_sbi_activiteit_code"),
181                 "Hoofdactiviteit in 'vestg_activiteit' klopt niet");
182         assertEquals(1, vestg_activiteit.getRowCount(),
183                 "Aantal hoofdactiviteiten in 'vestg_activiteit' klopt niet");
184     }

186     private void testNhrXMLToStagingToRsgb(String bestandNaam, long aantalBerichten, long aantalProcessen,
187                                            String vestgID, long aantalVestg_activiteit, long kvkNummer,
188                                            String[] sbiCodes, int verwerktNHr) throws Exception {
189 
190         assumeFalse(null == Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest.class.getResource(
191                 bestandNaam), "Het test bestand moet er zijn.");
192 
193         brmo.loadFromFile(BESTANDTYPE, Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest.class.getResource(
194                 bestandNaam).getFile(), null);
195         LOG.info("klaar met laden van berichten in staging DB.");
196 
197         // check aantallen brmo berichten
198         List<Bericht> berichten = brmo.listBerichten();
199         List<LaadProces> processen = brmo.listLaadProcessen();
200         assertNotNull(berichten, "De verzameling berichten bestaat niet.");
201         assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
202         assertNotNull(processen, "De verzameling processen bestaat niet.");
203         assertEquals(aantalProcessen, processen.size(), "Het aantal processen is niet als verwacht.");
204 
205         // alleen het eerste bericht van een nHR bericht heeft br_orgineel_xml, de rest niet, check dat
206         ITable bericht = staging.createQueryTable("bericht", "select * from bericht where volgordenummer=0 order by id");
207         assertEquals(verwerktNHr, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
208         LOG.debug("\n\n" + bericht.getValue(verwerktNHr - 1, "br_orgineel_xml") + "\n\n");
209         assertNotNull(bericht.getValue(verwerktNHr - 1, "br_orgineel_xml"), "BR origineel xml is null");
210         Object berichtId = bericht.getValue(verwerktNHr - 1, "id");
211 
212         LOG.info("Transformeren berichten naar rsgb DB.");
213         Thread t = brmo.toRsgb();
214         t.join();
215 
216         // na de verwerking moet soap payload er ook nog zijn
217         bericht = staging.createQueryTable("bericht", "select * from bericht where br_orgineel_xml is not null order by id");
218         assertEquals(verwerktNHr, bericht.getRowCount(), "Er zijn meer of minder dan verwacht rijen");
219         assertNotNull(bericht.getValue(verwerktNHr - 1, "br_orgineel_xml"),
220                 "BR origineel xml is null na transformatie");
221         assertEquals(berichtId, bericht.getValue(verwerktNHr - 1, "id"),
222                 "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
223 
224         // controle van transformatie
225         assertEquals(aantalBerichten, brmo.getCountBerichten(null, null, "nhr", "RSGB_OK"),
226                 "Niet alle berichten zijn OK getransformeerd");
227         berichten = brmo.listBerichten();
228         for (Bericht b : berichten) {
229             assertNotNull(b, "Bericht is 'null'");
230             assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
231         }
232 
233         // check kvk nummer
234         ITable subject = rsgb.createDataSet().getTable("subject");
235         boolean foundKvk = false;
236         for (int i = 0; i < subject.getRowCount(); i++) {
237             if (subject.getValue(i, "identif").toString().contains("nhr.maatschAct.kvk")) {
238                 assertEquals(kvkNummer + "", subject.getValue(i, "kvk_nummer") + "", "KVK nummer klopt niet");
239                 foundKvk = true;
240             }
241         }
242         if (!foundKvk) {
243             fail("KVK nummer maatschappelijke activiteit klopt niet, verwacht te vinden: " + kvkNummer);
244         }
245 
246         // check vestiging en activiteit
247         ITable vestg = rsgb.createDataSet().getTable("vestg");
248         assertEquals(vestgID, vestg.getValue(0, "sc_identif"),
249                 "De 'sc_identif' van hoofdvestiging klopt niet");
250         assertEquals(sbiCodes[0], vestg.getValue(0, "fk_sa_sbi_activiteit_sbi_code"),
251                 "De sbi code van (hoofd)vestiging klopt niet");
252         assertEquals("Ja", vestg.getValue(0, "sa_indic_hoofdactiviteit"),
253                 "De 'sa_indic_hoofdactiviteit' van (hoofd)vestiging klopt niet");
254 
255         // check vestingingsactiviteiten
256         ITable vestg_activiteit = rsgb.createDataSet().getTable("vestg_activiteit");
257         assertEquals(aantalVestg_activiteit, vestg_activiteit.getRowCount(),
258                 "Het aantal 'vestg_activiteit' records klopt niet");
259 
260         vestg_activiteit = rsgb.createQueryTable("vestg_activiteit",
261                 "select * from vestg_activiteit where indicatie_hoofdactiviteit=1");
262         assertEquals(sbiCodes[0], vestg_activiteit.getValue(0, "fk_sbi_activiteit_code"),
263                 "Hoofdactiviteit in 'vestg_activiteit' klopt niet");
264         assertEquals(1, vestg_activiteit.getRowCount(),
265                 "Aantal hoofdactiviteit in 'vestg_activiteit' klopt niet");
266 
267         ITable sbi_activiteit = rsgb.createDataSet().getTable("sbi_activiteit");
268         ITable herkomst_metadata = rsgb.createDataSet().getTable("herkomst_metadata");
269         for (String sbiCode : sbiCodes) {
270             boolean foundSbiCode = false;
271             boolean foundSbiCodeMeta = false;
272             for (int i = 0; i < sbi_activiteit.getRowCount(); i++) {
273                 if (sbiCode.equals(sbi_activiteit.getValue(i, "sbi_code").toString())) {
274                     LOG.debug("SBI code " + sbiCode + " gevonden in sbi_activiteit");
275                     foundSbiCode = true;
276                     break;
277                 }
278             }
279             if (!foundSbiCode) {
280                 fail("SBI code niet gevonden in sbi_activiteit, verwacht te vinden: " + sbiCode);
281             }
282             for (int i = 0; i < herkomst_metadata.getRowCount(); i++) {
283                 if (sbiCode.equals(herkomst_metadata.getValue(i, "waarde").toString())) {
284                     LOG.debug("SBI code " + sbiCode + " gevonden in herkomst_metadata");
285                     foundSbiCodeMeta = true;
286                     break;
287                 }
288             }
289             if (!foundSbiCodeMeta) {
290                 fail("SBI code niet gevonden in herkomst_metadata, verwacht te vinden: " + sbiCode);
291             }
292         }
