34  /**
35   * Integratie test om een nhr dataservice soap bericht te laden en te transformeren, bevat een
36   * testcase voor mantis15809 waarbij berichten van dezelfde inschrijving op verschillende
37   * tijdstippen worden geladen. <br>
38   * Draaien met: {@code mvn -Dit.test=Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest
39   * -Dtest.onlyITs=true verify -Ppostgresql -pl :brmo-loader > /tmp/postgresql.log} voor bijvoorbeeld
40   * PostgreSQL of {@code mvn -Dit.test=Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest
41   * -Dtest.onlyITs=true verify -Poracle -pl :brmo-loader > /tmp/oracle.log} voor Oracle.
42   *
43   * @author Mark Prins
44   */
45  public class Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest
46      extends AbstractDatabaseIntegrationTest {

60    @BeforeEach
61    @Override
62    public void setUp() throws Exception {
63      dsStaging = new BasicDataSource();
64      dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
65      dsStaging.setUsername(params.getProperty("staging.user"));
66      dsStaging.setPassword(params.getProperty("staging.passwd"));
67      dsStaging.setAccessToUnderlyingConnectionAllowed(true);
68      dsStaging.setConnectionProperties(params.getProperty("staging.options", ""));
69  
70      dsRsgb = new BasicDataSource();
71      dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
72      dsRsgb.setUsername(params.getProperty("rsgb.user"));
73      dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
74      dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
75      dsRsgb.setConnectionProperties(params.getProperty("rsgb.options", ""));
76  
77      staging = new DatabaseDataSourceConnection(dsStaging);
78      rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
79  
80      if (this.isOracle) {
81        staging =
82            new DatabaseConnection(
83                OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
84                params.getProperty("staging.user").toUpperCase());
85        staging
86            .getConfig()
87            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
88        staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
89  
90        rsgb =
91            new DatabaseConnection(
92                OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
93                params.getProperty("rsgb.user").toUpperCase());
94        rsgb.getConfig()
95            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
96        rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
97      } else if (this.isPostgis) {
98        // we hebben alleen nog postgres over
99        staging
100           .getConfig()
101           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
102       rsgb.getConfig()
103           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
104     }
105 
106     brmo = new BrmoFramework(dsStaging, dsRsgb, null);
107     brmo.setOrderBerichten(true);
108 
109     sequential.lock();
110 
111     // staging en rsgb tabellen leeg maken voorafgaan aan de test
112     CleanUtil.cleanSTAGING(staging, false);
113     CleanUtil.cleanRSGB_NHR(rsgb);
114 
115     Assumptions.assumeTrue(
116         0L == brmo.getCountBerichten("nhr", "STAGING_OK"), "Er zijn geen STAGING_OK berichten");
117     Assumptions.assumeTrue(
118         0L == brmo.getCountLaadProcessen("nhr", "STAGING_OK"),
119         "Er zijn geen STAGING_OK laadprocessen");
120   }

123    * staging en rsgb tabellen leeg maken
124    *
125    * @throws Exception is any
126    */
127   @AfterEach
128   public void cleanup() throws Exception {
129     brmo.closeBrmoFramework();
130 
131     CleanUtil.cleanSTAGING(staging, false);
132     staging.close();
133     dsStaging.close();
134 
135     CleanUtil.cleanRSGB_NHR(rsgb);
136     rsgb.close();
137     dsRsgb.close();
138 
139     sequential.unlock();
140   }

143   public void test() throws Exception {
144     // te laden berichten + kvknummer + vestg nummer + sbicode + aantallen
145     Object[][] testdataSet =
146         new Object[][] {
147           // {"filename", aantalBerichten, aantalProcessen, hoofdVestgID,
148           // aantalVestg_activiteit,
149           // kvkNummer, sbiCodes (eerste  sbiCode is hoofdactiviteit) },
150           {
151             "/mantis15809-nhr-bericht-generaties/78597161_20200903.anon.xml",
152             3,
153             1,
154             "nhr.comVestg.000006873758",
155             4,
156             78597161,
157             new String[] {"46738", "47789", "47918", "4329"}
158           },
159           {
160             "/mantis15809-nhr-bericht-generaties/78597161_20201029.anon.xml",
161             3,
162             1,
163             "nhr.comVestg.000006873758",
164             4,
165             78597161,
166             new String[] {"46738", "47789", "47918", "4329"}
167           },
168           {
169             "/mantis15809-nhr-bericht-generaties/78597161_20210127.anon.xml",
170             3,
171             1,
172             "nhr.comVestg.000006873758",
173             2,
174             78597161,
175             new String[] {"4329", "4339"}
176           }

179     String bestandNaam;
180     long aantalBerichten = 0;
181     long aantalProcessen = 0;
182     String vestgID;
183     long aantalVestg_activiteit;
184     long kvkNummer;

188     // berichten laden en transformeren
189     for (Object[] testdata : testdataSet) {
190       bestandNaam = (String) testdata[0];
191       aantalBerichten += Long.valueOf((Integer) testdata[1]);
192       aantalProcessen += Long.valueOf((Integer) testdata[2]);
193       vestgID = (String) testdata[3];
194       aantalVestg_activiteit = Long.valueOf((Integer) testdata[4]);
195       kvkNummer = Long.valueOf((Integer) testdata[5]);
196       sbiCodes = (String[]) testdata[6];

198       testNhrXMLToStagingToRsgb(

204           kvkNummer,

228         sbiCodes[0],
229         vestg_activiteit.getValue(0, "fk_sbi_activiteit_code"),
230         "Hoofdactiviteit in 'vestg_activiteit' klopt niet");
231     assertEquals(
232         1,
233         vestg_activiteit.getRowCount(),
234         "Aantal hoofdactiviteiten in 'vestg_activiteit' klopt niet");

237   private void testNhrXMLToStagingToRsgb(
238       String bestandNaam,
239       long aantalBerichten,
240       long aantalProcessen,
241       String vestgID,
242       long aantalVestg_activiteit,
243       long kvkNummer,
244       String[] sbiCodes,
245       int verwerktNHr)
246       throws Exception {
247 
248     assumeFalse(
249         null
250             == Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest.class.getResource(
251                 bestandNaam),
252         "Het test bestand moet er zijn.");
253 
254     brmo.loadFromFile(
255         BESTANDTYPE,
256         Mantis15809NhrBerichtGeneratiesToStagingToRsgbIntegrationTest.class
257             .getResource(bestandNaam)
258             .getFile(),
259         null);
260     LOG.info("klaar met laden van berichten in staging DB.");
261 
262     // check aantallen brmo berichten
263     List<Bericht> berichten = brmo.listBerichten();
264     List<LaadProces> processen = brmo.listLaadProcessen();
265     assertNotNull(berichten, "De verzameling berichten bestaat niet.");
266     assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
267     assertNotNull(processen, "De verzameling processen bestaat niet.");
268     assertEquals(aantalProcessen, processen.size(), "Het aantal processen is niet als verwacht.");
269 
270     // alleen het eerste bericht van een nHR bericht heeft br_orgineel_xml, de rest niet, check
271     // dat
272     ITable bericht =
273         staging.createQueryTable(
274             "bericht", "select * from bericht where volgordenummer=0 order by id");
275     assertEquals(verwerktNHr, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
276     LOG.debug("\n\n" + bericht.getValue(verwerktNHr - 1, "br_orgineel_xml") + "\n\n");
277     assertNotNull(bericht.getValue(verwerktNHr - 1, "br_orgineel_xml"), "BR origineel xml is null");
278     Object berichtId = bericht.getValue(verwerktNHr - 1, "id");
279 
280     LOG.info("Transformeren berichten naar rsgb DB.");
281     Thread t = brmo.toRsgb();
282     t.join();
283 
284     // na de verwerking moet soap payload er ook nog zijn
285     bericht =
286         staging.createQueryTable(
287             "bericht", "select * from bericht where br_orgineel_xml is not null order by id");
288     assertEquals(verwerktNHr, bericht.getRowCount(), "Er zijn meer of minder dan verwacht rijen");
289     assertNotNull(
290         bericht.getValue(verwerktNHr - 1, "br_orgineel_xml"),
291         "BR origineel xml is null na transformatie");
292     assertEquals(
293         berichtId,
294         bericht.getValue(verwerktNHr - 1, "id"),
295         "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
296 
297     // controle van transformatie
298     assertEquals(
299         aantalBerichten,
300         brmo.getCountBerichten("nhr", "RSGB_OK"),
301         "Niet alle berichten zijn OK getransformeerd");
302     berichten = brmo.listBerichten();
303     for (Bericht b : berichten) {
304       assertNotNull(b, "Bericht is 'null'");
305       assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
306     }
307 
308     // check kvk nummer
309     ITable subject = rsgb.createDataSet().getTable("subject");
310     boolean foundKvk = false;
311     for (int i = 0; i < subject.getRowCount(); i++) {
312       if (subject.getValue(i, "identif").toString().contains("nhr.maatschAct.kvk")) {
313         assertEquals(
314             kvkNummer + "", subject.getValue(i, "kvk_nummer") + "", "KVK nummer klopt niet");
315         foundKvk = true;
316       }
317     }
318     if (!foundKvk) {
319       fail("KVK nummer maatschappelijke activiteit klopt niet, verwacht te vinden: " + kvkNummer);
320     }
321 
322     // check vestiging en activiteit
323     ITable vestg = rsgb.createDataSet().getTable("vestg");
324     assertEquals(
325         vestgID, vestg.getValue(0, "sc_identif"), "De 'sc_identif' van hoofdvestiging klopt niet");
326     assertEquals(
327         sbiCodes[0],
328         vestg.getValue(0, "fk_sa_sbi_activiteit_sbi_code"),
329         "De sbi code van (hoofd)vestiging klopt niet");
330     assertEquals(
331         "Ja",
332         vestg.getValue(0, "sa_indic_hoofdactiviteit"),
333         "De 'sa_indic_hoofdactiviteit' van (hoofd)vestiging klopt niet");
334 
335     // check vestingingsactiviteiten
336     ITable vestg_activiteit = rsgb.createDataSet().getTable("vestg_activiteit");
337     assertEquals(
338         aantalVestg_activiteit,
339         vestg_activiteit.getRowCount(),
340         "Het aantal 'vestg_activiteit' records klopt niet");
341 
342     vestg_activiteit =
343         rsgb.createQueryTable(
344             "vestg_activiteit", "select * from vestg_activiteit where indicatie_hoofdactiviteit=1");
345     assertEquals(
346         sbiCodes[0],
347         vestg_activiteit.getValue(0, "fk_sbi_activiteit_code"),
348         "Hoofdactiviteit in 'vestg_activiteit' klopt niet");
349     assertEquals(
350         1,
351         vestg_activiteit.getRowCount(),
352         "Aantal hoofdactiviteit in 'vestg_activiteit' klopt niet");
353 
354     ITable sbi_activiteit = rsgb.createDataSet().getTable("sbi_activiteit");
355     ITable herkomst_metadata = rsgb.createDataSet().getTable("herkomst_metadata");
356     for (String sbiCode : sbiCodes) {
357       boolean foundSbiCode = false;
358       boolean foundSbiCodeMeta = false;
359       for (int i = 0; i < sbi_activiteit.getRowCount(); i++) {
360         if (sbiCode.equals(sbi_activiteit.getValue(i, "sbi_code").toString())) {
361           LOG.debug("SBI code " + sbiCode + " gevonden in sbi_activiteit");
362           foundSbiCode = true;
363           break;
364         }
365       }
366       if (!foundSbiCode) {
367         fail("SBI code niet gevonden in sbi_activiteit, verwacht te vinden: " + sbiCode);
368       }
369       for (int i = 0; i < herkomst_metadata.getRowCount(); i++) {
370         if (sbiCode.equals(herkomst_metadata.getValue(i, "waarde").toString())) {
371           LOG.debug("SBI code " + sbiCode + " gevonden in herkomst_metadata");
372           foundSbiCodeMeta = true;
373           break;
374         }
375       }
376       if (!foundSbiCodeMeta) {
377         fail("SBI code niet gevonden in herkomst_metadata, verwacht te vinden: " + sbiCode);
378       }
379     }
380   }
