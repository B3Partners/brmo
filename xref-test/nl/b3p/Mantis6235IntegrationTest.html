34  /**
35   * Testcases voor mantis-6235; incorrect parsen van VVE identificatie. Draaien met: {@code mvn
36   * -Dit.test=Mantis6235IntegrationTest -Dtest.onlyITs=true verify -Poracle > target/oracle.log} voor
37   * bijvoorbeeld Oracle.
38   *
39   * @author mprins
40   */
41  public class Mantis6235IntegrationTest extends AbstractDatabaseIntegrationTest {
42  
43    private static final Log LOG = LogFactory.getLog(Mantis6235IntegrationTest.class);
44  
45    private BrmoFramework brmo;
46    private IDatabaseConnection rsgb;
47    private IDatabaseConnection staging;
48    private final Lock sequential = new ReentrantLock();
49    private BasicDataSource dsStaging;
50    private BasicDataSource dsRsgb;
51  
52    @Override
53    @BeforeEach
54    public void setUp() throws Exception {
55      dsStaging = new BasicDataSource();
56      dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
57      dsStaging.setUsername(params.getProperty("staging.user"));
58      dsStaging.setPassword(params.getProperty("staging.passwd"));
59      dsStaging.setAccessToUnderlyingConnectionAllowed(true);
60  
61      dsRsgb = new BasicDataSource();
62      dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
63      dsRsgb.setUsername(params.getProperty("rsgb.user"));
64      dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
65      dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
66  
67      rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
68      staging = new DatabaseDataSourceConnection(dsStaging);
69  
70      if (this.isOracle) {
71        rsgb =
72            new DatabaseConnection(
73                OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
74                params.getProperty("rsgb.user").toUpperCase());
75        rsgb.getConfig()
76            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
77        rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
78        staging =
79            new DatabaseConnection(
80                OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
81                params.getProperty("staging.user").toUpperCase());
82        staging
83            .getConfig()
84            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
85        staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
86      } else if (this.isPostgis) {
87        // we hebben alleen nog postgres over
88        rsgb.getConfig()
89            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
90        staging
91            .getConfig()
92            .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
93      }
94  
95      IDataSet stagingDataSet =
96          new XmlDataSet(
97              new FileInputStream(
98                  new File(
99                      Mantis6235IntegrationTest.class
100                         .getResource("/mantis6235/staging.xml")
101                         .toURI())));
102 
103     sequential.lock();
104 
105     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
106 
107     brmo = new BrmoFramework(dsStaging, dsRsgb, null);
108 
109     assumeTrue(2l == brmo.getCountBerichten("brk", "STAGING_OK"), "Er zijn 2 STAGING_OK berichten");
110     assumeTrue(
111         1l == brmo.getCountLaadProcessen("brk", "STAGING_OK"), "Er is 1 STAGING_OK laadproces");
112   }
113 
114   @AfterEach
115   public void cleanup() throws Exception {
116     brmo.closeBrmoFramework();
117 
118     CleanUtil.cleanRSGB_BRK(rsgb, true);
119     CleanUtil.cleanSTAGING(staging, false);
120     rsgb.close();
121     dsRsgb.close();
122     staging.close();
123     dsStaging.close();
124 
125     sequential.unlock();
126   }

128   /**
129    * transformeer alle berichten.
130    *
131    * @throws Exception if any
132    */
133   @Test
134   public void testTransformStandberichten() throws Exception {
135     brmo.setOrderBerichten(false);
136     Thread t = brmo.toRsgb();
137     t.join();
138 
139     assertEquals(
140         2l,
141         brmo.getCountBerichten("brk", "RSGB_OK"),
142         "Niet alle berichten zijn OK getransformeerd");
143     assertEquals(
144         0l,
145         brmo.getCountBerichten("brk", "RSGB_NOK"),
146         "Niet alle berichten zijn OK getransformeerd");
147 
148     ITable brondocument = rsgb.createDataSet().getTable("brondocument");
149     assertTrue(brondocument.getRowCount() > 0, "Er zijn geen brondocumenten");
150 
151     ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
152     assertEquals(2, kad_onrrnd_zk.getRowCount(), "Er zijn geen actuele onroerende zaken");
153 
154     ITable app_re = rsgb.createDataSet().getTable("app_re");
155     assertEquals(2, app_re.getRowCount(), "Er zijn geen actuele appartmentsrechten");
156 
157     assertEquals(
158         "NL.KAD.Persoon.58193932",
159         app_re.getValue(0, "fk_2nnp_sc_identif").toString(),
160         "Eerste rij bevat niet verwachte VVE 'fk_2nnp_sc_identif'");
161     assertEquals(
162         "NL.KAD.Persoon.58201049",
163         app_re.getValue(1, "fk_2nnp_sc_identif").toString(),
164         "Tweede rij bevat niet verwachte VVE 'fk_2nnp_sc_identif'");
165   }
