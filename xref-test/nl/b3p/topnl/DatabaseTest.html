92          LOG.debug("save");
93          Geometry p = wkt.read("POINT (1 2)");
94          Hoogte e = new Hoogte();
95          getStandardTestTopNLEntity(e, type);
96          e.setGeometrie(p);
97          e.setReferentieVlak("uitgevlakt");
98          e.setTypeHoogte("superhoog");
99          e.setHoogte(16.06);
100 
101         instance.save(e);
102 
103         ResultSetHandler<Hoogte> h = new BeanHandler<>(Hoogte.class, new BasicRowProcessor(new DbUtilsGeometryColumnConverter(instance.getGjc())));
104 
105         Hoogte real = run.query("SELECT * FROM " + type.getType() + ".hoogte WHERE identificatie=?", h, identificatie);
106         assertNotNull("Insert failed", real);
107         testStandardTopNLEntity(real, e);
108 
109         assertEquals(e.getReferentieVlak(), real.getReferentieVlak());
110         assertEquals(e.getTypeHoogte(), real.getTypeHoogte());
111         assertEquals(e.getHoogte(), real.getHoogte());
112         assertEquals(p, real.getGeometrie());
113     }
