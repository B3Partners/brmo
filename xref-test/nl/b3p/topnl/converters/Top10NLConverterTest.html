422     public void testTopNLEntity(TopNLEntity expected, TopNLEntity real) {
423         assertEquals(expected.getTopnltype(), real.getTopnltype());
424 
425         assertEquals(expected.getId(), real.getId());
426         assertEquals(expected.getIdentificatie(), real.getIdentificatie());
427         assertEquals(expected.getBrontype(), real.getBrontype());
428         assertEquals(expected.getBronactualiteit(), real.getBronactualiteit());
429         assertEquals(expected.getBronbeschrijving(), real.getBronbeschrijving());
430         assertEquals(expected.getBronnauwkeurigheid(), real.getBronnauwkeurigheid());
431         assertEquals(expected.getObjectBeginTijd(), real.getObjectBeginTijd());
432         assertEquals(expected.getObjectEindTijd(), real.getObjectEindTijd());
433         assertEquals(expected.getVisualisatieCode(), real.getVisualisatieCode());

452     private TopNLEntity getEntity(String file) throws JAXBException, IOException, SAXException, ParserConfigurationException, TransformerException {
453         URL in = Top10NLConverterTest.class.getResource(file);
454         List jaxb = processor.parse(in);
455         TopNLEntity entity = instance.convertObject(jaxb.get(0));
456         return entity;
457     }
