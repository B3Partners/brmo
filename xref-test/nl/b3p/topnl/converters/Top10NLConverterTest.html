58    public void testConvertNoFeatureCollection() throws Exception {
59      Hoogte hoogte = new Hoogte();
60      URL in = Top10NLConverterTest.class.getResource("top10nl/Hoogte.xml");
61      List jaxb = processor.parse(in);
62  
63      List<TopNLEntity> expResult = Collections.singletonList(hoogte);
64      List<TopNLEntity> result = instance.convert(jaxb);
65      assertNotNull(result);
66      assertEquals(expResult.size(), result.size());
67      assertEquals(expResult.get(0).getClass(), result.get(0).getClass());
68    }

410   public void testTopNLEntity(TopNLEntity expected, TopNLEntity real) {
411     assertEquals(expected.getTopnltype(), real.getTopnltype());
412 
413     assertEquals(expected.getId(), real.getId());
414     assertEquals(expected.getIdentificatie(), real.getIdentificatie());
415     assertEquals(expected.getBrontype(), real.getBrontype());
416     assertEquals(expected.getBronactualiteit(), real.getBronactualiteit());
417     assertEquals(expected.getBronbeschrijving(), real.getBronbeschrijving());
418     assertEquals(expected.getBronnauwkeurigheid(), real.getBronnauwkeurigheid());
419     assertEquals(expected.getObjectBeginTijd(), real.getObjectBeginTijd());
420     assertEquals(expected.getObjectEindTijd(), real.getObjectEindTijd());
421     assertEquals(expected.getVisualisatieCode(), real.getVisualisatieCode());
422   }

440   private TopNLEntity getEntity(String file)
441       throws JAXBException, IOException, SAXException, ParserConfigurationException,
442           TransformerException {
443     URL in = Top10NLConverterTest.class.getResource(file);
444     List jaxb = processor.parse(in);
445     return instance.convertObject(jaxb.get(0));
446   }
