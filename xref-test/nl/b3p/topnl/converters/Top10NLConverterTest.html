61          Hoogte hoogte = new Hoogte();
62          URL in = Top10NLConverterTest.class.getResource("top10nl/Hoogte.xml");
63          List jaxb = processor.parse(in);
64  
65          List<TopNLEntity> expResult = Collections.singletonList(hoogte);
66          List<TopNLEntity> result = instance.convert(jaxb);
67          assertNotNull(result);
68          assertEquals(expResult.size(), result.size());
69          assertEquals(expResult.get(0).getClass(), result.get(0).getClass());

411     public void testTopNLEntity(TopNLEntity expected, TopNLEntity real) {
412         assertEquals(expected.getTopnltype(), real.getTopnltype());
413 
414         assertEquals(expected.getId(), real.getId());
415         assertEquals(expected.getIdentificatie(), real.getIdentificatie());
416         assertEquals(expected.getBrontype(), real.getBrontype());
417         assertEquals(expected.getBronactualiteit(), real.getBronactualiteit());
418         assertEquals(expected.getBronbeschrijving(), real.getBronbeschrijving());
419         assertEquals(expected.getBronnauwkeurigheid(), real.getBronnauwkeurigheid());
420         assertEquals(expected.getObjectBeginTijd(), real.getObjectBeginTijd());
421         assertEquals(expected.getObjectEindTijd(), real.getObjectEindTijd());
422         assertEquals(expected.getVisualisatieCode(), real.getVisualisatieCode());
423     }

441     private TopNLEntity getEntity(String file) throws JAXBException, IOException, SAXException, ParserConfigurationException, TransformerException {
442         URL in = Top10NLConverterTest.class.getResource(file);
443         List jaxb = processor.parse(in);
444         return instance.convertObject(jaxb.get(0));
445     }
