92    private static final Log LOG = LogFactory.getLog(GBAVXMLToStagingIntegrationTest.class);
93    private final Lock sequential = new ReentrantLock();
94    private BrmoFramework brmo;
95    // dbunit

98    private BasicDataSource dsStaging;

101   @BeforeEach
102   @Override
103   public void setUp() throws Exception {
104     dsStaging = new BasicDataSource();
105     dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
106     dsStaging.setUsername(params.getProperty("staging.user"));
107     dsStaging.setPassword(params.getProperty("staging.passwd"));
108     dsStaging.setAccessToUnderlyingConnectionAllowed(true);
109 
110     dsRsgb = new BasicDataSource();
111     dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
112     dsRsgb.setUsername(params.getProperty("rsgb.user"));
113     dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
114     dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
115 
116     staging = new DatabaseDataSourceConnection(dsStaging);
117     rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
118 
119     if (this.isOracle) {
120       staging =
121           new DatabaseConnection(
122               OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
123               params.getProperty("staging.user").toUpperCase());
124       staging
125           .getConfig()
126           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
127       staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
128 
129       rsgb =
130           new DatabaseConnection(
131               OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
132               params.getProperty("rsgb.user").toUpperCase());
133       rsgb.getConfig()
134           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
135       rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
136     } else if (this.isPostgis) {
137       // we hebben alleen nog postgres over
138       staging
139           .getConfig()
140           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
141       rsgb.getConfig()
142           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
143     }
144 
145     brmo = new BrmoFramework(dsStaging, dsRsgb, null);
146 
147     FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
148     fxdb.setCaseSensitiveTableNames(false);
149     IDataSet stagingDataSet =
150         fxdb.build(
151             new FileInputStream(
152                 new File(
153                     GBAVXMLToStagingIntegrationTest.class
154                         .getResource("/staging-empty-flat.xml")
155                         .toURI())));
156 
157     sequential.lock();
158 
159     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
160     assumeTrue(
161         0L == brmo.getCountBerichten("gbav", "STAGING_OK"), "Er zijn geen STAGING_OK berichten");
162     assumeTrue(
163         0L == brmo.getCountLaadProcessen("gbav", "STAGING_OK"),
164         "Er zijn geen STAGING_OK laadprocessen");
165   }
166 
167   @AfterEach
168   public void cleanup() throws Exception {
169     brmo.closeBrmoFramework();
170     CleanUtil.cleanSTAGING(staging, false);
171     CleanUtil.cleanRSGB_BAG(rsgb, true);
172     CleanUtil.cleanRSGB_BRP(rsgb);
173     rsgb.close();
174     dsRsgb.close();
175     staging.close();
176     dsStaging.close();
177     sequential.unlock();
178   }

180   @DisplayName("insertOntbrekendeKlantAfgiftenummers")
181   @ParameterizedTest(name = "{index}: type: {0}, bestand: {1}")
182   @MethodSource("argumentsProvider")
183   public void testGbavBerichtToStagingToRsgb(
184       String bestandType,
185       String bestandNaam,
186       long aantalBerichten,
187       long aantalProcessen,
188       long aantalSubject,
189       String aNummer,
190       String bsnNummer,
191       String achterNaam,
192       String aandNaamgebruik,
193       String geslacht)
194       throws Exception {
195 
196     try {
197       brmo.loadFromFile(
198           bestandType,
199           GBAVXMLToStagingIntegrationTest.class.getResource(bestandNaam).getFile(),
200           null);
201     } catch (BrmoLeegBestandException blbe) {
202       LOG.debug("Er is een bestand zonder berichten geladen (kan voorkomen...).");
203     }
204 
205     assertEquals(
206         aantalBerichten,
207         brmo.getCountBerichten(bestandType, "STAGING_OK"),
208         "Verwacht aantal berichten");
209     assertEquals(
210         aantalProcessen,
211         brmo.getCountLaadProcessen(bestandType, "STAGING_OK"),
212         "Verwacht aantal laadprocessen");
213 
214     LOG.debug("Transformeren berichten naar rsgb DB.");
215     brmo.setOrderBerichten(true);
216     Thread t = brmo.toRsgb();
217     t.join();
218 
219     assertEquals(
220         aantalBerichten,
221         brmo.getCountBerichten(bestandType, "RSGB_OK"),
222         "Niet alle berichten zijn OK getransformeerd");
223 
224     for (Bericht b : brmo.listBerichten()) {
225       Assertions.assertNotNull(b, "Bericht is 'null'");
226       Assertions.assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
227     }
228 
229     ITable subject = rsgb.createDataSet().getTable("subject");
230     assertEquals(aantalSubject, subject.getRowCount(), "Het aantal 'subject' klopt niet");
231 
232     ITable prs = rsgb.createDataSet().getTable("prs");
233     assertEquals(aantalSubject, prs.getRowCount(), "Het aantal 'prs' klopt niet");
234 
235     ITable nat_prs = rsgb.createDataSet().getTable("nat_prs");
236     assertEquals(aantalSubject, nat_prs.getRowCount(), "Het aantal 'nat_prs' klopt niet");
237     int rowNum = (int) (aantalSubject - 1);
238     assertEquals(
239         aandNaamgebruik,
240         nat_prs.getValue(rowNum, "aand_naamgebruik"),
241         "Aanduiding naamgebruik komt niet overeen");
242     assertEquals(
243         geslacht,
244         nat_prs.getValue(rowNum, "geslachtsaand"),
245         "Geslachtsaanduiding komt niet overeen");
246     assertEquals(
247         achterNaam, nat_prs.getValue(rowNum, "nm_geslachtsnaam"), "Achternaam komt niet overeen");
248 
249     ITable ingeschr_nat_prs = rsgb.createDataSet().getTable("ingeschr_nat_prs");
250     assertEquals(
251         aantalSubject, ingeschr_nat_prs.getRowCount(), "Het aantal 'ingeschr_nat_prs' klopt niet");
252     assertEquals(
253         new BigDecimal(bsnNummer),
254         ingeschr_nat_prs.getValue(rowNum, "bsn"),
255         "BSN komt niet overeen");
256     assertEquals(
257         new BigDecimal(aNummer),
258         ingeschr_nat_prs.getValue(rowNum, "a_nummer"),
259         "A-nummer komt niet overeen");
260   }
