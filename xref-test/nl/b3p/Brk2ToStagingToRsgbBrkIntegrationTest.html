295   @BeforeEach
296   @Override
297   public void setUp() throws Exception {
298     dsStaging = new BasicDataSource();
299     dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
300     dsStaging.setUsername(params.getProperty("staging.user"));
301     dsStaging.setPassword(params.getProperty("staging.passwd"));
302     dsStaging.setAccessToUnderlyingConnectionAllowed(true);
303 
304     dsRsgbBrk = new BasicDataSource();
305     dsRsgbBrk.setUrl(params.getProperty("rsgbbrk.jdbc.url"));
306     dsRsgbBrk.setUsername(params.getProperty("rsgbbrk.user"));
307     dsRsgbBrk.setPassword(params.getProperty("rsgbbrk.passwd"));
308     dsRsgbBrk.setAccessToUnderlyingConnectionAllowed(true);
309 
310     if (this.isOracle) {
311       staging =
312           new DatabaseConnection(
313               OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
314               params.getProperty("staging.user").toUpperCase());
315       staging
316           .getConfig()
317           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
318       staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
319 
320       rsgbBrk =
321           new DatabaseConnection(
322               OracleConnectionUnwrapper.unwrap(dsRsgbBrk.getConnection()),
323               params.getProperty("rsgbbrk.schema").toUpperCase());
324       rsgbBrk
325           .getConfig()
326           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
327       rsgbBrk.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
328     } else if (this.isPostgis) {
329       staging = new DatabaseDataSourceConnection(dsStaging);
330       staging
331           .getConfig()
332           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
333       rsgbBrk = new DatabaseDataSourceConnection(dsRsgbBrk, params.getProperty("rsgbbrk.schema"));
334       rsgbBrk
335           .getConfig()
336           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
337     }
338 
339     brmo = new BrmoFramework(dsStaging, null, dsRsgbBrk);
340 
341     FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
342     fxdb.setCaseSensitiveTableNames(false);
343     IDataSet stagingDataSet =
344         fxdb.build(
345             new FileInputStream(
346                 new File(
347                     Objects.requireNonNull(
348                             Brk2ToStagingToRsgbBrkIntegrationTest.class.getResource(
349                                 "/staging-empty-flat.xml"))
350                         .toURI())));
351 
352     sequential.lock();
353     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
354     // CleanUtil.cleanRSGB_BRK2(rsgbBrk);
355 
356     assumeTrue(
357         0L == brmo.getCountBerichten(BrmoFramework.BR_BRK2, "STAGING_OK"),
358         "Er zijn geen STAGING_OK berichten");
359     assumeTrue(
360         0L == brmo.getCountLaadProcessen(BrmoFramework.BR_BRK2, "STAGING_OK"),
361         "Er zijn geen STAGING_OK laadprocessen");
362   }
363 
364   @AfterEach
365   void cleanup() throws Exception {
366     brmo.closeBrmoFramework();
367 
368     CleanUtil.cleanSTAGING(staging, false);
369     CleanUtil.cleanRSGB_BRK2(rsgbBrk);
370     staging.close();
371     dsStaging.close();
372     rsgbBrk.close();
373     dsRsgbBrk.close();
374 
375     sequential.unlock();
376   }

378   @DisplayName("BRK2 XML in staging laden en transformeren")
379   @ParameterizedTest(name = "testBrk2XMLToStagingToRsgb #{index}: bestand: {0}, object ref: {1}")
380   @MethodSource("argumentsProvider")
381   void testBericht(
382       String bestandNaam,
383       String objectRef,
384       int aantalRecht,
385       int aantalStuk,
386       int aantalStukdeel,
387       int aantalNP,
388       int aantalNNP,
389       int aantalAdres,
390       int aantalKadObjLocatie,
391       int aantalPubliekRBeperking,
392       int aantalOnrndZkBeperking,
393       int aantalFiliatie,
394       int aantalAantekeningRecht,
395       int aantalIsbelastmetRecht,
396       int aantalIsbeperkttotRecht)
397       throws Exception {
398 
399     final boolean isPerceel = objectRef.endsWith("0000");
400 
401     assumeFalse(
402         null == Brk2ToStagingToRsgbBrkIntegrationTest.class.getResource(bestandNaam),
403         () -> "Het test bestand '" + bestandNaam + "' moet er zijn.");
404 
405     brmo.loadFromFile(
406         BrmoFramework.BR_BRK2,
407         Objects.requireNonNull(Brk2ToStagingToRsgbBrkIntegrationTest.class.getResource(bestandNaam))
408             .getFile(),
409         null);
410     LOG.debug("klaar met laden van berichten in staging DB.");
411 
412     List<Bericht> berichten = brmo.listBerichten();
413     List<LaadProces> processen = brmo.listLaadProcessen();
414     assertNotNull(berichten, "De verzameling berichten bestaat niet.");
415     assertEquals(1, berichten.size(), "Het aantal berichten is niet 1.");
416     assertEquals(
417         1,
418         brmo.getCountLaadProcessen(BrmoFramework.BR_BRK2, "STAGING_OK"),
419         "Het aantal berichten is niet 1.");
420     assertNotNull(processen, "De verzameling processen bestaat niet.");
421     assertEquals(1, processen.size(), "Het aantal processen is niet 1.");
422 
423     for (Bericht b : berichten) {
424       assertNotNull(b, "Bericht is 'null'");
425       assertNotNull(b.getBrXml(), "'br-xml' van bericht is 'null'");
426     }
427     assertEquals(
428         objectRef,
429         berichten.get(0).getObjectRef(),
430         "Het bericht uit de database heeft niet de juiste objectRef.");
431 
432     LOG.debug("Transformeren berichten naar rsgb DB.");
433     Thread t = brmo.toRsgb();
434     t.join();
435 
436     assertEquals(
437         1,
438         brmo.getCountBerichten(BrmoFramework.BR_BRK2, "RSGB_OK"),
439         "Niet alle berichten zijn OK getransformeerd");
440     berichten = brmo.listBerichten();
441     for (Bericht b : berichten) {
442       assertNotNull(b, "Bericht is 'null'");
443       assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
444     }
445 
446     ITable onroerendezaak = rsgbBrk.createDataSet().getTable("onroerendezaak");
447     assertEquals(1, onroerendezaak.getRowCount(), "Er is geen (of teveel) onroerendezaak");
448     assertEquals(
449         objectRef,
450         onroerendezaak.getValue(0, "identificatie"),
451         "identificatie is niet gelijk aan objectRef");
452 
453     ITable publiekrechtelijkebeperking =
454         rsgbBrk.createDataSet().getTable("publiekrechtelijkebeperking");
455     assertEquals(
456         aantalPubliekRBeperking,
457         publiekrechtelijkebeperking.getRowCount(),
458         "Aantal publiekrechtelijkebeperking klopt niet");
459 
460     ITable onroerendezaakbeperking = rsgbBrk.createDataSet().getTable("onroerendezaakbeperking");
461     assertEquals(
462         aantalOnrndZkBeperking,
463         onroerendezaakbeperking.getRowCount(),
464         "Aantal onroerendezaakbeperking klopt niet");
465 
466     ITable onroerendezaakfiliatie = rsgbBrk.createDataSet().getTable("onroerendezaakfiliatie");
467     assertEquals(
468         aantalFiliatie,
469         onroerendezaakfiliatie.getRowCount(),
470         "Aantal onroerendezaakfiliatie klopt niet");
471     // check dat alle records de objectRef als "onroerendezaak" hebben
472     for (int i = 0; i < onroerendezaakfiliatie.getRowCount(); i++) {
473       assertEquals(
474           objectRef,
475           onroerendezaakfiliatie.getValue(i, "onroerendezaak"),
476           "onroerendezaakfiliatie.onroerendezaak is niet gelijk aan objectRef");
477     }
478 
479     ITable perceelOfAppRe;
480     if (isPerceel) {
481       perceelOfAppRe = rsgbBrk.createDataSet().getTable("perceel");
482       assertEquals(
483           objectRef,
484           perceelOfAppRe.getValue(0, "identificatie"),
485           "Perceel identificatie is niet gelijk aan objectRef");
486       assertNotNull(
487           perceelOfAppRe.getValue(0, "begrenzing_perceel"),
488           "Perceel begrenzing geometrie is 'null'");
489       assertNotNull(
490           perceelOfAppRe.getValue(0, "plaatscoordinaten"), "Plaatscoordinaten geometrie is 'null'");
491     } else {
492       perceelOfAppRe = rsgbBrk.createDataSet().getTable("appartementsrecht");
493       assertEquals(
494           objectRef,
495           perceelOfAppRe.getValue(0, "identificatie"),
496           "Appartementsrecht identificatie is niet gelijk aan objectRef");
497     }
498 
499     ITable recht = rsgbBrk.createDataSet().getTable("recht");
500     ITable aantekeningrecht = rsgbBrk.createDataSet().getTable("recht_aantekeningrecht");
501     ITable isbelastmet = rsgbBrk.createDataSet().getTable("recht_isbelastmet");
502     ITable isbeperkttot = rsgbBrk.createDataSet().getTable("recht_isbeperkttot");
503     assertAll(
504         "rechten",
505         () -> assertEquals(aantalRecht, recht.getRowCount(), "Er is geen/teveel/te weinig recht"),
506         () ->
507             assertEquals(
508                 aantalAantekeningRecht,
509                 aantekeningrecht.getRowCount(),
510                 "Er zijn geen/teveel/te weinig aantekeningrecht relaties"),
511         () ->
512             assertEquals(
513                 aantalIsbelastmetRecht,
514                 isbelastmet.getRowCount(),
515                 "Er zijn geen/teveel/te weinig isbelastmet relaties"),
516         () ->
517             assertEquals(
518                 aantalIsbeperkttotRecht,
519                 isbeperkttot.getRowCount(),
520                 "Er zijn geen/teveel/te weinig isbeperkttot relaties"));
521 
522     ITable stuk = rsgbBrk.createDataSet().getTable("stuk");
523     assertEquals(aantalStuk, stuk.getRowCount(), "Er is geen/teveel/te weinig stuk");
524 
525     ITable stukdeel = rsgbBrk.createDataSet().getTable("stukdeel");
526     assertEquals(
527         aantalStukdeel, stukdeel.getRowCount(), "Het aantal stukdeel is niet als verwacht.");
528 
529     ITable persoon = rsgbBrk.createDataSet().getTable("persoon");
530     assertEquals(
531         aantalNP + aantalNNP, persoon.getRowCount(), "Het aantal persoon is niet als verwacht.");
532 
533     ITable natuurlijkpersoon = rsgbBrk.createDataSet().getTable("natuurlijkpersoon");
534     assertEquals(
535         aantalNP, natuurlijkpersoon.getRowCount(), "Het aantal nat.persoon is niet als verwacht.");
536 
537     ITable nietnatuurlijkpersoon = rsgbBrk.createDataSet().getTable("nietnatuurlijkpersoon");
538     assertEquals(
539         aantalNNP,
540         nietnatuurlijkpersoon.getRowCount(),
541         "Het aantal niet-nat.persoon is niet als verwacht.");
542 
543     ITable adres = rsgbBrk.createDataSet().getTable("adres");
544     assertEquals(aantalAdres, adres.getRowCount(), "Het aantal adressen is niet als verwacht.");
545 
546     ITable objectlocatie = rsgbBrk.createDataSet().getTable("objectlocatie");
547     assertAll(
548         "objectlocatie",
549         () ->
550             assertEquals(
551                 aantalKadObjLocatie,
552                 objectlocatie.getRowCount(),
553                 "Het aantal objectlocatie is niet als verwacht."),
554         () -> {
555           // check dat alle records de objectRef als "heeft" hebben
556           for (int i = 0; i < objectlocatie.getRowCount(); i++) {
557             assertEquals(
558                 objectRef,
559                 objectlocatie.getValue(i, "heeft"),
560                 "objectlocatie.heeft is niet gelijk aan objectRef");
561           }
562         });
