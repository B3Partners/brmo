59      @Override
60      @BeforeEach
61      public void setUp() throws Exception {
62          dsStaging = new BasicDataSource();
63          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
64          dsStaging.setUsername(params.getProperty("staging.user"));
65          dsStaging.setPassword(params.getProperty("staging.passwd"));
66          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
67  
68          dsRsgb = new BasicDataSource();
69          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
70          dsRsgb.setUsername(params.getProperty("rsgb.user"));
71          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
72          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
73  
74          rsgb = new DatabaseDataSourceConnection(dsRsgb);
75          staging = new DatabaseDataSourceConnection(dsStaging);
76  
77          if (this.isOracle) {
78              rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
79              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
80              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
81              staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
82              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
83              staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
84          } else if (this.isPostgis) {
85              // we hebben alleen nog postgres over
86              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
87              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
88          }
89  
90          IDataSet stagingDataSet = new XmlDataSet(new FileInputStream(new File(Mantis6380IntegrationTest.class.getResource("/mantis6380/staging.xml").toURI())));
91          sequential.lock();
92  
93          DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
94          brmo = new BrmoFramework(dsStaging, dsRsgb);
95  
96          // skip als de bron data er niet is
97          Assumptions.assumeTrue(2L == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
98                  "Er zijn geen 2 STAGING_OK berichten");
99          Assumptions.assumeTrue(2L == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"),
100                 "Er zijn geen 2 STAGING_OK laadproces");
101     }

105         brmo.closeBrmoFramework();

107         CleanUtil.cleanRSGB_BRK(rsgb, true);
108         rsgb.close();
109         dsRsgb.close();
110 
111         CleanUtil.cleanSTAGING(staging, false);
112         staging.close();
113         dsStaging.close();
114 
115         sequential.unlock();
116     }

118     @Test
119     public void testTransformStand() throws Exception {
120         brmo.setOrderBerichten(false);
121         Thread t = brmo.toRsgb();
122         t.join();
123 
124         assertEquals(

134 
135         ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");

141         assertEquals(1, subject.getRowCount(), "Aantal subjecten klopt niet");
142 
143         ITable ingeschr_niet_nat_prs = rsgb.createDataSet().getTable("ingeschr_niet_nat_prs");
144         assertEquals(1, ingeschr_niet_nat_prs.getRowCount(), "Aantal ingeschr_niet_nat_prs klopt niet");
145 
146         ITable zak_recht = rsgb.createDataSet().getTable("zak_recht");
147         assertEquals(2, zak_recht.getRowCount(), "Aantal zakelijke rechten klopt niet");

154         assertNull(zak_recht.getValue(1, "ar_noemer"), "noemer is niet null");
155         assertNull(zak_recht.getValue(1, "ar_teller"), "teller is niet null");
156     }

278      * daarna mutatie bericht en test of dat correct is gedaan. Verwacht een
279      * AssertionError omdat het aantal zakelijk rechten dan niet correct is
280      * (het oude zakelijk recht wordt niet opgeruimd).
281      *

283     @Test
284     public void testTransformDeleteStandDaarnaMutatieBerichten() {
285         assertThrows(AssertionError.class, ()->{
286             brmo.setOrderBerichten(false);
287             Thread t = brmo.toRsgb();
288             t.join();
289 
290             assertEquals(
291                     1L, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
292                     "Niet alle berichten zijn OK getransformeerd");

294                     1L, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"), "Niet alle berichten zijn OK getransformeerd");
295             assertEquals(0L, brmo.getCountBerichten(null, null, "brk", "RSGB_NOK"),
296                     "Er zijn berichten met status RSGB_NOK");
297 
298             // delete stand bericht
299             brmo.delete(1L);
300 
301             assertEquals(
302                     1L, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
