38  /**
39   * Testcases voor mantis-6380 / GH#332; incorrect verwerken van commanditair
40   * vennootschap rechten in mutatie. Draaien met:
41   * {@code mvn -Dit.test=Mantis6380IntegrationTest -Dtest.onlyITs=true verify -Poracle > target/oracle.log}
42   * voor bijvoorbeeld Oracle of
43   * {@code mvn -Dit.test=Mantis6380IntegrationTest -Dtest.onlyITs=true verify -Ppostgresql > target/postgresql.log}.
44   * <strong>NB.</strong> werkt niet op mssql, althans niet met de jTDS driver
45   * omdat die geen JtdsPreparedStatement#setNull() methode heeft.
46   *
47   * @author mprins
48   */
49  @Category(JTDSDriverBasedFailures.class)
50  public class Mantis6380IntegrationTest extends AbstractDatabaseIntegrationTest {
51  
52      private static final Log LOG = LogFactory.getLog(Mantis6380IntegrationTest.class);

57      private IDatabaseConnection staging;

59      private final Lock sequential = new ReentrantLock();

61      @Override
62      @Before
63      public void setUp() throws Exception {
64          BasicDataSource dsStaging = new BasicDataSource();
65          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
66          dsStaging.setUsername(params.getProperty("staging.user"));
67          dsStaging.setPassword(params.getProperty("staging.passwd"));
68          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
69  
70          BasicDataSource dsRsgb = new BasicDataSource();
71          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
72          dsRsgb.setUsername(params.getProperty("rsgb.user"));
73          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
74          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
75  
76          rsgb = new DatabaseDataSourceConnection(dsRsgb);
77          staging = new DatabaseDataSourceConnection(dsStaging);
78  
79          if (this.isMsSQL) {
80              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
81              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
82          } else if (this.isOracle) {
83              rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
84              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
85              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
86              staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
87              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
88              staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
89          } else if (this.isPostgis) {
90              // we hebben alleen nog postgres over
91              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
92              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
93          }
94  
95          IDataSet stagingDataSet = new XmlDataSet(new FileInputStream(new File(Mantis6380IntegrationTest.class.getResource("/mantis6380/staging.xml").toURI())));
96  
97          sequential.lock();
98  
99          if (this.isMsSQL) {
100             InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
101         } else {
102             DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
103         }
104         brmo = new BrmoFramework(dsStaging, dsRsgb);
105 
106         // skip als de bron data er niet is
107         assumeTrue("Er zijn geen 2 STAGING_OK berichten", 2l == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
108         assumeTrue("Er zijn geen 2 STAGING_OK laadproces", 2l == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"));
109     }

111     @After
112     public void cleanup() throws Exception {
113         brmo.closeBrmoFramework();
114 
115         CleanUtil.cleanRSGB_BRK(rsgb, true);
116         rsgb.close();
117 
118         CleanUtil.cleanSTAGING(staging, false);
119         staging.close();
120 
121         sequential.unlock();
122     }

124     @Test
125     public void testTransformStand() throws Exception {
126         brmo.setOrderBerichten(false);
127         Thread t = brmo.toRsgb();
128         t.join();
129 
130         assertEquals("Niet alle berichten zijn OK getransformeerd", 1l, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
131         assertEquals("Niet alle berichten zijn OK getransformeerd", 1l, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"));
132         assertEquals("Er zijn berichten met status RSGB_NOK", 0l, brmo.getCountBerichten(null, null, "brk", "RSGB_NOK"));
133 
134         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
135         assertEquals("Aantal actuele onroerende zaken is incorrect", 1, kad_onrrnd_zk.getRowCount());
136 
137         ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
138         assertEquals("Aantal actuele percelen is incorrect", 1, kad_perceel.getRowCount());
139         assertEquals("Perceel identif is incorrect", "37640054270000", kad_perceel.getValue(0, "sc_kad_identif").toString());
140 
141         ITable subject = rsgb.createDataSet().getTable("subject");
142         assertEquals("Aantal subjecten klopt niet", 1, subject.getRowCount());
143 
144         ITable ingeschr_niet_nat_prs = rsgb.createDataSet().getTable("ingeschr_niet_nat_prs");
145         assertEquals("Aantal ingeschr_niet_nat_prs klopt niet", 1, ingeschr_niet_nat_prs.getRowCount());
146 
147         ITable zak_recht = rsgb.createDataSet().getTable("zak_recht");
148         assertEquals("Aantal zakelijke rechten klopt niet", 2, zak_recht.getRowCount());
149         // rij 1
150         assertEquals("NL.KAD.Tenaamstelling.AKR1.100000007757634", zak_recht.getValue(0, "kadaster_identif").toString());
151         assertEquals("1", zak_recht.getValue(0, "ar_noemer").toString());
152         assertEquals("1", zak_recht.getValue(0, "ar_teller").toString());
153         // rij 2
154         assertEquals("NL.KAD.ZakelijkRecht.AKR1.100000007757634", zak_recht.getValue(1, "kadaster_identif").toString());
155         assertNull("noemer is niet null", zak_recht.getValue(1, "ar_noemer"));
156         assertNull("teller is niet null", zak_recht.getValue(1, "ar_teller"));
