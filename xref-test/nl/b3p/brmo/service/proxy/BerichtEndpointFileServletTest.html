25   *
26   * @author mprins
27   */
28  public class BerichtEndpointFileServletTest {
29  
30      private String saveDir = "/bericht-post-test-";
31      private ServletUnitClient client;
32      private ServletRunner sr;
33  
34      @Before
35      public void setUp() throws Exception {
36          saveDir = System.getProperty("java.io.tmpdir", "/tmp") + saveDir + new BigInteger(130, new SecureRandom()).toString(32);
37          // sr = new ServletRunner(BerichtEndpointFileServletTest.class.getResourceAsStream("/web.xml"));
38          sr = new ServletRunner();
39          Hashtable<String, String> servletParams = new Hashtable();
40          servletParams.put("save_dir", saveDir);
41          sr.registerServlet("/post/brk", BerichtEndpointFileServlet.class.getName(), servletParams);
42          client = sr.newClient();
43      }
44  
45      @After
46      public void tearDown() throws Exception {
47          sr.shutDown();
48          FileUtils.cleanDirectory(new File(saveDir));
49          FileUtils.deleteQuietly(new File(saveDir));
50      }
51  
52      @Test
53      public void testPostBRK() throws Exception {
54          WebRequest p = new PostMethodWebRequest("http://localhost:8080/post/brk", BerichtEndpointFileServletTest.class.getResourceAsStream("/web.xml"), "text/xml");
55          WebResponse response = client.getResponse(p);
56          assertNotNull("No response received", response);
57          assertEquals("Response not OK", 200, response.getResponseCode());
58  
59          // servlet scrijft naar saveDir, /tmp/brk/post-op_<yyyy-MM-dd_HH-mm-ss-SSS>_<randomuniek>.xml
60          File expected = new File(BerichtEndpointFileServletTest.class.getResource("/web.xml").getFile());
61          File actual = Files.newDirectoryStream(new File(saveDir).toPath(), "*.{xml}").iterator().next().toFile();
62  
63          assertTrue("Bestand begint niet met 'post-op", actual.getName().startsWith("post-op"));
64          assertTrue("File inhoud niet gelijk", FileUtils.contentEquals(
65                  expected,
66                  actual
67          ));
68      }
69  }
