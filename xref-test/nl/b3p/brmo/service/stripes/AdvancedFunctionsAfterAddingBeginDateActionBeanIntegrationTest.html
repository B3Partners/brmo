125   private void loadData(
126       String sBestandsNaam, long aantalBerichten, long aantalProcessen, String rBestandsNaam)
127       throws Exception {
128 
129     assumeTrue(
130         AdvancedFunctionsAfterAddingBeginDateActionBeanIntegrationTest.class.getResource(
131                 sBestandsNaam)
132             != null,
133         "Het bestand met staging testdata zou moeten bestaan.");
134     assumeTrue(
135         AdvancedFunctionsAfterAddingBeginDateActionBeanIntegrationTest.class.getResource(
136                 rBestandsNaam)
137             != null,
138         "Het bestand met rsgb testdata zou moeten bestaan.");
139 
140     FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
141     fxdb.setCaseSensitiveTableNames(false);
142     IDataSet stagingDataSet =
143         fxdb.build(
144             new FileInputStream(
145                 new File(
146                     AdvancedFunctionsAfterAddingBeginDateActionBeanIntegrationTest.class
147                         .getResource(sBestandsNaam)
148                         .toURI())));
149     IDataSet rsgbDataSet =
150         fxdb.build(
151             new FileInputStream(
152                 new File(
153                     AdvancedFunctionsAfterAddingBeginDateActionBeanIntegrationTest.class
154                         .getResource(rBestandsNaam)
155                         .toURI())));
156 
157     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
158     DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
159 
160     assumeTrue(
161         aantalBerichten == brmo.getCountBerichten("brk", "RSGB_OK"), "Er zijn x RSGB_OK berichten");
162     assumeTrue(
163         aantalProcessen == brmo.getCountLaadProcessen("brk", "STAGING_OK"),
164         "Er zijn x STAGING_OK laadprocessen");
165   }

167   @AfterEach
168   public void cleanup() throws Exception {
169     if (brmo != null) {
170       // in geval van niet waar gemaakte assumptions
171       brmo.closeBrmoFramework();
172       brmo = null;
173     }
174     if (rsgb != null) {
175       CleanUtil.cleanRSGB_BRK(rsgb, true);
176       rsgb.close();
177       rsgb = null;
178     }
179     if (staging != null) {
180       CleanUtil.cleanSTAGING(staging, true);
181       staging.close();
182       staging = null;
183     }
184     try {
185       sequential.unlock();
186     } catch (IllegalMonitorStateException e) {
187       // in geval van niet waar gemaakte assumptions
188       LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
189     }
190   }

192   @ParameterizedTest(
193       name =
194           "Replay BRK Verwijder berichten na update IngangsdatumRecht {index}: verwerken bestand: ''{0}''")
195   @MethodSource("argumentsProvider")
196   public void testReplayBRKVerwijderBerichtenNaUpdateIngangsdatumRecht(
197       String sBestandsNaam, long aantalBerichten, long aantalProcessen, String rBestandsNaam)
198       throws Exception {
199 
200     loadData(sBestandsNaam, aantalBerichten, aantalProcessen, rBestandsNaam);
201 
202     updatesBean.populateUpdateProcesses();
203     updatesBean.setUpdateProcessName("Bijwerken ingangsdatum_recht zakelijk recht");
204     updatesBean.update();
205     Date now = new Date();
206     long maxWait = 20000;
207     while (!updatesBean.isComplete() && (new Date().getTime() - now.getTime() < maxWait)) {
208       Thread.sleep(1000);
209       LOG.info("waiting...");
210     }
211     LOG.info("Finished waiting");
212 
213     final IDataSet rds = rsgb.createDataSet();
214     ITable zak_recht = rsgb.createDataSet().getTable("zak_recht");
215     for (int i = 0; i < zak_recht.getRowCount(); i++) {
216       Assertions.assertNotNull(
217           zak_recht.getValue(i, "ingangsdatum_recht"), "Ingangsdatum recht is niet gevuld");
218       Assertions.assertNotNull(
219           zak_recht.getValue(i, "fk_3avr_aand"), "fk_3avr_aand recht is niet gevuld");
220     }
221 
222     // deze gegevens komen uit de test data set
223     Assertions.assertEquals(
224         1,
225         rds.getTable("kad_onrrnd_zk").getRowCount(),
226         "Er is een spook record in de kad_onrrnd_zk tabel");
227     Assertions.assertEquals(
228         0, rds.getTable("kad_perceel").getRowCount(), "De perceel tabel is leeg");
229     Assertions.assertEquals(
230         aantalBerichten - (1 + 1),
231         rds.getTable("kad_onrrnd_zk_archief").getRowCount(),
232         "De kad_onrrnd_zk_archief komt een record te kort");
233     Assertions.assertTrue(
234         0 < rds.getTable("kad_perceel_archief").getRowCount(),
235         "Er is minstens een perceel in kad_perceel_archief");
236 
237     bean.replayBRKVerwijderBerichten(BrmoFramework.BR_BRK, Bericht.STATUS.RSGB_OK.toString());
238 
239     Assertions.assertEquals(
240         0,
241         rds.getTable("kad_onrrnd_zk").getRowCount(),
242         "Er zijn geen spook records in de kad_onrrnd_zk tabel");
243     Assertions.assertTrue(
244         0 < rds.getTable("kad_onrrnd_zk_archief").getRowCount(),
245         "De kad_onrrnd_zk_archief tabel is niet leeg");
246     Assertions.assertEquals(
247         aantalBerichten - 1,
248         rds.getTable("kad_perceel_archief").getRowCount(),
249         "Er zit voor ieder bericht met perceel een record in kad_perceel_archief");
250 
251     Assertions.assertEquals(
252         aantalBerichten,
253         brmo.getCountBerichten("brk", "RSGB_OK"),
254         "Alle berichten hebben status RSGB_OK");
255   }
