44  
45  
46      private final Lock sequential = new ReentrantLock(true);
47      /*
48       * test parameters.
49       */
50      private final String sBestandsNaam = "/rsgb.xml";
51      private final int aantalPercelen = 1;
52      private final String zoekLocatie = "POINT(230341 472237)";
53      private final int bufferAfstand = 100;
54      private final long eersteKadId = 66860489870000L;
55      private final double oppPerceel = 1050;
56      private final String aardCultuurOnbebouwd = "Wegen";
57      private final String gemCode = "MKL00";
58      private final int perceelNummer = 4898;
59      private IDatabaseConnection rsgb;
60  
61      @BeforeEach
62      @Override
63      public void setUp() throws Exception {
64          rsgb = new DatabaseDataSourceConnection(this.dsRsgb);
65          IDataSet rsgbDataSet = new XmlDataSet(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource(sBestandsNaam).toURI())));
66  
67          if (this.isMsSQL) {
68              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
69  
70              rsgbDataSet = new XmlDataSet(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource("/mssql-" + sBestandsNaam.substring(1)).toURI())));
71          } else if (this.isOracle) {
72              rsgb = new DatabaseConnection(dsRsgb.getConnection().unwrap(oracle.jdbc.OracleConnection.class), DBPROPS.getProperty("rsgb.username").toUpperCase());
73              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
74              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
75  
76              rsgbDataSet = new XmlDataSet(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource("/oracle-" + sBestandsNaam.substring(1)).toURI())));
77          } else if (this.isPostgis) {
78              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
79          }
80  
81          sequential.lock();
82  
83          DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
84      }
85  
86      @AfterEach
87      public void cleanup() throws Exception {
88          CleanUtil.cleanRSGB_BRK(rsgb, true);
89          rsgb.close();
90  
91          sequential.unlock();
92      }

105     }
