22  class BAG2LoaderUtilsTest {
23  
24      @ParameterizedTest(name="[{index}] {0}")
25      @MethodSource
26      void analyzeBAG2FileName(String name, BAG2LoaderUtils.BAG2FileName expected) {
27          assertEquals(expected, BAG2LoaderUtils.analyzeBAG2FileName(name));
28      }
29  
30      @Test
31      void analyzeBAG2FileNameError() {
32          assertThrows(IllegalArgumentException.class, () -> BAG2LoaderUtils.analyzeBAG2FileName("bla.zip"));
33      }
34  
35      private static Stream<Arguments> analyzeBAG2FileName() {
36          BAG2LoaderUtils.BAG2FileName nlStand = new BAG2LoaderUtils.BAG2FileName("lvbag-extract-nl.zip", true, false, false);
37          return Stream.of(
38                  Arguments.of("lvbag-extract-nl.zip", nlStand),
39                  Arguments.of("https://some.host/path/lvbag-extract-nl.zip", nlStand),
40                  Arguments.of("http://other.host/path/subpath/lvbag-extract-nl.zip", nlStand),
41                  Arguments.of("https://extracten.bag.kadaster.nl/lvbag/extracten/Nederland%20LVC/BAGNLDL-08112021.zip", new BAG2LoaderUtils.BAG2FileName("BAGNLDL-08112021.zip", true, false, false)),
42                  Arguments.of("some" + File.separator + "local" + File.separator + "path" + File.separator + "lvbag-extract-nl.zip", nlStand),
43                  Arguments.of("BAGGEM0344L-08112021.zip", new BAG2LoaderUtils.BAG2FileName("BAGGEM0344L-08112021.zip", true, true, false, "0344")),
44                  Arguments.of("BAGGEM0344M-08112021-08122021.zip", new BAG2LoaderUtils.BAG2FileName("BAGGEM0344M-08112021-08122021.zip", false, true, true, "0344", LocalDate.of(2021, 11, 8), LocalDate.of(2021, 12, 8))),
45                  Arguments.of("BAGNLDM-08112021-08122021.zip", new BAG2LoaderUtils.BAG2FileName("BAGNLDM-08112021-08122021.zip", false, false, true, null, LocalDate.of(2021, 11, 8), LocalDate.of(2021, 12, 8))),
46                  Arguments.of("BAGNLDM-08112021-09112021.zip", new BAG2LoaderUtils.BAG2FileName("BAGNLDM-08112021-09112021.zip", false, false, false, null, LocalDate.of(2021, 11, 8), LocalDate.of(2021, 11, 9))),
47                  Arguments.of("BAGNLDM-31102021-01112021.zip", new BAG2LoaderUtils.BAG2FileName("BAGNLDM-31102021-01112021.zip", false, false, false, null, LocalDate.of(2021, 10, 31), LocalDate.of(2021, 11, 1)))
48          );
49      }
