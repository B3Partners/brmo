105   @BeforeEach
106   void setUp(TestInfo info) throws Exception {
107     assumeFalse(StringUtils.isEmpty(dbUrl), "skipping integration test: missing database url");
108     URL u = BAGLoaderDatabaseIntegrationTest.class.getResource("/BAGGEM1904L-15102021.zip");
109     assumeFalse(null == u, "skipping integration test: missing testdata");
110     testFileName = u.getFile();
111     bag = new DatabaseConnection(DriverManager.getConnection(dbUrl, dbUser, dbPass));
112 
113     if (dbUrl.contains("postgresql")) {
114       bag.getConfig()
115           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
116       bag.getConfig().setProperty(DatabaseConfig.FEATURE_QUALIFIED_TABLE_NAMES, true);
117       tableQualifierPrefix = "bag.";
118       schema = "bag";
119     }
120     if (dbUrl.contains("oracle")) {
121       bag.getConfig()
122           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
123       bag.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
124 
125       expectedXmlDataSetSuffix = "-oracle";
126       // If we don't set the schema to the user the DatabaseMetaData.getTables() call without
127       // a schema filter will take 5 minutes
128       schema = dbUser.toUpperCase();
129     }
130 
131     bag2LoadOptions = new BAG2LoadOptions();
132     // dit ruim alle BAG tabelle op met een cascade; dat is niet wat je wilt omdat daar tabellen
133     // een views van het RSGB schema bij zitten
134     // bag2LoadOptions.setDropIfExists(true);
135 
136     databaseOptions = new BAG2DatabaseOptions();
137     databaseOptions.setConnectionString(dbUrl);
138     databaseOptions.setUser(dbUser);
139     databaseOptions.setPassword(dbPass);
140     bag2Database = new BAG2Database(databaseOptions);
141 
142     if (info.getTestMethod().isPresent()) {
143       if (info.getTestMethod().orElseThrow().getAnnotation(SkipDropTables.class) == null) {
144         try {
145           dropTables(bag.getConnection(), schema, dbUrl.contains("oracle"));
146         } catch (Exception e) {
147           LOG.error("Exception dropping tables before test", e);
148         }
149       }
150     }
151   }

153   private static void dropTables(Connection connection, String schema, boolean isOracle)
154       throws SQLException {
155     LoggingQueryRunner qr = new LoggingQueryRunner();
156     if (!isOracle) {
157       LOG.trace("Drop BAG schema");
158       qr.update(connection, "drop schema if exists " + schema + " cascade");
159     } else {
160       IMetadataHandler metadataHandler = new DefaultMetadataHandler();
161       try (ResultSet tablesRs =
162           metadataHandler.getTables(connection.getMetaData(), schema, new String[] {"TABLE"})) {
163         while (tablesRs.next()) {
164           String tableName = tablesRs.getString("TABLE_NAME");
165           try {
166             LOG.trace("Drop table: " + tableName);
167             qr.update(connection, "drop table " + tableName + " cascade constraints");
168           } catch (SQLException se) {
169             LOG.warn("Exception dropping table " + tableName + ": " + se.getLocalizedMessage());
170           }
171         }
172       }
173       try {
174         qr.update(connection, "drop sequence objectid_seq");
175       } catch (Exception ignored) {
176       }
177       qr.update(connection, "delete from user_sdo_geom_metadata");
178     }
179   }

186   private void loadBAGResourceFile(String file) {
187     try {
188       BAG2LoaderMain loader = new BAG2LoaderMain();
189       loader.loadFiles(
190           bag2Database,
191           databaseOptions,
192           bag2LoadOptions,
193           new BAG2ProgressReporter(),
194           new String[] {file},
195           null);
196     } catch (Exception e) {
197       fail("Laden BAG data uit resource " + file + " is mislukt: " + e.getLocalizedMessage(), e);
198     }
199   }

201   private void compareDataSet(String[] tables, String expectedXmlDataSetFileName) throws Exception {
202     IDatabaseTester databaseTester =
203         new JdbcDatabaseTester(
204             bag2Database.getDialect().getDriverClass(),
205             databaseOptions.getConnectionString(),
206             databaseOptions.getUser(),
207             databaseOptions.getPassword(),
208             schema);
209     IDatabaseConnection dbTestConnection = databaseTester.getConnection();
210     IDataSet actualDataSet = dbTestConnection.createDataSet(tables);
211     if (System.getProperty("db.writeActualDataSet") != null) {
212       XmlDataSet.write(actualDataSet, System.out);
213     }
214     IDataSet expectedDataSet =
215         new XmlDataSet(
216             Objects.requireNonNull(
217                     BAGLoaderDatabaseIntegrationTest.class.getResource(
218                         String.format(
219                             "/expected/%s%s.xml",
220                             expectedXmlDataSetFileName, expectedXmlDataSetSuffix)))
221                 .openStream());
222     for (String table : tables) {
223       Assertion.assertEqualsIgnoreCols(
224           expectedDataSet, actualDataSet, table, new String[] {"objectid"});
225     }
226   }

308   // Leave this as last integration test case, so the BAG tables loaded by it can be used to test
309   // the create view scripts
310   @Test
311   @Order(Integer.MAX_VALUE)
312   void testStandAllTablesAndViewsHaveRows() throws Exception {
313     if (bag2Database.getDialect() instanceof PostGISDialect) {
314       // For Oracle, BRMO_METADATA table will have been dropped, for PostGIS not because it's in
315       // public schema
316       bag2Database.setMetadataValue(BAG2SchemaMapper.Metadata.FILTER_GEOMETRIE, null);
317     }
318     loadBAGResourceFile(testFileName);
319 
320     // check tables
321     for (String t : BAGTABLES) {
322       // omdat sommige BAG tabellen ook in RSGB schema zitten bag qualifier gebruiken
323       assertTrue(bag.getRowCount(tableQualifierPrefix + t) > 0, "Onverwacht lege tabel: " + t);
324     }
325     // check views
326     for (String t : BAGACTUEELVIEWS) {
327       assertTrue(bag.getRowCount(tableQualifierPrefix + t) > 0, "Onverwacht lege view: " + t);
328     }
329   }
