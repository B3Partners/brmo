42      @ParameterizedTest(name="[{index}] {0}")
43      @MethodSource
44      void testAllFeatureTypes(String file, String objectName, Object[][] attributes) throws XMLStreamException {
45          BGTObjectStreamer streamer = new BGTObjectStreamer(BGTTestFiles.getTestInputStream(file));
46          Iterator<BGTObject> iterator = streamer.iterator();
47          BGTObject object = iterator.next();
48          assertNotNull(object);
49          assertEquals(objectName, object.getObjectType().getName());
50          Map<String,Object> objectAttributes = object.getAttributes();
51          //System.out.println(objectAttributes);
52          Set<String> allKeys = new HashSet<>(objectAttributes.keySet());
53          for(Object[] attribute: attributes) {
54              String key = (String) attribute[0];
55              Object value = attribute[1];
56              assertTrue(objectAttributes.containsKey(key), "Attribute should exist: " + key);
57              Object objectValue = objectAttributes.get(key);
58              assertEquals(value, objectValue, "Attribute equals: " + key);
59              allKeys.remove(key);
60          }
61          assertEquals(0, allKeys.size(), "Unexpected attributes: " + allKeys);
62      }

110     @Test
111     void testOpenbareRuimteLabel() throws XMLStreamException {
112         BGTObjectStreamer streamer = new BGTObjectStreamer(BGTTestFiles.getTestInputStream("bgt_openbareruimtelabel.gml"));
113         Iterator<BGTObject> iterator = streamer.iterator();
114         GeometryFactory gf = new GeometryFactory();
115         Point[] punten = new Point[] {
116                 gf.createPoint(new Coordinate(19993.166, 365316.442)),
117                 gf.createPoint(new Coordinate(19968.671, 365117.131)),
118                 gf.createPoint(new Coordinate(19944.305, 364918.490)),
119                 gf.createPoint(new Coordinate(19931.877, 364785.079)),
120         };
121         String[] hoeken = new String[] {
122                 "-82.9",
123                 "-83.1",
124                 "-82.9",
125                 "-82.8",
126         };
127         for(int i = 0; i < punten.length; i++) {
128             BGTObject object = iterator.next();
129             assertEquals("OpenbareRuimteLabel", object.getObjectType().getName());
130             assertNotNull(object);
131             assertEquals(i, object.getAttributes().get("idx"));
132             assertEquals(hoeken[i], object.getAttributes().get("hoek"));
133         }
134         AtomicInteger count = new AtomicInteger(punten.length);
135         iterator.forEachRemaining(o -> count.incrementAndGet());
136         assertEquals(20, count.get());
137     }

151     @Test
152     void testMutatiePand() throws XMLStreamException {
153         BGTObjectStreamer streamer = new BGTObjectStreamer(BGTTestFiles.getTestInputStream("bgt_mutatie_delta_pand.xml"));
154         Iterator<BGTObject> iterator = streamer.iterator();
155         BGTObject object = iterator.next();
156         assertNotNull(object);
157         assertEquals("BuildingPart", object.getObjectType().getName());
158         assertEquals(BGTObject.MutatieStatus.WAS_WORDT, object.getMutatieStatus());
159         assertEquals("ad402cde39e2fb79590fad390e6b78a5", object.getMutatiePreviousVersionGmlId());
160         Map<String,Object> attributes = object.getAttributes();
161         assertEquals("a318eb7108fd639cfe5ebfc36b723aec", attributes.get("gmlId"));
162         assertEquals("G0335.4c7766d48a82474abafc59d41707a232", attributes.get("identificatie"));
163         assertNotNull(attributes.get("geometrie2dGrondvlak"));
164         assertTrue(attributes.get("geometrie2dGrondvlak") instanceof MultiPolygon);
165         MultiPolygon polygon = (MultiPolygon)attributes.get("geometrie2dGrondvlak");
166         assertEquals(28992, polygon.getSRID());
167         assertEquals(1, polygon.getNumGeometries());
168         Polygon p = (Polygon) polygon.getGeometryN(0);
169         assertEquals(0, p.getNumInteriorRing());
170         assertEquals("LINEARRING (127439.592 452184.313, 127427.892 452186.396, 127426.384 452177.925, 127438.084 452175.843, 127439.592 452184.313)", p.getExteriorRing().toString());
171         assertNotNull(attributes.get("nummeraanduidingreeks"));
172         assertTrue(attributes.get("nummeraanduidingreeks") instanceof List);
173         List<BGTObject> nummeraanduidingreeksen = (List<BGTObject>) attributes.get("nummeraanduidingreeks");
174         assertEquals(1, nummeraanduidingreeksen.size());
175         object = nummeraanduidingreeksen.get(0);
176         assertEquals("nummeraanduidingreeks", object.getObjectType().getName());
177         assertEquals(object.getAttributes(), Stream.of(new Object[][] {
178                 {"identificatieBAGVBOLaagsteHuisnummer", "0335010000327253"},
179                 {"plaatsingspunt", new GeometryFactory().createPoint(new Coordinate(127434.037, 452182.942))},
180                 {"tekst", "27"},
181                 {"hoek", 11.1},
182         }).collect(Collectors.toMap(e -> (String)e[0], e -> e[1])));
183 
184         object = iterator.next();
185         assertNotNull(object);
186         assertEquals("BuildingPart", object.getObjectType().getName());
187         assertEquals(BGTObject.MutatieStatus.WORDT, object.getMutatieStatus());
188         assertEquals("6c729d86ce928280e0168412f3923c90", object.getAttributes().get("gmlId"));
189     }
