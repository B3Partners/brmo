58      private static void setDefaultDatabaseProperties() {
59          DatabaseOptions defaultDbOptions = new DatabaseOptions();
60          Stream.of(new String[][] {
61                  {"db.connectionString", defaultDbOptions.getConnectionString()},
62                  {"db.user", defaultDbOptions.getUser()},
63                  {"db.password", defaultDbOptions.getPassword()},
64          }).forEach(property -> {
65              String value = System.getProperty(property[0]);
66              if (value == null) {
67                  System.setProperty(property[0], property[1]);
68              }
69          });
70      }
71  
72      protected static DatabaseOptions getTestDatabaseOptions() {
73          // Get database connection options from test profile, but set defaults if no Maven profile specified
74          // With the defaults also set as properties these can be used for enabling/disabling tests for certain databases
75          setDefaultDatabaseProperties();
76          DatabaseOptions dbOptions = new DatabaseOptions();
77          dbOptions.setConnectionString(System.getProperty("db.connectionString"));
78          dbOptions.setUser(System.getProperty("db.user"));
79          dbOptions.setPassword(System.getProperty("db.password"));
80          return dbOptions;
81      }
82  
83      private static IDatabaseTester getDatabaseTester(DatabaseOptions dbOptions, String schema) throws ClassNotFoundException {
84          BGTDatabase db = new BGTDatabase(dbOptions);
85          return new JdbcDatabaseTester(db.getDialect().getDriverClass(), dbOptions.getConnectionString(), dbOptions.getUser(), dbOptions.getPassword(), schema);
86      }
87  
88      public static String getSchema(BGTDatabase db, DatabaseOptions dbOptions) {
89          if (db.getDialect() instanceof OracleDialect) {
90              // If we don't set the schema to the user the DatabaseMetaData.getTables() call without a schema filter will take 5 minutes
91              return dbOptions.getUser().toUpperCase();
92          }
93          return null;
94      }
95  
96      @BeforeEach
97      void setUp(TestInfo info) throws Exception {
98          dbOptions = getTestDatabaseOptions();
99          db = new BGTDatabase(dbOptions);
100         dbTest = getDatabaseTester(dbOptions, getSchema(db, dbOptions));
101         dbTestConnection = dbTest.getConnection();
102 
103         IDataTypeFactory dtf;
104         if (db.getDialect() instanceof PostGISDialect) {
105             dtf = new PostgresqlDataTypeFactory();
106             datasetFileSuffix = "_postgresql.xml";
107         } else if (db.getDialect() instanceof MSSQLDialect) {
108             dtf = new MsSqlDataTypeFactory();
109             datasetFileSuffix = "_mssql.xml";
110         } else if (db.getDialect() instanceof OracleDialect) {
111             dtf = new Oracle10DataTypeFactory();
112             datasetFileSuffix = "_oracle.xml";
113         } else {
114             throw new IllegalArgumentException();
115         }
116         dbTestConnection.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, dtf);
117         dbTestConnection.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
118 
119         if (info.getTestMethod().isPresent()) {
120             if(info.getTestMethod().get().getAnnotation(SkipDropTables.class) == null) {
121                 try {
122                     dropTables(dbTestConnection.getConnection(), getSchema(db, dbOptions), db.getDialect() instanceof OracleDialect);
123                 } catch(Exception e) {
124                     LOG.error("Exception dropping tables before test", e);
125                 }
126             }
127         }
128     }
129 
130     @AfterAll
131     static void dropTablesAfterAllTests() throws SQLException, ClassNotFoundException {
132         DatabaseOptions dbOptions = getTestDatabaseOptions();
133         try (BGTDatabase db = new BGTDatabase(dbOptions)) {
134             dropTables(db.getConnection(), getSchema(db, dbOptions), db.getDialect() instanceof OracleDialect);
135         }
136     }

138     protected void assertDataSetEquals(String tables, String dataSetXmlFiles) throws DatabaseUnitException, SQLException, IOException {
139         IDataSet actualDataSet = dbTestConnection.createDataSet(tables.split(","));
140         if (System.getProperty("db.writeActualDataSet") != null) {
141             write(actualDataSet);
142         }
143         List<IDataSet> dataSets = new ArrayList<>();
144         for(String dataSetXmlFile: dataSetXmlFiles.split(",")) {
145             dataSets.add(new XmlDataSet(BGTTestFiles.getTestInputStream("expected/" + dataSetXmlFile + datasetFileSuffix)));
146         }
147         IDataSet expectedDataSet = new CompositeDataSet(dataSets.toArray(new IDataSet[]{}));
148         Assertion.assertEquals(expectedDataSet, actualDataSet);
149     }
150 
151     protected void assertMinRowCounts(Object[][] featureTypesMinRowCounts) throws SQLException {
152         for(Object[] featureType: featureTypesMinRowCounts) {
153             String table = (String) featureType[0];
154             int minRowCount = (int) featureType[1];
155             int actualRowCount = dbTestConnection.getRowCount(table);
156             assertTrue(actualRowCount >= minRowCount,
157                     String.format("Expected a minimum row count of %d for table %s, actual %d",
158                             minRowCount,
159                             table,
160                             actualRowCount)
161             );
162         }
163     }

166         XmlDataSet.write(dataSet, System.out);
167     }

174     private static void dropTables(Connection connection, String schema, boolean isOracle) throws SQLException {
175         for (String tableName: Stream.concat(
176                 BGTSchema.getInstance().getAllObjectTypes().map(objectType -> BGTSchemaMapper.getInstance().getTableNameForObjectType(objectType, "")),
177                 Stream.of(BGTSchemaMapper.METADATA_TABLE_NAME)).collect(Collectors.toList())) {
178             if (isOracle) {
179                 tableName = tableName.toUpperCase();
180             }
181             if(new DefaultMetadataHandler().tableExists(connection.getMetaData(), schema, tableName)) {
182                 try {
183                     LOG.trace("Drop table: " + tableName);
184                     new LoggingQueryRunner().update(connection,"drop table " + tableName);
185                 } catch (SQLException se) {
186                     LOG.warn("Exception dropping table " + tableName + ": " + se.getLocalizedMessage());
187                 }
188             }
189         }
190         if (isOracle) {
191             new LoggingQueryRunner().update(connection,"delete from user_sdo_geom_metadata");
192         }
193     }
