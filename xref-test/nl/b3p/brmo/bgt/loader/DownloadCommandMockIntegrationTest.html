35  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
36  public class DownloadCommandMockIntegrationTest extends CommandLineTestBase {
37      @Test
38      @Order(1)
39      void downloadInitialCompareThreeRows() throws IOException, InterruptedException, SQLException, DatabaseUnitException {
40          mockWebServer.enqueue(new MockResponse()
41                  .setBody("{\"_links\":{\"status\":{\"href\":\"/lv/bgt/download/v1_0/delta/custom/d396d842-3963-4377-55e1-0e9a91e1de01/status\"}},\"downloadRequestId\":\"d396d842-3963-4377-55e1-0e9a91e1de01\"}")
42                  .setResponseCode(202)
43          );
44          mockWebServer.enqueue(new MockResponse()
45                  .setBody("{\"_links\":{\"self\":{\"href\":\"/lv/bgt/download/v1_0/delta/custom/d396d842-3963-4377-55e1-0e9a91e1de01/status\"}},\"progress\":0,\"status\":\"PENDING\"}")
46                  .setResponseCode(200)
47          );
48          mockWebServer.enqueue(new MockResponse()
49                  .setBody("{\"_links\":{\"self\":{\"href\":\"/lv/bgt/download/v1_0/delta/custom/d396d842-3963-4377-55e1-0e9a91e1de01/status\"}},\"progress\":38,\"status\":\"RUNNING\"}")
50                  .setResponseCode(200)
51          );
52          mockWebServer.enqueue(new MockResponse()
53                  .setBody("{\"_links\":{\"download\":{\"href\":\"/lv/bgt/download/v1_0/extract/c31c1c7e-a59d-4357-991d-dcc68069d05e/extract.zip\"}},\"progress\":100,\"status\":\"COMPLETED\"}")
54                  .setResponseCode(201)
55          );
56          mockWebServer.enqueue(new MockResponse()
57                  .setBody(new Buffer().readFrom(getTestInputStream("extract.zip")))
58                  .setResponseCode(200)
59          );
60          String url = mockWebServer.url("/").toString();
61          run("download initial", "--no-http-zip-random-access", "--download-service=" + url, "--feature-types=all,plaatsbepalingspunt", "--geo-filter=POLYGON ((131021 458768, 131021 459259, 131694 459259, 131694 458768, 131021 458768))", "--max-objects=3", "--include-history");
62  
63          RecordedRequest recordedRequest =  mockWebServer.takeRequest(0, TimeUnit.SECONDS);
64          assertNotNull(recordedRequest);
65          String body = recordedRequest.getBody().readString(Charset.defaultCharset());
66          assertTrue(body.contains("\"format\":\"citygml\""));
67          assertDataSetEquals("kast,ondersteunendwaterdeel,paal,sensor,begroeidterreindeel,overbruggingsdeel,spoor," +
68                  "bord,pand,nummeraanduidingreeks,functioneelgebied,onbegroeidterreindeel,gebouwinstallatie,weginrichtingselement," +
69                  "kunstwerkdeel,waterinrichtingselement,installatie,vegetatieobject,waterdeel,put,scheiding,ondersteunendwegdeel" +
70                  "straatmeubilair,wegdeel,plaatsbepalingspunt,bak,openbareruimtelabel,overigbouwwerk,overigescheiding",
71                  "extract");
72      }

92      @Test
93      @Order(3)
94      void catchApiException() {
95          mockWebServer.enqueue(new MockResponse()
96                  .setBody("invalid wktString. a polygon ring must have at least 4 points, got 1")
97                  .setHeader("Content-Type", "text/plain; charset=utf-8")
98                  .setResponseCode(400)
99          );
100         String url = mockWebServer.url("/").toString();
101         StringWriter sw = new StringWriter();
102         cmd.setErr(new PrintWriter(sw));
103         int exitCode = cmd.execute(getArgs("download initial", "--geo-filter=POLYGON((123 456))", "--download-service=" + url));
104         assertEquals(1, exitCode);
105         assertThat(sw.toString(), containsString("deltaCustomDownload call failed with: 400 - invalid wktString"));
106     }
