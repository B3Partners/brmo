66      /**
67       * Test of een combi download van twee kaartbladen dezlfde aantallen
68       * features oplevert als twee losse zipfiles.
69       *
70       * @throws Exception if any
71       */
72      @Test
73      public void testCombiKaartbladTegenLosseKaartbladenInZipFiles() throws Exception {
74          ldr.setScanDirectory(BGTGMLLightLoaderIntegrationTest.class.getResource("/gmllight/schiermonnikoog/").getFile());
75          List<File> zips = ldr.scanDirectory();
76          // Verwacht aantal zipfiles
77          assumingThat(zips.size() == 3, () -> {
78  
79              final int expected = 399;
80              int actualCombi = 0;
81              int actualLos = 0;
82  
83              for (File zip : zips) {
84                  // kaaartbladen combi-zip laden
85                  if (zip.getName().contains("57830-57831.zip")) {
86                      LOG.info("Laden van 'combi' zipfile/kaartbladen: " + zip.getName());
87                      actualCombi = ldr.processZipFile(zip);
88                  }
89              }
90              final Map<String, Long> countsCombi = countRows();
91  
92              // maak een nieuwe loader omdat de interne cache van de loader anders niet leeg is
93              // en leeg de tabellen
94              ldr = new BGTGMLLightLoader();
95              ldr.setDbConnProps(params);
96              ldr.truncateTables();
97  
98              for (File zip : zips) {
99                  // losse kaart bladen laden uit de anadres 2 zips
100                 if (!zip.getName().contains("57830-57831.zip")) {
101                     LOG.info("Laden van 'losse' zipfile/kaartblad: " + zip.getName());
102                     // los is 345 + 54 = 399
103                     actualLos += ldr.processZipFile(zip);
104                 }
105             }
106             final Map<String, Long> countsLos = countRows();
107 
108             assertEquals(expected, actualCombi, "Verwacht aantal geschreven features klopt niet");
109             assertEquals(expected, actualLos, "Verwacht aantal geschreven features klopt niet");
110             assertEquals(countsCombi, countsLos, "Gecombineerde kaartblad set komt niet overeen met losse geladen kaartbladen");
111         });
112     }
