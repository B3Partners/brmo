30  public class BGTGMLLightValidatieTest {
31  
32      private static final Log LOG = LogFactory.getLog(BGTGMLLightValidatieTest.class);
33  
34      /**
35       * probeer gml te valideren met ons schema.
36       *
37       * @throws Exception if any
38       */
39      @Test
40      @Disabled("Als er kruinlijnen elementen zijn die als LineString in de GML zitten mislukt deze test")
41      public void validateFile() throws Exception {
42          DocumentBuilder parser = DocumentBuilderFactory.newInstance().newDocumentBuilder();
43          String gmlFileName = "/gmllight/kruinlijntest/bgt_ondersteunendwegdeel.gml";
44          Document document = parser.parse(new File(BGTLightKruinlijnIntegrationTest.class.getResource(gmlFileName).toURI()));
45  
46          SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
47          Source schemaFile = new StreamSource(
48                  new File(BGTLightKruinlijnIntegrationTest.class.getResource("/imgeo-simple_resolved.xsd").toURI())
49          );
50          Schema schema = factory.newSchema(schemaFile);
51          Validator validator = schema.newValidator();
52  
53          // validate DOM tree
54          try {
55              validator.validate(new DOMSource(document));
56          } catch (SAXException e) {
57              // instance document is invalid!
58              LOG.error("Valitatiefout: ", e);
59              fail("Validation failed: " + e.getMessage());
60          }
61      }
62  
63  }
