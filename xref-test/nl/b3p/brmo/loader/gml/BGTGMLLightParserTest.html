36  /**
37   * Geparametriseerde test om features uit BGT/PDOK gmllight formaat te parsen.
38   *
39   * @author mprins
40   */
41  @RunWith(Parameterized.class)
42  public class BGTGMLLightParserTest {
43  
44      private static final Log LOG = LogFactory.getLog(BGTGMLLightParserTest.class);
45      private Parser parser;
46  
47      private final String gmlFileName;
48      private final String typeNamespace;
49      private final String elementName;
50      private final int expectedNumOfElements;
51      private final int expectedNumOfAtrr;
52  
53      /**
54       *
55       * @param gmlFileName filename of the test resource (GML)
56       * @param typeNamespace element type namespace
57       * @param elementName element
58       * @param expectedNumOfElements number of elements in resources
59       *
60       * @see #params()
61       */
62      public BGTGMLLightParserTest(String gmlFileName, String typeNamespace, String elementName, int expectedNumOfElements, int expectedNumOfAtrr) {
63          this.gmlFileName = gmlFileName;
64          this.typeNamespace = typeNamespace;
65          this.elementName = elementName;
66          this.expectedNumOfElements = expectedNumOfElements;
67          this.expectedNumOfAtrr = expectedNumOfAtrr;
68      }
69  
70      @Parameters
71      public static Collection<Object[]> params() {
72          return Arrays.asList(new Object[][]{
73              //arrays of: {"gmlFileName", "typeNamespace", "elementName", expectedNumOfElements, expectedNumOfAtrr},
74              {"/gmllight/bgt_ondersteunendwaterdeel.gml", "http://www.geostandaarden.nl/imgeo/2.1/simple/gml31", "OndersteunendWaterdeel", 47, 20},
75              {"/gmllight/bgt_ongeclassificeerdobject.gml", "http://www.geostandaarden.nl/imgeo/2.1/simple/gml31", "OngeclassificeerdObject", 0, 18}
76          });
77      }
78  
79      @Before
80      public void setUpParser() {
81          final String schemaLocation = BGTGMLLightParserTest.class.getResource("/imgeo-simple_resolved.xsd").getFile();
82  
83          Configuration configuration = new ApplicationSchemaConfiguration("http://www.geostandaarden.nl/imgeo/2.1/simple/gml31", schemaLocation);
84          configuration.getContext().registerComponentInstance(new GeometryFactory(new PrecisionModel(), 28992));
85  
86          parser = new Parser(configuration);
87          parser.setValidating(true);
88          parser.setStrict(true);
89          parser.setFailOnValidationError(false);
90      }
91  
92      @Test
93      public void testGMLFile() throws IOException, SAXException, ParserConfigurationException {
94          LOG.debug("Start testing for: " + typeNamespace + ":" + elementName);
95  
96          InputStream is = BGTGMLLightParserTest.class.getResourceAsStream(gmlFileName);
97  
98          FeatureCollection<?, ?> featureCollection = (FeatureCollection<?, ?>) parser.parse(is);
99          assertEquals("Collection size should be as expected.", expectedNumOfElements, featureCollection.size());
100 
101         @SuppressWarnings("unchecked")
102         List<Exception> errors = parser.getValidationErrors();
103         assertTrue("There should be no parsing errors", errors.isEmpty());
104 
105         final Name elemTypeName = new NameImpl(typeNamespace, elementName);
106         featureCollection.accepts(new AbstractFeatureVisitor() {
107             @Override
108             public void visit(Feature feature) {
109                 assertEquals("TypeName should match", elemTypeName, feature.getType().getName());
110 
111                 Collection<Property> props = feature.getProperties();
112                 assertEquals("There should be a number of properties", expectedNumOfAtrr, props.size());
113 //                String s = "Feature " + feature.getIdentifier() + "\n";
114 //                for (Property p : props) {
115 //                    s = s.concat(" - " + p.getName() + " (" + p.getType() + "):\t " + p.getValue() + "\n");
116 //                }
117 //                LOG.debug(s);
118             }
119 
120         }, new NullProgressListener());
121     }
