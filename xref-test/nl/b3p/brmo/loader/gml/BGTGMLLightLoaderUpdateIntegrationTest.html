55      /**
56       * test scannen en laden van een directory zipfiles in bestaande tabellen,
57       * de data wordt twee keer geladen zodat alle objecten de nieuwe datum
58       * hebben.
59       *
60       * @throws Exception if any
61       */
62      @Test
63      public void testUpdateFromDirectory() throws Exception {
64          int load_one = 0;
65          int load_two = 0;
66  
67          ldr.setScanDirectory(BGTGMLLightLoaderUpdateIntegrationTest.class.getResource("/gmllight/zips/").getFile());
68          List<File> zips = ldr.scanDirectory();
69          assertEquals(1, zips.size(), "Verwacht aantal zipfiles");
70  
71          // eerste set laden met datum 21 dagen voor vandaag
72          Calendar cal = Calendar.getInstance();
73          cal.add(Calendar.DATE, -21);
74  
75          for (File zip : zips) {
76              load_one = ldr.processZipFile(zip);
77              LOG.info("Totaal aantal ingevoegde features voor: " + zip.getName() + " is: " + load_one);
78              assertTrue((load_one > 1), "Verwacht meer dan 1 geschreven feature");
79              if (zip.getName().equalsIgnoreCase("extract-gmllight.zip")) {
80                  assertEquals(114, load_one, "Er zitten 114 objecten in de gml bestanden");
81              }
82          }
83      }
84

88       *

91      @Test
92      @Disabled("De update zipfile zit nog te dicht op de bron, dus geen verschil.")

99          assumeFalse(null == updateUrl, "Verwacht de zipfile met data te bestaan.");

109     }
