17  package nl.b3p.brmo.loader.xml;
18  
19  import java.util.ArrayList;
20  import java.util.Date;
21  import java.util.List;
22  import nl.b3p.brmo.loader.BrmoFramework;
23  import nl.b3p.brmo.loader.entity.GbavBericht;
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  import static org.junit.Assert.assertArrayEquals;
27  import static org.junit.Assert.assertEquals;
28  import static org.junit.Assert.assertTrue;
29  import org.junit.Test;
30  
31  /**
32   * Testcase voor GbavXMLReader.
33   *
34   * @author Mark Prins
35   */
36  public class GbavXMLReaderTest {
37  
38      private static final Log LOG = LogFactory.getLog(GbavXMLReaderTest.class);
39  
40      @Test
41      public void berichtTest() throws Exception {
42          GbavXMLReader r;
43          GbavBericht b = null;
44  
45          r = new GbavXMLReader(GbavXMLReader.class.getResourceAsStream("gbav-voorbeeld.xml"));
46          assertTrue(r.hasNext());
47          int total = 0;
48          List<String> objectRefs = new ArrayList();
49          while (r.hasNext()) {
50              b = r.next();
51              objectRefs.add(b.getObjectRef());
52              LOG.debug(String.format("Bericht #%d van %tF %2$tT, object ref %s",
53                      b.getVolgordeNummer(),
54                      b.getDatum(),
55                      b.getObjectRef()
56              ));
57              total++;
58              assertEquals("Soort komt niet overeen", BrmoFramework.BR_GBAV, b.getSoort());
59              assertEquals("Volgordenummer komt niet overeen", 0, b.getVolgordeNummer().intValue());
60              assertEquals("Datum komt niet overeen", new Date(2018 - 1900, 1, 1), b.getDatum());
61          }
62  
63          assertEquals(1, total);
64          assertArrayEquals(new String[]{GbavXMLReader.PREFIX + "b3e513383d0454de798b29c6900643130e3b2d6b",}, objectRefs.toArray(new String[]{}));
65      }
66  }
