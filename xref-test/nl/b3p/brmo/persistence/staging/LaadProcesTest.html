28          p.setStatus(LaadProces.STATUS.STAGING_OK);
29          entityManager.persist(p);
30          Bericht b = new Bericht();
31          b.setLaadprocesid(p);
32          b.setStatus(Bericht.STATUS.STAGING_OK);
33          entityManager.persist(b);
34          entityManager.getTransaction().commit();
35      }

37      @Test
38      public void testDuplicateQuery() {
39          // zit in nl.b3p.brmo.service.scanner.AbstractExecutableProces#isDuplicaatLaadProces(...)
40          CriteriaBuilder criteriaBuilder = entityManager.getCriteriaBuilder();
41          CriteriaQuery<LaadProces> criteriaQuery = criteriaBuilder.createQuery(LaadProces.class);
42          Root<LaadProces> from = criteriaQuery.from(LaadProces.class);
43          CriteriaQuery<LaadProces> select = criteriaQuery.select(from);
44          Predicate _bestand_naam = criteriaBuilder.equal(from.get("bestand_naam"), "naam");
45          Predicate _soort = criteriaBuilder.equal(from.get("soort"), "soort");
46          criteriaQuery.where(criteriaBuilder.and(_bestand_naam, _soort));
47          TypedQuery<LaadProces> typedQuery = entityManager.createQuery(select);
48          // verwacht dat er geen resultaten zijn / lege lijst
49          assertTrue(typedQuery.getResultList().isEmpty());
50      }

57      }
