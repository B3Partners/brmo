18      /**
19       * round trip test van maken, opslaan en uitlezen van een BRKScannerProces.
20       */
21      @Test
22      public void roundtrip() {
23          BRKScannerProces p = new BRKScannerProces();
24          p.setScanDirectory(DIR);
25          p.getConfig().put("isActive", new ClobElement("true"));
26          entityManager.persist(p);
27  
28          final long id = p.getId();
29  
30          BRKScannerProces c = entityManager.find(BRKScannerProces.class, id);
31          assertEquals("De directory is zoals geconfigureerd.", DIR, c.getScanDirectory());
32          assertEquals("Verwacht dat de parameter is zoals geconfigureerd.", "true", c.getConfig().get("isActive").getValue());
33  
34          entityManager.remove(c);
35          entityManager.getTransaction().commit();
36      }
37  
38      /**
39       * testcase voor {@link nl.b3p.brmo.persistence.staging.AutomatischProces#updateSamenvattingEnLogfile(java.lang.String)}
40       */
41      @Test
42      public void testUpdateSamenvattingEnLogfile() {
43          BRKScannerProces p = new BRKScannerProces();
44          p.updateSamenvattingEnLogfile(NAAM_BESCHIJVING);
45          entityManager.persist(p);
46          final long id = p.getId();
47  
48          BRKScannerProces c = entityManager.find(BRKScannerProces.class, id);
49          assertEquals("Verwacht dat de logfile en de samenvatting hetzelfde zijn.", c.getLogfile(), c.getSamenvatting());
50  
51          c.updateSamenvattingEnLogfile(NAAM);
52  
53          final String TWEEDE = "2e entry-";
54  
55          c.updateSamenvattingEnLogfile(TWEEDE + NAAM_BESCHIJVING);
56  
57          assertEquals(NAAM_BESCHIJVING + LOG_NEWLINE + NAAM + LOG_NEWLINE + TWEEDE + NAAM_BESCHIJVING, c.getLogfile());
58          entityManager.merge(c);
59  
60          String[] s = p.getLogfile().split(NAAM);
61          String expected = s[0].replace(LOG_NEWLINE, "");
62          String actual = s[1].substring(TWEEDE.length() + 1).replace(LOG_NEWLINE, "");
63          assertEquals("Verwacht dat de logfile twee dezelfde delen bevat.", expected, actual);
64  
65          entityManager.remove(c);
66          entityManager.getTransaction().commit();
67      }
