16  public class BAGScannerProcesTest extends TestUtil {
17  
18      /**
19       * round trip test van maken en uitlezen van een BAGScannerProces.
20       */
21      @Test
22      public void roundtrip() {
23          BAGScannerProces p = new BAGScannerProces();
24          p.setScanDirectory(DIR);
25          p.setArchiefDirectory(DIR);
26          p.getConfig().put("isActive", new ClobElement("true"));
27          entityManager.persist(p);
28  
29          final long id = p.getId();
30          BAGScannerProces c = entityManager.find(BAGScannerProces.class, id);
31  
32          assertEquals(DIR, c.getScanDirectory(), "Verwacht dat de scan directory is zoals geconfigureerd.");
33          assertEquals(DIR, c.getArchiefDirectory(), "Verwacht dat de archief directory is zoals geconfigureerd.");
34          assertEquals("true", c.getConfig().get("isActive").getValue(), "Verwacht dat de parameter is zoals geconfigureerd.");
35  
36          entityManager.remove(p);
37          entityManager.getTransaction().commit();
38      }
39  }
