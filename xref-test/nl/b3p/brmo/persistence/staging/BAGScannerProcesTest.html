6   import static org.junit.jupiter.api.Assertions.assertEquals;

8   import nl.b3p.brmo.persistence.TestUtil;

11  /**
12   * Testcase voor {@link nl.b3p.brmo.persistence.staging.BAGScannerProces}.
13   *
14   * @author mprins
15   */
16  public class BAGScannerProcesTest extends TestUtil {
17  
18    /** round trip test van maken en uitlezen van een BAGScannerProces. */
19    @Test
20    public void roundtrip() {
21      BAGScannerProces p = new BAGScannerProces();
22      p.setScanDirectory(DIR);
23      p.setArchiefDirectory(DIR);
24      p.getConfig().put("isActive", new ClobElement("true"));
25      entityManager.persist(p);
26  
27      final long id = p.getId();
28      BAGScannerProces c = entityManager.find(BAGScannerProces.class, id);
29  
30      assertEquals(
31          DIR, c.getScanDirectory(), "Verwacht dat de scan directory is zoals geconfigureerd.");
32      assertEquals(
33          DIR, c.getArchiefDirectory(), "Verwacht dat de archief directory is zoals geconfigureerd.");
34      assertEquals(
35          "true",
36          c.getConfig().get("isActive").getValue(),
37          "Verwacht dat de parameter is zoals geconfigureerd.");
38  
39      entityManager.remove(p);
40      entityManager.getTransaction().commit();
41    }
42  }
