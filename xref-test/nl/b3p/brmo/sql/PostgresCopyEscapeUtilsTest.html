7   package nl.b3p.brmo.sql;

9   import org.apache.commons.text.StringEscapeUtils;
10  import org.junit.jupiter.params.ParameterizedTest;
11  import org.junit.jupiter.params.provider.Arguments;
12  import org.junit.jupiter.params.provider.MethodSource;
13  
14  import java.util.stream.Stream;
15  
16  import static org.junit.jupiter.api.Assertions.assertEquals;
17  
18  public class PostgresCopyEscapeUtilsTest {
19      /**
20       * This class is used as a parametrized test parameter and displays the value escaped for Java so non-whitespace
21       * characters are legible.
22       */
23      public static class EscapedForDisplayParam {
24          String param;
25  
26          EscapedForDisplayParam(String param) {
27              this.param = param;
28          }
29  
30          public String getParam() {
31              return param;
32          }
33  
34          public String toString() {
35              return StringEscapeUtils.escapeJava(param);
36          }
37      }
38  
39      @ParameterizedTest(name="[{index}] {0} == {1}")
40      @MethodSource
41      public void escape(EscapedForDisplayParam input, String expected) {
42          String output = PostgresCopyEscapeUtils.builder().escape(input.getParam()).toString();
43          assertEquals(expected, output);
44      }
45  
46      private static Stream<Arguments> escape() {
47          return Stream.of(new String[][] {
48                  {"test", "test"},
49                  {"te\\st","te\\\\st",},
50                  {"t\nst", "t\\nst",},
51                  {"\\t\nst", "\\\\t\\nst"},
52                  {"t\nst\n", "t\\nst\\n"},
53                  {"\ttest\\\n", "\\ttest\\\\\\n"},
54          }).map(arguments -> Arguments.of(new EscapedForDisplayParam(arguments[0]), arguments[1]));
55      }
56  }
