27  public class NestedZipIntegrationTest extends TestUtil {
28  
29      private static final Log LOG = LogFactory.getLog(NestedZipIntegrationTest.class);
30      private BrmoFramework brmo;
31  
32      @BeforeEach
33      public void setUp() throws Exception {
34          brmo = new BrmoFramework(dsStaging, null);
35      }
36  
37      @AfterEach
38      public void cleanup() throws Exception {
39          brmo.emptyStagingDb();
40          brmo.closeBrmoFramework();
41          brmo = null;
42      }
43  
44      @Test
45      public void testLadenNestedZipFiles() throws Exception {
46          InputStream input = NestedZipIntegrationTest.class.getResourceAsStream("/GH-317-BAG-van-GDS2/DNLDLXAM02-15967-5014117-31032017-01042017.zip");
47          ZipInputStream zip = new ZipInputStream(input);
48          ZipEntry entry = zip.getNextEntry();
49          while (entry != null) {
50              LOG.debug("gevonden " + entry.getName());
51              if (!entry.getName().toLowerCase().startsWith("gem-wpl-relatie")
52                      && !entry.getName().equalsIgnoreCase("Leveringsdocument-BAG-Mutaties.xml")) {
53                  if (entry.getName().toLowerCase().endsWith("zip")) {
54                      LOG.debug("start verwerken " + entry.getName());
55                      // alleen mutaties oppakken
56                      ZipInputStream innerzip = new ZipInputStream(zip);
57                      ZipEntry innerentry = innerzip.getNextEntry();
58                      while (innerentry != null && innerentry.getName().toLowerCase().endsWith(".xml")) {
59                          LOG.info("verwerken " + innerentry.getName());
60                          brmo.loadFromStream("bag", CloseShieldInputStream.wrap(innerzip), "DNLDLXAM02-15967-5014117-31032017-01042017.zip/" + innerentry.getName(), (Long)null);
61                          innerentry = innerzip.getNextEntry();
62                      }
63                      LOG.debug("einde verwerken " + entry.getName());
64                  } else {
65                      LOG.warn("Overslaan van onbekend bestand in bag mutaties: " + entry.getName());
66                  }
67              }
68              entry = zip.getNextEntry();
69          }
70          zip.close();
71  
72          Assertions.assertEquals(
73                  2l,
74                  brmo.getCountLaadProcessen(null, null, "bag", null),
75                  "het aantal LP mag niet afwijken"
76          );
77          Assertions.assertEquals(
78                  /*<product_LVC:Nieuw> - 5723l + 3210l= 8933*/ 8921l /*<product_LVC:Mutatie-product>  10000l + 6301l*/,
79                  brmo.getCountBerichten(null, null, "bag", null),
80                  "het aantal Bericht mag niet afwijken"
81          );
82      }
