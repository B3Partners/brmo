310   public void setUp() throws Exception {
311     dsStaging = new BasicDataSource();
312     dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
313     dsStaging.setUsername(params.getProperty("staging.user"));
314     dsStaging.setPassword(params.getProperty("staging.passwd"));

316     dsStaging.setConnectionProperties(params.getProperty("staging.options", ""));
317 
318     dsRsgb = new BasicDataSource();
319     dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
320     dsRsgb.setUsername(params.getProperty("rsgb.user"));
321     dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
322     dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
323     dsRsgb.setConnectionProperties(params.getProperty("rsgb.options", ""));
324 
325     staging = new DatabaseDataSourceConnection(dsStaging);
326     rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
327 
328     if (this.isOracle) {
329       staging =
330           new DatabaseConnection(
331               OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
332               params.getProperty("staging.user").toUpperCase());
333       staging
334           .getConfig()
335           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
336       staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
337 
338       rsgb =
339           new DatabaseConnection(
340               OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
341               params.getProperty("rsgb.user").toUpperCase());
342       rsgb.getConfig()
343           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
344       rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
345     } else if (this.isPostgis) {
346       // we hebben alleen nog postgres over
347       staging
348           .getConfig()
349           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
350       rsgb.getConfig()
351           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
352     }
353 
354     brmo = new BrmoFramework(dsStaging, dsRsgb, null);
355     brmo.setOrderBerichten(true);
356 
357     FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
358     fxdb.setCaseSensitiveTableNames(false);
359     IDataSet stagingDataSet =
360         fxdb.build(
361             new FileInputStream(
362                 new File(
363                     NhrToStagingToRsgbIntegrationTest.class
364                         .getResource("/staging-empty-flat.xml")
365                         .toURI())));
366 
367     sequential.lock();
368 
369     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
370 
371     assumeTrue(
372         0L == brmo.getCountBerichten("nhr", "STAGING_OK"), "Er zijn geen STAGING_OK berichten");
373     assumeTrue(
374         0L == brmo.getCountLaadProcessen("nhr", "STAGING_OK"),
375         "Er zijn geen STAGING_OK laadprocessen");
376   }

390   @DisplayName("NHR to STAGING to RSGB")
391   @ParameterizedTest(name = "case #{index}: bestand: {0}")
392   @MethodSource("argumentsProvider")
393   public void testNhrXMLToStagingToRsgb(String bestandNaam, Map<String, Integer> rowCounts)
394       throws Exception {
395 
396     assumeFalse(
397         null
398             == BedrijfAlsFuntionarisNhrToStagingToRsgbIntegrationTest.class.getResource(
399                 bestandNaam),
400         "Het test bestand moet er zijn.");
401 
402     brmo.loadFromFile(
403         "nhr",
404         BedrijfAlsFuntionarisNhrToStagingToRsgbIntegrationTest.class
405             .getResource(bestandNaam)
406             .getFile(),
407         null);
408     LOG.info("klaar met laden van berichten in staging DB.");
409 
410     // alleen het eerste bericht heeft br_orgineel_xml, de rest niet
411     ITable bericht =
412         staging.createQueryTable("bericht", "select * from bericht where volgordenummer=0");
413     assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
414     LOG.debug("\n\n" + bericht.getValue(0, "br_orgineel_xml") + "\n\n");
415     assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null");
416     Object berichtId = bericht.getValue(0, "id");
417 
418     bericht =
419         staging.createQueryTable(
420             "bericht", "select * from bericht where object_ref like 'nhr.%Vestg%'");
421     assertEquals(
422         rowCounts.get("vestg"),
423         Integer.valueOf(bericht.getRowCount()),
424         "aantal (niet)commerciele vestiging berichten onjuist");
425 
426     LOG.info("Transformeren berichten naar rsgb DB.");
427     Thread t = brmo.toRsgb();
428     t.join();
429 
430     // na de verwerking moet soap payload er ook nog zijn
431     bericht =
432         staging.createQueryTable(
433             "bericht", "select * from bericht where br_orgineel_xml is not null");
434     assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
435     assertNotNull(
436         bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null na transformatie");
437     assertEquals(
438         berichtId,
439         bericht.getValue(0, "id"),
440         "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
441 
442     // check RSGB tabellen voor aantallen
443     for (String table : rowCounts.keySet()) {
444       checkRowCount(table, rowCounts.get(table));
445     }
446   }
447 
448   private void checkRowCount(String table, int aantal) throws SQLException, DataSetException {
449     ITable _tbl;
450     if (table.startsWith("select ")) {
451       List<String> words = Arrays.asList(table.split(" "));
452       _tbl = rsgb.createQueryTable(words.get(words.indexOf(table) + 1), table);
453     } else {
454       _tbl = rsgb.createTable(table);
455     }
456 
457     assertEquals(aantal, _tbl.getRowCount(), "aantal '" + table + "' records klopt niet");
458   }
