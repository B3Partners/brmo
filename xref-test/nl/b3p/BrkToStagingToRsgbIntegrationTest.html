103     @Before
104     @Override
105     public void setUp() throws Exception {
106         BasicDataSource dsStaging = new BasicDataSource();
107         dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
108         dsStaging.setUsername(params.getProperty("staging.user"));
109         dsStaging.setPassword(params.getProperty("staging.passwd"));
110         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
111 
112         BasicDataSource dsRsgb = new BasicDataSource();
113         dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
114         dsRsgb.setUsername(params.getProperty("rsgb.user"));
115         dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
116         dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
117 
118         staging = new DatabaseDataSourceConnection(dsStaging);
119         rsgb = new DatabaseDataSourceConnection(dsRsgb);
120 
121         if (this.isMsSQL) {
122             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
123             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
124         } else if (this.isOracle) {
125             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
126             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
127             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
128 
129             rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
130             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
131             rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
132         } else if (this.isPostgis) {
133             // we hebben alleen nog postgres over
134             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
135             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
136         }
137 
138         brmo = new BrmoFramework(dsStaging, dsRsgb);
139 
140         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
141         fxdb.setCaseSensitiveTableNames(false);
142         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(BrkToStagingToRsgbIntegrationTest.class.getResource("/staging-empty-flat.xml").toURI())));
143 
144         sequential.lock();
145 
146         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
147 
148         assumeTrue("Er zijn geen STAGING_OK berichten", 0l == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
149         assumeTrue("Er zijn geen STAGING_OK laadprocessen", 0l == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"));
150     }

165     @Test
166     public void testBrkXMLToStaging() throws Exception {
167         assumeNotNull("Het test bestand moet er zijn.", BrkToStagingToRsgbIntegrationTest.class.getResource(bestandNaam));
168 
169         brmo.loadFromFile(bestandType, BrkToStagingToRsgbIntegrationTest.class.getResource(bestandNaam).getFile());
170         LOG.debug("klaar met laden van berichten in staging DB.");
171 
172         List<Bericht> berichten = brmo.listBerichten();
173         List<LaadProces> processen = brmo.listLaadProcessen();
174         assertNotNull("De verzameling berichten bestaat niet.", berichten);
175         assertEquals("Het aantal berichten is niet als verwacht.", aantalBerichten, berichten.size());
176         assertNotNull("De verzameling processen bestaat niet.", processen);
177         assertEquals("Het aantal processen is niet als verwacht.", aantalProcessen, processen.size());
178 
179         LOG.debug("Transformeren berichten naar rsgb DB.");
180         Thread t = brmo.toRsgb();
181         t.join();
182 
183         assertEquals("Niet alle berichten zijn OK getransformeerd", aantalBerichten, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"));
184         berichten = brmo.listBerichten();
185         for (Bericht b : berichten) {
186             assertNotNull("Bericht is 'null'", b);
187             assertNotNull("'db-xml' van bericht is 'null'", b.getDbXml());
188         }
189 
190         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
191         assertEquals("Het aantal on", aantalBerichten, kad_onrrnd_zk.getRowCount());
192         assertEquals("Datum eerste record komt niet overeen", datumEersteMutatie, kad_onrrnd_zk.getValue(0, "dat_beg_geldh"));
193     }
