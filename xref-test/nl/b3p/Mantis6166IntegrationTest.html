44  /**
45   * testcases voor mantis 6166; incorrecte verwijdering van percelen. Draaien
46   * met:
47   * {@code mvn -Dit.test=Mantis6166IntegrationTest -Dtest.onlyITs=true verify -Poracle > target/oracle.log}
48   * voor bijvoorbeeld Oracle.
49   *
50   * <strong>NB. werkt niet op mssql, althans niet met de jTDS driver omdat die
51   * geen JtdsPreparedStatement#setNull() methode heeft.</strong>
52   *
53   * @author mprins
54   */
55  @RunWith(Parameterized.class)
56  @Category(JTDSDriverBasedFailures.class)
57  public class Mantis6166IntegrationTest extends AbstractDatabaseIntegrationTest {

59      private static final Log LOG = LogFactory.getLog(Mantis6166IntegrationTest.class);

62      public static Collection params() {
63          return Arrays.asList(new Object[][]{
64              // {"bestandsNaam", aantalProcessen, aantalBerichten, standId, verwijderId, mutatieIds },
65              {"/mantis6166/staging-69660669770000.xml", 4, 4, 722959, 946717, new long[]{47134330, 47134331}},
66              // dit bestand zit niet in de repository vanwege de inhoud van de data
67              {"/mantis6166/staging-66860489870000.xml", 4, 4, 488741, 948118, new long[]{47125275, 47125276}}
68          });
69      }

87      public Mantis6166IntegrationTest(final String bestandsNaam, final long aantalProcessen, final long aantalBerichten, final long stand, final long verwijder, final long[] mutaties) {
88          this.bestandsNaam = bestandsNaam;
89          this.aantalProcessen = aantalProcessen;
90          this.aantalBerichten = aantalBerichten;
91          this.stand = stand;

96      @Override
97      @Before
98      public void setUp() throws Exception {
99          BasicDataSource dsStaging = new BasicDataSource();
100         dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
101         dsStaging.setUsername(params.getProperty("staging.user"));
102         dsStaging.setPassword(params.getProperty("staging.passwd"));
103         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
104 
105         BasicDataSource dsRsgb = new BasicDataSource();
106         dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
107         dsRsgb.setUsername(params.getProperty("rsgb.user"));
108         dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
109         dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
110 
111         rsgb = new DatabaseDataSourceConnection(dsRsgb);
112         staging = new DatabaseDataSourceConnection(dsStaging);
113 
114         assumeTrue("Deze test werkt niet met de jTDS driver omdat die geen JtdsPreparedStatement#setNull() methode heeft.", !this.isMsSQL);
115 
116         if (this.isMsSQL) {
117             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
118             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
119         } else if (this.isOracle) {
120             rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
121             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
122             rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
123             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
124             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
125             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
126         } else if (this.isPostgis) {
127             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
128             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
129         } else {
130             fail("Geen onstersteunde database aangegegeven");
131         }
132 
133         assumeTrue("Het bestand met testdata zou moeten bestaan.", Mantis6166IntegrationTest.class.getResource(bestandsNaam) != null);
134 
135         IDataSet stagingDataSet = new XmlDataSet(new FileInputStream(new File(Mantis6166IntegrationTest.class.getResource(bestandsNaam).toURI())));
136 
137         sequential.lock();
138 
139         if (this.isMsSQL) {
140             // SET IDENTITY_INSERT op ON
141             InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
142         } else {
143             DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
144         }
145 
146         brmo = new BrmoFramework(dsStaging, dsRsgb);
147         brmo.setOrderBerichten(true);
148 
149         assumeTrue("Er zijn meer dan 0 berichten in het bestand om te laden", aantalBerichten > 0);
150         assumeTrue("Er zijn meer dan 0 laadprocessen in het bestand om te laden", aantalProcessen > 0);
151         assumeTrue("Er zijn x STAGING_OK berichten om te verwerken", aantalBerichten == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
152         assumeTrue("Er zijn x STAGING_OK laadprocessen in de database", aantalProcessen == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"));
153     }
154 
155     @After
156     public void cleanup() throws Exception {
157         if (brmo != null) {
158             // in geval van niet waar gemaakte assumptions
159             brmo.closeBrmoFramework();
160         }
161         assumeTrue("Deze test werkt niet met de jTDS driver omdat die geen JtdsPreparedStatement#setNull() methode heeft.", !this.isMsSQL);
162 
163         CleanUtil.cleanRSGB(rsgb);
164         rsgb.close();
165 
166         DatabaseOperation.DELETE_ALL.execute(staging, new DefaultDataSet(new DefaultTable[]{
167             new DefaultTable("job")}
168         ));
169         staging.close();
170 
171         try {
172             sequential.unlock();
173         } catch (IllegalMonitorStateException e) {
174             // in geval van niet waar gemaakte assumptions
175             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed");
176         }
177     }

184     @Test
185     public void testStand() throws Exception {
186         Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_IDS, new long[]{stand}, null);
187         t.join();
188 
189         assertEquals("Mutatie berichten zijn niet getransformeerd", aantalBerichten - 1, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
190         assertEquals("Een bericht is OK getransformeerd", 1l, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"));
191 
192         ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
193         assertEquals("Er is 1 perceel geladen", 1, kad_perceel.getRowCount());
194 
195         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_perceel");
196         assertEquals("Er is 1 onroerende zaak geladen", 1, kad_onrrnd_zk.getRowCount());
197     }

199     /**
200      * transformeer stand bericht en mutaties.
201      *
202      * @throws Exception if any
203      */
204     @Test
205     public void testStandMutatie() throws Exception {
206         long[] transformIds = new long[mutaties.length + 1];
207         transformIds[0] = stand;
208         for (int i = 1; i < transformIds.length; i++) {
209             transformIds[i] = mutaties[i - 1];
210         }
211         Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_IDS, transformIds, null);
212         t.join();
213 
214         assertEquals("Alle berichten behalve verwijderen zijn OK getransformeerd", aantalBerichten - 1, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"));
215         assertEquals("Een (verwijder) bericht is niet getransformeerd en heeft STAGING_OK status", 1l, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"));
216 
217         ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
218         assertEquals("Er is 1 perceel geladen", 1, kad_perceel.getRowCount());
219 
220         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
221         assertEquals("Er is 1 onroerende zaak geladen", 1, kad_onrrnd_zk.getRowCount());
222 
223         ITable kad_perceel_archief = rsgb.createDataSet().getTable("kad_perceel_archief");
224         assertEquals("Er zijn (verwerkt aantal -1) percelen gearchiveerd", transformIds.length - 1, kad_perceel_archief.getRowCount());
225 
226         ITable kad_onrrnd_zk_archief = rsgb.createDataSet().getTable("kad_onrrnd_zk_archief");
227         assertEquals("Er zijn (verwerkt aantal -1) onroerende zaken gearchiveerd", transformIds.length - 1, kad_onrrnd_zk_archief.getRowCount());
228     }
