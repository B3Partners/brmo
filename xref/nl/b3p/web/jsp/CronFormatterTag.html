17  package nl.b3p.web.jsp;
18  
19  import com.cronutils.descriptor.CronDescriptor;
20  import com.cronutils.model.definition.CronDefinitionBuilder;
21  import com.cronutils.parser.CronParser;
22  
23  import javax.servlet.jsp.JspException;
24  import javax.servlet.jsp.tagext.SimpleTagSupport;
25  import java.io.IOException;
26  import java.util.Locale;
27  
28  import static com.cronutils.model.CronType.QUARTZ;
29  
30  /**
31   * Formatter voor cron expressies. The default locale for this tag is
32   * {@code nl}.
33   *
34   * @author mprins
35   */
36  public class CronFormatterTag extends SimpleTagSupport {
37  
38      private String cronExpression;
39      private Locale locale;
40  
41      public CronFormatterTag() {
42          setLocale("nl");
43      }
44  
45      public void setCronExpression(String cronExpression) {
46          this.cronExpression = cronExpression;
47      }
48  
49      public void setLocale(String locale) {
50          try {
51              this.locale = new Locale(locale);
52          } catch (NullPointerException npe) {
53              this.locale = null;
54          }
55      }
56  
57      @Override
58      public void doTag() throws JspException, IOException {
59          getJspContext().getOut().write(buildOutput());
60      }
61  
62      public String buildOutput() {
63          String formatted = "";
64          if (cronExpression != null && !cronExpression.isEmpty()) {
65              try {
66                  CronParser cronParser = new CronParser(CronDefinitionBuilder.instanceDefinitionFor(QUARTZ));
67                  if (locale != null) {
68                      formatted = CronDescriptor.instance(locale).describe(cronParser.parse(cronExpression));
69                  } else {
70                      formatted = CronDescriptor.instance(Locale.getDefault()).describe(cronParser.parse(cronExpression));
71                  }
72              } catch (Exception e) {
73                  formatted = e.getLocalizedMessage();
74              }
75          }
76          return formatted;
77      }
78  
79  }
