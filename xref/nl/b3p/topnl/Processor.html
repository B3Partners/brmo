63      public Processor(DataSource ds) throws JAXBException, SQLException{
64          database = new Database(ds);

68      public void importIntoDb(URL in, TopNLType type) throws JDOMException {
69          XMLInputFactory xif = XMLInputFactory.newFactory();
70          this.resetStatus();
71          try {
72              log.info("Importing file " + in.toExternalForm() + ", type: " + type.getType());
73              Unmarshaller jaxbUnmarshaller = converterFactory.getContext(type).createUnmarshaller();
74  
75              XMLStreamReader xsr = xif.createXMLStreamReader(in.openStream());
76              LocationListener ll = new LocationListener(xsr);
77              jaxbUnmarshaller.setListener(ll);
78  
79              while (xsr.hasNext()) {
80                  int eventType = xsr.next();
81  
82                  if (eventType == XMLStreamReader.START_ELEMENT) {
83                      String localname = xsr.getLocalName();
84                      if (xsr.getLocalName().equals("FeatureMember")) {
85                          JAXBElement jb = null;
86                          try {
87                              jb = (JAXBElement) jaxbUnmarshaller.unmarshal(xsr);
88                              Object obj = jb.getValue();
89                              ArrayList list = new ArrayList();
90                              list.add(obj);
91                              List<TopNLEntity> entities = convert(list, type);
92                              save(entities, type);
93                          } catch (SQLException ex) {
94                              log.error("Error inserting", ex);
95                              this.status = STATUS.NOK;
96                          } catch (JAXBException | IOException | SAXException | ParserConfigurationException | TransformerException | ParseException | IllegalArgumentException ex) {
97                              log.error("Error parsing", ex);
98                              this.status = STATUS.NOK;
99                          } catch (ClassCastException cce) {
100                             log.error(String.format(Locale.ROOT, "Verwerkingsfout van element %s locatie %s", localname, (jb == null ? "onbekend" : ll.getLocation(jb))), cce);
101                         }
102                     }
103                 }
104             }
105 
106             xsr.close();
107         } catch (XMLStreamException | IOException | JAXBException ex) {
108             this.status = STATUS.NOK;
109             log.error("cannot correctly stream xml file:", ex);
110         } 
111     }
112 
113     public List parse (URL in) throws  JAXBException,  IOException{
114         List list = new ArrayList();
115         try {
116             TopNLType type = TopNLTypeFactory.getTopNLType(in);
117             Unmarshaller jaxbUnmarshaller = converterFactory.getContext(type).createUnmarshaller();
118             
119             XMLInputFactory xif = XMLInputFactory.newFactory();
120             
121             XMLStreamReader xsr = xif.createXMLStreamReader(in.openStream());
122             
123             while (xsr.hasNext()) {
124                 int eventType = xsr.next();
125                 
126                 if (eventType == XMLStreamReader.START_ELEMENT) {
127                     String localname = xsr.getLocalName();
128                     if (xsr.getLocalName().equals("FeatureMember")) {
129                         JAXBElement jb = (JAXBElement)jaxbUnmarshaller.unmarshal(xsr);
130                         list.add(jb.getValue());
131                     }
132                 }
133             }
134             
135             xsr.close();
136         } catch (XMLStreamException ex) {
137             log.error("cannot correctly stream xml file:", ex);
138         } catch (JDOMException ex) {
139             log.error("Cannot retrieve topnltype: ",ex);
140         }
141         return list;
142     }

144     /**
145      *
146      * @throws ClassCastException Als er een onverwacht type geometrie in de
147      * data zit, bijvoorbeeld een punt ipv een lijn
148      */
149     public List<TopNLEntity> convert(List listOfJaxbObjects, TopNLType type) throws IOException, SAXException,
150             ParserConfigurationException, TransformerException, ClassCastException {
151 
152         Converter converter = converterFactory.getConverter(type);
153         List<TopNLEntity> entity = converter.convert(listOfJaxbObjects);
154         return entity;
155     }
156     
157     public void save(TopNLEntity entity, TopNLType type) throws ParseException, SQLException {
158         database.save(entity);
159     }
160     
161     public void save(List<TopNLEntity> entities, TopNLType type) throws ParseException, SQLException {
162         for (TopNLEntity entity : entities) {
163             save(entity, type);
164         }
165     }

189     private class LocationListener extends Listener {
190         private XMLStreamReader xsr;
191         private Map<Object, Location> locations;
192 
193         public LocationListener(XMLStreamReader xsr) {
194             this.xsr = xsr;
195             this.locations = new HashMap<Object, Location>();
196         }
197 
198         @Override
199         public void beforeUnmarshal(Object target, Object parent) {
200             locations.put(target, xsr.getLocation());
201         }
202 
203         public Location getLocation(Object o) {
204             return locations.get(o);
205         }

207     }
