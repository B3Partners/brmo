65      public void importIntoDb(URL in, TopNLType type) throws JDOMException {
66          XMLInputFactory xif = XMLInputFactory.newFactory();
67          try {
68              log.info("Importing file " + in.toExternalForm() + ", type: " + type.getType());
69              Unmarshaller jaxbUnmarshaller = converterFactory.getContext(type).createUnmarshaller();
70  
71              XMLStreamReader xsr = xif.createXMLStreamReader(in.openStream());
72              LocationListener ll = new LocationListener(xsr);
73              jaxbUnmarshaller.setListener(ll);
74  
75              while (xsr.hasNext()) {
76                  int eventType = xsr.next();
77  
78                  if (eventType == XMLStreamReader.START_ELEMENT) {
79                      String localname = xsr.getLocalName();
80                      if (xsr.getLocalName().equals("FeatureMember")) {
81                          JAXBElement jb = null;
82                          try {
83                              jb = (JAXBElement) jaxbUnmarshaller.unmarshal(xsr);
84                              Object obj = jb.getValue();
85                              ArrayList list = new ArrayList();
86                              list.add(obj);
87                              List<TopNLEntity> entities = convert(list, type);
88                              save(entities, type);
89                          } catch (JAXBException | IOException | SAXException | ParserConfigurationException | TransformerException | ParseException ex) {
90                              log.error("Error parsing", ex);
91                          } catch (ClassCastException cce) {
92                              log.error(String.format(Locale.ROOT, "Verwerkingsfout van element %s locatie %s", localname, (jb == null ? "onbekend" : ll.getLocation(jb))), cce);
93                          }
94                      }
95                  }
96              }
97  
98              xsr.close();
99          } catch (XMLStreamException | IOException | JAXBException ex) {
100             log.error("cannot correctly stream xml file:", ex);
101         } 
102     }
103 
104     public List parse (URL in) throws  JAXBException,  IOException{
105         List list = new ArrayList();
106         try {
107             TopNLType type = TopNLTypeFactory.getTopNLType(in);
108             Unmarshaller jaxbUnmarshaller = converterFactory.getContext(type).createUnmarshaller();
109             
110             XMLInputFactory xif = XMLInputFactory.newFactory();
111             
112             XMLStreamReader xsr = xif.createXMLStreamReader(in.openStream());
113             
114             while (xsr.hasNext()) {
115                 int eventType = xsr.next();
116                 
117                 if (eventType == XMLStreamReader.START_ELEMENT) {
118                     String localname = xsr.getLocalName();
119                     if (xsr.getLocalName().equals("FeatureMember")) {
120                         JAXBElement jb = (JAXBElement)jaxbUnmarshaller.unmarshal(xsr);
121                         list.add(jb.getValue());
122                     }
123                 }
124             }
125             
126             xsr.close();
127         } catch (XMLStreamException ex) {
128             log.error("cannot correctly stream xml file:", ex);
129         } catch (JDOMException ex) {
130             log.error("Cannot retrieve topnltype: ",ex);
131         }
132         return list;
133     }

135     /**
136      *
137      * @throws ClassCastException Als er een onverwacht type geometrie in de
138      * data zit, bijvoorbeeld een punt ipv een lijn
139      */
140     public List<TopNLEntity> convert(List listOfJaxbObjects, TopNLType type) throws IOException, SAXException,
141             ParserConfigurationException, TransformerException, ClassCastException {
142 
143         Converter converter = converterFactory.getConverter(type);
144         List<TopNLEntity> entity = converter.convert(listOfJaxbObjects);
145         return entity;
146     }
147     
148     public void save(TopNLEntity entity, TopNLType type) throws ParseException{
149         database.save(entity);
150     }
151     
152     public void save(List<TopNLEntity> entities, TopNLType type) throws ParseException {
153         for (TopNLEntity entity : entities) {
154             save(entity, type);
155         }
156     }

158     private class LocationListener extends Listener {
159         private XMLStreamReader xsr;
160         private Map<Object, Location> locations;
161 
162         public LocationListener(XMLStreamReader xsr) {
163             this.xsr = xsr;
164             this.locations = new HashMap<Object, Location>();
165         }
166 
167         @Override
168         public void beforeUnmarshal(Object target, Object parent) {
169             locations.put(target, xsr.getLocation());
170         }
171 
172         public Location getLocation(Object o) {
173             return locations.get(o);
174         }
175 
176     }
