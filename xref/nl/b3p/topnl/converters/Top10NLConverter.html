75      @Override
76      public List<TopNLEntity> convert(List jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException {
77          List<TopNLEntity> entities = null;
78          if (jaxbObject instanceof ArrayList) {
79              entities = convertFeatureCollection(jaxbObject);
80          } else {
81              entities = new ArrayList<>();
82              entities.add(convertObject(jaxbObject));
83          }
84          return entities;
85      }

87      @Override
88      public List<TopNLEntity> convertFeatureCollection(List jaxbFeatureCollection) throws IOException, SAXException, ParserConfigurationException, TransformerException {
89          List<TopNLEntity> entities = new ArrayList<>();
90  
91          for (Object fm : jaxbFeatureCollection) {
92              FeatureMemberType featureMember = (FeatureMemberType)fm;
93              JAXBElement jaxbObject = featureMember.getTop10NlObject();
94              Object obj = jaxbObject.getValue();
95              TopNLEntity entity = convertObject(obj);
96              entities.add(entity);
97          }
98  
99          return entities;
100     }

102     @Override
103     public TopNLEntity convertObject(Object featureMember) throws IOException, SAXException, ParserConfigurationException, TransformerException,IllegalArgumentException {
104         TopNLEntity entity = null;
105 
106          if(featureMember instanceof FeatureMemberType){
107             FeatureMemberType fm = (FeatureMemberType)featureMember;
108             JAXBElement jaxbObject = fm.getTop10NlObject();
109             return convertObject(jaxbObject.getValue());
110         }else if (featureMember instanceof HoogteType) {
111             entity = convertHoogte(featureMember);
112         } else if (featureMember instanceof FunctioneelGebiedType) {
113             entity = convertFunctioneelGebied(featureMember);
114         } else if (featureMember instanceof GebouwType) {
115             entity = convertGebouw(featureMember);
116         } else if (featureMember instanceof GeografischGebiedType) {
117             entity = convertGeografischGebied(featureMember);
118         } else if (featureMember instanceof InrichtingselementType) {
119             entity = convertInrichtingselement(featureMember);
120         } else if (featureMember instanceof PlaatsType) {
121             entity = convertPlaats(featureMember);
122         } else if (featureMember instanceof RegistratiefGebiedType) {
123             entity = convertRegistratiefGebied(featureMember);
124         } else if (featureMember instanceof ReliefType) {
125             entity = convertRelief(featureMember);
126         } else if (featureMember instanceof SpoorbaandeelType) {
127             entity = convertSpoorbaandeel(featureMember);
128         } else if (featureMember instanceof TerreinType) {
129             entity = convertTerrein(featureMember);
130         } else if (featureMember instanceof WaterdeelType) {
131             entity = convertWaterdeel(featureMember);
132         } else if (featureMember instanceof WegdeelType) {
133             entity = convertWegdeel(featureMember);
134         }else if (featureMember instanceof PlanTopografieType) {
135             entity = convertPlanTopografie(featureMember);
136         } else {
137             throw new IllegalArgumentException("Type not recognized: " + featureMember.getClass());
138         }
139 
140         log.debug("Converted type " + entity.getClass() + ", identificatie: " + entity.getIdentificatie());
141         entity.setTopnltype(TopNLType.TOP10NL.getType());
142         return entity;
143     }

145     @Override
146     public Hoogte convertHoogte(Object jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException {
147         HoogteType h = (HoogteType) jaxbObject;
148 
149         Hoogte hoogte = new Hoogte();
150         convertTop10NlObjectType(h, hoogte);
151 
152         hoogte.setTypeHoogte(h.getTypeHoogte().getValue());
153 
154         hoogte.setHoogte(h.getHoogte());
155         hoogte.setObjectEindTijd(h.getObjectEindTijd() != null ? h.getObjectEindTijd().getTime() : null);
156 
157         Element el = h.getGeometrie();
158         Geometry geom = gc.convertGeometry(el);
159         hoogte.setGeometrie(geom);
160         hoogte.setReferentieVlak(h.getReferentievlak() != null ? h.getReferentievlak().getValue() : null );
161         return hoogte;
162     }

164     public String convertIdentificatie(Identificatie identificatie) {
165         String idString = identificatie.getNEN3610ID().getNamespace().trim() + "." + identificatie.getNEN3610ID().getLokaalID();
166         return idString;
167     }

169     private void convertTop10NlObjectType(Top10NlObjectType type, TopNLEntity entity) {
170         entity.setIdentificatie(convertIdentificatie(type.getIdentificatie()));
171         entity.setBrontype(type.getBrontype().getValue());
172         entity.setBronactualiteit(type.getBronactualiteit().getTime());
173         entity.setBronbeschrijving(type.getBronbeschrijving());
174         entity.setBronnauwkeurigheid(type.getBronnauwkeurigheid());
175         entity.setObjectBeginTijd(type.getObjectBeginTijd().getTime());
176         entity.setVisualisatieCode(type.getVisualisatieCode() != null ? type.getVisualisatieCode().longValue() : null);
177     }

193     @Override
194     public Gebouw convertGebouw(Object jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException {
195         GebouwType g = (GebouwType) jaxbObject;
196 
197         Gebouw gb = new Gebouw();
198         convertTop10NlObjectType(g, gb);
199         String types = "";
200         for (CodeType typeGebouwT10Type : g.getTypeGebouw()) {
201             if (types.length() != 0) {
202                 types += ",";
203             }
204             types += typeGebouwT10Type.getValue();
205         }
206         gb.setTypeGebouw(types);
207         gb.setFysiekVoorkomen(g.getFysiekVoorkomen() != null ? g.getFysiekVoorkomen().getValue() : null);
208         gb.setHoogteklasse(g.getHoogteklasse() != null ? g.getHoogteklasse().getValue() : null);
209         gb.setHoogte(g.getHoogte());
210         gb.setNaam(String.join(",", g.getNaam()));
211         gb.setSoortnaam(g.getSoortnaam());
212         gb.setStatus(g.getStatus() != null ? g.getStatus().getValue() : null);
213         gb.setGeometrie(gc.convertGeometry(g.getGeometrie()));
214         return gb;
215     }

281     @Override
282     public Relief convertRelief(Object jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException {
283         ReliefType r = (ReliefType) jaxbObject;
284 
285         Relief rg = new Relief();
286         convertTop10NlObjectType(r, rg);
287         
288         if(r.getLijnGeometrie() != null){
289             rg.setGeometrie((LineString) gc.convertGeometry(r.getLijnGeometrie()));
290         }
291         
292         if(r.getTaludGeometrie() != null){
293             BRTHogeEnLageZijdeType hl = r.getTaludGeometrie().getBRTHogeEnLageZijde();
294             Element[] els = hl.getHogeZijde();
295             for (Element el : els) {
296                 LineString g = (LineString) gc.convertGeometry(el);
297                 if(el.getLocalName().equals("lageZijde")){
298                     rg.setTaludLageZijde(g);
299                 }else if(el.getLocalName().equals("hogeZijde")){
300                     rg.setTaludHogeZijde(g);
301                 }
302             }
303         }
304 
305         rg.setHoogteklasse(r.getHoogteklasse().getValue());
306         rg.setTypeRelief(r.getTypeRelief() != null ? r.getTypeRelief().getValue() : null);
307         rg.setHoogteniveau(r.getHoogteniveau().longValue());
308         
309         return rg;
310     }

312     @Override
313     public Spoorbaandeel convertSpoorbaandeel(Object jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException, ClassCastException {
314         SpoorbaandeelType r = (SpoorbaandeelType) jaxbObject;
315 
316         Spoorbaandeel rg = new Spoorbaandeel();
317         convertTop10NlObjectType(r, rg);
318 
319         final Geometry g = gc.convertGeometry(r.getGeometrie());
320         if (g instanceof LineString) {
321             rg.setGeometrie((LineString) g);
322         } else {
323             rg.setPuntGeometrie((Point) g);
324         }
325 
326         String fysieks = "";
327         for (CodeType fk : r.getFysiekVoorkomen()) {
328             if(fysieks.length() != 0){
329                 fysieks += ",";
330             }
331             fysieks += fk.getValue();
332         }
333         rg.setFysiekVoorkomen(fysieks);
334         
335         rg.setTypeSpoorbaan(r.getTypeSpoorbaan() != null ? r.getTypeSpoorbaan().getValue() : null);
336         rg.setAantalSporen(r.getAantalSporen() != null ? r.getAantalSporen().getValue() : null);
337         rg.setStatus(r.getStatus().getValue());
338         rg.setHoogteniveau(r.getHoogteniveau().longValue());
339         rg.setTypeInfrastructuur(r.getTypeInfrastructuur() != null ? r.getTypeInfrastructuur().getValue() : null);
340         rg.setSpoorbreedte(r.getSpoorbreedte() != null ? r.getSpoorbreedte().getValue() : null);
341         rg.setVervoerfunctie(r.getVervoerfunctie()!= null ? r.getVervoerfunctie().getValue() : null);
342         rg.setElektrificatie(r.getElektrificatie() != null ? r.getElektrificatie().equals(BRTJaNeeWaardeType.JA) : null);
343         return rg;
344     }

401     @Override
402     public Wegdeel convertWegdeel(Object jaxbObject) throws IOException, SAXException, ParserConfigurationException, TransformerException {
403         WegdeelType r = (WegdeelType) jaxbObject;
404 
405         Wegdeel rg = new Wegdeel();
406         convertTop10NlObjectType(r, rg);
407 
408         String fysieks = "";
409         for (CodeType fk : r.getFysiekVoorkomen()) {
410             if(fysieks.length() != 0){
411                 fysieks += ",";
412             }
413             fysieks += fk.getValue();
414         }
415         rg.setFysiekVoorkomen(fysieks);
416         
417         String typeweg = "";
418         for (CodeType codeType : r.getTypeWeg()) {
419             if (typeweg.length() != 0) {
420                 typeweg += ",";
421             }
422             typeweg += codeType.getValue();
423         }
424 
425         rg.setTypeWeg(typeweg);
426         
427         
428         String hoofdverkeersgebruikt = "";
429         for (CodeType codeType : r.getHoofdverkeersgebruik()) {
430             if (hoofdverkeersgebruikt.length() != 0) {
431                 hoofdverkeersgebruikt += ",";
432             }
433             hoofdverkeersgebruikt += codeType.getValue();
434         }
435 
436         rg.setHoofdverkeersgebruik(hoofdverkeersgebruikt);
437         
438         rg.setNaam(String.join(",",r.getNaam()));
439         
440         Element [] els = r.getHartGeometrie();
441         for (Element el : els) {
442             String localname = el.getLocalName();
443             Geometry g = gc.convertGeometry(el);
444             if(localname.equals("hoofdGeometrie")){
445                 rg.setGeometrie(g);
446             }else if(localname.equals("hartGeometrie")){
447                 rg.setHartGeometrie(g);
448             }
449         }
450         rg.setTypeInfrastructuur(r.getTypeInfrastructuur() != null ? r.getTypeInfrastructuur().getValue() : null);
451         rg.setVerhardingstype(r.getVerhardingstype().getValue());
452         rg.setVerhardingsbreedteklasse(r.getVerhardingsbreedteklasse() != null ? r.getVerhardingsbreedteklasse().getValue() : null);
453         rg.setGescheidenRijbaan(r.getGescheidenRijbaan() ==BRTJaNeeWaardeType.JA);
454         rg.setaWegnummer(String.join(",", r.getAWegnummer()));
455         rg.setnWegnummer(String.join(",", r.getNWegnummer()));
456         rg.seteWegnummer(String.join(",", r.getEWegnummer()));
457         rg.setsWegnummer(String.join(",", r.getSWegnummer()));
458         rg.setAfritnummer(r.getAfritnummer());
459         rg.setAfritnaam(String.join(",",r.getAfritnaam()));
460         rg.setKnooppuntnaam(String.join(",",r.getKnooppuntnaam()));
461         rg.setBrugnaam(String.join(",",r.getBrugnaam()));
462         rg.setTunnelnaam(String.join(",",r.getTunnelnaam()));
463         rg.setHoogteniveau(r.getHoogteniveau().longValue());
464         rg.setStatus(r.getStatus().getValue());
465         return rg;
