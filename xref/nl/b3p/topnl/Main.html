67      private static void loadtopnl(String dir, Processor p, TopNLType type)  throws ParseException, IOException, SAXException, ParserConfigurationException, JAXBException, TransformerException, JDOMException {
68          File f = new File (dir);
69          FilenameFilter filter = (dir1, name) -> name.toLowerCase().endsWith(".gml");
70          /*File f = new File("/mnt/data/Documents/TopNL/TOP100NL_GML_Filechunks_november_2016/TOP100NL_GML_Filechunks/Top100NL_000002.gml");
71          p.importIntoDb(f.toURL(), TopNLType.TOP100NL);*/

73          for (File file : files) {
74              // String fileString = file.getCanonicalPath();
75              p.importIntoDb(file.toURI().toURL(), type);
76          }
77      }
