7   package nl.b3p.brmo.sql.dialect;
8   
9   import oracle.jdbc.OracleConnection;
10  import org.geotools.data.oracle.sdo.GeometryConverter;
11  import org.locationtech.jts.geom.Geometry;
12  
13  import java.sql.Connection;
14  import java.sql.PreparedStatement;
15  import java.sql.SQLException;
16  import java.sql.Types;
17  import java.util.regex.Matcher;
18  import java.util.regex.Pattern;
19  
20  public class OracleDialect implements SQLDialect {
21  
22      public OracleDialect() {
23      }
24  
25      @Override
26      public String getDriverClass() {
27          return "oracle.jdbc.OracleDriver";
28      }
29  
30      @Override
31      public boolean supportsDropTableIfExists() {
32          return false;
33      }
34  
35      @Override
36      public String getType(String type) {
37          Matcher varchar = Pattern.compile("varchar\\((.+)\\)").matcher(type);
38          if(varchar.matches()) {
39              return "varchar2(" + varchar.group(1) + " char)";
40          }
41          if(type.equals("integer")) {
42              return "number(10)";
43          }
44          if(type.equals("boolean")) {
45              return "varchar2(5)";
46          }
47          if(type.startsWith("geometry(")) {
48              return "MDSYS.SDO_GEOMETRY";
49          }
50          if(type.equals("text")) {
51              return "clob";
52          }
53          return type;
54      }
55  
56      @Override
57      public Object getGeometryParameter(Connection c, org.locationtech.jts.geom.Geometry geometry, boolean linearizeCurves) throws SQLException {
58          OracleConnection connection = c.unwrap(OracleConnection.class);
59          return new GeometryConverter(connection).toSDO(geometry);
60      }
61  
62      @Override
63      public void setGeometryParameter(Connection c, PreparedStatement ps, int parameterIndex, int pmdType, Geometry geometry, boolean linearizeCurves) throws SQLException {
64          if (geometry == null || geometry.isEmpty()) {
65              ps.setNull(parameterIndex, Types.STRUCT, "MDSYS.SDO_GEOMETRY");
66          } else {
67              ps.setObject(parameterIndex, getGeometryParameter(c, geometry, linearizeCurves));
68          }
69      }
70  
71      @Override
72      public String getCreateGeometryMetadataSQL(String tableName, String geometryColumn, String type) {
73          return String.format("insert into user_sdo_geom_metadata values ('%S', '%S', MDSYS.SDO_DIM_ARRAY(" +
74                  "MDSYS.SDO_DIM_ELEMENT('X', 12000, 280000, .1), " +
75                  "MDSYS.SDO_DIM_ELEMENT('Y', 304000, 620000, .1) " +
76                  "), 28992)",
77                  tableName, geometryColumn);
78      }
79  
80      @Override
81      public String getCreateGeometryIndexSQL(String tableName, String geometryColumn, String type) {
82          return String.format("create index idx_%s_%s on %s (%s) indextype is mdsys.spatial_index",
83                  tableName, geometryColumn, tableName, geometryColumn);
84      }
85  
86      @Override
87      public int getDefaultOptimalBatchSize() {
88          return 500;
89      }
90  }
