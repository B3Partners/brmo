7   package nl.b3p.brmo.sql.dialect;
8   
9   import org.locationtech.jts.geom.Geometry;
10  
11  import java.sql.Connection;
12  import java.sql.PreparedStatement;
13  import java.sql.SQLException;
14  
15  public interface SQLDialect {
16      String getDriverClass() throws ClassNotFoundException;
17  
18      default void loadDriver() throws ClassNotFoundException {
19          Class.forName(getDriverClass());
20      }
21  
22      default String getType(String type) {
23          return type;
24      }
25  
26      default boolean supportsDropTableIfExists() {
27          return true;
28      }
29  
30      default Object getGeometryParameter(Connection c, Geometry geometry) throws SQLException {
31          return getGeometryParameter(c, geometry, false);
32      }
33  
34      Object getGeometryParameter(Connection c, Geometry geometry, boolean linearizeCurves) throws SQLException;
35  
36      default void setGeometryParameter(Connection c, PreparedStatement ps, int parameterIndex, int pmdType, Geometry geometry, boolean linearizeCurves) throws SQLException {
37          if (geometry == null) {
38              ps.setNull(parameterIndex, pmdType);
39          } else {
40              ps.setObject(parameterIndex, getGeometryParameter(c, geometry, linearizeCurves));
41          }
42      }
43  
44      default String getCreateGeometryMetadataSQL(String tableName, String geometryColumn, String type) {
45          return "";
46      }
47  
48      default String getCreateGeometryIndexSQL(String tableName, String geometryColumn, String type) {
49          return "";
50      }
51  
52      int getDefaultOptimalBatchSize();
53  }
