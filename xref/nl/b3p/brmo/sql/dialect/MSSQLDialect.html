7   package nl.b3p.brmo.sql.dialect;
8   
9   import com.microsoft.sqlserver.jdbc.Geometry;
10  import nl.b3p.brmo.util.StandardLinearizedWKTWriter;
11  import org.geotools.geometry.jts.MultiSurface;
12  import org.geotools.geometry.jts.WKTWriter2;
13  import org.locationtech.jts.geom.GeometryCollection;
14  import org.locationtech.jts.geom.GeometryFactory;
15  
16  import java.sql.Connection;
17  import java.sql.PreparedStatement;
18  import java.sql.SQLException;
19  import java.sql.Types;
20  
21  public class MSSQLDialect implements SQLDialect {
22      private final StandardLinearizedWKTWriter wktWriter = new StandardLinearizedWKTWriter();
23      private final WKTWriter2 wktWriter2 = new WKTWriter2();
24  
25      @Override
26      public String getDriverClass() {
27          return "com.microsoft.sqlserver.jdbc.SQLServerDriver";
28      }
29  
30      @Override
31      public String getType(String type) {
32          if(type.startsWith("varchar")) {
33              return "n" + type;
34          }
35          if(type.equals("timestamp")) {
36              return "datetime2";
37          }
38          if(type.equals("boolean")) {
39              return "bit";
40          }
41          if(type.startsWith("geometry(")) {
42              return "geometry";
43          }
44          if(type.equals("text")) {
45              return "ntext";
46          }
47          return type;
48      }
49  
50      @Override
51      public Object getGeometryParameter(Connection c, org.locationtech.jts.geom.Geometry geometry, boolean linearizeCurves) throws SQLException {
52          String wkt;
53          if (linearizeCurves) {
54              wkt = wktWriter.write(geometry);
55          } else {
56              if (geometry instanceof MultiSurface) {
57                  // MSSQL does not support MULTISURFACE but does support CURVEPOLYGON and CIRCULARSTRING. Convert the
58                  // geometry to a CURVEPOLYGON if there is only a single geometry component and otherwise a
59                  // GEOMETRYCOLLECTION, which is only subtly different from a MULTISURFACE.
60  
61                  // If we had a WKB writer supporting curves we could use Geometry.STGeomFromWKB()
62  
63                  MultiSurface multiSurface = (MultiSurface) geometry;
64                  if (multiSurface.getNumGeometries() == 1) {
65                      wkt = wktWriter2.write(multiSurface.getGeometryN(0));
66                  } else {
67                      GeometryFactory geometryFactory = new GeometryFactory();
68                      org.locationtech.jts.geom.Geometry[] geometries = new org.locationtech.jts.geom.Geometry[multiSurface.getNumGeometries()];
69                      for(int i = 0; i < multiSurface.getNumGeometries(); i++) {
70                          geometries[i] = multiSurface.getGeometryN(i);
71                      }
72                      GeometryCollection collection = geometryFactory.createGeometryCollection(geometries);
73                      wkt = wktWriter2.write(collection);
74  //                        // Convert after conversion to WKT to avoid using a GeometryFactory
75  //                        wkt = wktWriter2.write(multiSurface);
76  //                        wkt = "GEOMETRYCOLLECTION" + wkt.substring("MULTISURFACE".length());
77                  }
78              } else {
79                  wkt = wktWriter2.write(geometry);
80              }
81          }
82          try {
83              return Geometry.STGeomFromText(wkt, geometry.getSRID());
84          } catch(SQLException e) {
85              throw new SQLException("Error converting geometry to MSSQL geometry for WKT: " + wkt, e);
86          }
87      }
88  
89  
90      @Override
91      public void setGeometryParameter(Connection c, PreparedStatement ps, int parameterIndex, int pmdType, org.locationtech.jts.geom.Geometry geometry, boolean linearizeCurves) throws SQLException {
92          if (geometry == null) {
93              // Note that using the pmdType (-157 for GEOMETRY) does not work (using mssql-jdbc 9.2.1.jre8), throwing
94              // com.microsoft.sqlserver.jdbc.SQLServerException: The conversion from OBJECT to GEOMETRY is unsupported.
95              ps.setNull(parameterIndex, Types.OTHER);
96          } else {
97              ps.setObject(parameterIndex, getGeometryParameter(c, geometry, linearizeCurves));
98          }
99      }
100 
101     @Override
102     public String getCreateGeometryIndexSQL(String tableName, String geometryColumn, String type) {
103         return String.format("create spatial index idx_%s_%s on %s (%s) with (bounding_box = (12000,304000,280000,620000))",
104                 tableName, geometryColumn, tableName, geometryColumn);
105     }
106 
107     @Override
108     public int getDefaultOptimalBatchSize() {
109         return 250;
110     }
111 
112     @Override
113     public boolean supportsDropTableIfExists() {
114         // MSSQL does support "drop table if exists", but not with "cascade" added
115         return false;
116     }
117 }
