7   package nl.b3p.brmo.sql;
8   
9   import nl.b3p.brmo.sql.dialect.PostGISDialect;
10  import nl.b3p.brmo.sql.dialect.SQLDialect;
11  import org.apache.commons.logging.Log;
12  import org.apache.commons.logging.LogFactory;
13  import org.apache.commons.text.StringEscapeUtils;
14  import org.locationtech.jts.geom.Geometry;
15  import org.postgresql.PGConnection;
16  import org.postgresql.copy.CopyIn;
17  
18  import java.nio.charset.StandardCharsets;
19  import java.sql.Connection;
20  import java.sql.SQLException;
21  
22  /**
23   * The PostgreSQL JDBC driver does not support parallel copy operations, even with multiple connections. So this class
24   * can cache the copy stream in memory so only one copy operation is active at a time, although this is significantly
25   * slower.
26   */
27  public class PostGISCopyInsertBatch implements QueryBatch {
28      private static final Log LOG = LogFactory.getLog(PostGISCopyInsertBatch.class);
29  
30      protected Connection connection;
31      protected String sql;
32      protected PostGISDialect dialect;
33      protected int batchSize;
34      protected boolean linearizeCurves;
35  
36      protected int count = 0;
37      protected StringEscapeUtils.Builder copyData = PostgresCopyEscapeUtils.builder();
38      protected CopyIn copyIn = null;
39      protected CopyIn lastCopyIn = null;
40      protected boolean buffer;
41  
42      public PostGISCopyInsertBatch(Connection connection, String sql, int batchSize, SQLDialect dialect, boolean buffer, boolean linearizeCurves) {
43          if (!(dialect instanceof PostGISDialect)) {
44              throw new IllegalArgumentException();
45          }
46          this.connection = connection;
47          this.sql = sql;
48          this.dialect = (PostGISDialect)dialect;
49          this.batchSize = batchSize;
50          this.buffer = buffer;
51          this.linearizeCurves = linearizeCurves;
52      }
53  
54      protected void createCopyIn() throws SQLException {
55          PGConnection pgConnection = connection.unwrap(PGConnection.class);
56          this.copyIn = pgConnection.getCopyAPI().copyIn(sql);
57      }
58  
59      protected void writeToCopy() throws SQLException {
60          if (copyIn == null) {
61              createCopyIn();
62          }
63          byte[] bytes = copyData.toString().getBytes(StandardCharsets.UTF_8);
64          copyIn.writeToCopy(bytes, 0, bytes.length);
65          copyData = PostgresCopyEscapeUtils.builder();
66      }
67  
68      @Override
69      public boolean addBatch(Object[] params) throws Exception {
70          for(int i = 0; i < params.length; i++) {
71              if (i != 0) {
72                  copyData.append("\t");
73              }
74              Object param = params[i];
75              if (param == null) {
76                  copyData.append("\\N");
77              } else if (param instanceof Geometry) {
78                  Geometry geometry = (Geometry) param;
79                  copyData.append(dialect.getEWkt(geometry, linearizeCurves));
80              } else if (param instanceof Boolean) {
81                  copyData.append((Boolean)param ? "t" : "f");
82              } else {
83                  // TODO any more types need special conversion?
84                  copyData.escape(param.toString());
85              }
86          }
87          copyData.append("\n");
88  
89          if (!buffer) {
90              writeToCopy();
91          }
92  
93          count++;
94          if (count == batchSize) {
95              this.executeBatch();
96              return true;
97          }
98          return false;
99      }
100 
101 
102     @Override
103     public void executeBatch() throws Exception {
104         if (count > 0) {
105             if (buffer) {
106                 if (LOG.isDebugEnabled()) {
107                     // To get the number of bytes this call is duplicated from writeToCopy()...
108                     int bytes = copyData.toString().getBytes(StandardCharsets.UTF_8).length;
109                     LOG.debug(String.format("execute buffered copy batch, %d bytes, %d rows, sql: %s", bytes, count, sql));
110                 }
111                 writeToCopy();
112             } else {
113                 LOG.debug(String.format("execute copy batch, %d rows, sql: %s", count, sql));
114             }
115             copyIn.endCopy();
116             lastCopyIn = copyIn;
117             // reset so writeToCopy will create a new one
118             copyIn = null;
119             count = 0;
120         }
121     }
122 
123     @Override
124     public void close() {
125     }
126 }
