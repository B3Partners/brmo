7   package nl.b3p.brmo.bgt.loader;
8   
9   import java.text.MessageFormat;
10  import java.time.Duration;
11  import java.time.Instant;
12  import java.util.ResourceBundle;
13  
14  public class Utils {
15  
16      public static String formatTimeSince(Instant start) {
17          Duration d = Duration.between(start, Instant.now());
18          String days = d.toDaysPart() > 0 ? d.toDaysPart() + "d " : "";
19          if (d.toHoursPart() == 0 && d.toMinutesPart() == 0) {
20              return days + d.toSecondsPart() + "s";
21          } else if(d.toHoursPart() == 0) {
22              return String.format("%s%dm %2ds", days, d.toMinutesPart(), d.toSecondsPart());
23          }
24          return String.format("%s%dh %2dm %2ds", days, d.toHoursPart(), d.toMinutesPart(), d.toSecondsPart());
25      }
26  
27      public static final String BUNDLE_NAME = "BGTLoader";
28  
29      private static final ResourceBundle bundle = ResourceBundle.getBundle(BUNDLE_NAME);
30  
31      public static ResourceBundle getBundle() {
32          return bundle;
33      }
34  
35      public static String getBundleString(String key) {
36          return bundle.getString(key);
37      }
38  
39      public static String getMessageFormattedString(String key, Object... args) {
40          ResourceBundle bundle = getBundle();
41          return new MessageFormat(bundle.getString(key), bundle.getLocale()).format(args);
42      }
43  
44      public static String getLoaderVersion() {
45          return getBundleString("app.version");
46      }
47  
48      public static String getBrmoVersion() {
49          return getBundleString("brmo.version");
50      }
51  
52      public static String getUserAgent() {
53          return String.format("%s, %s (%s)/%s, %s/%s",
54                  getBundleString("app.user-agent"),
55                  System.getProperty("os.name"),
56                  System.getProperty("os.arch"),
57                  System.getProperty("os.version"),
58                  System.getProperty("java.vm.name"),
59                  System.getProperty("java.vm.version")
60          );
61      }
62  }
