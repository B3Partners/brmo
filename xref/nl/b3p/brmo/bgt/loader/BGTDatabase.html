34  public class BGTDatabase implements AutoCloseable {
35    private static final Log log = LogFactory.getLog(BGTDatabase.class);

37    public enum SQLDialectEnum {
38      postgis,
39      oracle
40    }
41  
42    private SQLDialect dialect;
43    private final DatabaseOptions dbOptions;
44    private Connection connection;
45    private final boolean allowConnectionCreation;
46  
47    /**
48     * Create a BGTDatabase that will load the driver and create a database connection.
49     *
50     * @param dbOptions Database connection options.
51     * @throws ClassNotFoundException If the JDBC driver could not be loaded.
52     */
53    public BGTDatabase(DatabaseOptions dbOptions) throws ClassNotFoundException {
54      this.dbOptions = dbOptions;
55      dialect = createDialect(dbOptions.getConnectionString());
56      dialect.loadDriver();
57      this.allowConnectionCreation = true;
58    }
59  
60    /**
61     * Create a BGTDatabase with a supplied Connection without loading the driver and making
62     * connections here. The connection string in DatabaseOptions is not used, it is retrieved from
63     * the supplied Connection to determine the dialect.
64     *
65     * @param dbOptions Database options - not used for making connections
66     * @param connection Connection to use, must not be closed. Connection will be closed by CLI
67     *     classes after processing, override the close() method to avoid that.
68     */
69    public BGTDatabase(DatabaseOptions dbOptions, Connection connection) throws SQLException {
70      this.dbOptions = dbOptions;
71      this.dialect = createDialect(connection.getMetaData().getURL());
72      this.connection = connection;
73      this.allowConnectionCreation = false;
74    }
75  
76    public SQLDialect getDialect() {
77      return dialect;
78    }
79  
80    public Connection getConnection() throws SQLException {
81      if (this.connection == null || this.connection.isClosed()) {
82        this.connection = createConnection();
83      }
84      return this.connection;
85    }
86  
87    public void setConnection(Connection connection) {
88      this.connection = connection;
89    }
90  
91    public void setDialect(SQLDialect dialect) {
92      this.dialect = dialect;
93    }
94  
95    @Override
96    public void close() throws SQLException {
97      if (this.connection != null && !this.connection.isClosed()) {
98        this.connection.close();
99      }
100   }

102   private Connection createConnection() {
103     if (!allowConnectionCreation) {
104       throw new RuntimeException(
105           "New connection required but supplied connection is null or closed");
106     }
107     try {
108       return DriverManager.getConnection(
109           dbOptions.getConnectionString(), dbOptions.getUser(), dbOptions.getPassword());
110     } catch (SQLException e) {
111       throw new RuntimeException(
112           getMessageFormattedString("db.connection_error", dbOptions.getConnectionString()), e);
113     }
114   }

116   public BGTObjectTableWriter createObjectTableWriter(
117       LoadOptions loadOptions, DatabaseOptions dbOptions) throws SQLException {
118     BGTObjectTableWriter writer =
119         new BGTObjectTableWriter(getConnection(), this.getDialect(), BGTSchemaMapper.getInstance());
120 
121     if (loadOptions == null) {
122       loadOptions = new LoadOptions();
123     }
124     writer.setBatchSize(
125         dbOptions.getBatchSize() != null
126             ? dbOptions.getBatchSize()
127             : this.getDialect().getDefaultOptimalBatchSize());
128     writer.setMultithreading(loadOptions.isMultithreading());
129     writer.setUsePgCopy(dbOptions.isUsePgCopy());
130     writer.setObjectLimit(loadOptions.getMaxObjects());
131     writer.setLinearizeCurves(loadOptions.isLinearizeCurves());
132     writer.setCurrentObjectsOnly(!loadOptions.isIncludeHistory());
133     writer.setCreateSchema(loadOptions.isCreateSchema());
134     writer.setDropIfExists(loadOptions.isDropIfExists());
135     writer.setTablePrefix(loadOptions.getTablePrefix());
136     return writer;
137   }

139   public static SQLDialect createDialect(String connectionString) {
140     SQLDialectEnum sqlDialectEnum;
141     if (connectionString.startsWith("jdbc:postgresql:")) {
142       sqlDialectEnum = SQLDialectEnum.postgis;
143     } else if (connectionString.startsWith("jdbc:oracle:thin:")) {
144       sqlDialectEnum = SQLDialectEnum.oracle;
145     } else {
146       throw new IllegalArgumentException(
147           getMessageFormattedString("db.unknown_connection_string_dialect", connectionString));
148     }
149     return createDialect(sqlDialectEnum);
150   }
151 
152   public static SQLDialect createDialect(SQLDialectEnum dialectEnum) {
153     switch (dialectEnum) {
154       case postgis:
155         return new PostGISDialect();
156       case oracle:
157         return new OracleDialect();
158     }
159     throw new IllegalArgumentException(
160         getMessageFormattedString("db.dialect_invalid", dialectEnum));
161   }

163   public void createMetadataTable(LoadOptions loadOptions) throws SQLException {
164     log.info(getBundleString("db.create_metadata"));
165     for (String sql :
166         BGTSchemaMapper.getInstance()
167             .getCreateMetadataTableStatements(
168                 getDialect(), loadOptions.getTablePrefix(), loadOptions.isDropIfExists())) {
169       new LoggingQueryRunner().update(getConnection(), sql);
170     }
171   }

173   public String getMetadata(Metadata key) throws SQLException {
174     Object value =
175         new LoggingQueryRunner()
176             .query(
177                 getConnection(),
178                 "select waarde from " + METADATA_TABLE_NAME + " where naam = ?",
179                 new ScalarHandler<>(),
180                 key.getDbKey());
181     if (value == null) {
182       return null;
183     }
184     if (value instanceof Clob) {
185       Clob clob = (Clob) value;
186       return clob.getSubString(1, (int) clob.length());
187     }
188     return value.toString();
189   }
190 
191   public void setMetadataValue(Metadata key, String value) throws Exception {
192     try {
193       int updated =
194           new LoggingQueryRunner()
195               .update(
196                   getConnection(),
197                   "update " + METADATA_TABLE_NAME + " set waarde = ? where naam = ?",
198                   value,
199                   key.getDbKey());
200       if (updated == 0) {
201         new LoggingQueryRunner()
202             .update(
203                 getConnection(),
204                 "insert into " + METADATA_TABLE_NAME + "(naam, waarde) values(?,?)",
205                 key.getDbKey(),
206                 value);
207       }
208     } catch (SQLException e) {
209       throw new Exception(
210           getMessageFormattedString("db.metadata_error", key.getDbKey(), value, e.getMessage()), e);
211     }
212   }

214   public void setMetadataForMutaties(BGTObjectStreamer.MutatieInhoud mutatieInhoud)
215       throws Exception {
216     setMetadataValue(Metadata.DELTA_TIME_TO, null);
217     if (mutatieInhoud == null || mutatieInhoud.getLeveringsId() == null) {
218       setMetadataValue(Metadata.INITIAL_LOAD_DELTA_ID, null);
219       setMetadataValue(Metadata.INITIAL_LOAD_TIME, null);
220       setMetadataValue(Metadata.DELTA_ID, null);
221     } else {
222       String deltaId = mutatieInhoud.getLeveringsId();
223 
224       if ("initial".equals(mutatieInhoud.getMutatieType())) {
225         setMetadataValue(Metadata.INITIAL_LOAD_DELTA_ID, deltaId);
226         setMetadataValue(Metadata.INITIAL_LOAD_TIME, Instant.now().toString());
227       }
228       setMetadataValue(Metadata.DELTA_ID, deltaId);
229     }
230   }
