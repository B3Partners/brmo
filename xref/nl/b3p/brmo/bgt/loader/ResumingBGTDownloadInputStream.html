30  public class ResumingBGTDownloadInputStream extends ResumingInputStream {
31      private static final Log log = LogFactory.getLog(ResumingBGTDownloadInputStream.class);
32  
33      public ResumingBGTDownloadInputStream(URI uri, BGTObjectTableWriter writer) {
34          this(uri, writer, false);
35      }
36  
37      public ResumingBGTDownloadInputStream(URI uri, BGTObjectTableWriter writer, boolean logRetries) {
38          super(new HttpStartRangeInputStreamProvider(uri, new Java11HttpClientWrapper() {
39              @Override
40              public void beforeRequest(HttpRequest.Builder requestBuilder) {
41                  requestBuilder.headers("User-Agent", getUserAgent());
42              }
43  
44              @Override
45              public HttpResponseWrapper wrapResponse(HttpResponse<InputStream> response) {
46                  HttpResponseWrapper wrapper = super.wrapResponse(response);
47  
48                  // The direct download https://api.pdok.nl/lv/bgt/download/v1_0/full/predefined/bgt-citygml-nl-nopbp.zip
49                  // does not support the HEAD method to read the Content-Length -- it first sends a redirect. Read the
50                  // Content-Length later:
51  
52                  // Only read Content-Length from the first response starting at position 0, not a response from a
53                  // retried request starting at a later position (we could parse the Content-Range response though).
54                  if (wrapper.getHeader("Content-Range") == null) {
55                      String contentLength = wrapper.getHeader("Content-Range");
56                      if (contentLength != null) {
57                          ((ProgressReporter) writer.getProgressUpdater()).setTotalBytes(Long.parseLong(contentLength));
58                      }
59                  }
60                  return wrapper;
61              }
62          }) {
63              @Override
64              public InputStream get(long position, int totalRetries, Exception causeForRetry) throws IOException {
65                  if (causeForRetry != null && logRetries) {
66                      String msg = getMessageFormattedString("download.retry",
67                              totalRetries,
68                              position,
69                              ExceptionUtils.getRootCause(causeForRetry).getMessage()
70                      );
71                      if (writer.getProgressUpdater() instanceof ConsoleProgressReporter) {
72                          System.out.println("\r" + msg);
73                      } else {
74                          log.warn(msg);
75                          log.trace(causeForRetry);
76                      }
77                  }
78                  return super.get(position, totalRetries, causeForRetry);
79              }
80          });
81      }
82  }
