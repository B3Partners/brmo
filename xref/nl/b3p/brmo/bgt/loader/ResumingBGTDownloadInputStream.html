7   package nl.b3p.brmo.bgt.loader;
8   
9   import static nl.b3p.brmo.bgt.loader.Utils.getMessageFormattedString;
10  import static nl.b3p.brmo.bgt.loader.Utils.getUserAgent;
11  
12  import java.io.IOException;
13  import java.io.InputStream;
14  import java.net.URI;
15  import java.net.http.HttpRequest;
16  import java.net.http.HttpResponse;
17  import nl.b3p.brmo.bgt.loader.cli.ConsoleProgressReporter;
18  import nl.b3p.brmo.bgt.schema.BGTObjectTableWriter;
19  import nl.b3p.brmo.util.ResumingInputStream;
20  import nl.b3p.brmo.util.http.HttpResponseWrapper;
21  import nl.b3p.brmo.util.http.HttpStartRangeInputStreamProvider;
22  import nl.b3p.brmo.util.http.wrapper.Java11HttpClientWrapper;
23  import org.apache.commons.lang3.exception.ExceptionUtils;
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  
27  public class ResumingBGTDownloadInputStream extends ResumingInputStream {
28    private static final Log log = LogFactory.getLog(ResumingBGTDownloadInputStream.class);
29  
30    public ResumingBGTDownloadInputStream(URI uri, BGTObjectTableWriter writer) {
31      this(uri, writer, false);
32    }
33  
34    public ResumingBGTDownloadInputStream(URI uri, BGTObjectTableWriter writer, boolean logRetries) {
35      super(
36          new HttpStartRangeInputStreamProvider(
37              uri,
38              new Java11HttpClientWrapper() {
39                @Override
40                public void beforeRequest(HttpRequest.Builder requestBuilder) {
41                  requestBuilder.headers("User-Agent", getUserAgent());
42                }
43  
44                @Override
45                public HttpResponseWrapper wrapResponse(HttpResponse<InputStream> response) {
46                  HttpResponseWrapper wrapper = super.wrapResponse(response);
47  
48                  // The direct download
49                  // https://api.pdok.nl/lv/bgt/download/v1_0/full/predefined/bgt-citygml-nl-nopbp.zip
50                  // does not support the HEAD method to read the Content-Length -- it
51                  // first sends a redirect. Read the
52                  // Content-Length later:
53  
54                  // Only read Content-Length from the first response starting at
55                  // position 0, not a response from a
56                  // retried request starting at a later position (we could parse the
57                  // Content-Range response though).
58                  if (wrapper.getHeader("Content-Range") == null) {
59                    String contentLength = wrapper.getHeader("Content-Range");
60                    if (contentLength != null) {
61                      ((ProgressReporter) writer.getProgressUpdater())
62                          .setTotalBytes(Long.parseLong(contentLength));
63                    }
64                  }
65                  return wrapper;
66                }
67              }) {
68            @Override
69            public InputStream get(long position, int totalRetries, Exception causeForRetry)
70                throws IOException {
71              if (causeForRetry != null && logRetries) {
72                String msg =
73                    getMessageFormattedString(
74                        "download.retry",
75                        totalRetries,
76                        position,
77                        ExceptionUtils.getRootCause(causeForRetry).getMessage());
78                if (writer.getProgressUpdater() instanceof ConsoleProgressReporter) {
79                  System.out.println("\r" + msg);
80                } else {
81                  log.warn(msg);
82                  log.trace(causeForRetry);
83                }
84              }
85              return super.get(position, totalRetries, causeForRetry);
86            }
87          });
88    }
89  }
