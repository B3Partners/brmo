10  import nl.b3p.brmo.bgt.schema.BGTSchema;

14  import java.util.Arrays;
15  import java.util.HashSet;
16  import java.util.List;
17  import java.util.Objects;
18  import java.util.Set;
19  import java.util.stream.Collectors;
20  import java.util.stream.Stream;
21  
22  import static nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadRequest.FeaturetypesEnum.PLAATSBEPALINGSPUNT;
23  
24  public class FeatureTypeSelectionOptions {
25      @Option(names={"--feature-types"}, split=",", defaultValue="all", paramLabel = "<name>")
26      List<String> featureTypes;
27  
28      public List<String> getFeatureTypes() {
29          return featureTypes;
30      }
31  
32      public void setFeatureTypes(List<String> featureTypes) {
33          this.featureTypes = featureTypes;
34      }
35  
36  
37      public Set<DeltaCustomDownloadRequest.FeaturetypesEnum> getFeatureTypesList() {
38          Set<DeltaCustomDownloadRequest.FeaturetypesEnum> types = new HashSet<>();
39          if (featureTypes.contains("all")) {
40              types.addAll(Arrays.asList(DeltaCustomDownloadRequest.FeaturetypesEnum.values()));
41          } else {
42              BGTSchema bgtSchema = BGTSchema.getInstance();
43              BGTSchemaMapper bgtSchemaMapper = BGTSchemaMapper.getInstance();
44              if (featureTypes.contains("bgt")) {
45                  types.addAll(BGTSchema.getInstance().getOnlyBGTObjectTypes().map(objectType -> {
46                      try {
47                          // Just add the table names as featureType enum value, these are the same
48                          return DeltaCustomDownloadRequest.FeaturetypesEnum.fromValue(bgtSchemaMapper.getTableNameForObjectType(objectType, ""));
49                      } catch(IllegalArgumentException e) {
50                          // Ignore nummeraanduidingreeks as it is a one-to-many table for pand
51                          return null;
52                      }
53                  }).filter(Objects::nonNull).collect(Collectors.toSet()));
54              }
55              if (featureTypes.contains("plus")) {
56                  types.addAll(bgtSchema.getIMGeoPlusObjectTypes().map(objectType ->
57                          DeltaCustomDownloadRequest.FeaturetypesEnum.fromValue(bgtSchemaMapper.getTableNameForObjectType(objectType, ""))
58                  ).collect(Collectors.toSet()));
59              }
60          }
61  
62          // This feature type is mostly not useful and very big so must be added explicitly, not included with 'all' or 'bgt'
63          types.remove(PLAATSBEPALINGSPUNT);
64  
65          for(String featureType: featureTypes) {
66              if (Stream.of("all", "bgt", "plus").noneMatch(t -> t.equals(featureType))) {
67                  try {
68                      types.add(DeltaCustomDownloadRequest.FeaturetypesEnum.fromValue(featureType));
69                  } catch(IllegalArgumentException e) {
70                      throw new IllegalArgumentException(String.format("Invalid feature type: \"%s\"", featureType));
71                  }
72              }
73          }
74          return types;
75      }
76  }
