7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import picocli.CommandLine.Option;
10  
11  public class LoadOptions {
12      @Option(names={"--no-create-schema"}, negatable = true)
13      boolean createSchema = true;
14  
15      @Option(names={"--include-history"})
16      boolean includeHistory = false;
17  
18      @Option(names="--table-prefix", defaultValue = "", hidden = true)
19      String tablePrefix = "";
20  
21      @Option(names="--max-objects", paramLabel= "<number>", hidden = true)
22      Integer maxObjects;
23  
24      @Option(names={"--linearize-curves"})
25      boolean linearizeCurves = false;
26  
27      @Option(names={"--no-multithreading"}, negatable = true, hidden = true)
28      boolean multithreading = true;
29  
30      @Option(names="--no-http-zip-random-access", negatable = true, hidden = true)
31      boolean httpZipRandomAccess = true;
32  
33      @Option(names="--debug-http-seeks", hidden = true)
34      boolean debugHttpSeeks = false;
35  
36      public boolean isCreateSchema() {
37          return createSchema;
38      }
39  
40      public void setCreateSchema(boolean createSchema) {
41          this.createSchema = createSchema;
42      }
43  
44      public boolean isIncludeHistory() {
45          return includeHistory;
46      }
47  
48      public void setIncludeHistory(boolean includeHistory) {
49          this.includeHistory = includeHistory;
50      }
51  
52      public String getTablePrefix() {
53          return tablePrefix;
54      }
55  
56      public void setTablePrefix(String tablePrefix) {
57          this.tablePrefix = tablePrefix;
58      }
59  
60      public Integer getMaxObjects() {
61          return maxObjects;
62      }
63  
64      public void setMaxObjects(Integer maxObjects) {
65          this.maxObjects = maxObjects;
66      }
67  
68      public boolean isLinearizeCurves() {
69          return linearizeCurves;
70      }
71  
72      public void setLinearizeCurves(boolean linearizeCurves) {
73          this.linearizeCurves = linearizeCurves;
74      }
75  
76      public boolean isMultithreading() {
77          return multithreading;
78      }
79  
80      public void setMultithreading(boolean multithreading) {
81          this.multithreading = multithreading;
82      }
83  
84      public boolean isHttpZipRandomAccess() {
85          return httpZipRandomAccess;
86      }
87  
88      public void setHttpZipRandomAccess(boolean httpZipRandomAccess) {
89          this.httpZipRandomAccess = httpZipRandomAccess;
90      }
91  
92      public boolean isDebugHttpSeeks() {
93          return debugHttpSeeks;
94      }
95  
96      public void setDebugHttpSeeks(boolean debugHttpSeeks) {
97          this.debugHttpSeeks = debugHttpSeeks;
98      }
99  }
