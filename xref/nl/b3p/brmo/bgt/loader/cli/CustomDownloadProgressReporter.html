7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import nl.b3p.brmo.bgt.download.api.CustomDownloadProgress;
10  import nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadStatusResponse;
11  import org.apache.commons.logging.Log;
12  import org.apache.commons.logging.LogFactory;
13  
14  import java.time.Duration;
15  import java.time.Instant;
16  import java.util.function.Consumer;
17  
18  import static nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadStatusResponse.StatusEnum.COMPLETED;
19  import static nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadStatusResponse.StatusEnum.PENDING;
20  import static nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadStatusResponse.StatusEnum.RUNNING;
21  import static nl.b3p.brmo.bgt.loader.Utils.getBundleString;
22  import static nl.b3p.brmo.bgt.loader.Utils.getMessageFormattedString;
23  
24  class CustomDownloadProgressReporter implements Consumer<CustomDownloadProgress> {
25      private static final Log log = LogFactory.getLog(CustomDownloadProgressReporter.class);
26  
27      private final boolean printProgress;
28  
29      private DeltaCustomDownloadStatusResponse.StatusEnum lastStatus;
30  
31      private Instant lastProgress = null;
32  
33      public CustomDownloadProgressReporter(boolean printProgress) {
34          this.printProgress = printProgress;
35      }
36  
37      @Override
38      public void accept(CustomDownloadProgress progress) {
39          if (progress.getStatusApiCalls() == 0 && progress.getDownloadRequestId() != null) {
40              log.debug(getMessageFormattedString("download.request_id", progress.getDownloadRequestId()));
41              log.info(getBundleString("download.requesting_status"));
42          } else {
43              DeltaCustomDownloadStatusResponse.StatusEnum status = progress.getStatusResponse().getStatus();
44              if (!printProgress) {
45                  if (lastStatus != status) {
46                      log.info(getMessageFormattedString("download.extract_status", status));
47                  }
48                  if (status == RUNNING) {
49                      // Log progress in logfile every 5 minutes
50                      if (lastProgress == null) {
51                          lastProgress = Instant.now();
52                      } else if (Duration.between(lastProgress, Instant.now()).getSeconds() > 300) {
53                          log.info(getMessageFormattedString("download.extract_status", String.format("%s (%2d%%)", status, progress.getStatusResponse().getProgress())));
54                          lastProgress = Instant.now();
55                      }
56                  }
57              } else {
58                  if (status == PENDING) {
59                      System.out.print("\r" + getMessageFormattedString("download.extract_pending", progress.getTimeSinceStart()));
60                  } else if (status == RUNNING) {
61                      System.out.print("\r" + getMessageFormattedString("download.extract_running", String.format("%3d", progress.getStatusResponse().getProgress()), progress.getTimeSinceStart()));
62                  } else if (status == COMPLETED) {
63                      log.debug(getMessageFormattedString("download.extract_ready", progress.getTimeSinceStart()));
64                      System.out.printf("\r%s\r", " ".repeat(30));
65                  }
66              }
67              lastStatus = status;
68          }
69      }
70  }
