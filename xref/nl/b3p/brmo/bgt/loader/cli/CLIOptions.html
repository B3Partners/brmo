7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import picocli.CommandLine.Option;
10  
11  public class CLIOptions {
12  
13      @Option(names="--progress", hidden = true)
14      boolean progress;
15  
16      public boolean isConsoleProgressEnabled() {
17          if (progress) {
18              return true;
19          }
20          // Imperfect but good enough without using native library like Jansi's isatty()
21          // https://stackoverflow.com/questions/1403772/how-can-i-check-if-a-java-programs-input-output-streams-are-connected-to-a-term
22          return System.console() != null;
23      }
24  }
