93    @Command(name = "schema", sortOptions = false)
94    public int schema(
95        @Option(names = "--dialect", paramLabel = "<dialect>", defaultValue = "postgis")
96            BGTDatabase.SQLDialectEnum dialectEnum,
97        @Mixin FeatureTypeSelectionOptions featureTypeSelectionOptions,
98        @Option(names = "--table-prefix", defaultValue = "", hidden = true) String tablePrefix,
99        @Option(
100               names = {"-h", "--help"},
101               usageHelp = true)
102           boolean showHelp)
103       throws SQLException {
104     SQLDialect dialect = BGTDatabase.createDialect(dialectEnum);
105     // For schema generation include plaatsbepalingspunt with 'all' and 'bgt'
106     if (featureTypeSelectionOptions.featureTypes.contains("all")
107         || featureTypeSelectionOptions.featureTypes.contains("bgt")) {
108       featureTypeSelectionOptions
109           .getFeatureTypes()
110           .add(DeltaCustomDownloadRequest.FeaturetypesEnum.PLAATSBEPALINGSPUNT.getValue());
111     }
112     Set<String> tableNames =
113         featureTypeSelectionOptions.getFeatureTypesList().stream()
114             .map(DeltaCustomDownloadRequest.FeaturetypesEnum::getValue)
115             .collect(Collectors.toSet());
116     BGTSchemaMapper bgtSchemaMapper = BGTSchemaMapper.getInstance();
117     bgtSchemaMapper.printSchema(
118         dialect,
119         tablePrefix,
120         objectType ->
121             tableNames.contains(bgtSchemaMapper.getTableNameForObjectType(objectType, "")));
122     return ExitCode.OK;
123   }

125   @Command(name = "load", sortOptions = false)
126   public int load(
127       @Mixin DatabaseOptions dbOptions,
128       @Mixin LoadOptions loadOptions,
129       @Mixin FeatureTypeSelectionOptions featureTypeSelectionOptions,
130       @Parameters(paramLabel = "<file>") String file,
131       @Mixin CLIOptions cliOptions,
132       @Option(
133               names = {"-h", "--help"},
134               usageHelp = true)
135           boolean showHelp)
136       throws Exception {
137 
138     log.info(getUserAgent());
139 
140     try (BGTDatabase db = new BGTDatabase(dbOptions)) {
141       BGTObjectTableWriter writer = db.createObjectTableWriter(loadOptions, dbOptions);
142 
143       ProgressReporter progressReporter =
144           cliOptions.isConsoleProgressEnabled()
145               ? new ConsoleProgressReporter()
146               : new ProgressReporter();
147       writer.setProgressUpdater(progressReporter);
148 
149       if (loadOptions.createSchema) {
150         db.createMetadataTable(loadOptions);
151       }
152 
153       if (file.endsWith(".zip") && (file.startsWith("http://") || file.startsWith("https://"))) {
154         loadZipFromURI(new URI(file), writer, featureTypeSelectionOptions, loadOptions, true);
155       } else if (file.endsWith(".zip")) {
156         loadZip(new File(file), writer, featureTypeSelectionOptions);
157       } else if (file.matches(".*\\.[xg]ml")) {
158         loadXml(new File(file), writer);
159       } else {
160         log.error(getMessageFormattedString("load.invalid_extension", file));
161         return ExitCode.USAGE;
162       }
163 
164       if (writer.getProgress() == null) {
165         log.error(getBundleString("error.no_feature_types"));
166         return ExitCode.SOFTWARE;
167       }
168       db.setMetadataValue(Metadata.LOADER_VERSION, getLoaderVersion());
169       // Set feature types list from options, not MutatieInhoud (if input has it)...
170       // FIXME if downloaded initial extract has less object types, update will fail -- should
171       // set to only encountered
172       // feature types
173       db.setFeatureTypesEnumMetadata(featureTypeSelectionOptions.getFeatureTypesList());
174       db.setMetadataValue(Metadata.INCLUDE_HISTORY, loadOptions.includeHistory + "");
175       db.setMetadataValue(Metadata.LINEARIZE_CURVES, loadOptions.linearizeCurves + "");
176       db.setMetadataValue(Metadata.TABLE_PREFIX, loadOptions.tablePrefix);
177       BGTObjectTableWriter.BGTProgress progress = writer.getProgress();
178       if (progress.getMutatieInhoud() != null) {
179         db.setMetadataForMutaties(progress.getMutatieInhoud());
180         db.setMetadataValue(Metadata.GEOM_FILTER, progress.getMutatieInhoud().getGebied());
181 
182         log.info(
183             getMessageFormattedString(
184                 "load.mutatie",
185                 progress.getMutatieInhoud().getMutatieType(),
186                 progress.getMutatieInhoud().getLeveringsId()));
187       }
188       progressReporter.reportTotalSummary();
189       db.getConnection().commit();
190     }
191 
192     return ExitCode.OK;
193   }

195   private static boolean isBGTZipEntrySelected(
196       String entryName,
197       FeatureTypeSelectionOptions featureTypeSelectionOptions,
198       boolean logSkipAsInfo) {
199     Set<DeltaCustomDownloadRequest.FeaturetypesEnum> featureTypes =
200         featureTypeSelectionOptions.getFeatureTypesList();
201     Pattern p = Pattern.compile("bgt_(.+).[xg]ml");
202 
203     Matcher m = p.matcher(entryName);
204     if (!m.matches()) {
205       log.warn(getMessageFormattedString("load.skip_entry", entryName));
206       return false;
207     }
208     String tableName = m.group(1);
209     try {
210       DeltaCustomDownloadRequest.FeaturetypesEnum featureType =
211           DeltaCustomDownloadRequest.FeaturetypesEnum.fromValue(tableName);
212       if (!featureTypes.contains(featureType)) {
213         String msg = getMessageFormattedString("load.skip_unselected", tableName);
214         if (logSkipAsInfo) {
215           log.info(msg);
216         } else {
217           log.debug(msg);
218         }
219         return false;
220       } else {
221         return true;
222       }
223     } catch (IllegalArgumentException e) {
224       log.warn(getMessageFormattedString("load.skip_unknown_feature_type", entryName));
225       return false;
226     }
227   }

229   public void loadZipFromURI(
230       URI uri,
231       BGTObjectTableWriter writer,
232       FeatureTypeSelectionOptions featureTypeSelectionOptions,
233       LoadOptions loadOptions,
234       boolean showSelected)
235       throws Exception {
236     log.info(getMessageFormattedString("download.downloading_from", uri));
237     if (loadOptions.isHttpZipRandomAccess()) {
238       loadZipFromURIUsingRandomAccess(
239           uri, writer, featureTypeSelectionOptions, showSelected, loadOptions.isDebugHttpSeeks());
240     } else {
241       loadZipFromURIUsingStreaming(uri, writer, featureTypeSelectionOptions);
242     }
243   }

245   public void loadZipFromURIUsingRandomAccess(
246       URI uri,
247       BGTObjectTableWriter writer,
248       FeatureTypeSelectionOptions featureTypeSelectionOptions,
249       boolean showSelected,
250       boolean debugHttpSeeks)
251       throws Exception {
252     Instant start = Instant.now();
253 
254     // NOTE: it can happen that not all entries from a ZIP are read because of
255     // https://issues.apache.org/jira/browse/COMPRESS-584
256     // This happened with
257     // https://api.pdok.nl/lv/bgt/download/v1_0/cache/2/ebe787b3-e113-4331-ab96-edd1e9bf5aa7/bgt-citygml-nl-nopbp.zip
258 
259     try (HttpSeekableByteChannel channel =
260             new HttpSeekableByteChannel(uri).withDebug(debugHttpSeeks);
261         CountingSeekableByteChannel loggingChannel = new CountingSeekableByteChannel(channel);
262         org.apache.commons.compress.archivers.zip.ZipFile zipFile =
263             new org.apache.commons.compress.archivers.zip.ZipFile(
264                 loggingChannel, uri.toString(), "UTF8", false, true)) {
265       if (debugHttpSeeks) {
266         System.out.println();
267       }
268       int count = 0;
269       long uncompressed = 0;
270       for (Iterator<ZipArchiveEntry> it = zipFile.getEntries().asIterator(); it.hasNext(); ) {
271         ZipArchiveEntry entry = it.next();
272         count++;
273         uncompressed += entry.getSize();
274       }
275       log.info(
276           getMessageFormattedString(
277               "download.zip.read",
278               formatTimeSince(start),
279               count,
280               byteCountToDisplaySize(channel.size()),
281               byteCountToDisplaySize(uncompressed)));
282       if (debugHttpSeeks) {
283         log.info(
284             getMessageFormattedString(
285                 "download.zip.debug-http-seeks.entries",
286                 loggingChannel.getNonConsecutiveIops(),
287                 channel.getHttpRequestCount(),
288                 channel.getBytesRead()));
289       }
290       loggingChannel.setLoggingEnabled(false);
291 
292       ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
293       List<ZipArchiveEntry> selected = new ArrayList<>();
294       zipFile
295           .getEntries()
296           .asIterator()
297           .forEachRemaining(
298               entry -> {
299                 if (isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, false)) {
300                   selected.add(entry);
301                 }
302               });
303 
304       if (selected.size() > 1) {
305         // Only report total percentage when more than one entry
306         long totalSize = selected.stream().map(ZipArchiveEntry::getSize).reduce(0L, Long::sum);
307         Long totalCompressedSize =
308             selected.stream().map(ZipArchiveEntry::getCompressedSize).reduce(0L, Long::sum);
309         progressReporter.setTotalBytes(totalSize);
310         if (showSelected) {
311           log.info(
312               getMessageFormattedString(
313                   "download.zip.selected",
314                   selected.size(),
315                   byteCountToDisplaySize(totalCompressedSize),
316                   byteCountToDisplaySize(totalSize)));
317         }
318       }
319       Long[] previousEntriesBytesRead = new Long[] {0L};
320       progressReporter.setTotalBytesReadFunction(
321           () -> previousEntriesBytesRead[0] + writer.getProgress().getBytesRead());
322 
323       for (ZipArchiveEntry entry : selected) {
324         progressReporter.startNewFile(entry.getName(), entry.getSize());
325         writer.write(zipFile.getInputStream(entry));
326       }
327       if (debugHttpSeeks) {
328         log.info(
329             getMessageFormattedString(
330                 "download.zip.debug-http-seeks.totals",
331                 loggingChannel.getNonConsecutiveIops(),
332                 channel.getHttpRequestCount(),
333                 channel.getBytesRead(),
334                 byteCountToDisplaySize(channel.getBytesRead())));
335       }
336     }
337   }

339   public void loadZipFromURIUsingStreaming(
340       URI downloadURI,
341       BGTObjectTableWriter writer,
342       FeatureTypeSelectionOptions featureTypeSelectionOptions)
343       throws Exception {
344     ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
345 
346     try (InputStream input = new ResumingBGTDownloadInputStream(downloadURI, writer)) {
347       CountingInputStream countingInputStream = new CountingInputStream(input);
348       progressReporter.setTotalBytesReadFunction(countingInputStream::getByteCount);
349 
350       try (ZipInputStream zis = new ZipInputStream(countingInputStream)) {
351         ZipEntry entry = zis.getNextEntry();
352         while (entry != null) {
353           if (isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, true)) {
354             progressReporter.startNewFile(entry.getName(), null);
355             writer.write(CloseShieldInputStream.wrap(zis));
356           }
357           entry = zis.getNextEntry();
358         }
359       }
360     }
361   }

363   public void loadZip(
364       File file,
365       BGTObjectTableWriter writer,
366       FeatureTypeSelectionOptions featureTypeSelectionOptions)
367       throws Exception {
368     try (ZipFile zipFile = new ZipFile(file)) {
369       List<ZipEntry> entries =
370           zipFile.stream()
371               .filter(
372                   entry ->
373                       isBGTZipEntrySelected(entry.getName(), featureTypeSelectionOptions, false))
374               .collect(Collectors.toList());
375 
376       ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
377       if (entries.size() > 1) {
378         // Only report total percentage when more than one entry
379         Long totalSize = entries.stream().map(ZipEntry::getSize).reduce(0L, Long::sum);
380         progressReporter.setTotalBytes(totalSize);
381       }
382       Long[] previousEntriesBytesRead = new Long[] {0L};
383       progressReporter.setTotalBytesReadFunction(
384           () -> previousEntriesBytesRead[0] + writer.getProgress().getBytesRead());
385 
386       for (ZipEntry entry : entries) {
387         try (InputStream in = zipFile.getInputStream(entry)) {
388           // getSize() will not return -1 because ZipFile uses random access to read the
389           // ZIP central directory
390           loadInputStream(entry.getName(), in, entry.getSize(), writer);
391           previousEntriesBytesRead[0] += entry.getSize();
392         }
393       }
394       progressReporter.reportTotalSummary();
395     }
396   }

398   public void loadXml(File file, BGTObjectTableWriter writer) throws Exception {
399     try (FileInputStream in = new FileInputStream(file)) {
400       loadInputStream(file.getName(), in, file.length(), writer);
401     }
402   }

404   public void loadInputStream(
405       String name, InputStream input, long size, BGTObjectTableWriter writer) throws Exception {
406     ProgressReporter progressReporter = (ProgressReporter) writer.getProgressUpdater();
407     progressReporter.startNewFile(name, size);
408     writer.write(input);
409   }
