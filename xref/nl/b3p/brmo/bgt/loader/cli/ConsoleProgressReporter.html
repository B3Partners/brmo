7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import nl.b3p.brmo.bgt.schema.BGTObjectTableWriter;
10  import nl.b3p.brmo.bgt.loader.ProgressReporter;
11  import nl.b3p.brmo.schema.ObjectTableWriter;
12  
13  import java.time.Instant;
14  
15  import static nl.b3p.brmo.bgt.loader.Utils.formatTimeSince;
16  import static nl.b3p.brmo.bgt.loader.Utils.getBundleString;
17  import static nl.b3p.brmo.schema.ObjectTableWriter.Stage.FINISHED;
18  
19  public class ConsoleProgressReporter extends ProgressReporter {
20      Instant totalStart = Instant.now();
21  
22      protected void log(String msg) {
23          System.out.print("\r" + msg);
24      }
25  
26      protected void status(String msg) {
27          System.out.print("\r" + msg + " ".repeat(40));
28      }
29  
30      public void reportTotalSummary() {
31          super.reportTotalSummary();
32          System.out.println();
33      }
34  
35      public void accept(ObjectTableWriter.Progress genericProgress) {
36          super.accept(genericProgress);
37          BGTObjectTableWriter.BGTProgress progress = (BGTObjectTableWriter.BGTProgress)genericProgress;
38  
39          if (progress.getStage() == FINISHED) {
40              System.out.println();
41          } else if (progress.getStage() == BGTObjectTableWriter.Stage.LOAD_OBJECTS) {
42  
43              String total = "";
44              if (getTotalBytes() != null) {
45                  total = String.format(" - %s %4.1f%%   %s",
46                          getBundleString("progress.total"),
47                          100.0 / getTotalBytes() * getTotalBytesReadFunction().get(),
48                          formatTimeSince(totalStart));
49              }
50  
51              String current;
52              if (getCurrentFileSize() != null) {
53                  current = String.format("%4.1f%% ", 100.0 / getCurrentFileSize() * progress.getBytesRead());
54              } else {
55                  current = String.format("%,6d MB", progress.getBytesRead() / 1024 / 1024);
56              }
57              System.out.printf("\r%s: %s  %s, %,10d %s%s",
58                      getCurrentFileName(),
59                      current,
60                      formatTimeSince(getCurrentFileStart()),
61                      progress.getObjectCount(),
62                      getBundleString("progress.objects"),
63                      total
64              );
65          }
66      }
67  }
