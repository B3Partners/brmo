7   package nl.b3p.brmo.bgt.loader.cli;
8   
9   import nl.b3p.brmo.bgt.schema.BGTObjectTableWriter;
10  import nl.b3p.brmo.bgt.loader.ProgressReporter;
11  import nl.b3p.brmo.schema.ObjectTableWriter;
12  
13  import java.time.Instant;
14  
15  import static nl.b3p.brmo.bgt.loader.Utils.formatTimeSince;
16  import static nl.b3p.brmo.bgt.loader.Utils.getBundleString;
17  import static nl.b3p.brmo.schema.ObjectTableWriter.Stage.FINISHED;
18  
19  public class ConsoleProgressReporter extends ProgressReporter {
20      Instant totalStart = Instant.now();
21  
22      @Override
23      protected void log(String msg) {
24          System.out.print("\r" + msg);
25      }
26  
27      @Override
28      protected void status(String msg) {
29          System.out.print("\r" + msg + " ".repeat(40));
30      }
31  
32      @Override
33      public void reportTotalSummary() {
34          super.reportTotalSummary();
35          System.out.println();
36      }
37  
38      @Override
39      public void accept(ObjectTableWriter.Progress genericProgress) {
40          super.accept(genericProgress);
41          BGTObjectTableWriter.BGTProgress progress = (BGTObjectTableWriter.BGTProgress)genericProgress;
42  
43          if (progress.getStage() == FINISHED) {
44              System.out.println();
45          } else if (progress.getStage() == BGTObjectTableWriter.Stage.LOAD_OBJECTS) {
46  
47              String total = "";
48              if (getTotalBytes() != null) {
49                  total = String.format(" - %s %4.1f%%   %s",
50                          getBundleString("progress.total"),
51                          100.0 / getTotalBytes() * getTotalBytesReadFunction().get(),
52                          formatTimeSince(totalStart));
53              }
54  
55              String current;
56              if (getCurrentFileSize() != null) {
57                  current = String.format("%4.1f%% ", 100.0 / getCurrentFileSize() * progress.getBytesRead());
58              } else {
59                  current = String.format("%,6d MB", progress.getBytesRead() / 1024 / 1024);
60              }
61              System.out.printf("\r%s: %s  %s, %,10d %s%s",
62                      getCurrentFileName(),
63                      current,
64                      formatTimeSince(getCurrentFileStart()),
65                      progress.getObjectCount(),
66                      getBundleString("progress.objects"),
67                      total
68              );
69          }
70      }
71  }
