20  import static nl.b3p.brmo.bgt.download.model.DeltaCustomDownloadStatusResponse.StatusEnum.COMPLETED;
21  import static nl.b3p.brmo.bgt.loader.Utils.formatTimeSince;
22  
23  public class DownloadApiUtils {
24      public static URI getCustomDownloadURL(ApiClient client, Delta delta, ExtractSelectionOptions extractSelectionOptions, Consumer<CustomDownloadProgress> progressConsumer) throws ApiException, InterruptedException {
25          DeltaCustomApi deltaCustomApi = new DeltaCustomApi(client);
26          DeltaCustomDownloadRequest deltaCustomDownloadRequest = new DeltaCustomDownloadRequest();
27          deltaCustomDownloadRequest.setDeltaId(delta == null ? null : delta.getId());
28          deltaCustomDownloadRequest.featuretypes(extractSelectionOptions.getFeatureTypesList());
29          deltaCustomDownloadRequest.setFormat(DeltaCustomDownloadRequest.FormatEnum.CITYGML);
30          deltaCustomDownloadRequest.setGeofilter(extractSelectionOptions.getGeoFilterWkt());
31  
32          progressConsumer = progressConsumer == null ? p -> {} : progressConsumer;
33  
34          DeltaCustomDownloadResponse downloadResponse = deltaCustomApi.deltaCustomDownload(deltaCustomDownloadRequest);
35          String downloadRequestId = downloadResponse.getDownloadRequestId();
36          CustomDownloadProgress progress = new CustomDownloadProgress();
37          progress.downloadRequestId = downloadRequestId;
38          progress.start = Instant.now();
39          progressConsumer.accept(progress);
40  
41          // TODO make argument
42          long waitTime = 1000;
43          do {
44              progress.statusResponse = deltaCustomApi.deltaCustomDownloadStatus(downloadRequestId);
45              progress.statusApiCalls++;
46              progress.timeSinceStart = formatTimeSince(progress.start);
47              progressConsumer.accept(progress);
48  
49              if (progress.statusResponse.getStatus() == COMPLETED) {
50                  break;
51              }
52  
53              if (progress.statusApiCalls > 100) {
54                  waitTime += 1000;
55                  waitTime = Math.min(30000, waitTime);
56              } else if (progress.statusApiCalls > 10) {
57                  waitTime += 1000;
58                  waitTime = Math.min(5000, waitTime);
59              }
60              Thread.sleep(waitTime);
61  
62              // TODO timeout
63          } while (true);
64  
65          if (progress.statusResponse.getStatus() != COMPLETED) {
66              throw new IllegalStateException(String.format("Download status for request id \"%s\" is not COMPLETED but \"%s\"", downloadRequestId, progress.statusResponse.getStatus()));
67          }
68  
69          String downloadUrl = progress.statusResponse.getLinks().getDownload().getHref();
70          URI baseUri = URI.create(client.getBaseUri());
71          return baseUri.resolve(downloadUrl);
72      }
73  }
