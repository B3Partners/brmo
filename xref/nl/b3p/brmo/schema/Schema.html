8   package nl.b3p.brmo.schema;
9   
10  import nl.b3p.brmo.bgt.schema.BGTObjectType;
11  
12  import java.util.HashMap;
13  import java.util.Map;
14  import java.util.stream.Stream;
15  
16  public class Schema {
17      public static final String INDEX = "idx";
18  
19      private final Map<String, ObjectType> objectTypes = new HashMap<>();
20  
21      protected Schema() {
22      }
23  
24      protected void addObjectType(BGTObjectType objectType) {
25          objectTypes.put(objectType.getName(), objectType);
26      }
27  
28      protected Map<String, ObjectType> getObjectTypes() {
29          return objectTypes;
30      }
31  
32      public Stream<? extends ObjectType> getAllObjectTypes() {
33          return objectTypes.values().stream();
34      }
35  
36      public ObjectType getObjectTypeByName(String name) {
37          return objectTypes.get(name);
38      }
39  }
