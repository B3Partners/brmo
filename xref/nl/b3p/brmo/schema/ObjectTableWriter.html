232   protected void prepareDatabaseForObject(SchemaObjectInstance object) throws Exception {
233     // Do creation of QueryBatch on main thread, so exceptions stop processing immediately
234     // This creates/truncates the table
235     getInsertBatch(object);
236   }

238   protected synchronized QueryBatch getInsertBatch(SchemaObjectInstance object) throws Exception {
239     Map<ObjectType, QueryBatch> insertBatches = progress.insertBatches;
240     if (insertBatches.isEmpty()) {
241       if (progress.initialLoad) {
242         if (isCreateSchema()) {
243           QueryRunner qr = new LoggingQueryRunner();
244           for (String sql :
245               Stream.concat(
246                       schemaSQLMapper.getCreateTableStatements(
247                           object.getObjectType(), dialect, tablePrefix, dropIfExists),
248                       schemaSQLMapper.getCreateGeometryMetadataStatements(
249                           object.getObjectType(), dialect, tablePrefix))
250                   .collect(Collectors.toList())) {
251             qr.update(connection, sql);
252           }
253         } else {
254           truncateTable(connection, object.getObjectType());
255           for (ObjectType oneToManyObjectType :
256               object.getObjectType().getOneToManyAttributeObjectTypes()) {
257             truncateTable(connection, oneToManyObjectType);
258           }
259         }
260       }
261     }
262 
263     if (!insertBatches.containsKey(object.getObjectType())) {
264       QueryBatch queryBatch;
265       if (this.dialect instanceof PostGISDialect && this.usePgCopy) {
266         String sql = buildPgCopySql(object, progress.initialLoad);
267         // Using Postgres COPY while buffering the copy stream is not faster than using
268         // batched inserts with the
269         // reWriteBatchedInserts=true connection parameter. Buffering is required when a
270         // one-to-many attribute
271         // exists, because simultaneous COPY statements (even with separate connections) are
272         // not supported by
273         // the JDBC driver
274         boolean bufferCopy = !progress.singleTableInserts;
275         queryBatch =
276             new PostGISCopyInsertBatch(
277                 connection, sql, batchSize, dialect, bufferCopy, linearizeCurves);
278       } else {
279         String sql = buildInsertSql(object);
280         Boolean[] parameterIsGeometry =
281             object.getObjectType().getDirectNonDefaultInsertAttributes().stream()
282                 .map(
283                     attributeColumnMapping ->
284                         attributeColumnMapping instanceof GeometryAttributeColumnMapping)
285                 .toArray(Boolean[]::new);
286         queryBatch =
287             new GeometryHandlingPreparedStatementBatch(
288                 connection, sql, batchSize, dialect, parameterIsGeometry, linearizeCurves);
289       }
290       insertBatches.put(object.getObjectType(), queryBatch);
291     }
292     return insertBatches.get(object.getObjectType());
293   }

329   protected void addObjectToBatch(SchemaObjectInstance object) throws Exception {
330     // Determine if we can use only a single table COPY without buffering
331     if (progress.firstObject) {
332       progress.setSingleTableInserts(
333           progress.initialLoad && object.getObjectType().hasOnlyDirectAttributes());
334       progress.firstObject = false;
335     }
336 
337     try {
338       addObjectToBatch(object, false);
339     } catch (Throwable e) {
340       String message = "Exception writing object to database, object: ";
341       if (getBatchSize() > 1) {
342         message =
343             "Exception adding parameters to database write batch, may be caused by previous batches. Object: ";
344       }
345       throw new Exception(message + object, e);
346     }
347   }

349   private void addObjectToBatch(SchemaObjectInstance object, boolean fromWorkerThread)
350       throws Throwable {
351 
352     if (multithreading && !fromWorkerThread) {
353       while (exceptionFromWorkerThread == null) {
354         // We can't use the blocking put() method because the worker thread may be
355         // interrupted because of an
356         // exception and never drain the full queue
357         if (progress.objectsToWrite.offer(object, 500, TimeUnit.MILLISECONDS)) {
358           return;
359         }
360       }
361       throw exceptionFromWorkerThread;
362     }
363 
364     // This always returns a value without creating the table because it has been called before
365     // (so no table
366     // creation in worker thread)
367     QueryBatch batch = getInsertBatch(object);
368 
369     Map<String, Object> attributes = object.getAttributes();
370     List<Object> params = new ArrayList<>();
371     for (AttributeColumnMapping attributeColumnMapping :
372         object.getObjectType().getDirectNonDefaultInsertAttributes()) {
373       Object attribute = attributes.get(attributeColumnMapping.getName());
374       params.add(attributeColumnMapping.toQueryParameter(attribute));
375     }
376 
377     // The main object batch must be executed before oneToMany and arrayAttribute batches which
378     // reference the main
379     // object when foreign key constraints are in place. During "stand" loading this is
380     // deferrered to after all
381     // batches are fully executed so we don't need to care about order. For BAG2 mutaties batch
382     // size is set to 1 to
383     // correctly order all batches for foreign key integrity
384     batch.addBatch(params.toArray());
385 
386     for (ObjectType oneToManyAttribute :
387         object.getObjectType().getOneToManyAttributeObjectTypes()) {
388       List<SchemaObjectInstance> objects =
389           (List<SchemaObjectInstance>) attributes.get(oneToManyAttribute.getName());
390       if (objects != null && !objects.isEmpty()) {
391         for (int i = 0; i < objects.size(); i++) {
392           SchemaObjectInstance oneToMany = objects.get(i);
393           // Add FK and index
394           String tableName = schemaSQLMapper.getTableNameForObjectType(object.getObjectType(), "");
395           String idColumnName = object.getObjectType().getPrimaryKeys().get(0).getName();
396           oneToMany
397               .getAttributes()
398               .put(
399                   schemaSQLMapper.getColumnNameForObjectType(
400                       oneToMany.getObjectType(), tableName + idColumnName),
401                   object.getAttributes().get(idColumnName));
402           oneToMany.getAttributes().put(Schema.INDEX, i);
403           // TODO call overridable method to add attributes to one to many
404           Object eindRegistratie = object.getAttributes().get("eindRegistratie");
405           oneToMany
406               .getAttributes()
407               .put(
408                   schemaSQLMapper.getColumnNameForObjectType(
409                       oneToMany.getObjectType(), tableName + "eindRegistratie"),
410                   eindRegistratie != null);
411           addObjectToBatch(oneToMany, fromWorkerThread);
412         }
413       }
414     }
415 
416     for (ArrayAttributeMapping arrayAttribute : object.getObjectType().getArrayAttributes()) {
417       insertArrayAttribute(object, arrayAttribute);
418     }
419 
420     updateProgress();
421   }

423   private void insertArrayAttribute(SchemaObjectInstance object, ArrayAttributeMapping attribute)
424       throws Exception {
425     Set values = (Set) object.getAttributes().get(attribute.getName());
426     if (values != null && !values.isEmpty()) {
427       QueryBatch insertBatch = getArrayAttributeInsertBatch(object, attribute);
428       Object[] keys =
429           object.getObjectType().getPrimaryKeys().stream()
430               .map(
431                   key -> {
432                     try {
433                       return key.toQueryParameter(object.getAttributes().get(key.getName()));
434                     } catch (Exception e) {
435                       throw new RuntimeException(e);
436                     }
437                   })
438               .toArray();
439       for (Object value : values) {
440         Object[] params = new Object[keys.length + 1];
441         System.arraycopy(keys, 0, params, 0, keys.length);
442         params[params.length - 1] = value;
443         insertBatch.addBatch(params);
444       }
445     }
446   }

458   public void abortWorkerThread() throws Exception {
459     if (workerThread != null) {
460       // Remove all objects already queued
461       progress.objectsToWrite.clear();
462       endOfObjects();
463     }
464   }

466   private final Runnable worker =
467       () -> {
468         try {
469           while (true) {
470             List<SchemaObjectInstance> objects = new ArrayList<>(progress.objectsToWrite.size());
471             objects.add(progress.objectsToWrite.take());
472             progress.objectsToWrite.drainTo(objects);
473             for (SchemaObjectInstance object : objects) {
474               // Check for end of objects marker
475               if (object.getObjectType() == null) {
476                 return;
477               }
478               addObjectToBatch(object, true);
479             }
480           }
481         } catch (InterruptedException e) {
482           Thread.currentThread().interrupt();
483         } catch (Throwable e) {
484           log.error("Exception in object writing thread", e);
485           setExceptionFromWorkerThread(e);
486         }
487       };

489   protected void updateProgress(Stage stage) {
490     if (this.progress != null) {

493     }

496   protected void updateProgress() {

500   private void updateProgress(boolean always) {
501     if (progressUpdater == null) {
502       return;
503     }
504     boolean timeForProgress =
505         progress.lastProgressUpdate == null
506             || (progress.objectCount % 500 == 0
507                 && Duration.between(progress.lastProgressUpdate, Instant.now()).getNano() > 250e6);
508     if (always || timeForProgress) {
509       progressUpdater.accept(progress);
510       progress.lastProgressUpdate = Instant.now();
511     }
512   }

514   private void truncateTable(Connection c, ObjectType objectType) throws SQLException {
515     // TODO like jdbc-util, truncate may fail but 'delete from' may succeed
516     new LoggingQueryRunner()
517         .execute(
518             c,
519             String.format(
520                 "truncate table %s",
521                 schemaSQLMapper.getTableNameForObjectType(objectType, tablePrefix)));
522   }

524   private String buildInsertSql(SchemaObjectInstance object) {
525     StringBuilder sql = new StringBuilder("insert into ");
526     String tableName =
527         schemaSQLMapper.getTableNameForObjectType(object.getObjectType(), tablePrefix);
528     sql.append(tableName).append("(");
529     sql.append(buildColumnList(object.getObjectType()));
530     sql.append(") values (");
531     String paramPlaceholders =
532         object.getObjectType().getDirectNonDefaultInsertAttributes().stream()
533             .map(c -> "?")
534             .collect(Collectors.joining(", "));
535     sql.append(paramPlaceholders).append(")");
536     return sql.toString();
537   }
538 
539   private String buildInsertSql(ObjectType objectType, ArrayAttributeMapping attribute) {
540     String referencingKeys =
541         objectType.getPrimaryKeys().stream()
542             .map(pk -> schemaSQLMapper.getColumnNameForObjectType(objectType, pk.getName()))
543             .collect(Collectors.joining(", "));
544     return String.format(
545         "insert into %s (%s, %s) values (%s)",
546         schemaSQLMapper.getTableNameForArrayAttribute(objectType, attribute, tablePrefix),
547         referencingKeys,
548         schemaSQLMapper.getColumnNameForObjectType(objectType, attribute.getName()),
549         String.join(", ", Collections.nCopies(objectType.getPrimaryKeys().size() + 1, "?")));
550   }
551 
552   private String buildPgCopySql(SchemaObjectInstance object, boolean initialLoad) {
553     String tableName =
554         schemaSQLMapper.getTableNameForObjectType(object.getObjectType(), tablePrefix);
555     String copySql =
556         "copy " + tableName + "(" + buildColumnList(object.getObjectType()) + ") from stdin";
557     return copySql + (initialLoad ? " with freeze" : "");
558   }
559 
560   private String buildColumnList(ObjectType objectType) {
561     return objectType.getDirectNonDefaultInsertAttributes().stream()
562         .map(column -> schemaSQLMapper.getColumnNameForObjectType(objectType, column.getName()))
563         .collect(Collectors.joining(", "));
564   }

566   private Thread workerThread = null;

568   protected void start(Progress progress) throws SQLException {
569     this.progress = progress;
570 
571     getConnection().setAutoCommit(false);
572 
573     if (multithreading) {
574       workerThread = new Thread(worker);
575       workerThread.start();
576     }
577   }
578 
579   protected void endOfObjects() throws Exception {
580     if (workerThread != null) {
581       // Signal end of objects to thread
582       progress.objectsToWrite.put(new SchemaObjectInstance(null, Collections.emptyMap()));
583       workerThread.join();
584     }
585   }
586 
587   protected void complete() throws Exception {
588     for (QueryBatch batch : progress.insertBatches.values()) {
589       batch.executeBatch();
590     }
591 
592     for (QueryBatch batch : progress.arrayAttributeInsertBatches.values()) {
593       batch.executeBatch();
594     }
595 
596     if (isCreateSchema() && isCreateKeysAndIndexes() && progress.initialLoad) {
597       for (ObjectType objectType : progress.insertBatches.keySet()) {
598         createKeys(objectType);
599       }
600       for (ObjectType objectType : progress.insertBatches.keySet()) {
601         createIndexes(objectType);
602       }
603     }
604 
605     connection.commit();
606     updateProgress(Stage.FINISHED);
607   }

609   public void createKeys(ObjectType objectType) throws Exception {
610     QueryRunner qr = new LoggingQueryRunner();
611     updateProgress(Stage.CREATE_PRIMARY_KEY);
612     // XXX why includeOneToMany is false here?
613     for (String sql :
614         schemaSQLMapper
615             .getCreatePrimaryKeyStatements(objectType, dialect, tablePrefix, false)
616             .collect(Collectors.toList())) {
617       qr.update(connection, sql);
618     }
619   }
620 
621   public void createIndexes(ObjectType objectType) throws Exception {
622     QueryRunner qr = new LoggingQueryRunner();
623     updateProgress(Stage.CREATE_GEOMETRY_INDEX);
624     // XXX why includeOneToMany is false here?
625     for (String sql :
626         schemaSQLMapper
627             .getCreateGeometryIndexStatements(objectType, dialect, tablePrefix, false)
628             .collect(Collectors.toList())) {
629       qr.update(connection, sql);
630     }
631   }
632 
633   protected void closeBatches() {
634     progress.insertBatches.values().forEach(QueryBatch::closeQuietly);
635     progress.arrayAttributeInsertBatches.values().forEach(QueryBatch::closeQuietly);
636   }
