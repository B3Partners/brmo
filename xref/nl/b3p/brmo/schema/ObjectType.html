8   package nl.b3p.brmo.schema;
9   
10  import nl.b3p.brmo.schema.mapping.AttributeColumnMapping;
11  import nl.b3p.brmo.schema.mapping.GeometryAttributeColumnMapping;
12  import nl.b3p.brmo.schema.mapping.OneToManyColumnMapping;
13  
14  import java.util.List;
15  import java.util.stream.Collectors;
16  
17  public class ObjectType {
18      private final String name;
19      private final Schema schema;
20  
21      private List<AttributeColumnMapping> attributes;
22  
23      private final List<AttributeColumnMapping> primaryKeys;
24      private final List<AttributeColumnMapping> directAttributes;
25      private final List<GeometryAttributeColumnMapping> geometryAttributes;
26      private List<ObjectType> oneToManyAttributeObjectTypes;
27  
28      protected ObjectType(Schema schema, String name, List<AttributeColumnMapping> attributes) {
29          this.schema = schema;
30          this.name = name;
31          this.attributes = attributes;
32  
33          this.primaryKeys = attributes.stream()
34                  .filter(AttributeColumnMapping::isPrimaryKey)
35                  .collect(Collectors.toList());
36          this.directAttributes = attributes.stream()
37                  .filter(attributeColumnMapping -> !(attributeColumnMapping instanceof OneToManyColumnMapping))
38                  .collect(Collectors.toList());
39          this.geometryAttributes = attributes.stream()
40                  .filter(attributeColumnMapping -> (attributeColumnMapping instanceof GeometryAttributeColumnMapping))
41                  .map(attributeColumnMapping -> (GeometryAttributeColumnMapping) attributeColumnMapping)
42                  .collect(Collectors.toList());
43      }
44  
45      public String getName() {
46          return name;
47      }
48  
49      public List<AttributeColumnMapping> getAllAttributes() {
50          return attributes;
51      }
52  
53      public List<AttributeColumnMapping> getPrimaryKeys() {
54          return primaryKeys;
55      }
56  
57      public List<AttributeColumnMapping> getDirectAttributes() {
58          return directAttributes;
59      }
60  
61      public List<ObjectType> getOneToManyAttributeObjectTypes() {
62          // Create on-demand because objectTypes map must be completely filled
63          if (oneToManyAttributeObjectTypes == null) {
64              oneToManyAttributeObjectTypes = attributes.stream()
65                      .filter(attributeColumnMapping -> (attributeColumnMapping instanceof OneToManyColumnMapping))
66                      .map(attributeColumnMapping -> schema.getObjectTypes().get(attributeColumnMapping.getName()))
67                      .collect(Collectors.toList());
68          }
69          return oneToManyAttributeObjectTypes;
70      }
71  
72      public List<GeometryAttributeColumnMapping> getGeometryAttributes() {
73          return geometryAttributes;
74      }
75  }
