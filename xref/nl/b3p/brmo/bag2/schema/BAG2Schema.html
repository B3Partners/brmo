47      private static final String WHERE_CLAUSE_ACTUEEL = "(begingeldigheid <= current_date and (eindgeldigheid is null or eindgeldigheid > current_date) and tijdstipinactief is null)";

49      private static final List<AttributeColumnMapping> bag2BaseAttributes = Arrays.asList(
50              new AttributeColumnMapping("objectid",  "serial", true, false, true),
51              new AttributeColumnMapping("identificatie",  "char(16)", true, true),
52              new IntegerAttributeColumnMapping("voorkomenidentificatie", true, true),
53              new SimpleDateFormatAttributeColumnMapping("beginGeldigheid", "date", PATTERN_XML_DATE),
54              new SimpleDateFormatAttributeColumnMapping("eindGeldigheid", "date", false, PATTERN_XML_DATE),
55              new SimpleDateFormatAttributeColumnMapping("tijdstipRegistratie", "timestamp", PATTERN_XML_TIMESTAMP),
56              new SimpleDateFormatAttributeColumnMapping("eindRegistratie", "timestamp", false, PATTERN_XML_TIMESTAMP),
57              new SimpleDateFormatAttributeColumnMapping("tijdstipInactief", "timestamp", false, PATTERN_XML_TIMESTAMP),
58              new SimpleDateFormatAttributeColumnMapping("tijdstipRegistratieLV", "timestamp", PATTERN_XML_TIMESTAMP),
59              new SimpleDateFormatAttributeColumnMapping("tijdstipEindRegistratieLV", "timestamp", false, PATTERN_XML_TIMESTAMP),
60              new SimpleDateFormatAttributeColumnMapping("tijdstipInactiefLV", "timestamp", false, PATTERN_XML_TIMESTAMP),
61              // "Niet BAG" is not supported because it would have to be part of the primary key, which would complicate
62              // things, and it is not very useful anyway. "Niet BAG" would be set for versions which were overwritten 
63              // during "synchronisation" when a bronhouder detects previous versions were not correctly registered.
64              // An update setting tijdstipNietBagLV would lead to deletion of the record and insertion of the newly
65              // synchronized versions.
66              // new SimpleDateFormatAttributeColumnMapping(TIJDSTIP_NIETBAGLV, "timestamp", true, PATTERN_XML_TIMESTAMP),
67              new SimpleDateFormatAttributeColumnMapping("documentdatum", "date", PATTERN_XML_DATE),
68              new AttributeColumnMapping("documentnummer"),
69              new BooleanAttributeColumnMapping("geconstateerd"),
70              new AttributeColumnMapping("status")
71      );
72  
73      private List<AttributeColumnMapping> withBaseAttributes(AttributeColumnMapping... attributes) {
74          return Stream.concat(bag2BaseAttributes.stream(), Stream.of(attributes)).collect(Collectors.toList());
75      }
76  
77      public BAG2Schema() {
78          super();
79  
80          // For Woonplaats, use CHAR(4) instead of CHAR(16). For Oracle we can't use CHAR(16) because then we would need
81          // to query with space-padded values to correctly delete previous versions on mutaties (issue 13151).
82          List<AttributeColumnMapping> woonplaatsAttributes = bag2BaseAttributes.stream().map(attribute -> {
83              if ("identificatie".equals(attribute.getName())) {
84                  return new AttributeColumnMapping("identificatie",  "char(4)", true, true);
85              }
86              return attribute;
87          }).collect(Collectors.toList());
88          woonplaatsAttributes.addAll(List.of(
89                  new AttributeColumnMapping("naam"),
90                  new GeometryAttributeColumnMapping("geometrie", "geometry(GEOMETRY, 28992)")
91          ));
92          addObjectType(new BAG2ObjectType(this, "Woonplaats", woonplaatsAttributes
93          ).addExtraDataDefinitionSQL(List.of(
94                  "create or replace view v_woonplaats_actueel as select * from woonplaats where " + WHERE_CLAUSE_ACTUEEL
95          )));
96  
97          addObjectType(new BAG2ObjectType(this, "OpenbareRuimte", withBaseAttributes(
98                  new AttributeColumnMapping("naam"),
99                  new AttributeColumnMapping("verkorteNaam", "varchar(255)", false),
100                 new AttributeColumnMapping("type"),
101                 new AttributeColumnMapping("ligtIn", "char(4)", false)
102         )).addExtraDataDefinitionSQL(List.of(
103                 "create or replace view v_openbareruimte_actueel as select * from openbareruimte where " + WHERE_CLAUSE_ACTUEEL
104         )));
105 
106         addObjectType(new BAG2ObjectType(this, "Nummeraanduiding", withBaseAttributes(
107                 new IntegerAttributeColumnMapping("huisnummer"),
108                 new AttributeColumnMapping("huisletter", "varchar(255)", false),
109                 new AttributeColumnMapping("huisnummertoevoeging", "varchar(255)", false),
110                 new AttributeColumnMapping("postcode", "char(6)", false),
111                 new AttributeColumnMapping("typeAdresseerbaarObject", "varchar(255)", false),
112                 new AttributeColumnMapping("ligtIn", "char(4)", false),
113                 new AttributeColumnMapping("ligtAan", "char(16)", false)
114         )).addExtraDataDefinitionSQL(List.of(
115                 "create or replace view v_nummeraanduiding_actueel as select * from nummeraanduiding where " + WHERE_CLAUSE_ACTUEEL
116         )));
117 
118         addObjectType(new BAG2ObjectType(this, "Verblijfsobject", withBaseAttributes(
119                 new ArrayAttributeMapping("gebruiksdoel", "gebruiksdoel", "varchar(255)"),
120                 new IntegerAttributeColumnMapping("oppervlakte"),
121                 new AttributeColumnMapping("heeftAlsHoofdadres", "char(16)", false),
122                 new ArrayAttributeMapping("heeftAlsNevenadres", "nevenadres", "char(16)"),
123                 new ArrayAttributeMapping("maaktDeelUitVan", "maaktdeeluitvan", "char(16)"),
124                 new GeometryAttributeColumnMapping("geometrie", "geometry(GEOMETRY, 28992)")
125         )).addExtraDataDefinitionSQL(List.of(
126                 "create or replace view v_verblijfsobject_actueel as select * from verblijfsobject where " + WHERE_CLAUSE_ACTUEEL,
127                 "create index verblijfsobject_nevenadres_idx on verblijfsobject_nevenadres (identificatie, voorkomenidentificatie)",
128                 "create index verblijfsobject_maaktdeeluitvan_idx on verblijfsobject_maaktdeeluitvan (identificatie, voorkomenidentificatie)",
129                 "create index verblijfsobject_gebruiksdoel_idx on verblijfsobject_gebruiksdoel (identificatie, voorkomenidentificatie)"
130         )));
131 
132         addObjectType(new BAG2ObjectType(this, "Ligplaats", withBaseAttributes(
133                 new GeometryAttributeColumnMapping("geometrie", "geometry(POLYGON, 28992)"),
134                 new AttributeColumnMapping("heeftAlsHoofdadres", "char(16)", false),
135                 new ArrayAttributeMapping("heeftAlsNevenadres", "nevenadres", "char(16)")
136         )).addExtraDataDefinitionSQL(List.of(
137                 "create or replace view v_ligplaats_actueel as select * from ligplaats where " + WHERE_CLAUSE_ACTUEEL,
138                 "create index ligplaats_nevenadres_idx on ligplaats_nevenadres (identificatie, voorkomenidentificatie)"
139         )));
140 
141         addObjectType(new BAG2ObjectType(this, "Standplaats", withBaseAttributes(
142                 new GeometryAttributeColumnMapping("geometrie", "geometry(POLYGON, 28992)"),
143                 new AttributeColumnMapping("heeftAlsHoofdadres", "char(16)", false),
144                 new ArrayAttributeMapping("heeftAlsNevenadres", "nevenadres", "char(16)")
145         )).addExtraDataDefinitionSQL(List.of(
146                 "create or replace view v_standplaats_actueel as select * from standplaats where " + WHERE_CLAUSE_ACTUEEL,
147                 "create index standplaats_nevenadres_idx on standplaats_nevenadres (identificatie, voorkomenidentificatie)"
148         )));
149 
150         addObjectType(new BAG2ObjectType(this, "Pand", withBaseAttributes(
151                 new IntegerAttributeColumnMapping("oorspronkelijkBouwjaar"),
152                 new GeometryAttributeColumnMapping("geometrie", "geometry(POLYGON, 28992)")
153         )).addExtraDataDefinitionSQL(List.of(
154                 "create or replace view v_pand_actueel as select * from pand where " + WHERE_CLAUSE_ACTUEEL
155         )));
156     }

158     @Override
159     protected void addObjectType(ObjectType objectType) {
160         // Always add a unique index for the OBJECTID column, which is not the primary key but only for ArcGIS
161 
162         // XXX table prefixes not supported here and also not in extra DDL above
163 
164         String tableName = objectType.getName().toLowerCase();
165         String sql = String.format("create index %s_objectid_idx on %s (objectid)", tableName, tableName);
166         objectType.addExtraDataDefinitionSQL(List.of(sql));
167         super.addObjectType(objectType);
168     }
