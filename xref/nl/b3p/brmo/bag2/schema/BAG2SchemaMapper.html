8   package nl.b3p.brmo.bag2.schema;
9   
10  import nl.b3p.brmo.bag2.loader.BAG2LoaderUtils;
11  import nl.b3p.brmo.schema.SchemaSQLMapper;
12  import nl.b3p.brmo.sql.dialect.SQLDialect;
13  
14  import java.util.List;
15  import java.util.Map;
16  import java.util.stream.Collectors;
17  import java.util.stream.Stream;
18  
19  public class BAG2SchemaMapper extends SchemaSQLMapper {
20      private static BAG2SchemaMapper instance;
21  
22      private static final String SCHEMA_VERSION_VALUE = "1";
23  
24      public static final String METADATA_TABLE_NAME = "brmo_metadata";
25  
26      public enum Metadata {
27          SCHEMA_VERSION,
28          LOADER_VERSION,
29          STAND_LOAD_TIME,
30          STAND_LOAD_TECHNISCHE_DATUM,
31          CURRENT_TECHNISCHE_DATUM,
32          GEMEENTE_CODES,
33          FILTER_MUTATIES_WOONPLAATS;
34  
35          public String getDbKey() {
36              return "bag2_" + this.name().toLowerCase();
37          }
38      }
39  
40      public BAG2SchemaMapper() {
41          super(BAG2Schema.getInstance());
42      }
43  
44      public static BAG2SchemaMapper getInstance() {
45          if (instance == null) {
46              instance = new BAG2SchemaMapper();
47          }
48          return instance;
49      }
50  
51      @Override
52      public String getMetadataTableName() {
53          return METADATA_TABLE_NAME;
54      }
55  
56      public List<String> getCreateMetadataTableStatements(SQLDialect dialect, String tablePrefix, boolean dropIfExists) {
57          List<String> statements = super.getCreateMetadataTableStatements(dialect, tablePrefix, dropIfExists);
58  
59          Map<BAG2SchemaMapper.Metadata, String> defaultMetadata = Stream.of(new Object[][]{
60                  {BAG2SchemaMapper.Metadata.SCHEMA_VERSION, SCHEMA_VERSION_VALUE},
61                  {BAG2SchemaMapper.Metadata.LOADER_VERSION, BAG2LoaderUtils.getLoaderVersion()},
62          }).collect(Collectors.toMap(entry -> (BAG2SchemaMapper.Metadata) entry[0], entry -> (String) entry[1]));
63  
64          Stream.of(BAG2SchemaMapper.Metadata.values()).forEach(metadata -> {
65              String value = defaultMetadata.get(metadata);
66              statements.add(String.format("insert into %s (naam, waarde) values ('%s', %s)",
67                      getMetadataTableName(),
68                      metadata.getDbKey(),
69                      value == null ? "null" : "'" + value + "'"));
70          });
71  
72          return statements;
73      }
74  }
