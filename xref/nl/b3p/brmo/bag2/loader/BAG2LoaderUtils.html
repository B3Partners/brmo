74    public static BAGExtractSelectie getBAGExtractSelectieFromZip(String zipFileName)
75        throws IOException {
76      if (zipFileName.startsWith("http://") || zipFileName.startsWith("https://")) {
77        // Try to parse filename in URL to avoid having to download the ZIP and extract the
78        // leveringsdocument --
79        // because this may happen many times to check whether a mutatie ZIP is applicable
80        try {
81          return BAGExtractSelectieFromFilename.parse(zipFileName);
82        } catch (IllegalArgumentException iae) {
83          // Could not parse BAG2 extract selection from filename, use HTTP random access to
84          // parse only
85          // leveringsdocument without downloading/streaming the entire ZIP file
86          return getBAGExtractSelectieFromHttpZip(URI.create(zipFileName));
87        }
88      } else {
89        return getBAGExtractSelectieFromZipFile(new File(zipFileName));
90      }
91    }

93    private static BAGExtractSelectie getBAGExtractSelectieFromHttpZip(URI uri) throws IOException {
94      String message = getMessageFormattedString("load.leveringsdocument.readzip", uri);
95      try (HttpSeekableByteChannel channel = new HttpSeekableByteChannel(uri);
96          org.apache.commons.compress.archivers.zip.ZipFile zipFile =
97              new org.apache.commons.compress.archivers.zip.ZipFile(
98                  channel, uri.toString(), "UTF8", false, true)) {
99        message = getMessageFormattedString("load.leveringsdocument.zipentries", uri);
100       Iterator<ZipArchiveEntry> iterator =
101           zipFile.getEntries(LEVERINGSDOCUMENT_FILENAME).iterator();
102       if (!iterator.hasNext()) {
103         throw new IOException(getMessageFormattedString("load.leveringsdocument.notfound", uri));
104       }
105       message = getMessageFormattedString("load.leveringsdocument.unmarshal", uri);
106       JAXBContext jaxbContext = JAXBContext.newInstance(BAGExtractLevering.class);
107       Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
108       try (InputStream in = zipFile.getInputStream(iterator.next())) {
109         BAGExtractLevering levering = (BAGExtractLevering) unmarshaller.unmarshal(in);
110         return new BAGExtractSelectieFromLeveringsdocument(levering);
111       }
112     } catch (Exception e) {
113       throw new IOException(message, e);
114     }
115   }

117   public static BAGExtractSelectie getBAGExtractSelectieFromZipFile(File zipFile)
118       throws IOException {
119     String zipFileName = zipFile.getName();
120     String message = getMessageFormattedString("load.leveringsdocument.readzip", zipFileName);
121     try (ZipFile zf = new ZipFile(zipFile)) {
122       message = getMessageFormattedString("load.leveringsdocument.zipentries", zipFileName);
123       Enumeration<? extends ZipEntry> entries = zf.entries();
124       while (entries.hasMoreElements()) {
125         ZipEntry entry = entries.nextElement();
126         if (LEVERINGSDOCUMENT_FILENAME.equals(entry.getName())) {
127           message = getMessageFormattedString("load.leveringsdocument.unmarshal", zipFileName);
128           JAXBContext jaxbContext = JAXBContext.newInstance(BAGExtractLevering.class);
129           Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
130           try (InputStream in = zf.getInputStream(entry)) {
131             BAGExtractLevering levering = (BAGExtractLevering) unmarshaller.unmarshal(in);
132             return new BAGExtractSelectieFromLeveringsdocument(levering);
133           }
134         }
135       }
136     } catch (Exception e) {
137       throw new IOException(message, e);
138     }
139     throw new IOException(
140         getMessageFormattedString("load.leveringsdocument.notfound", zipFileName));
141   }

155   public static class BAGExtractSelectieFromFilename implements BAGExtractSelectie {
156     private String name;
157     private boolean isStand;
158     private boolean isGebiedNLD;
159     private String gemeenteCode;
160     private LocalDate mutatiesFrom;
161     private LocalDate mutatiesTot;
162 
163     /**
164      * @param name BAG2 filename to parse, can be URL or full path.
165      * @throws IllegalArgumentException If the filename is not a BAG2 filename (outer zip only)
166      */
167     public static BAGExtractSelectie parse(String name) throws IllegalArgumentException {
168 
169       if (name.startsWith("http://") || name.startsWith("https://")) {
170         URI uri = URI.create(name);
171         String[] parts = uri.getPath().split("/");
172         name = parts[parts.length - 1];
173       }
174       if (name.contains(File.separator)) {
175         Path p = Path.of(name);
176         name = p.getFileName().toString();
177       }
178 
179       BAGExtractSelectieFromFilename selectie = new BAGExtractSelectieFromFilename();
180       selectie.name = name;
181 
182       if (name.equals("lvbag-extract-nl.zip") || name.matches("BAGNLDL-\\d{8}\\.zip")) {
183         selectie.isStand = true;
184         selectie.isGebiedNLD = true;
185         return selectie;
186       }
187 
188       if (name.matches("BAGGEM\\d{4}L-\\d{8}\\.zip")) {
189         selectie.isStand = true;
190         selectie.isGebiedNLD = false;
191         selectie.gemeenteCode = name.substring(6, 10);
192         return selectie;
193       }
194 
195       DateTimeFormatter dtf = DateTimeFormatter.ofPattern("ddMMyyyy");
196 
197       if (name.matches("BAGGEM\\d{4}M-\\d{8}-\\d{8}\\.zip")) {
198         selectie.isStand = false;
199         selectie.isGebiedNLD = false;
200         selectie.gemeenteCode = name.substring(6, 10);
201         // Gemeente can only have monthly updates
202         selectie.mutatiesFrom = LocalDate.parse(name.substring(12, 20), dtf);
203         selectie.mutatiesTot = LocalDate.parse(name.substring(21, 29), dtf);
204         return selectie;
205       }
206 
207       if (name.matches("BAGNLDM-\\d{8}-\\d{8}\\.zip")) {
208         selectie.isStand = false;
209         selectie.isGebiedNLD = true;
210         selectie.mutatiesFrom = LocalDate.parse(name.substring(8, 16), dtf);
211         selectie.mutatiesTot = LocalDate.parse(name.substring(17, 25), dtf);
212         return selectie;
213       }
214 
215       throw new IllegalArgumentException("Ongeldige BAG2 bestandsnaam: " + name);
216     }
217 
218     public String getName() {
219       return name;
220     }
221 
222     @Override
223     public boolean isStand() {
224       return isStand;
225     }
226 
227     @Override
228     public boolean isGebiedNLD() {
229       return isGebiedNLD;
230     }
231 
232     @Override
233     public Set<String> getGemeenteCodes() {
234       return Collections.singleton(gemeenteCode);
235     }
236 
237     @Override
238     public LocalDate getMutatiesFrom() {
239       return mutatiesFrom;
240     }
241 
242     @Override
243     public LocalDate getMutatiesTot() {
244       return mutatiesTot;
245     }

248   public static class BAGExtractSelectieFromLeveringsdocument implements BAGExtractSelectie {
249     BAGExtractLevering levering;
250 
251     private BAGExtractSelectieFromLeveringsdocument(BAGExtractLevering levering) {
252       this.levering = levering;
253     }
254 
255     @Override
256     public boolean isStand() {
257       return levering.getSelectieGegevens().getLVCExtract() != null;
258     }
259 
260     @Override
261     public boolean isGebiedNLD() {
262       return levering.getSelectieGegevens().getGebiedRegistratief().getGebiedNLD() != null;
263     }
264 
265     @Override
266     public LocalDate getMutatiesFrom() {
267       XMLGregorianCalendar xmlGregorianCalendar =
268           levering.getSelectieGegevens().getMUTExtract().getMutatieperiode().getMutatiedatumVanaf();
269       return LocalDate.of(
270           xmlGregorianCalendar.getYear(),
271           xmlGregorianCalendar.getMonth(),
272           xmlGregorianCalendar.getDay());
273     }
274 
275     @Override
276     public LocalDate getMutatiesTot() {
277       XMLGregorianCalendar xmlGregorianCalendar =
278           levering.getSelectieGegevens().getMUTExtract().getMutatieperiode().getMutatiedatumTot();
279       return LocalDate.of(
280           xmlGregorianCalendar.getYear(),
281           xmlGregorianCalendar.getMonth(),
282           xmlGregorianCalendar.getDay());
283     }
284 
285     @Override
286     public Set<String> getGemeenteCodes() {
287       return levering
288           .getSelectieGegevens()
289           .getGebiedRegistratief()
290           .getGebiedGEM()
291           .getGemeenteCollectie()
292           .getGemeente()
293           .stream()
294           .map(Gemeente::getGemeenteIdentificatie)
295           .collect(Collectors.toSet());
296     }
