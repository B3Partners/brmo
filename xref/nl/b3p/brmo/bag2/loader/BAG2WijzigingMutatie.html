8   package nl.b3p.brmo.bag2.loader;
9   
10  import nl.b3p.brmo.bag2.schema.BAG2Object;
11  
12  import javax.xml.stream.Location;
13  
14  public class BAG2WijzigingMutatie extends BAG2Mutatie {
15      private BAG2Object was;
16      private BAG2Object wordt;
17  
18      BAG2WijzigingMutatie(Location location, BAG2Object was, BAG2Object wordt) {
19          super(location);
20          this.was = was;
21          this.wordt = wordt;
22      }
23  
24      public BAG2Object getWas() {
25          return was;
26      }
27  
28      public BAG2Object getWordt() {
29          return wordt;
30      }
31  
32      @Override
33      public String toString() {
34          return "BAG2WijzigingMutatie{" +
35                  "location=" + getLocation() +
36                  ", was=" + was +
37                  ", wordt=" + wordt +
38                  '}';
39      }
40  }
