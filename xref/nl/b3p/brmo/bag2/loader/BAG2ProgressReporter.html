8   package nl.b3p.brmo.bag2.loader;
9   
10  import nl.b3p.brmo.bag2.schema.BAG2ObjectTableWriter;
11  import nl.b3p.brmo.schema.ObjectTableWriter;
12  import org.apache.commons.logging.Log;
13  import org.apache.commons.logging.LogFactory;
14  
15  import java.time.Instant;
16  import java.util.function.Consumer;
17  
18  import static nl.b3p.brmo.bag2.loader.BAG2LoaderUtils.getMessageFormattedString;
19  import static nl.b3p.brmo.bgt.loader.Utils.formatTimeSince;
20  
21  public class BAG2ProgressReporter implements Consumer<ObjectTableWriter.Progress> {
22      private static final Log log = LogFactory.getLog(BAG2ProgressReporter.class);
23  
24      private final Instant start = Instant.now();
25  
26      protected String currentFileName;
27      protected Instant currentFileStart;
28  
29      protected int splitFileCount;
30  
31      protected void log(String msg) {
32          log.info(msg);
33      }
34  
35      protected void status(String msg) {
36          log.debug(msg);
37      }
38  
39      public void startNewFile(String name) {
40          this.currentFileName = name;
41          this.currentFileStart = Instant.now();
42          this.splitFileCount = 0;
43      }
44  
45      public void startNextSplitFile(String entry) {
46          this.splitFileCount++;
47      }
48  
49      @Override
50      public void accept(ObjectTableWriter.Progress progress) {
51          BAG2ObjectTableWriter.BAG2Progress bag2Progress = (BAG2ObjectTableWriter.BAG2Progress) progress;
52  
53          switch(progress.getStage()) {
54              case LOAD_OBJECTS:
55                  if (progress.getObjectCount() == 0) {
56                      log(getMessageFormattedString("progress.loading", currentFileName));
57                  }
58                  break;
59              case CREATE_PRIMARY_KEY:
60                  log(getMessageFormattedString("progress.create_primary_key", bag2Progress.getCurrentObjectType().getName()));
61                  break;
62              case CREATE_GEOMETRY_INDEX:
63                  log(getMessageFormattedString("progress.create_geometry_indexes", bag2Progress.getCurrentObjectType().getName()));
64                  break;
65              case FINISHED:
66                  log(getMessageFormattedString("progress.finished_file", currentFileName, splitFileCount, bag2Progress.getObjectCount(), formatTimeSince(currentFileStart)));
67                  //log(String.format("%s: loaded %,d files, %,d total objects in %s", currentFileName, splitFileCount, bag2Progress.getObjectCount(), formatTimeSince(currentFileStart)));
68                  break;
69          }
70      }
71  
72      public void reportTotalSummary() {
73          log(getMessageFormattedString("progress.finished", formatTimeSince(start)));
74      }
75  }
