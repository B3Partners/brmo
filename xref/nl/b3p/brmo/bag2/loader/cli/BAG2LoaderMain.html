113     @Command(name = "load", sortOptions = false)
114     public int load(
115             @Mixin BAG2DatabaseOptions dbOptions,
116             @Mixin BAG2LoadOptions loadOptions,
117             @Mixin BAG2ProgressOptions progressOptions,
118             @Parameters(paramLabel = "<file>") String[] filenames,
119             @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp) throws Exception {
120 
121         log.info(BAG2LoaderUtils.getUserAgent());
122 
123         try(BAG2Database db = getBAG2Database(dbOptions)) {
124             BAG2ProgressReporter progressReporter = progressOptions.isConsoleProgressEnabled()

126                     : new BAG2ProgressReporter();
127 
128             loadFiles(db, dbOptions, loadOptions, progressReporter, filenames, null);
129             return ExitCode.OK;
130         }
131     }

144     public void loadFiles(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, CookieManager cookieManager) throws Exception {
145 
146         if (filenames.length == 1 && Files.isDirectory(Path.of(filenames[0]))) {
147             log.info("Directory opgegeven, kijken naar toepasbare mutaties...");
148             filenames = Files.list(Path.of(filenames[0]))
149                     .filter(p -> !Files.isDirectory(p) && p.getFileName().toString().endsWith(".zip"))
150                     .map(p -> p.toAbsolutePath().toString())
151                     .toArray(String[]::new);
152             if (filenames.length == 0) {
153                 log.info("Geen ZIP bestanden gevonden, niets te doen");
154             } else {
155                 applyMutaties(db, dbOptions, loadOptions, progressReporter, filenames, null);
156             }
157             return;
158         }
159 
160         BAG2LoaderUtils.BAGExtractSelectie bagExtractLevering = BAG2LoaderUtils.getBAGExtractSelectieFromZip(filenames[0]);
161 
162         if (bagExtractLevering.isStand()) {
163             if (bagExtractLevering.isGebiedNLD()) {
164                 if (filenames.length > 1) {
165                     throw new IllegalArgumentException("Inladen stand heel Nederland: teveel bestanden opgegeven");
166                 }
167             } else {
168                 // Verify all filenames are gemeentestanden
169                 Set<String> gemeenteCodes = new HashSet<>();
170                 for (int i = 1; i < filenames.length; i++) {
171                     BAG2LoaderUtils.BAGExtractSelectie nextBagExtractLevering = BAG2LoaderUtils.getBAGExtractSelectieFromZip(filenames[i]);
172 
173                     if (!nextBagExtractLevering.isStand() || nextBagExtractLevering.isGebiedNLD()) {
174                         throw new IllegalArgumentException("Inladen stand gemeentes, ongeldig bestand opgegeven (geen gemeentestand): " + filenames[i]);
175                     }
176                     Set<String> nextBagExtractLeveringGemeenteCodes = nextBagExtractLevering.getGemeenteCodes();
177                     if (gemeenteCodes.stream().anyMatch(nextBagExtractLeveringGemeenteCodes::contains)) {
178                         throw new IllegalArgumentException("Inladen stand gemeentes, dubbele gemeentecode in bestand: " + filenames[i]);
179                     }
180                     gemeenteCodes.addAll(nextBagExtractLeveringGemeenteCodes);
181                 }
182             }
183 
184             loadStandFiles(db, dbOptions, loadOptions, progressReporter, filenames, cookieManager);
185         } else {
186             // Process mutaties while ignoring files not applicable
187             applyMutaties(db, dbOptions, loadOptions, progressReporter, filenames, null);
188         }
189     }

191     /**
192      * Only called after list of files have been checked to only have been entire NL stand or unique gemeente standen.
193      */
194     private void loadStandFiles(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, CookieManager cookieManager) throws Exception {
195         try {
196             // When loading multiple standen (for gemeentes), set ignore duplicates so the seen object keys are kept in
197             // memory so duplicates can be ignored. Don't keep keys in memory for entire NL stand.
198             loadOptions.setIgnoreDuplicates(filenames.length > 1);
199 
200             // Multiple gemeentes can also be provided in a single ZIP file, check the Leveringsdocument whether that is
201             // the case
202             if (filenames.length == 1) {
203                 BAG2LoaderUtils.BAGExtractSelectie levering = BAG2LoaderUtils.getBAGExtractSelectieFromZip(filenames[0]);
204                 if (!levering.isGebiedNLD()) {
205                     loadOptions.setIgnoreDuplicates(levering.getGemeenteCodes().size() > 1);
206                 }
207             }
208 
209             BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
210             String lastFilename = null;
211 
212             // Keep track of which gemeentes are loaded so the correct mutations can be processed later
213             Set<String> gemeenteIdentificaties = new HashSet<>();
214 
215             for (String filename: filenames) {
216                 BAG2GMLMutatieGroepStream.BagInfo latestBagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, filename);
217                 if (bagInfo != null) {
218                     // For gemeentes the BagInfo must be the same so the standen are of the same date
219                     if (!latestBagInfo.equalsExceptGemeenteIdentificaties(bagInfo)) {
220                         throw new IllegalArgumentException(String.format("Incompatible BagInfo for file \"%s\" (%s) compared to last file \"%s\" (%s)",
221                                 filename,
222                                 latestBagInfo,
223                                 lastFilename,
224                                 bagInfo));
225                     }
226                 }
227                 bagInfo = latestBagInfo;
228 
229                 // For NL stand this will be "9999"
230                 gemeenteIdentificaties.addAll(bagInfo.getGemeenteIdentificaties());
231                 lastFilename = filename;
232             }
233             if (bagInfo != null) {
234                 // TODO: when loading gemeente without rare objects such as ligplaatsen/standplaatsen, table will not be created
235                 // and a future change with such an object will fail. Should create entire schema up-front instead of when first
236                 // encountering object type
237                 createKeysAndIndexes(db, loadOptions, dbOptions, progressReporter);
238 
239                 updateMetadata(db, loadOptions, true, gemeenteIdentificaties, bagInfo.getStandTechnischeDatum());
240             }
241 
242             db.getConnection().commit();
243         } finally {
244             progressReporter.reportTotalSummary();
245         }
246     }

248     public void applyMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] urls, CookieManager cookieManager) throws Exception {
249         if (urls.length == 0) {
250             return;
251         }
252         BAG2LoaderUtils.BAGExtractSelectie bagExtractLevering = BAG2LoaderUtils.getBAGExtractSelectieFromZip(urls[0]);
253         if (bagExtractLevering.isGebiedNLD()) {
254             applyNLMutaties(db, dbOptions, loadOptions, progressReporter, urls, cookieManager);
255         } else {
256             applyGemeenteMutaties(db, dbOptions, loadOptions, progressReporter, urls, cookieManager);
257         }
258     }

260     private void applyGemeenteMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] urls, CookieManager cookieManager) throws Exception {
261         LocalDate currentTechnischeDatum = db.getCurrentTechnischeDatum();
262         Set<String> gemeenteCodes = db.getGemeenteCodes();
263 
264         Set<Integer> applicableMutatieIndexes;
265         do {
266             applicableMutatieIndexes = new HashSet<>();
267 
268             Set<String> missingGemeentes = new HashSet<>(gemeenteCodes);
269             for(int i = 0; i < urls.length; i++) {
270                 BAG2LoaderUtils.BAGExtractSelectie bagExtractLevering = BAG2LoaderUtils.getBAGExtractSelectieFromZip(urls[i]);
271 
272                 if (!bagExtractLevering.isGebiedNLD()
273                         && !bagExtractLevering.isStand()
274                         && bagExtractLevering.getMutatiesFrom().equals(currentTechnischeDatum)) {
275 
276                     for(String gemeenteCode: bagExtractLevering.getGemeenteCodes()) {
277                         if (gemeenteCodes.contains(gemeenteCode)) {
278                             applicableMutatieIndexes.add(i);
279                             missingGemeentes.remove(gemeenteCode);
280                         }
281                     }
282                 }
283             }
284 
285             if (applicableMutatieIndexes.isEmpty()) {
286                 log.info(String.format("Geen nieuw toe te passen gemeentemutatiebestanden gevonden voor huidige stand technische datum %s, klaar", currentTechnischeDatum));
287                 break;
288             }
289 
290             // Check whether applicable mutaties are available for all gemeentecodes because they need to be processed
291             // at the same time to ignore duplicates
292             if (!missingGemeentes.isEmpty()) {
293                 throw new IllegalArgumentException(String.format("Kan geen gemeente mutaties toepassen voor gemeentes %s vanaf stand technische datum %s, in opgegeven mutatiebestanden ontbreken gemeentecodes %s",
294                         gemeenteCodes,
295                         currentTechnischeDatum,
296                         missingGemeentes));
297             }
298 
299             log.info(String.format("Toepassen gemeentemutaties voor %d gemeentes vanaf stand technische datum %s...", gemeenteCodes.size(), currentTechnischeDatum));
300 
301             BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
302 
303             loadOptions.setIgnoreDuplicates(gemeenteCodes.size() > 1);
304             for (int index: applicableMutatieIndexes) {
305                 bagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, urls[index], cookieManager);
306             }
307             currentTechnischeDatum = new java.sql.Date(bagInfo.getStandTechnischeDatum().getTime()).toLocalDate();
308             updateMetadata(db, loadOptions, false, null, bagInfo.getStandTechnischeDatum());
309             db.getConnection().commit();
310             // Duplicates need only be checked for mutaties for a single from date, clear cache to reduce memory usage
311             clearDuplicatesCache();
312             log.info("Mutaties verwerkt, huidige stand technische datum: " + currentTechnischeDatum);
313 
314         } while(true);
315     }

317     private void applyNLMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] urls, CookieManager cookieManager) throws Exception {
318         LocalDate currentTechnischeDatum = db.getCurrentTechnischeDatum();
319         do {
320             String applicatieMutatieURL = null;
321 
322             for (String url: urls) {
323                 BAG2LoaderUtils.BAGExtractSelectie bagExtractSelectie = BAG2LoaderUtils.getBAGExtractSelectieFromZip(url);
324 
325                 if (bagExtractSelectie.isGebiedNLD()
326                         && !bagExtractSelectie.isStand()
327                         && bagExtractSelectie.getMutatiesFrom().equals(currentTechnischeDatum)) {
328                     applicatieMutatieURL = url;
329                 }
330             }
331 
332             if (applicatieMutatieURL == null) {
333                 log.info(String.format("Geen nieuw toe te passen mutatiebestanden gevonden voor huidige stand technische datum %s, klaar", currentTechnischeDatum));
334                 break;
335             }
336 
337             log.info(String.format("Toepassen mutaties vanaf stand technische datum %s...", currentTechnischeDatum));
338 
339             BAG2GMLMutatieGroepStream.BagInfo bagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, applicatieMutatieURL, cookieManager);
340             currentTechnischeDatum = new java.sql.Date(bagInfo.getStandTechnischeDatum().getTime()).toLocalDate();
341             updateMetadata(db, loadOptions, false, null, bagInfo.getStandTechnischeDatum());
342             db.getConnection().commit();
343             log.info("Mutaties verwerkt, huidige stand technische datum: " + currentTechnischeDatum);
344         } while(true);
345     }

347     private void createKeysAndIndexes(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions databaseOptions, BAG2ProgressReporter progressReporter) throws Exception {
348         BAG2ObjectTableWriter writer = db.createObjectTableWriter(loadOptions, databaseOptions);
349         writer.setProgressUpdater(progressReporter);
350         for(BAG2ObjectType objectType: objectTypesWithSchemaCreated) {
351             writer.createKeys(objectType); // BAG2 writer is always a single ObjectType unlike BGT
352             writer.createIndexes(objectType);
353         }
354     }

360     private BAG2GMLMutatieGroepStream.BagInfo loadBAG2ExtractFromURLorFile(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions dbOptions, BAG2ProgressReporter progressReporter, String url, CookieManager cookieManager) throws Exception {
361         HttpClientWrapper<HttpRequest.Builder, HttpResponse<InputStream>> httpClientWrapper = cookieManager == null
362                 ? new Java11HttpClientWrapper()
363                 : new Java11HttpClientWrapper(HttpClient.newBuilder().cookieHandler(cookieManager));
364 
365         if (url.startsWith("http://") || url.startsWith("https://")) {
366             try (InputStream in = new ResumingInputStream(new HttpStartRangeInputStreamProvider(URI.create(url), httpClientWrapper))) {
367                 return loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, url, in);
368             }
369         }
370         if (url.endsWith(".zip")) {
371             try (InputStream in = new FileInputStream(url)) {
372                 return loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, url, in);
373             }
374         }
375 
376         throw new IllegalArgumentException(getMessageFormattedString("load.invalid_file", url));
377     }

379     private BAG2GMLMutatieGroepStream.BagInfo loadBAG2ExtractFromStream(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions dbOptions, BAG2ProgressReporter progressReporter, String name, InputStream input) throws Exception {
380         BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
381         Set<String> gemeenteIdentificaties = new HashSet<>();
382         try (ZipArchiveInputStream zip = new ZipArchiveInputStream(input)) {
383             ZipArchiveEntry entry = zip.getNextZipEntry();
384             while(entry != null) {
385                 if (entry.getName().matches("[0-9]{4}(STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.xml")) {
386                     // Load extracted zipfile
387                     bagInfo = loadXmlEntriesFromZipFile(db, loadOptions, dbOptions, progressReporter, name, zip, entry);
388                     break;
389                 }
390 
391                 if (entry.getName().matches("[0-9]{4}GEM[0-9]{8}\\.zip")) {
392                     bagInfo = loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, name, CloseShieldInputStream.wrap(zip));
393                 }
394 
395                 // Process single and double-nested ZIP files
396 
397                 if (entry.getName().matches("[0-9]{4}(STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.zip")
398                 || entry.getName().matches("[0-9]{4}MUT[0-9]{8}-[0-9]{8}\\.zip")) {
399                     ZipArchiveInputStream nestedZip = new ZipArchiveInputStream(zip);
400                     bagInfo = loadXmlEntriesFromZipFile(db, loadOptions, dbOptions, progressReporter, entry.getName(), nestedZip, nestedZip.getNextZipEntry());
401                 }
402 
403                 if (entry.getName().matches("[0-9]{4}Inactief.*\\.zip")) {
404                     ZipArchiveInputStream nestedZip = new ZipArchiveInputStream(zip);
405                     ZipArchiveEntry nestedEntry = nestedZip.getNextZipEntry();
406                     while(nestedEntry != null) {
407                         if (nestedEntry.getName().matches("[0-9]{4}IA.*\\.zip")) {
408                             ZipArchiveInputStream moreNestedZip = new ZipArchiveInputStream(nestedZip);
409                             bagInfo = loadXmlEntriesFromZipFile(db,  loadOptions, dbOptions, progressReporter, nestedEntry.getName(), moreNestedZip, moreNestedZip.getNextZipEntry());
410                         }
411                         nestedEntry = nestedZip.getNextZipEntry();
412                     }
413                 }
414 
415                 if (bagInfo != null) {
416                     gemeenteIdentificaties.addAll(bagInfo.getGemeenteIdentificaties());
417                 }
418 
419                 try {
420                     entry = zip.getNextZipEntry();
421                 } catch(IOException e) {
422                     // Reading the ZIP from HTTP may give this error, but it is a normal end...
423                     if ("Truncated ZIP file".equals(e.getMessage())) {
424                         break;
425                     }
426                 }
427             }
428         }
429         if (bagInfo != null) {
430             // Update BagInfo with all seen gemeenteIdentificaties in case we processed a ZIP with multiple gemeentestanden
431             bagInfo.setGemeenteIdentificaties(gemeenteIdentificaties);
432         }
433         return bagInfo;
434     }

436     private static BAG2ObjectType getObjectTypeFromFilename(String filename) {
437         Matcher m = Pattern.compile(".*[0-9]{4}(IA)?(MUT|STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.(xml|zip)").matcher(filename);
438         if (!m.matches()) {
439             throw new IllegalArgumentException("Invalid BAG2 filename: " + filename);
440         }
441         String objectTypeName = null;
442         switch(m.group(2)) {
443             case "MUT": break;
444             case "STA": objectTypeName = "Standplaats"; break;
445             case "OPR": objectTypeName = "OpenbareRuimte"; break;
446             case "VBO": objectTypeName = "Verblijfsobject"; break;
447             case "NUM": objectTypeName = "Nummeraanduiding"; break;
448             case "LIG": objectTypeName = "Ligplaats"; break;
449             case "PND": objectTypeName = "Pand"; break;
450             case "WPL": objectTypeName = "Woonplaats"; break;
451         }
452         if (objectTypeName == null) {
453             return null;
454         } else {
455             return BAG2Schema.getInstance().getObjectTypeByName(objectTypeName);
456         }
457     }

459     private void updateMetadata(BAG2Database db, BAG2LoadOptions loadOptions, boolean stand, Set<String> gemeenteIdentificaties, Date standTechnischeDatum) throws Exception {
460         // Check if metadata table already exists. For PostgreSQL we can use the metadata table in the public schema
461         if (!db.getDialect().tableExists(db.getConnection(), METADATA_TABLE_NAME)) {
462             // Create a new metadata table, for Oracle as BAG is in separate schema, for PostgreSQL if loading BAG
463             // into a non-brmo RSGB database
464             db.createMetadataTable(loadOptions);
465         }
466 
467         db.setMetadataValue(BAG2SchemaMapper.Metadata.LOADER_VERSION, BAG2LoaderUtils.getLoaderVersion());
468         SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
469         if (stand) {
470             db.setMetadataValue(STAND_LOAD_TIME, new SimpleDateFormat("yyyy-MM-dd HH:mm:ss").format(new Date()));
471             db.setMetadataValue(STAND_LOAD_TECHNISCHE_DATUM, df.format(standTechnischeDatum));
472             db.setMetadataValue(GEMEENTE_CODES, String.join(",", gemeenteIdentificaties));
473             db.setMetadataValue(FILTER_MUTATIES_WOONPLAATS, "false");
474         }
475         db.setMetadataValue(CURRENT_TECHNISCHE_DATUM, df.format(standTechnischeDatum));
476     }

482     private BAG2GMLMutatieGroepStream.BagInfo loadXmlEntriesFromZipFile(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions databaseOptions, BAG2ProgressReporter progressReporter, String name, ZipArchiveInputStream zip, ZipArchiveEntry entry) throws Exception {
483         BAG2ObjectType objectType = getObjectTypeFromFilename(name);
484         // objectType is null for mutaties, which contain mixed object types instead of a single object type with stand
485         boolean schemaCreated = objectType == null || objectTypesWithSchemaCreated.contains(objectType);
486         BAG2ObjectTableWriter writer = db.createObjectTableWriter(loadOptions, databaseOptions);
487         writer.setProgressUpdater(progressReporter);
488         writer.setCreateSchema(!schemaCreated);
489         writer.setCreateKeysAndIndexes(false);
490         writer.setKeysPerObjectType(keysPerObjectType);
491         writer.start(); // sets InitialLoad to true
492         writer.getProgress().setInitialLoad(!schemaCreated); // For a COPY in transaction, table must be created or truncated in it
493         if (objectType == null) {
494             // When processing mutaties, set batch size to 1 so all mutaties are processed sequentially and can not
495             // conflict with deleting and inserting of old/new versions
496             writer.setBatchSize(1);
497             // Disable multithreading so deletion of previous versions and new inserts are processed sequentially
498             writer.setMultithreading(false);
499         }
500         progressReporter.startNewFile(name);
501         try {
502             while(entry != null) {
503                 progressReporter.startNextSplitFile(entry.getName());
504                 writer.write(CloseShieldInputStream.wrap(zip));
505                 if (loadOptions.getMaxObjects() != null && writer.getProgress().getObjectCount() == loadOptions.getMaxObjects()) {
506                     break;
507                 }
508                 entry = zip.getNextZipEntry();
509             }
510             writer.complete();
511 
512             if (writer.getProgress().getObjectCount() > 0 && objectType != null) {
513                 objectTypesWithSchemaCreated.add(objectType);
514             }
515 
516             return writer.getProgress().getMutatieInfo();
517         } catch(Exception e) {
518             writer.abortWorkerThread();
519             throw e;
520         }
521     }
522 }
