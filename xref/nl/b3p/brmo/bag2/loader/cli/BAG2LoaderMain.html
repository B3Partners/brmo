113     @Command(name = "load", sortOptions = false)
114     public int load(
115             @Mixin BAG2DatabaseOptions dbOptions,
116             @Mixin BAG2LoadOptions loadOptions,
117             @Mixin BAG2ProgressOptions progressOptions,
118             @Parameters(paramLabel = "<file>") String[] filenames,
119             @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp) throws Exception {
120 
121         log.info(BAG2LoaderUtils.getUserAgent());
122 
123         try(BAG2Database db = getBAG2Database(dbOptions)) {
124             BAG2ProgressReporter progressReporter = progressOptions.isConsoleProgressEnabled()
125                     ? new BAG2ConsoleProgressReporter()
126                     : new BAG2ProgressReporter();
127 
128             loadFiles(db, dbOptions, loadOptions, progressReporter, filenames, null);
129             return ExitCode.OK;
130         }
131     }

148     public void loadFiles(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, CookieManager cookieManager) throws Exception {
149 
150         if (filenames.length == 1 && Files.isDirectory(Path.of(filenames[0]))) {
151             log.info("Directory opgegeven, kijken naar toepasbare mutaties...");
152             filenames = Files.list(Path.of(filenames[0]))
153                     .filter(p -> !Files.isDirectory(p) && p.getFileName().toString().endsWith(".zip"))
154                     .map(p -> p.toAbsolutePath().toString())
155                     .toArray(String[]::new);
156             if (filenames.length == 0) {
157                 log.info("Geen ZIP bestanden gevonden, niets te doen");
158             } else {
159                 applyMutaties(db, dbOptions, loadOptions, progressReporter, filenames, null,null);
160             }
161             return;
162         }
163 
164         BAG2LoaderUtils.BAG2FileName bag2FileName = BAG2LoaderUtils.analyzeBAG2FileName(filenames[0]);
165 
166         if (bag2FileName.isStand()) {
167 
168             if (!bag2FileName.isGemeente()) {
169                 if (filenames.length > 1) {
170                     throw new IllegalArgumentException("Inladen stand heel Nederland: teveel bestanden opgegeven");
171                 }
172             } else {
173                 // Verify all filenames are gemeentestanden
174                 Set<String> gemeenteCodes = new HashSet<>();
175                 for (int i = 1; i < filenames.length; i++) {
176                     BAG2LoaderUtils.BAG2FileName nextBag2FileName = BAG2LoaderUtils.analyzeBAG2FileName(filenames[i]);
177                     if (!nextBag2FileName.isStand() || !nextBag2FileName.isGemeente() || gemeenteCodes.contains(nextBag2FileName.getGemeenteCode())) {
178                         throw new IllegalArgumentException("Inladen stand gemeentes, ongeldig bestand opgegeven: " + filenames[i]);
179                     }
180                     gemeenteCodes.add(nextBag2FileName.getGemeenteCode());
181                 }
182             }
183 
184             loadStandFiles(db, dbOptions, loadOptions, progressReporter, filenames, cookieManager);
185         } else {
186             // Process mutaties while ignoring files not applicable
187             applyMutaties(db, dbOptions, loadOptions, progressReporter, filenames, null,null);
188         }
189     }

191     /**
192      * Only called after list of files have been checked to only have been entire NL stand or unique gemeente standen.
193      */
194     private void loadStandFiles(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, CookieManager cookieManager) throws Exception {
195         try {
196             // When loading multiple standen (for gemeentes), set ignore duplicates so the seen object keys are kept in
197             // memory so duplicates can be ignored. Don't keep keys in memory for entire NL stand.
198             loadOptions.setIgnoreDuplicates(filenames.length > 1);
199 
200             BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
201             String lastFilename = null;
202 
203             // Keep track of which gemeentes are loaded so the correct mutations can be processed
204             Set<String> gemeenteIdentificaties = new HashSet<>();
205 
206             for (String filename: filenames) {
207                 BAG2GMLMutatieGroepStream.BagInfo latestBagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, filename);
208                 if (bagInfo != null) {
209                     // For gemeentes the BagInfo must be the same so the standen are of the same date
210                     if (!latestBagInfo.equals(bagInfo)) {
211                         throw new IllegalArgumentException(String.format("Incompatible BagInfo for file \"%s\" (%s) compared to last file \"%s\" (%s)",
212                                 filename,
213                                 latestBagInfo,
214                                 lastFilename,
215                                 bagInfo));
216                     }
217                 }
218                 bagInfo = latestBagInfo;
219 
220                 // For NL stand this will be "9999"
221                 gemeenteIdentificaties.add(bagInfo.getGemeenteIdentificatie());
222                 lastFilename = filename;
223             }
224             if (bagInfo != null) {
225                 // TODO: when loading gemeente without rare objects such as ligplaatsen/standplaatsen, table will not be created
226                 // and a future change with such an object will fail. Should create entire schema up-front instead of when first
227                 // encountering object type
228                 createKeysAndIndexes(db, loadOptions, dbOptions, progressReporter);
229 
230                 updateMetadata(db, loadOptions, true, gemeenteIdentificaties, bagInfo.getStandTechnischeDatum());
231             }
232 
233             db.getConnection().commit();
234         } finally {
235             progressReporter.reportTotalSummary();
236         }
237     }

239     public void applyMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, String[] urls, CookieManager cookieManager) throws Exception {
240         if (filenames.length == 0) {
241             return;
242         }
243         BAG2LoaderUtils.BAG2FileName bag2FileName = BAG2LoaderUtils.analyzeBAG2FileName(filenames[0]);
244         if (bag2FileName.isGemeente()) {
245             applyGemeenteMutaties(db, dbOptions, loadOptions, progressReporter, filenames, urls, cookieManager);
246         } else {
247             applyNLMutaties(db, dbOptions, loadOptions, progressReporter, filenames, urls, cookieManager);
248         }
249     }

251     private void applyGemeenteMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, String[] urls, CookieManager cookieManager) throws Exception {
252         LocalDate currentTechnischeDatum = db.getCurrentTechnischeDatum();
253         Set<String> gemeenteCodes = db.getGemeenteCodes();
254 
255         Set<Integer> applicableMutatieIndexes;
256         do {
257             applicableMutatieIndexes = new HashSet<>();
258 
259             Set<String> missingGemeentes = new HashSet<>(gemeenteCodes);
260             for(int i = 0; i < filenames.length; i++) {
261                 BAG2LoaderUtils.BAG2FileName bag2FileName = BAG2LoaderUtils.analyzeBAG2FileName(filenames[i]);
262 
263                 if (bag2FileName.isGemeente()
264                         && !bag2FileName.isStand()
265                         && bag2FileName.getMutatiesFrom().equals(currentTechnischeDatum)
266                         && gemeenteCodes.contains(bag2FileName.getGemeenteCode())) {
267                     applicableMutatieIndexes.add(i);
268                     missingGemeentes.remove(bag2FileName.getGemeenteCode());
269                 }
270             }
271 
272             if (applicableMutatieIndexes.isEmpty()) {
273                 log.info(String.format("Geen nieuw toe te passen gemeentemutatiebestanden gevonden voor huidige stand technische datum %s, klaar", currentTechnischeDatum));
274                 break;
275             }
276 
277             // Check whether applicable mutaties are available for all gemeentecodes because they need to be processed
278             // at the same time to ignore duplicates
279             if (!missingGemeentes.isEmpty()) {
280                 throw new IllegalArgumentException(String.format("Kan geen gemeente mutaties toepassen voor gemeentes %s vanaf stand technische datum %s, in opgegeven mutatiebestanden ontbreken gemeentecodes %s",
281                         gemeenteCodes,
282                         currentTechnischeDatum,
283                         missingGemeentes));
284             }
285 
286             log.info(String.format("Toepassen gemeentemutaties voor %d gemeentes vanaf stand technische datum %s...", gemeenteCodes.size(), currentTechnischeDatum));
287 
288             BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
289 
290             loadOptions.setIgnoreDuplicates(gemeenteCodes.size() > 1);
291             for (int index: applicableMutatieIndexes) {
292                 String filename = filenames[index];
293                 String url = urls == null ? filenames[index] : urls[index];
294                 bagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, filename, url, cookieManager);
295             }
296             currentTechnischeDatum = new java.sql.Date(bagInfo.getStandTechnischeDatum().getTime()).toLocalDate();
297             updateMetadata(db, loadOptions, false, null, bagInfo.getStandTechnischeDatum());
298             db.getConnection().commit();
299             // Duplicates need only be checked for mutaties for a single from date, clear cache to reduce memory usage
300             clearDuplicatesCache();
301             log.info("Mutaties verwerkt, huidige stand technische datum: " + currentTechnischeDatum);
302 
303         } while(true);
304     }

306     private void applyNLMutaties(BAG2Database db, BAG2DatabaseOptions dbOptions, BAG2LoadOptions loadOptions, BAG2ProgressReporter progressReporter, String[] filenames, String[] urls, CookieManager cookieManager) throws Exception {
307         LocalDate currentTechnischeDatum = db.getCurrentTechnischeDatum();
308         do {
309             String applicableMutatie = null;
310             String applicatieMutatieURL = null;
311 
312             for(int i = 0; i < filenames.length; i++) {
313                 String filename = filenames[i];
314                 BAG2LoaderUtils.BAG2FileName bag2FileName = BAG2LoaderUtils.analyzeBAG2FileName(filename);
315 
316                 if (!bag2FileName.isGemeente()
317                         && !bag2FileName.isStand()
318                         && bag2FileName.getMutatiesFrom().equals(currentTechnischeDatum)) {
319                     applicableMutatie = filename;
320                     applicatieMutatieURL = urls == null ? filename : urls[i];
321                 }
322             }
323 
324             if (applicableMutatie == null) {
325                 log.info(String.format("Geen nieuw toe te passen mutatiebestanden gevonden voor huidige stand technische datum %s, klaar", currentTechnischeDatum));
326                 break;
327             }
328 
329             log.info(String.format("Toepassen mutaties vanaf stand technische datum %s...", currentTechnischeDatum));
330 
331             BAG2GMLMutatieGroepStream.BagInfo bagInfo = loadBAG2ExtractFromURLorFile(db, loadOptions, dbOptions, progressReporter, applicableMutatie, applicatieMutatieURL, cookieManager);
332             currentTechnischeDatum = new java.sql.Date(bagInfo.getStandTechnischeDatum().getTime()).toLocalDate();
333             updateMetadata(db, loadOptions, false, null, bagInfo.getStandTechnischeDatum());
334             db.getConnection().commit();
335             log.info("Mutaties verwerkt, huidige stand technische datum: " + currentTechnischeDatum);
336         } while(true);
337     }

339     private void createKeysAndIndexes(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions databaseOptions, BAG2ProgressReporter progressReporter) throws Exception {
340         BAG2ObjectTableWriter writer = db.createObjectTableWriter(loadOptions, databaseOptions);
341         writer.setProgressUpdater(progressReporter);
342         for(BAG2ObjectType objectType: objectTypesWithSchemaCreated) {
343             writer.createKeys(objectType); // BAG2 writer is always a single ObjectType unlike BGT
344             writer.createIndexes(objectType);
345         }
346     }

352     private BAG2GMLMutatieGroepStream.BagInfo loadBAG2ExtractFromURLorFile(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions dbOptions, BAG2ProgressReporter progressReporter, String filename, String url, CookieManager cookieManager) throws Exception {
353         HttpClientWrapper<HttpRequest.Builder, HttpResponse<InputStream>> httpClientWrapper = cookieManager == null
354                 ? new Java11HttpClientWrapper()
355                 : new Java11HttpClientWrapper(HttpClient.newBuilder().cookieHandler(cookieManager));
356 
357         if (url.startsWith("http://") || url.startsWith("https://")) {
358             try (InputStream in = new ResumingInputStream(new HttpStartRangeInputStreamProvider(URI.create(url), httpClientWrapper))) {
359                 return loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, filename, in);
360             }
361         }
362         if (url.endsWith(".zip")) {
363             try (InputStream in = new FileInputStream(url)) {
364                 return loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, filename, in);
365             }
366         }
367 
368         throw new IllegalArgumentException(getMessageFormattedString("load.invalid_file", url));
369     }

371     private BAG2GMLMutatieGroepStream.BagInfo loadBAG2ExtractFromStream(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions dbOptions, BAG2ProgressReporter progressReporter, String name, InputStream input) throws Exception {
372         BAG2GMLMutatieGroepStream.BagInfo bagInfo = null;
373         try (ZipArchiveInputStream zip = new ZipArchiveInputStream(input)) {
374             ZipArchiveEntry entry = zip.getNextZipEntry();
375             while(entry != null) {
376                 if (entry.getName().matches("[0-9]{4}(STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.xml")) {
377                     // Load extracted zipfile
378                     bagInfo = loadXmlEntriesFromZipFile(db, loadOptions, dbOptions, progressReporter, name, zip, entry);
379                     break;
380                 }
381 
382                 if (entry.getName().matches("[0-9]{4}GEM[0-9]{8}\\.zip")) {
383                     return loadBAG2ExtractFromStream(db, loadOptions, dbOptions, progressReporter, name, zip);
384                 }
385 
386                 // Process single and double-nested ZIP files
387 
388                 if (entry.getName().matches("[0-9]{4}(STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.zip")
389                 || entry.getName().matches("[0-9]{4}MUT[0-9]{8}-[0-9]{8}\\.zip")) {
390                     ZipArchiveInputStream nestedZip = new ZipArchiveInputStream(zip);
391                     bagInfo = loadXmlEntriesFromZipFile(db, loadOptions, dbOptions, progressReporter, entry.getName(), nestedZip, nestedZip.getNextZipEntry());
392                 }
393 
394                 if (entry.getName().matches("[0-9]{4}Inactief.*\\.zip")) {
395                     ZipArchiveInputStream nestedZip = new ZipArchiveInputStream(zip);
396                     ZipArchiveEntry nestedEntry = nestedZip.getNextZipEntry();
397                     while(nestedEntry != null) {
398                         if (nestedEntry.getName().matches("[0-9]{4}IA.*\\.zip")) {
399                             ZipArchiveInputStream moreNestedZip = new ZipArchiveInputStream(nestedZip);
400                             bagInfo = loadXmlEntriesFromZipFile(db,  loadOptions, dbOptions, progressReporter, nestedEntry.getName(), moreNestedZip, moreNestedZip.getNextZipEntry());
401                         }
402                         nestedEntry = nestedZip.getNextZipEntry();
403                     }
404                 }
405 
406                 try {
407                     entry = zip.getNextZipEntry();
408                 } catch(IOException e) {
409                     // Reading the ZIP from HTTP may give this error, but it is a normal end...
410                     if ("Truncated ZIP file".equals(e.getMessage())) {
411                         break;
412                     }
413                 }
414             }
415         }
416         return bagInfo;
417     }

419     private static BAG2ObjectType getObjectTypeFromFilename(String filename) {
420         Matcher m = Pattern.compile(".*[0-9]{4}(IA)?(MUT|STA|VBO|OPR|NUM|LIG|PND|WPL).*\\.(xml|zip)").matcher(filename);
421         if (!m.matches()) {
422             throw new IllegalArgumentException("Invalid BAG2 filename: " + filename);
423         }
424         String objectTypeName = null;
425         switch(m.group(2)) {
426             case "MUT": break;
427             case "STA": objectTypeName = "Standplaats"; break;
428             case "OPR": objectTypeName = "OpenbareRuimte"; break;
429             case "VBO": objectTypeName = "Verblijfsobject"; break;
430             case "NUM": objectTypeName = "Nummeraanduiding"; break;
431             case "LIG": objectTypeName = "Ligplaats"; break;
432             case "PND": objectTypeName = "Pand"; break;
433             case "WPL": objectTypeName = "Woonplaats"; break;
434         }
435         if (objectTypeName == null) {
436             return null;
437         } else {
438             return BAG2Schema.getInstance().getObjectTypeByName(objectTypeName);
439         }
440     }

442     private void updateMetadata(BAG2Database db, BAG2LoadOptions loadOptions, boolean stand, Set<String> gemeenteIdentificaties, Date standTechnischeDatum) throws Exception {
443         // Check if metadata table already exists. For PostgreSQL we can use the metadata table in the public schema
444         if (!db.getDialect().tableExists(db.getConnection(), METADATA_TABLE_NAME)) {
445             // Create a new metadata table, for Oracle as BAG is in separate schema, for PostgreSQL if loading BAG
446             // into a non-brmo RSGB database
447             db.createMetadataTable(loadOptions);
448         }
449 
450         db.setMetadataValue(BAG2SchemaMapper.Metadata.LOADER_VERSION, BAG2LoaderUtils.getLoaderVersion());
451         SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
452         if (stand) {
453             db.setMetadataValue(STAND_LOAD_TIME, new SimpleDateFormat("yyyy-MM-dd HH:mm:ss").format(new Date()));
454             db.setMetadataValue(STAND_LOAD_TECHNISCHE_DATUM, df.format(standTechnischeDatum));
455             db.setMetadataValue(GEMEENTE_CODES, String.join(",", gemeenteIdentificaties));
456             db.setMetadataValue(FILTER_MUTATIES_WOONPLAATS, "false");
457         }
458         db.setMetadataValue(CURRENT_TECHNISCHE_DATUM, df.format(standTechnischeDatum));
459     }

465     private BAG2GMLMutatieGroepStream.BagInfo loadXmlEntriesFromZipFile(BAG2Database db, BAG2LoadOptions loadOptions, BAG2DatabaseOptions databaseOptions, BAG2ProgressReporter progressReporter, String name, ZipArchiveInputStream zip, ZipArchiveEntry entry) throws Exception {
466         BAG2ObjectType objectType = getObjectTypeFromFilename(name);
467         // objectType is null for mutaties, which contain mixed object types instead of a single object type with stand
468         boolean schemaCreated = objectType == null || objectTypesWithSchemaCreated.contains(objectType);
469         BAG2ObjectTableWriter writer = db.createObjectTableWriter(loadOptions, databaseOptions);
470         writer.setProgressUpdater(progressReporter);
471         writer.setCreateSchema(!schemaCreated);
472         writer.setCreateKeysAndIndexes(false);
473         writer.setKeysPerObjectType(keysPerObjectType);
474         writer.start(); // sets InitialLoad to true
475         writer.getProgress().setInitialLoad(!schemaCreated); // For a COPY in transaction, table must be created or truncated in it
476         if (objectType == null) {
477             // When processing mutaties, set batch size to 1 so all mutaties are processed sequentially and can not
478             // conflict with deleting and inserting of old/new versions
479             writer.setBatchSize(1);
480             // Disable multithreading so deletion of previous versions and new inserts are processed sequentially
481             writer.setMultithreading(false);
482         }
483         progressReporter.startNewFile(name);
484         try {
485             while(entry != null) {
486                 progressReporter.startNextSplitFile(entry.getName());
487                 writer.write(CloseShieldInputStream.wrap(zip));
488                 if (loadOptions.getMaxObjects() != null && writer.getProgress().getObjectCount() == loadOptions.getMaxObjects()) {
489                     break;
490                 }
491                 entry = zip.getNextZipEntry();
492             }
493             writer.complete();
494 
495             if (writer.getProgress().getObjectCount() > 0 && objectType != null) {
496                 objectTypesWithSchemaCreated.add(objectType);
497             }
498 
499             return writer.getProgress().getMutatieInfo();
500         } catch(Exception e) {
501             writer.abortWorkerThread();
502             throw e;
