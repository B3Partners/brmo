70      @Command(name="download", sortOptions = false)
71      public int download(
72              @Option(names="--no-delete", negatable = true) boolean noDelete,
73              @Option(names="--kadaster-user") String kadasterUser,
74              @Option(names="--kadaster-password") String kadasterPassword,
75              @Option(names="--url", defaultValue = LVBAG_BESTANDEN_API_URL) String url,
76              @Option(names="--query-params", defaultValue = "artikelnummers=2529") String queryParams,
77              @Option(names="--path", defaultValue = "") String downloadPath,
78              @Option(names="--mirror-base-url") String mirrorBaseUrl,
79              @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp
80              ) throws Exception {
81  
82          log.info(BAG2LoaderUtils.getUserAgent());
83          Instant start = Instant.now();
84  
85          CookieManager kadasterCookieManager = new CookieManager(null, CookiePolicy.ACCEPT_ALL);
86          mijnKadasterLogin(URI.create(url), kadasterUser, kadasterPassword, kadasterCookieManager);
87  
88          JSONArray bestanden = getBagBestanden(url, queryParams, kadasterCookieManager);
89  
90          log.info("Aantal beschikbare bestanden: " + bestanden.length());
91  
92          List<JSONObject> toDownload = new ArrayList<>();
93          for(int i = 0; i < bestanden.length(); i++) {
94              JSONObject bestand = bestanden.getJSONObject(i);
95  
96              File f = Path.of(downloadPath, bestand.getString("naam")).toFile();
97              if (!f.exists() || f.length() != bestand.getLong("grootte")) {
98                  toDownload.add(bestand);
99              }
100         }
101 
102         if (!noDelete) {
103             deleteZipFilesNotInBestanden(downloadPath, bestanden);
104         }
105 
106         long totalBytes = toDownload.stream().map(bestand -> bestand.getLong("grootte")).reduce(Long::sum).orElse(0L);
107         log.info(String.format("Aantal te downloaden naar directory \"%s\": %d bestanden (%s)",
108                 Path.of(downloadPath).toAbsolutePath(),
109                 toDownload.size(),
110                 byteCountToDisplaySize(totalBytes)));
111 
112         int count = 0;
113         long bytesRead = 0;
114         for(JSONObject bestand: toDownload) {
115             url = bestand.getString("url");
116             String name = bestand.getString("naam");
117 
118             try(ResumingInputStream input = new ResumingInputStream(new HttpStartRangeInputStreamProvider(URI.create(url),
119                     new Java11HttpClientWrapper(HttpClient.newBuilder().cookieHandler(kadasterCookieManager))));
120                 OutputStream out = new FileOutputStream(Path.of(downloadPath, name).toFile())) {
121 
122                 log.info(String.format("Bestand %2d/%d (%.1f%%): downloaden %s...",
123                         ++count,
124                         toDownload.size(),
125                         (100.0 / totalBytes) * bytesRead,
126                         name));
127                 IOUtils.copyLarge(input, out);
128             }
129             bytesRead += bestand.getLong("grootte");
130         }
131         if (!toDownload.isEmpty()) {
132             String msg = "";
133             if (mirrorBaseUrl != null) {
134                 for (int i = 0; i < bestanden.length(); i++) {
135                     JSONObject bestand = bestanden.getJSONObject(i);
136                     String name = bestand.getString("naam");
137                     bestand.put("url", URI.create(mirrorBaseUrl).resolve(name));
138                 }
139                 File bestandenJSONMirror = Path.of(downloadPath, "bestanden.json").toFile();
140                 try(OutputStream out = new FileOutputStream(bestandenJSONMirror)) {
141                     IOUtils.write(bestanden.toString(2), out, StandardCharsets.UTF_8);
142                     msg = String.format(", JSON voor mirror \"%s\" geschreven naar \"%s\"", mirrorBaseUrl, bestandenJSONMirror);
143                 }
144             }
145             log.info("Alle bestanden gedownload in " + formatTimeSince(start) + msg);
146         }
147 
148         return ExitCode.OK;
149     }

151     @Command(name="apply", sortOptions = false)
152     public int apply(
153             @Mixin BAG2DatabaseOptions dbOptions,
154             @Mixin BAG2ProgressOptions progressOptions,
155             @Option(names="--kadaster-user") String kadasterUser,
156             @Option(names="--kadaster-password") String kadasterPassword,
157             @Option(names="--url", defaultValue = LVBAG_BESTANDEN_API_URL) String url,
158             @Option(names="--query-params", defaultValue = "artikelnummers=2529") String queryParams,
159             @Option(names={"-h","--help"}, usageHelp = true) boolean showHelp
160     ) throws Exception {
161 
162         log.info(BAG2LoaderUtils.getUserAgent());
163         Instant start = Instant.now();
164 
165         CookieManager kadasterCookieManager = new CookieManager(null, CookiePolicy.ACCEPT_ALL);
166         mijnKadasterLogin(URI.create(url), kadasterUser, kadasterPassword, kadasterCookieManager);
167 
168         JSONArray bestanden = getBagBestanden(url, queryParams, kadasterCookieManager);
169 
170         log.info("Aantal beschikbare bestanden: " + bestanden.length());
171 
172         List<String> names = new ArrayList<>();
173         List<String> urls = new ArrayList<>();
174         for(int i = 0; i < bestanden.length(); i++) {
175             JSONObject bestand = bestanden.getJSONObject(i);
176             names.add(bestand.getString("naam"));
177             urls.add(bestand.getString("url"));
178         }
179 
180         try(BAG2Database db = parent.getBag2Database()) {
181             BAG2ProgressReporter progressReporter = progressOptions.isConsoleProgressEnabled()
182                     ? new BAG2ConsoleProgressReporter()
183                     : new BAG2ProgressReporter();
184 
185             parent.applyMutaties(db, dbOptions, new BAG2LoadOptions(), progressReporter, names.toArray(String[]::new), urls.toArray(String[]::new), kadasterCookieManager);
186             log.info("Alle mutatiebestanden verwerkt in " + formatTimeSince(start));
187             return ExitCode.OK;
188         }
189     }

191     private static void mijnKadasterLogin(URI forUri, String username, String password, CookieManager cookieManager) throws IOException, InterruptedException {
192         if (!forUri.getHost().endsWith("kadaster.nl")) {
193             return;
194         }
195         if (username == null || password == null) {
196             throw new IllegalArgumentException("Gebruikersnaam en wachtwoord zijn verplicht");
197         }
198 
199         HttpRequest request = HttpRequest.newBuilder().uri(URI.create(KADASTER_LOGIN_URL))
200                 .header("Content-Type", "application/x-www-form-urlencoded")
201                 .POST(HttpRequest.BodyPublishers.ofString(String.format(
202                         "user=%s&password=%s",
203                         URLEncoder.encode(username, StandardCharsets.UTF_8),
204                         URLEncoder.encode(password, StandardCharsets.UTF_8))))
205                 .build();
206 
207         HttpClient httpClient = HttpClient.newBuilder()
208                 .cookieHandler(cookieManager)
209                 .build();
210 
211         HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
212 
213         if (response.statusCode() != 302) {
214             throw new IllegalArgumentException(String.format("Fout bij inloggen op Mijn Kadaster met gebruikersnaam \"%s\"", username));
215         }
216 
217         Map<String,List<String>> cookies = cookieManager.get(URI.create(KADASTER_LOGIN_URL), new HashMap<>());
218 
219         Optional<String> kadasterTicketIdCookie = cookies.getOrDefault("Cookie", List.of())
220                 .stream().filter(c -> c.startsWith("KadasterTicketId=")).findFirst();
221 
222         if (kadasterTicketIdCookie.isEmpty()) {
223             throw new IllegalArgumentException("Geen KadasterTicketId cookie ontvangen na inloggen");
224         }
225     }

227     private static JSONArray getBagBestanden(String url, String queryParams, CookieManager kadasterCookieManager) throws Exception {
228         url = url + "?" + queryParams;
229         log.info("Opvragen bestanden JSON vanaf URL " + url);
230         HttpRequest request = HttpRequest.newBuilder().uri(URI.create(url)).build();
231         HttpClient httpClient = HttpClient.newBuilder()
232                 .cookieHandler(kadasterCookieManager)
233                 .build();
234         HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
235 
236         JSONArray bestanden;
237         try {
238             bestanden = new JSONArray(response.body());
239         } catch(Exception e) {
240             throw new Exception("Fout bij parsen BAG bestanden JSON, body: " + response.body(), e);
241         }
242         return bestanden;
243     }

245     private static void deleteZipFilesNotInBestanden(String downloadPath, JSONArray bestanden) throws IOException {
246         final Set<String> names = new HashSet<>();
247         for(int i = 0; i < bestanden.length(); i++) {
248             names.add(bestanden.getJSONObject(i).getString("naam"));
249         }
250         try(Stream<Path> stream = Files.list(Path.of(downloadPath))) {
251             stream.filter(p -> !Files.isDirectory(p) && p.getFileName().toString().endsWith(".zip") && !names.contains(p.getFileName().toString()))
252                     .forEach(p -> {
253                         log.info("Verwijderen ZIP bestand niet in bestandenlijst: " + p.getFileName());
254                         try {
255                             Files.delete(p);
256                         } catch(Exception e) {
257                             log.error(String.format("Fout bij verwijderen ZIP bestand %s: %s", p.getFileName(), e));
258                         }
259                     });
260         }
261     }
