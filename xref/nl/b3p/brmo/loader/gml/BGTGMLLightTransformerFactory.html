81      private static final Map<String, BGTGMLLightTransformerFactory> lookup = new HashMap<>();
82  
83      static {
84          for (BGTGMLLightTransformerFactory d : BGTGMLLightTransformerFactory.values()) {
85              //lookup.put(d.getGmlFileName(), d);
86              lookup.put(d.getTransformerClassName(), d);
87          }
88      }
89  
90      private BGTGMLLightTransformerFactory(String gmlFileName, String transformerClassName) {
91          this.gmlFileName = gmlFileName;
92          this.transformerClassName = transformerClassName;
93      }
94  
95      public String getGmlFileName() {

97      }
98  
99      public String getTransformerClassName() {
100         return transformerClassName;
101     }

103     /**
104      * zoek de tabelnaam voor dit GML bestand.
105      *
106      * @param gmlFileName GML bestandsnaam
107      * @return tabelnaam
108      */
109     public static String getTableName(String gmlFileName) {
110         return lookup.get(gmlFileName).name();
111     }
112 
113     /**
114      * Zoek de juiste feature transformer op voor het (gml light) bestand.
115      *
116      * @param gmlFileName bestandsnaam
117      * @return transformer of {@code null} als er geen transformer beschikbaar
118      * danwel gevonden is
119      */
120     public static GMLLightFeatureTransformer getTransformer(String gmlFileName) {
121         try {
122             Class c = Class.forName(GMLLightFeatureTransformer.class.getPackage().getName() + ".light." + lookup.get(gmlFileName).getTransformerClassName());
123             return (GMLLightFeatureTransformer) c.newInstance();
124         } catch (ClassNotFoundException | InstantiationException | IllegalAccessException | NullPointerException ex) {
125             return null;
126         }
127     }
