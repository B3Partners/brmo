158     /**
159      * Verwerk een gmllight extract zipfile.
160      *
161      * @param zipExtract zipfile met gml bestanden
162      * @return het aantal geschreven features voor de zipfile
163      *
164      * @throws FileNotFoundException als zipExtract niet gevonden kan worden
165      * @throws IOException als ophalen next zipentry mislukt of als de database
166      * verbinding wegvalt
167      */
168     public int processZipFile(File zipExtract) throws FileNotFoundException, IOException {
169         this.omhullendeVanZipFile = null;
170         this.resetStatus();
171         if (this.bijwerkDatum == null) {
172             this.bijwerkDatum = new Date();
173         }
174 
175         int result = 0, total = 0;
176         String eName = "";
177         if (this.isValidZipFile(zipExtract)) {
178             try (ZipInputStream zip = new ZipInputStream(new FileInputStream(zipExtract))) {
179                 ZipEntry entry = zip.getNextEntry();
180                 if (entry == null) {
181                     LOG.error("Geen bestanden in zipfile (" + zipExtract + ") gevonden.");
182                 }
183                 // for each gml in zip
184                 while (entry != null) {
185                     if (!entry.getName().toLowerCase().endsWith(".gml")) {
186                         LOG.warn("Overslaan zip entry geen GML bestand: " + entry.getName());
187                     } else {
188                         eName = entry.getName();
189                         LOG.info("Lezen GML bestand: " + eName + " uit zip file: " + zipExtract.getCanonicalPath());
190                         result = storeFeatureCollection(new CloseShieldInputStream(zip), eName.toLowerCase());
191                         opmerkingen.append(result)
192                                 .append(" features geladen uit: ")
193                                 .append(eName).append(", zipfile: ")
194                                 .append(zipExtract.getCanonicalPath())
195                                 .append("\n");
196                         total += result;
197                     }
198                     entry = zip.getNextEntry();
199                 }
200 
201                 if (this.loadingUpdate) {
202                     // nadat zipfile is geladen
203                     //                LOG.debug("omhullende van zipfile: " + omhullendeVanZipFile);
204                     //                LOG.debug("convex omhullende van zipfile(" + zipExtract + "): " + omhullendeVanZipFile.convexHull());
205                     //                LOG.debug("union omhullende van zipfile(" + zipExtract + "): " + omhullendeVanZipFile.union());
206 
207                     // org.opensphere.geometry.algorithm.ConcaveHull, zie ook: pom.xml
208                     //ConcaveHull ch = new ConcaveHull(omhullendeVanZipFile, 1d);
209                     //LOG.debug("concave omhullende van zipfile(" + zipExtract + "): " + ch.getConcaveHull());
210                     // verwijderen van de onderliggend aan omhullendeVanZipFile, verouderde objecten in iedere tabel
211                     // self union lijkt vooranlog het beste masker te geven voor verwijderen
212                     deleteOldData(this.bijwerkDatum, omhullendeVanZipFile.union());
213                 }
214             } catch (SAXException | ParserConfigurationException ex) {
215                 LOG.error("Er is een parse fout opgetreden tijdens verwerken van " + eName + " uit " + zipExtract.getCanonicalPath(), ex);
216                 opmerkingen.append("Er is een parse fout opgetreden tijdens verwerken van ").append(eName)
217                         .append(" Foutmelding: ").append(ex).append("\n");
218                 this.status = STATUS.NOK;
219             }
220         } else {
221             LOG.error("Ongeldige of corrupte zipfile: " + zipExtract.getCanonicalPath());
222             opmerkingen.append("Ongeldige of corrupte zipfile: ").append(zipExtract.getCanonicalPath())
223                     .append("\nHet bestand kan niet verwerkt worden. Download het bestand opnieuw.");
224             this.status = STATUS.NOK;
225         }
226         return total;
227     }

229     private void deleteOldData(Date deleteBeforeDatum, Geometry deleteOverlaps) throws IOException {
230         JDBCDataStore dataStore = (JDBCDataStore) DataStoreFinder.getDataStore(dbConnProps);
231         if (dataStore == null) {
232             throw new IllegalStateException("Datastore mag niet 'null' zijn voor verwijderen van van data.");
233         }
234 
235         FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2();
236         Filter filter = ff.and(
237                 ff.before(ff.property(
238                         (this.isOracle ? BIJWERKDATUM_NAME.toUpperCase() : BIJWERKDATUM_NAME)
239                 ), ff.literal(deleteBeforeDatum)),
240                 ff.overlaps(ff.property(
241                         (this.isOracle ? DEFAULT_GEOM_NAME.toUpperCase() : DEFAULT_GEOM_NAME)
242                 ), ff.literal(deleteOverlaps))
243         );
244         LOG.info("Opruimen van verouderde data uit tabellen met filters: " + filter);
245 
246         FeatureStore store;
247         try (Transaction deletetransaction = new DefaultTransaction("delete-bgt")) {
248             for (BGTGMLLightTransformerFactory t : BGTGMLLightTransformerFactory.values()) {
249                 if (t.getGmlFileName().isEmpty()) {
250                     continue;
251                 }
252                 String tableName = this.isOracle ? t.name().toUpperCase() : t.name();
253                 opmerkingen.append("Opruimen van verouderde data uit tabel: ").append(tableName).append("\n");
254                 LOG.info("Opruimen van verouderde data uit tabel: " + tableName);
255                 store = (FeatureStore) dataStore.getFeatureSource(tableName, deletetransaction);
256                 store.removeFeatures(filter);
257                 deletetransaction.commit();
258             }
259         } finally {
260             dataStore.dispose();
261         }
262     }

264     /**
265      * Verwerk een enkel GML bestand. <strong>Let op:</strong> alleen voor
266      * "stand"     * verwerking omdat de onderliggende geometrische data niet betrouwbaar kan
267      * worden verwijderd in dit geval. Waarschijnlijk wil je ook de status
268      * resetten voor deze Loader, gebruik {@link #resetStatus() }.
269      *
270      * @param gml GML Light bestand
271      * @return aantal geschreven features
272      *
273      * @throws IOException als ophalen gml mislukt
274      */
275     public int processGMLFile(File gml) throws IOException {
276         int result = 0;
277         try {
278             result = storeFeatureCollection(new FileInputStream(gml), gml.getName().toLowerCase());
279         } catch (SAXException | ParserConfigurationException ex) {
280             LOG.error("Er is een parse fout opgetreden tijdens verwerken van: " + gml.getCanonicalPath(), ex);
281             this.status = STATUS.NOK;
282         }
283         return result;
284     }

286     /**
287      * transformeert de input en slaat op in database.
288      *
289      * @param gmlFeatCollection te laden features
290      * @param gmlFileName naam input gml bestand
291      * @return aantal geschreven features
292      *
293      * @throws SAXException als parsen van input mislukt
294      * @throws IOException als er een database fout optreedt
295      * @throws ParserConfigurationException als gml parser config niet deugd
296      * @throws IllegalStateException als er iets mis is in de configuratie
297      */
298     private int storeFeatureCollection(InputStream in, String gmlFileName) throws IOException, IllegalStateException, SAXException, ParserConfigurationException {
299         int writtenFeatures = 0;
300         JDBCDataStore dataStore = (JDBCDataStore) DataStoreFinder.getDataStore(dbConnProps);
301         if (dataStore == null) {
302             throw new IllegalStateException("Datastore mag niet 'null' zijn voor opslaan van data.");
303         }
304         // TODO boolean mapping voor mssql en oracle
305         if (this.isOracle) {
306             dataStore.getClassToSqlTypeMappings().put(java.lang.Boolean.class, Types.VARCHAR);
307         }
308         if (this.isMSSQL) {
309             dataStore.getClassToSqlTypeMappings().put(java.lang.Boolean.class, Types.VARCHAR);
310         }
311 
312         SimpleFeatureCollection gmlFeatCollection = (SimpleFeatureCollection) parser.parse(in);
313         
314         List<Exception> validationErrors = parser.getValidationErrors();
315         if (!validationErrors.isEmpty()) {
316             LOG.warn("Er zijn validatie fouten opgetreden tijdens verwerking van bestand: " + gmlFileName);
317             for (Exception e : validationErrors) {
318                 LOG.warn("validatie fout: " + e.getMessage());
319             }
320         }
321 
322         if (gmlFeatCollection.isEmpty()) {
323             opmerkingen.append("Geen features gevonden in bestand: ").append(gmlFileName).append("\n");
324             LOG.info("Geen features gevonden in bestand: " + gmlFileName);
325             dataStore.dispose();
326             return writtenFeatures;
327         }
328 
329         SimpleFeatureType sft = gmlFeatCollection.getSchema();
330         GMLLightFeatureTransformer featTransformer = BGTGMLLightTransformerFactory.getTransformer(sft.getTypeName());
331         if (featTransformer == null) {
332             LOG.error("Opzoeken van FeatureTransformer voor " + sft.getTypeName() + " is mislukt; er geen transformer beschikbaar voor dit bestand.");
333             dataStore.dispose();
334             return writtenFeatures;
335         }
336 
337         // check table exists
338         String tableName = BGTGMLLightTransformerFactory.getTableName(sft.getTypeName());
339         boolean exists = false;
340         String[] typeNames = dataStore.getTypeNames();
341         for (String name : typeNames) {
342             if (tableName.equalsIgnoreCase(name)) {
343                 LOG.debug("De tabel '" + tableName + "' is gevonden in de database als: " + name);
344                 exists = true;
345             }
346         }
347 
348         SimpleFeatureType targetSchema = featTransformer.getTargetSchema(sft, tableName, this.isOracle);
349         LOG.debug("Doel tabel schema: " + targetSchema);
350 
351         if (!exists) {
352             if (createTables) {
353                 dataStore.createSchema(targetSchema);
354                 LOG.warn("De volgende tabel is aangemaakt in de database: " + targetSchema.getTypeName());
355                 if (this.isOracle) {
356                     try {
357                         dataStore.getConnection(Transaction.AUTO_COMMIT).createStatement().execute("UPDATE USER_SDO_GEOM_METADATA SET SRID=28992");
358                     } catch (SQLException ex) {
359                         LOG.error("Bijwerken ruimtelijke metadata is mislukt", ex);
360                     }
361                 }
362             } else {
363                 LOG.error("Tabel: " + tableName + " is niet beschikbaar in de database.");
364                 throw new IllegalStateException("De tabel " + tableName + " ontbreekt in de database; opslaan van gegevens uit GML " + gmlFileName + " is niet mogelijk.");
365             }
366         } else {
367             // als tabel bestaat aannemen dat deze de juiste primary key heeft
368             dataStore.setExposePrimaryKeyColumns(true);
369         }
370 
371         Transaction transaction = new DefaultTransaction("add-bgt");
372         Transaction updatetransaction = new DefaultTransaction("update-bgt");
373         try (FeatureIterator<SimpleFeature> feats = gmlFeatCollection.features()) {
374             FeatureStore store = (FeatureStore) dataStore.getFeatureSource(targetSchema.getTypeName(), transaction);
375             FeatureStore store2 = (FeatureStore) dataStore.getFeatureSource(targetSchema.getTypeName(), updatetransaction);
376 
377             SimpleFeature gmlSF;
378             SimpleFeature transformed;
379             while (feats.hasNext()) {
380                 gmlSF = feats.next();
381                 // als object niet meer bestaat of nog niet bestaat overslaan!
382                 if ((/* bestaat nog niet*/((Date) gmlSF.getAttribute("objectBeginTijd")).after(new Date()))
383                         | (/* bestaat niet meer */gmlSF.getAttribute("objectEindTijd") != null && ((Date) gmlSF.getAttribute("objectEindTijd")).before(new Date()))) {
384                     // mogelijk toch nog de geom eruit halen voor de delete...
385                     LOG.info("Vervallen of nog niet bestaand object gevonden met GML ID: " + gmlSF.getID());
386                     continue;
387                 }
388 
389                 transformed = featTransformer.transform(gmlSF, targetSchema, this.isOracle, dataStore.isExposePrimaryKeyColumns(), this.bijwerkDatum);
390                 if (transformed == null) {
391                     continue;
392                 }
393                 if (this.loadingUpdate) {
394                     // bijwerken omhullende omhullendeVanZipFile met vlakken in geval van een update
395                     Geometry geom = (Geometry) transformed.getDefaultGeometry();
396                     if (geom != null && geom.getDimension() > Dimension.L) {
397                         if (omhullendeVanZipFile != null) {
398                             try {
399                                 omhullendeVanZipFile = omhullendeVanZipFile.union(geom);
400                             } catch (IllegalArgumentException | TopologyException ex) {
401                                 LOG.error("Union fout bij verwerking van " + transformed.getID() + " (GML ID: " + gmlSF.getID() + ", bestand: " + gmlFileName
402                                         + ")", ex);
403                                 LOG.debug("geom voor union :" + geom);
404                                 LOG.debug("omhullendeVanZipFile voor union: " + omhullendeVanZipFile);
405                             }
406                         } else {
407                             omhullendeVanZipFile = geom.union();
408                         }
409                     }
410                 }
411 
412                 try {
413                     store.addFeatures(DataUtilities.collection(transformed));
414                     writtenFeatures++;
415                     // commit per feature
416                     transaction.commit();
417                     LOG.debug("Feature toegevoegd in database met NEN3610ID: " + transformed.getID());
418                 } catch (IOException ioe) {
419                     // als primary key violation bij insert dan afvangen
420                     if (isDuplicateKeyViolationMessage(ioe.getCause().getMessage())) {
421                         LOG.info("Duplicaat gevonden tijdens insert van feature: " + transformed.getID());
422                         // bij mssql transactie sluiten ander blijft de boel hangen, voor orcl + pg ook
423                         transaction.close();
424                         // object opzoeken,
425                         Filter filter = CommonFactoryFinder.getFilterFactory2(null).id(transformed.getIdentifier());
426                         FeatureIterator<SimpleFeature> bestaandeFeats = store2.getFeatures(filter).features();
427                         SimpleFeature bestaandeFeat = bestaandeFeats.next();
428 
429                         final String beginAttr = this.isOracle ? BEGINTIJD_NAME.toUpperCase() : BEGINTIJD_NAME;
430                         final String idAttr = this.isOracle ? ID_NAME.toUpperCase() : ID_NAME;
431                         // update als datum jonger dan bestaand
432                         if (((Date) transformed.getAttribute(beginAttr)).after((Date) bestaandeFeat.getAttribute(beginAttr))) {
433                             for (AttributeDescriptor attr : transformed.getFeatureType().getAttributeDescriptors()) {
434                                 Name attrName = attr.getName();
435                                 if (!attrName.getLocalPart().equals(idAttr)) {
436                                     store2.modifyFeatures(attrName, transformed.getAttribute(attrName), filter);
437                                 }
438                             }
439                             LOG.debug("Klaar met update van feature: " + transformed.getID());
440                             updatetransaction.commit();
441                         } else {
442                             LOG.debug("Geen update van feature: " + transformed.getID());
443                         }
444                         bestaandeFeats.close();
445                         // maak een nieuw transactie voor toevoegen, de eerdere is aborted
446                         transaction = new DefaultTransaction("add-bgt");
447                         store.setTransaction(transaction);
448                     } else {
449                         throw ioe;
450                     }
451                 }
452             }
453             opmerkingen.append("Aantal ingevoegde features voor: ").append(gmlFileName).append(": ").append(writtenFeatures).append("\n");
454             LOG.info("Aantal ingevoegde features voor " + gmlFileName + ": " + writtenFeatures);
455         } catch (IOException ioe) {
456             LOG.error("I/O database probleem tijdens insert van features", ioe);
457             this.status = STATUS.NOK;
458             transaction.rollback();
459         } finally {
460             transaction.close();
461             updatetransaction.close();
462             dataStore.dispose();
463         }
464         return writtenFeatures;
465     }

467     private boolean isDuplicateKeyViolationMessage(String message) {
468         return message != null
469                 /* oracle */
470                 && (message.startsWith("ORA-00001:")
471                 /* postgresql */
472                 | message.startsWith("ERROR: duplicate key value violates unique constraint")
473                 /* mssql */
474                 | message.contains("Cannot insert duplicate key in object"));
475     }

477     /**
478      * scan directory for zipfiles in de geconfigueerde directory.
479      *
480      * @return lijst met zipfiles, mogelijk leeg
481      * @see #setScanDirectory(java.io.File)
482      * @see #setScanDirectory(java.lang.String)
483      */
484     public List<File> scanDirectory() {
485         List<File> zipfiles = Collections.emptyList();
486 
487         if (scanDirectory != null && scanDirectory.isDirectory() && scanDirectory.canExecute()) {
488             FilenameFilter filter = new FilenameFilter() {
489                 /**
490                  * accepteer alleen zip files.
491                  *
492                  * @param dir the directory in which the file was found.
493                  * @param name the name of the file
494                  * @return {@code true} if and only if the name should be
495                  * included in the file list; {@code false} otherwise.
496                  */
497                 @Override
498                 public boolean accept(File dir, String name) {
499                     return name.toLowerCase().endsWith("zip");
500                 }
501             };
502             File files[] = scanDirectory.listFiles(filter);
503             zipfiles = Arrays.asList(files);
504         } else {
505             LOG.fatal("De directory (" + scanDirectory + ") kan niet worden gelezen of doorbladerd.");
506             this.status = STATUS.NOK;
507         }
508         return zipfiles;
509     }

589     /**
590      * Test of de zipFile een geldige zipfile is.
591      *
592      * @param file de te testen zipfile
593      * @return {@code true} als de zipfile OK is, aders {@code false}
594      */
595     private boolean isValidZipFile(final File file) {
596         ZipFile zipfile = null;
597         ZipInputStream zis = null;
598         String path = null;
599         try {
600             path = file.getCanonicalPath();
601             zipfile = new ZipFile(file, ZipFile.OPEN_READ);
602             zis = new ZipInputStream(new FileInputStream(file));
603             ZipEntry ze = zis.getNextEntry();
604             if (ze == null) {
605                 return false;
606             }
607             while (ze != null) {
608                 // als er een exception bij 1 van volgende optreed is het bestand corrupt
609                 zipfile.getInputStream(ze);
610                 ze.getCrc();
611                 ze.getCompressedSize();
612                 ze.getName();
613                 ze = zis.getNextEntry();
614             }
615             return true;
616         } catch (IOException e) {
617             LOG.error("Ongeldige zipfile: " + path, e);
618             return false;
619         } finally {
620             try {
621                 if (zipfile != null) {
622                     zipfile.close();
623                     zipfile = null;
624                 }
625             } catch (IOException e) {
626                 LOG.warn("Fout tijdens sluiten zipfile, mogelijk corrupt bestand: " + path, e);
627                 // return false;
628             }
629             try {
630                 if (zis != null) {
631                     zis.close();
632                     zis = null;
633                 }
634             } catch (IOException e) {
635                 LOG.warn("Fout tijdens sluiten zipstream, mogelijk corrupt bestand: " + path, e);
636                 // return false;
637             }
638         }
639     }
