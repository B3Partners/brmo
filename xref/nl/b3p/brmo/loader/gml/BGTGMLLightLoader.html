153     /**
154      * Verwerk een gmllight extract zipfile.
155      *
156      * @param zipExtract zipfile met gml bestanden
157      * @return het aantal geschreven features voor de zipfile
158      *
159      * @throws FileNotFoundException als zipExtract niet gevonden kan worden
160      * @throws IOException als ophalen next zipentry mislukt of als de database
161      * verbinding wegvalt
162      */
163     public int processZipFile(File zipExtract) throws FileNotFoundException, IOException {
164         this.omhullendeVanZipFile = null;
165         this.resetStatus();
166 
167         int result = 0, total = 0;
168         String eName = "";
169         if (this.isValidZipFile(zipExtract)) {
170             try (ZipInputStream zip = new ZipInputStream(new FileInputStream(zipExtract))) {
171                 ZipEntry entry = zip.getNextEntry();
172                 if (entry == null) {
173                     LOG.error("Geen bestanden in zipfile (" + zipExtract + ") gevonden.");
174                 }
175                 // for each gml in zip
176                 while (entry != null) {
177                     if (!entry.getName().toLowerCase().endsWith(".gml")) {
178                         LOG.warn("Overslaan zip entry geen GML bestand: " + entry.getName());
179                     } else {
180                         eName = entry.getName();
181                         LOG.info("Lezen GML bestand: " + eName + " uit zip file: " + zipExtract.getCanonicalPath());
182                         result = storeFeatureCollection(new CloseShieldInputStream(zip), eName.toLowerCase());
183                         opmerkingen.append(result)
184                                 .append(" features geladen uit: ")
185                                 .append(eName).append(", zipfile: ")
186                                 .append(zipExtract.getCanonicalPath())
187                                 .append("\n");
188                         total += result;
189                     }
190                     entry = zip.getNextEntry();
191                 }
192 
193             } catch (SAXException | ParserConfigurationException ex) {
194                 LOG.error("Er is een parse fout opgetreden tijdens verwerken van " + eName + " uit " + zipExtract.getCanonicalPath(), ex);
195                 opmerkingen.append("Er is een parse fout opgetreden tijdens verwerken van ").append(eName)
196                         .append(" Foutmelding: ").append(ex).append("\n");
197                 this.status = STATUS.NOK;
198             }
199         } else {
200             LOG.error("Ongeldige of corrupte zipfile: " + zipExtract.getCanonicalPath());
201             opmerkingen.append("Ongeldige of corrupte zipfile: ").append(zipExtract.getCanonicalPath())
202                     .append("\nHet bestand kan niet verwerkt worden. Download het bestand opnieuw.");
203             this.status = STATUS.NOK;
204         }
205         return total;
206     }

208     /**
209      * Verwerk een enkel GML bestand. <strong>Let op:</strong> alleen voor
210      * "stand"     * verwerking omdat de onderliggende geometrische data niet betrouwbaar kan
211      * worden verwijderd in dit geval. Waarschijnlijk wil je ook de status
212      * resetten voor deze Loader, gebruik {@link #resetStatus() }.
213      *
214      * @param gml GML Light bestand
215      * @return aantal geschreven features
216      *
217      * @throws IOException als ophalen gml mislukt
218      */
219     public int processGMLFile(File gml) throws IOException {
220         int result = 0;
221         try {
222             result = storeFeatureCollection(new FileInputStream(gml), gml.getName().toLowerCase());
223         } catch (SAXException | ParserConfigurationException ex) {
224             LOG.error("Er is een parse fout opgetreden tijdens verwerken van: " + gml.getCanonicalPath(), ex);
225             this.status = STATUS.NOK;
226         }
227         return result;
228     }

230     /**
231      * transformeert de input en slaat op in database.
232      *
233      * @param gmlFeatCollection te laden features
234      * @param gmlFileName naam input gml bestand
235      * @return aantal geschreven features
236      *
237      * @throws SAXException als parsen van input mislukt
238      * @throws IOException als er een database fout optreedt
239      * @throws ParserConfigurationException als gml parser config niet deugd
240      * @throws IllegalStateException als er iets mis is in de configuratie
241      */
242     private int storeFeatureCollection(InputStream in, String gmlFileName) throws IOException, IllegalStateException, SAXException, ParserConfigurationException {
243         int inserts = 0, updates = 0, deletes = 0, features = 0;
244         JDBCDataStore dataStore = (JDBCDataStore) DataStoreFinder.getDataStore(dbConnProps);
245         if (dataStore == null) {
246             throw new IllegalStateException("Datastore mag niet 'null' zijn voor opslaan van data.");
247         }
248         // TODO boolean mapping voor mssql en oracle
249         if (this.isOracle) {
250             dataStore.getClassToSqlTypeMappings().put(java.lang.Boolean.class, Types.VARCHAR);
251         }
252         if (this.isMSSQL) {
253             dataStore.getClassToSqlTypeMappings().put(java.lang.Boolean.class, Types.VARCHAR);
254         }
255 
256         SimpleFeatureCollection gmlFeatCollection = (SimpleFeatureCollection) parser.parse(in);
257         
258         List<Exception> validationErrors = parser.getValidationErrors();
259         if (!validationErrors.isEmpty()) {
260             LOG.warn("Er zijn validatie fouten opgetreden tijdens verwerking van bestand: " + gmlFileName);
261             for (Exception e : validationErrors) {
262                 LOG.warn("validatie fout: " + e.getMessage());
263             }
264         }
265 
266         if (gmlFeatCollection.isEmpty()) {
267             opmerkingen.append("Geen features gevonden in bestand: ").append(gmlFileName).append("\n");
268             LOG.info("Geen features gevonden in bestand: " + gmlFileName);
269             dataStore.dispose();
270             return 0;
271         }
272 
273         SimpleFeatureType sft = gmlFeatCollection.getSchema();
274         GMLLightFeatureTransformer featTransformer = BGTGMLLightTransformerFactory.getTransformer(sft.getTypeName());
275         if (featTransformer == null) {
276             LOG.error("Opzoeken van FeatureTransformer voor " + sft.getTypeName() + " is mislukt; er geen transformer beschikbaar voor dit bestand.");
277             dataStore.dispose();
278             return 0;
279         }
280 
281         // check table exists
282         String tableName = BGTGMLLightTransformerFactory.getTableName(sft.getTypeName());
283         boolean exists = false;
284         String[] typeNames = dataStore.getTypeNames();
285         for (String name : typeNames) {
286             if (tableName.equalsIgnoreCase(name)) {
287                 LOG.debug("De tabel '" + tableName + "' is gevonden in de database als: " + name);
288                 exists = true;
289             }
290         }
291 
292         SimpleFeatureType targetSchema = featTransformer.getTargetSchema(sft, tableName, this.isOracle);
293         LOG.debug("Doel tabel schema: " + targetSchema);
294 
295         if (!exists) {
296             if (createTables) {
297                 dataStore.createSchema(targetSchema);
298                 LOG.warn("De volgende tabel is aangemaakt in de database: " + targetSchema.getTypeName());
299                 if (this.isOracle) {
300                     try {
301                         dataStore.getConnection(Transaction.AUTO_COMMIT).createStatement().execute("UPDATE USER_SDO_GEOM_METADATA SET SRID=28992");
302                     } catch (SQLException ex) {
303                         LOG.error("Bijwerken ruimtelijke metadata is mislukt", ex);
304                     }
305                 }
306             } else {
307                 LOG.error("Tabel: " + tableName + " is niet beschikbaar in de database.");
308                 throw new IllegalStateException("De tabel " + tableName + " ontbreekt in de database; opslaan van gegevens uit GML " + gmlFileName + " is niet mogelijk.");
309             }
310         } else {
311             // als tabel bestaat aannemen dat deze de juiste primary key heeft
312             dataStore.setExposePrimaryKeyColumns(true);
313 
314             // huidige records verwijderen
315             LOG.info("Probeer tabel leeg te maken: " + targetSchema.getTypeName());
316 
317             try {
318                 StringBuffer sql = new StringBuffer("TRUNCATE TABLE ");
319                 Method encodeTableName = dataStore.getClass().getDeclaredMethod("encodeTableName", String.class, StringBuffer.class, Hints.class);
320                 encodeTableName.setAccessible(true);
321                 encodeTableName.invoke(dataStore, targetSchema.getTypeName(), sql, null);
322 
323                 if(this.isOracle) {
324                     sql.append(" PURGE MATERIALIZED VIEW LOG REUSE STORAGE");
325                 }
326                 LOG.debug("SQL: " + sql.toString());
327                 new QueryRunner(dataStore.getDataSource()).update(sql.toString());
328                 LOG.info("Tabel leeggemaakt");
329             } catch(Exception e) {
330                 LOG.debug("Exception bij TRUNCATE", e);
331                 LOG.error("Fout bij TRUNCATE, probeer met DELETE FROM: " + e.getClass() + ": " + e.getMessage());
332 
333                 try {
334                     StringBuffer sql = new StringBuffer("DELETE FROM ");
335                     Method encodeTableName = dataStore.getClass().getDeclaredMethod("encodeTableName", String.class, StringBuffer.class, Hints.class);
336                     encodeTableName.setAccessible(true);
337                     encodeTableName.invoke(dataStore, targetSchema.getTypeName(), sql, null);
338                     LOG.debug("SQL: " + sql.toString());
339                     new QueryRunner(dataStore.getDataSource()).update(sql.toString());
340                 LOG.info("Tabel leeggemaakt met DELETE FROM");
341                 } catch(Exception e2) {
342                     LOG.error("Fout bij DELETE FROM, tabel kon niet leeggemaakt worden! ", e2);
343                 }
344             }
345         }
346 
347         // Map om per ID bij te houden wat meest recente tijdstipRegistratie en beeindigd status is
348         // Niet in map: niet in tabel
349         // TRUE: beeindigd en niet in tabel
350         // FALSE: niet beeindigd, en in tabel
351         Map<String,Pair<Date,Boolean>> records = new HashMap();
352 
353         Transaction transaction = new DefaultTransaction("add-bgt");
354         Transaction updatetransaction = new DefaultTransaction("update-bgt");
355         FilterFactory2 ff = CommonFactoryFinder.getFilterFactory2(null);
356         try (FeatureIterator<SimpleFeature> feats = gmlFeatCollection.features()) {
357             FeatureStore store = (FeatureStore) dataStore.getFeatureSource(targetSchema.getTypeName(), transaction);
358             FeatureStore store2 = (FeatureStore) dataStore.getFeatureSource(targetSchema.getTypeName(), updatetransaction);
359 
360             SimpleFeature gmlSF;
361             SimpleFeature transformed;
362             while (feats.hasNext()) {
363                 gmlSF = feats.next();
364                 features++;
365 
366                 transformed = featTransformer.transform(gmlSF, targetSchema, this.isOracle, dataStore.isExposePrimaryKeyColumns());
367                 if (transformed == null) {
368                     LOG.warn("Fout bij transformeren feature op index " + features + ", null geretourneerd");
369                     continue;
370                 }
371 
372                 String id = transformed.getID();
373                 Pair<Date,Boolean> record = records.get(id);
374                 Date tijdstipRegistratie = (Date) gmlSF.getAttribute("tijdstipRegistratie");
375                 if(tijdstipRegistratie == null) {
376                     LOG.error("Feature voor object " + id + " zonder tijdstipRegistratie gevonden, genegeerd");
377                     continue;
378                 }
379                 boolean beeindigd = gmlSF.getAttribute("objectEindTijd") != null;
380                 if(LOG.isDebugEnabled()) {
381                     LOG.info(String.format("Object %s, beeindigd=%s, tijdstipRegistratie=%tc, eerder tijdstip %tc, eerder beeindigd %s", id, beeindigd ? "ja":"nee", tijdstipRegistratie, record != null ? record.getLeft() : null, record != null ? (record.getRight() ? "ja":"nee") : "-"));
382                 }
383                 if(record != null && tijdstipRegistratie.before(record.getLeft())) {
384                     LOG.warn(String.format("Feature voor object %s gevonden met tijdstipRegistratie %tc eerder dan vorig record van %tc, genegeerd", id, tijdstipRegistratie, record.getLeft()));
385                     continue;
386                 }
387 
388                 if(beeindigd) {
389                     Date eindTijd = (Date) gmlSF.getAttribute("objectEindTijd");
390                     if(eindTijd.after(new Date())) {
391                         LOG.warn(String.format("Object %s vervallen in de toekomst op %tc, wordt alvast verwijderd!", id, eindTijd));
392                     }
393 
394                     if(record != null) {
395                         if(!record.getRight()) {
396                             // Verwijder eerder geinsert record
397                             LOG.info(String.format("Object %s vervallen met tijdstipRegistratie %tc, verwijder eerder geinsert record met laatste tijdstipRegistratie %tc", id, tijdstipRegistratie, record.getLeft()));
398 
399                             // bij mssql transactie sluiten ander blijft de boel hangen, voor orcl + pg ook
400                             transaction.close();
401                             Filter filter = ff.id(transformed.getIdentifier());
402                             store2.removeFeatures(filter);
403                             updatetransaction.commit();
404                             // maak een nieuw transactie voor toevoegen, de eerdere is aborted
405                             transaction = new DefaultTransaction("add-bgt");
406                             store.setTransaction(transaction);
407                             deletes++;
408                         }
409                     }
410                     records.put(id, new ImmutablePair<>(tijdstipRegistratie, true));
411                 } else {
412 
413                     if(record != null) {
414                         if(record.getRight()) {
415                             LOG.error(String.format("Object %s was eerder vervallen met tijdstipRegistratie %tc, nu weer niet met tijdstipRegistratie %tc, genegeerd", id, record.getLeft(), tijdstipRegistratie));
416                             // Alternatief weer inserten?
417                             continue;
418                         } else {
419                             // bij mssql transactie sluiten ander blijft de boel hangen, voor orcl + pg ook
420                             transaction.close();
421                             // object opzoeken,
422                             Filter filter = CommonFactoryFinder.getFilterFactory2(null).id(transformed.getIdentifier());
423                             FeatureIterator<SimpleFeature> bestaandeFeats = store2.getFeatures(filter).features();
424                             SimpleFeature bestaandeFeat = bestaandeFeats.next();
425 
426                             final String idAttr = this.isOracle ? ID_NAME.toUpperCase() : ID_NAME;
427                             for (AttributeDescriptor attr : transformed.getFeatureType().getAttributeDescriptors()) {
428                                 Name attrName = attr.getName();
429                                 if (!attrName.getLocalPart().equals(idAttr)) {
430                                     store2.modifyFeatures(attrName, transformed.getAttribute(attrName), filter);
431                                 }
432                             }
433                             updatetransaction.commit();
434                             LOG.info(String.format("Object %s geupdate van tijdstip %tc naar nieuw tijdstipRegistratie %tc", id, record.getLeft(), tijdstipRegistratie));
435                             bestaandeFeats.close();
436                             // maak een nieuw transactie voor toevoegen, de eerdere is aborted
437                             transaction = new DefaultTransaction("add-bgt");
438                             store.setTransaction(transaction);
439                             updates++;
440                         }
441                     } else {
442                         store.addFeatures(DataUtilities.collection(transformed));
443                         inserts++;
444                         // commit per feature
445                         transaction.commit();
446                         LOG.debug(String.format("Object toegevoegd in database met NEN3610ID: %s en tijdstipRegistratie %tc", id, tijdstipRegistratie));
447                     }
448                     records.put(id, new ImmutablePair<>(tijdstipRegistratie, false));
449                 }
450             }
451             String s = String.format("Totaal verwerkte features voor %s: %d, inserts: %d, updates: %d, deletes: %d\n", gmlFileName, features, inserts, updates, deletes);
452             opmerkingen.append(s).append("\n");
453             LOG.info(s);
454         } catch (IOException ioe) {
455             String s = String.format("Fout opgetreden, hiervoor verwerkte features voor %s: %d, inserts: %d, updates: %d, deletes: %d\n", gmlFileName, features, inserts, updates, deletes);
456             opmerkingen.append(s).append("\n");
457             LOG.info(s);
458 
459             LOG.error("I/O database probleem tijdens insert van features", ioe);
460             this.status = STATUS.NOK;
461             transaction.rollback();
462         } finally {
463             try {
464                 transaction.close();
465                 updatetransaction.close();
466                 dataStore.dispose();
467             } catch(Exception e) {
468                 LOG.error("Fout sluiten datastores", e);
469             }
470         }
471         return inserts;
472     }

474     private boolean isDuplicateKeyViolationMessage(String message) {
475         return message != null
476                 /* oracle */
477                 && (message.startsWith("ORA-00001:")
478                 /* postgresql */
479                 | message.startsWith("ERROR: duplicate key value violates unique constraint")
480                 /* mssql */
481                 | message.contains("Cannot insert duplicate key in object"));
482     }

484     /**
485      * scan directory for zipfiles in de geconfigueerde directory.
486      *
487      * @return lijst met zipfiles, mogelijk leeg
488      * @see #setScanDirectory(java.io.File)
489      * @see #setScanDirectory(java.lang.String)
490      */
491     public List<File> scanDirectory() {
492         List<File> zipfiles = Collections.emptyList();
493 
494         if (scanDirectory != null && scanDirectory.isDirectory() && scanDirectory.canExecute()) {
495             FilenameFilter filter = new FilenameFilter() {
496                 /**
497                  * accepteer alleen zip files.
498                  *
499                  * @param dir the directory in which the file was found.
500                  * @param name the name of the file
501                  * @return {@code true} if and only if the name should be
502                  * included in the file list; {@code false} otherwise.
503                  */
504                 @Override
505                 public boolean accept(File dir, String name) {
506                     return name.toLowerCase().endsWith("zip");
507                 }
508             };
509             File files[] = scanDirectory.listFiles(filter);
510             zipfiles = Arrays.asList(files);
511         } else {
512             LOG.fatal("De directory (" + scanDirectory + ") kan niet worden gelezen of doorbladerd.");
513             this.status = STATUS.NOK;
514         }
515         return zipfiles;
516     }

583     /**
584      * Test of de zipFile een geldige zipfile is.
585      *
586      * @param file de te testen zipfile
587      * @return {@code true} als de zipfile OK is, aders {@code false}
588      */
589     private boolean isValidZipFile(final File file) {
590         ZipFile zipfile = null;
591         ZipInputStream zis = null;
592         String path = null;
593         try {
594             path = file.getCanonicalPath();
595             zipfile = new ZipFile(file, ZipFile.OPEN_READ);
596             zis = new ZipInputStream(new FileInputStream(file));
597             ZipEntry ze = zis.getNextEntry();
598             if (ze == null) {
599                 return false;
600             }
601             while (ze != null) {
602                 // als er een exception bij 1 van volgende optreed is het bestand corrupt
603                 zipfile.getInputStream(ze);
604                 ze.getCrc();
605                 ze.getCompressedSize();
606                 ze.getName();
607                 ze = zis.getNextEntry();
608             }
609             return true;
610         } catch (IOException e) {
611             LOG.error("Ongeldige zipfile: " + path, e);
612             return false;
613         } finally {
614             try {
615                 if (zipfile != null) {
616                     zipfile.close();
617                     zipfile = null;
618                 }
619             } catch (IOException e) {
620                 LOG.warn("Fout tijdens sluiten zipfile, mogelijk corrupt bestand: " + path, e);
621                 // return false;
622             }
623             try {
624                 if (zis != null) {
625                     zis.close();
626                     zis = null;
627                 }
628             } catch (IOException e) {
629                 LOG.warn("Fout tijdens sluiten zipstream, mogelijk corrupt bestand: " + path, e);
630                 // return false;
631             }
632         }
633     }
