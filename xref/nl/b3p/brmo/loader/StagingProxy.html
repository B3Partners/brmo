636      * @param fileName naam van het bestand (ter identificatie)

642     public void loadBr(InputStream stream, String type, String fileName, Date d, ProgressUpdateListener listener, Long automatischProces) throws Exception {
643 
644         CountingInputStream cis = new CountingInputStream(stream);
645 
646         BrmoXMLReader brmoXMLReader = null;
647         if (type.equals(BrmoFramework.BR_BRK)) {
648             brmoXMLReader = new BrkSnapshotXMLReader(cis);
649         } else if (type.equals(BrmoFramework.BR_BAG)) {
650             brmoXMLReader = new BagXMLReader(cis);
651         } else if (type.equals(BrmoFramework.BR_NHR)) {
652             brmoXMLReader = new NhrXMLReader(cis);
653         } else if (type.equals(BrmoFramework.BR_BGTLIGHT)) {
654             brmoXMLReader = new BGTLightFileReader(fileName);
655         } else if (TopNLType.isTopNLType(type)) {
656             brmoXMLReader = new TopNLFileReader(fileName, type);
657         } else if(type.equals(BrmoFramework.BR_BRP)){
658             brmoXMLReader = new BRPXMLReader(cis, d);
659         } else if (type.equals(BrmoFramework.BR_GBAV)) {
660             brmoXMLReader = new GbavXMLReader(cis);
661         } else {
662             throw new UnsupportedOperationException("Ongeldige basisregistratie: " + type);
663         }
664 
665         if (brmoXMLReader.getBestandsDatum()==null) {
666             throw new BrmoException("Header van bestand bevat geen datum, verkeerd formaat?");
667         }
668 
669         LaadProcesadProces.html#LaadProces">LaadProces lp = new LaadProces();
670         lp.setBestandNaam(fileName);
671         lp.setBestandDatum(brmoXMLReader.getBestandsDatum());
672         lp.setSoort(type);
673         lp.setGebied(brmoXMLReader.getGebied());
674         lp.setStatus(LaadProces.STATUS.STAGING_OK);
675         lp.setStatusDatum(new Date());
676         lp.setAutomatischProcesId(automatischProces);
677 
678         if (laadProcesExists(lp)) {
679             throw new BrmoDuplicaatLaadprocesException("Laadproces al gerund voor bestand "+ fileName);
680         }
681         lp = writeLaadProces(lp);
682 
683         if (type.equalsIgnoreCase(BrmoFramework.BR_BGTLIGHT)) {
684             // van een BGT Light bestand maken we alleen een LP, geen bericht,
685             // de datum halen we van een GML uit het zip bestand
686             if (listener != null) {
687                 listener.total(((BGTLightFileReader) brmoXMLReader).getFileSize());
688                 listener.progress(((BGTLightFileReader) brmoXMLReader).getFileSize());
689             }
690         } else if (TopNLType.isTopNLType(type)) {
691             // van een TopNL GML bestand maken we alleen een LP, geen bericht,
692             // de datum halen we van het zip bestand
693             if (listener != null) {
694                 listener.total(((TopNLFileReader) brmoXMLReader).getFileSize());
695                 listener.progress(((TopNLFileReader) brmoXMLReader).getFileSize());
696             }
697         } else {
698             if (!brmoXMLReader.hasNext()) {
699                 throw new BrmoLeegBestandException("Leeg bestand, geen berichten gevonden in " + fileName);
700             }
701 
702             boolean isBerichtGeschreven = false;
703             int berichten = 0;
704             int foutBerichten = 0;
705             String lastErrorMessage = null;
706 
707             while (brmoXMLReader.hasNext()) {
708                 Bericht b = null;
709                 try {
710                     b = brmoXMLReader.next();
711                     b.setLaadProcesId(lp.getId().intValue());
712                     b.setStatus(Bericht.STATUS.STAGING_OK);
713                     b.setStatusDatum(new Date());
714                     b.setSoort(type);
715 
716                     if (b.getDatum() == null) {
717                         throw new BrmoException("Datum bericht is null");
718                     }
719                     log.debug(b);
720 
721                     Bericht existingBericht = getExistingBericht(b);
722                     if (existingBericht == null) {
723                         writeBericht(b);
724                         isBerichtGeschreven = true;
725                     } else if (existingBericht.getStatus().equals(Bericht.STATUS.STAGING_OK)) {
726                         //als bericht nog niet getransformeerd is, dan overschrijven.
727                         //als een BAG bericht inactief wordt gezet dan zal het
728                         //oorspronkelijke bericht nog getransformeerd
729                         //moeten worden, door overschrijven wordt dit bericht inactief
730                         //en zal nooit getransformeerd worden.
731                         b.setId(existingBericht.getId());
732                         this.updateBericht(b);
733                     }
734                     if (listener != null) {
735                         listener.progress(cis.getByteCount());
736                     }
737                     berichten++;
738                 } catch (Exception e) {
739                     lastErrorMessage = String.format("Laden bericht uit %s mislukt vanwege: %s",
740                             fileName, e.getLocalizedMessage());
741                     log.error(lastErrorMessage);
742                     if (listener != null) {
743                         listener.exception(e);
744                     }
745                     foutBerichten++;
746                 }
747             }
748             if (listener != null) {
749                 listener.total(berichten);
750             }
751             if (foutBerichten > 0) {
752                 String opmerking = "Laden van " + foutBerichten
753                         + " bericht(en) mislukt, laatste melding: "
754                         + lastErrorMessage + ", zie logs voor meer info.";
755                 this.updateLaadProcesStatus(lp, LaadProces.STATUS.STAGING_NOK, opmerking);
756             } else if (!isBerichtGeschreven) {
757                 String opmerking = "Dit bestand is waarschijnlijk al eerder geladen.";
758                 this.updateLaadProcesStatus(lp, LaadProces.STATUS.STAGING_DUPLICAAT, opmerking);
759             }
760         }

944     private String buildFilterSql(int page, String sort, String dir, String filterSoort, String filterStatus, List<String> params) {
945 
946         StringBuilder builder = new StringBuilder();
947 
948         List<String> conjunctions = new ArrayList();
949 
950         if(!StringUtils.isBlank(filterSoort)) {
951             String[] soorten = filterSoort.split(",");
952             params.addAll(Arrays.asList(soorten));
953             conjunctions.add("soort in (" + StringUtils.repeat("?", ", ", soorten.length) + ")");
954         }
955 
956         if(!StringUtils.isBlank(filterStatus)) {
957             String[] statussen = filterStatus.split(",");
958             params.addAll(Arrays.asList(statussen));
959             conjunctions.add("status in (" + StringUtils.repeat("?", ", ", statussen.length) + ")");
960         }
961 
962         // build where part
963         if(!conjunctions.isEmpty()) {
964             builder.append(" where ");
965             builder.append(StringUtils.join(conjunctions.toArray(), " and "));
966         }
967 
968         // build order by part
969         if (sort != null && dir != null) {
970             builder.append(" ORDER BY ");
971             builder.append(sort);
972             builder.append(" ");
973             builder.append(dir);
974         }
975 
976         return builder.toString();
977     }
