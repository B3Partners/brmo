13  public class ColumnMetadata implements Comparable {
14      private String name;
15      private int position;
16      private int dataType;
17      private String typeName;
18      private int size;
19      private Integer decimalDigits;
20      private boolean nullable;
21      private Integer charOctetLength;
22      private String defaultValue;
23      private boolean autoIncrement;
24      
25      public ColumnMetadata(ResultSet columnMetadataRs) throws SQLException {
26          name = columnMetadataRs.getString("COLUMN_NAME");
27          position = columnMetadataRs.getInt("ORDINAL_POSITION");
28          dataType = columnMetadataRs.getInt("DATA_TYPE");
29          typeName = columnMetadataRs.getString("TYPE_NAME");
30          size = columnMetadataRs.getInt("COLUMN_SIZE");
31          Object o = columnMetadataRs.getObject("DECIMAL_DIGITS");
32          if(o != null) {
33              if(o instanceof BigDecimal) {
34                  decimalDigits = ((BigDecimal)o).intValue();
35              } else if(o instanceof Integer) {
36                  decimalDigits = (Integer)o;
37              }
38          }
39          nullable = DatabaseMetaData.columnNullable == columnMetadataRs.getInt("NULLABLE");
40          o = columnMetadataRs.getObject("CHAR_OCTET_LENGTH");
41          if(o != null) {
42              if(o instanceof BigDecimal) {
43                  charOctetLength = ((BigDecimal)o).intValue();
44              } else if(o instanceof Integer) {
45                  charOctetLength = (Integer)o;
46              }
47          }
48          defaultValue = columnMetadataRs.getString("COLUMN_DEF");
49          //autoIncrement = "YES".equals(columnMetadataRs.getString("IS_AUTOINCREMENT"));
50      }

124     @Override
125     public String toString() {
126         return toString("%s %4d %s %s bytes=%d %s %s");
127     }
128     
129     public String typeToString() {
130         return typeName + "(" + size + 
131                 (decimalDigits == null ? "" : "," + decimalDigits) + ")";
132     }
133     
134     public String toStringFixedWidth(Collection<ColumnMetadata> allColumns) {
135         int nameWidth = name.length(), typeWidth = typeToString().length();
136         for(ColumnMetadata c: allColumns) {
137             nameWidth = Math.max(c.getName().length(), nameWidth);
138             typeWidth = Math.max(c.typeToString().length(), typeWidth);
139         }
140         return toString("%-" + nameWidth + "s %4d %-" + typeWidth + "s %8s bytes=%d %s %s");
141     }
142     
143     private String toString(String formatString) {
144         
145         return String.format(formatString,
146                 name,
147                 dataType,
148                 typeToString(),
149                 nullable ? "NULL" : "NOT NULL",
150                 charOctetLength,
151                 autoIncrement ? "AUTO INCREMENT" : "",
152                 defaultValue == null ? "" : "DEFAULT '" + defaultValue + "'");
153     }
154 
155     public int compareTo(Object rhs) {
156         return new Integer(position).compareTo(((ColumnMetadata)rhs).position);
157     }
