189   public void init() throws SQLException {
190     ResultSet tablesRs = null;
191     if (null != this.dataSourceRsgb) {
192       connRsgb = dataSourceRsgb.getConnection();
193       geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(connRsgb);
194 
195       try {
196         dbMetadata = connRsgb.getMetaData();
197         tablesRs = dbMetadata.getTables(null, geomToJdbc.getSchema(), "%", new String[] {"TABLE"});
198         while (tablesRs.next()) {
199           tables.put(
200               tablesRs.getString("TABLE_NAME").toLowerCase(), tablesRs.getString("TABLE_NAME"));
201         }
202       } catch (SQLException sqlEx) {
203         throw new SQLException("Fout bij ophalen rsgb tabellen: ", sqlEx);
204       } finally {
205         DbUtils.closeQuietly(tablesRs);
206       }
207     }
208 
209     if (null != dataSourceRsgbBrk) {
210       connRsgbBrk = dataSourceRsgbBrk.getConnection();
211       geomToJdbcBrk = GeometryJdbcConverterFactory.getGeometryJdbcConverter(connRsgbBrk);
212 
213       try {
214         dbMetadataBrk = connRsgbBrk.getMetaData();
215         tablesRs =
216             dbMetadataBrk.getTables(null, geomToJdbcBrk.getSchema(), "%", new String[] {"TABLE"});
217         while (tablesRs.next()) {
218           tables.put(
219               tablesRs.getString("TABLE_NAME").toLowerCase(), tablesRs.getString("TABLE_NAME"));
220         }
221       } catch (SQLException sqlEx) {
222         throw new SQLException("Fout bij ophalen rsgb brk tabellen: ", sqlEx);
223       } finally {
224         DbUtils.closeQuietly(tablesRs);
225       }
226     }

278   public void run() {
279     try {
280       init();

282       long total = setWaitingStatus();
283       if (listener != null) {
284         listener.total(total);
285       }
286       // Do the work by querying all berichten, berichten are passed to
287       // handle() method
288       if (total > 0) {
289         stagingProxy.handleBerichtenByJob(
290             total, this, enablePipeline, pipelineCapacity, orderBerichten);
291       }
292     } catch (Exception e) {
293       // user is informed via status in database
294       log.error("Fout tijdens verwerken berichten", e);
295 
296       if (listener != null) {
297         listener.exception(e);
298       }
299     }
300 
301     close();
302   }

304   private long setWaitingStatus() throws SQLException, BrmoException {
305     // verwijder al verwerkte berichten uit afgebroken job
306     stagingProxy.cleanJob();
307     // zijn er nog berichten over uit de vorige job
308     long count = stagingProxy.getCountJob();
309     if (count > 0 && !mode.equals(BerichtSelectMode.RETRY_WAITING)) {
310       throw new BrmoException(
311           "Vorige transformatie is afgebroken," + " verwerk eerst die berichten (zie job-tabel)!");
312     }
313 
314     switch (mode) {
315       case BY_STATUS:
316         return stagingProxy.setBerichtenJobByStatus(status, orderBerichten);
317       case BY_IDS:
318         return stagingProxy.setBerichtenJobByIds(berichtIds, orderBerichten);
319       case BY_LAADPROCES:
320         return stagingProxy.setBerichtenJobByLaadprocessen(laadprocesIds, orderBerichten);
321       case FOR_UPDATE:
322         return stagingProxy.setBerichtenJobForUpdate(updateProcess.getSoort(), orderBerichten);
323       case RETRY_WAITING:
324         return count;
325       default:
326         return 0L;
327     }
328   }

390 
391     RsgbTransformer transformer = rsgbTransformers.get(updateProcess.getName());
392 
393     if (transformer == null) {

432   /**
433    * verwerk een bericht.
434    *
435    * @param ber te verwerken bericht
436    * @param pretransformedTableData tabel data uit staging
437    * @param updateResult {@code true} update verwerkings resultaat in staging
438    * @throws BrmoException if any
439    */
440   @Override
441   public void handle(Bericht ber, List<TableData> pretransformedTableData, boolean updateResult)
442       throws BrmoException {
443     if (updateProcess != null) {
444       update(ber, pretransformedTableData);
445       return;
446     }
447     Connection conn = this.connRsgb;
448     GeometryJdbcConverter jdbcConverter = this.geomToJdbc;
449     DatabaseMetaData metaData = this.dbMetadata;
450     if (ber.getSoort().equalsIgnoreCase(BrmoFramework.BR_BRK2)) {
451       conn = this.connRsgbBrk;
452       jdbcConverter = this.geomToJdbcBrk;
453       metaData = this.dbMetadataBrk;
454     }
455 
456     Bericht.STATUS newStatus = Bericht.STATUS.RSGB_OK;
457 
458     SimpleDateFormat dateTimeFormat = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss:SSS");
459     SimpleDateFormat dateFormat = new SimpleDateFormat("dd-MM-yyyy");
460 
461     log.debug(
462         String.format(
463             "RSGB verwerking van %s bericht met id %s, object_ref %s, datum %tc",
464             ber.getSoort(), ber.getId(), ber.getObjectRef(), ber.getDatum()));
465     StringBuilder loadLog;
466     if (pretransformedTableData == null) {
467       loadLog = new StringBuilder();
468     } else {
469       loadLog = new StringBuilder(ber.getOpmerking());
470     }
471     loadLog.append(
472         String.format(
473             "%s: start verwerking %s bericht object ref %s, met id %s, berichtdatum %s\n",
474             dateTimeFormat.format(new Date()),
475             ber.getSoort(),
476             ber.getObjectRef(),
477             ber.getId(),
478             dateFormat.format(ber.getDatum())));
479 
480     setHerkomstMetadata(ber.getSoort());
481 
482     long startTime = 0;
483     try {
484 
485       if (conn.getAutoCommit()) {
486         conn.setAutoCommit(false);
487       }
488 
489       List<TableData> newList;
490       if (pretransformedTableData != null) {
491         loadLog.append("Bericht was al getransformeerd in pipeline\n");
492         newList = pretransformedTableData;
493       } else {
494         loadLog.append("Transformeren naar RSGB database-xml en lezen resultaat... ");
495         startTime = System.currentTimeMillis();
496         newList = transformToTableData(ber);
497         loadLog.append(
498             String.format("%4.1fs\n", (System.currentTimeMillis() - startTime) / 1000.0));
499       }
500       // nieuwe starttijd voor laden in RSGB
501       startTime = System.currentTimeMillis();
502 
503       // per bericht kijken of er een oud bericht is
504       Bericht oud = null;
505       if (orderBerichten) {
506         oud = stagingProxy.getOldBericht(ber, loadLog);
507       } else {
508         // als geen orderberichten dan wordt stand ingelezen
509         // en indien volgordenummer >=0 (geen stand) dan afbreken
510         if (ber.getVolgordeNummer() >= 0) {
511           throw new BrmoException("Standverwerking, maar bericht is mutatie: niet verwerken!");
512         }
513         loadLog.append(
514             "Standverwerking, gegevens worden alleen toegevoegd als ze nog niet bestaan!\n");
515       }
516 
517       // oud bericht
518       if (oud != null) {
519         loadLog.append(
520             String.format(
521                 "Eerder bericht (id: %d) voor zelfde object gevonden van datum %s, volgnummer %d,status %s op %s\n",
522                 oud.getId(),
523                 dateTimeFormat.format(oud.getDatum()),
524                 oud.getVolgordeNummer(),
525                 oud.getStatus().toString(),
526                 dateTimeFormat.format(oud.getStatusDatum())));
527 
528         if (ber.getDatum().equals(oud.getDatum())
529             && ber.getVolgordeNummer().equals(oud.getVolgordeNummer())) {
530           if (Objects.equals(ber.getId(), oud.getId())) {
531             // dit kan voorkomen bij herstart van job terwijl bericht nog in pipeline
532             // van andere job zat
533             // niets doen, log overnemen.
534             loadLog.append(oud.getOpmerking());
535           } else {
536             loadLog.append(
537                 "Datum en volgordenummer van nieuw bericht hetzelfde als de oude, negeer dit bericht!\n");
538             boolean dbXmlEquals = ber.getDbXml().equals(oud.getDbXml());
539             if (!dbXmlEquals) {
540               String s =
541                   String.format(
542                       "Bericht %d met zelfde datum en volgnummer als eerder verwerkt bericht %d heeft andere db xml! Object ref %s",
543                       ber.getId(), oud.getId(), ber.getObjectRef());
544               log.warn(s);
545               loadLog.append("Waarschuwing: ").append(s).append("\n");
546             }
547           }
548         } else {
549           // Check of eerder bericht toevallig niet nieuwere datum
550           if (oud.getDatum().after(ber.getDatum())) {
551             newStatus = Bericht.STATUS.RSGB_OUTDATED;
552             loadLog.append(
553                 "Bericht bevat oudere data dan eerder verwerkt bericht, status RSGB_OUTDATED\n");
554             // Check of eerder bericht toevallig niet zelfde datum met hoger volgorde
555             // nummer, mantis-6098
556           } else if (oud.getDatum().equals(ber.getDatum())
557               && oud.getVolgordeNummer() > ber.getVolgordeNummer()) {
558             newStatus = Bericht.STATUS.RSGB_OUTDATED;
559             loadLog.append(
560                 "Bericht bevat oudere data dan eerder verwerkt bericht met hoger volgnummer, status RSGB_OUTDATED\n");
561           } else {
562             StringReader oReader = new StringReader(oud.getDbXml());
563             List<TableData> oudList = oldDbXmlReader.readDataXML(new StreamSource(oReader));
564 
565             parseNewData(oudList, newList, loadLog, jdbcConverter, conn, metaData);
566             parseOldData(oudList, newList, ber.getDatum(), loadLog, jdbcConverter, conn, metaData);
567           }
568         }
569       } else {
570         parseNewData(null, newList, loadLog, jdbcConverter, conn, metaData);
571       }
572 
573       Split commit = SimonManager.getStopwatch(simonNamePrefix + "commit").start();
574       conn.commit();
575       commit.stop();
576       ber.setStatus(newStatus);
577 
578       this.processed++;
579       if (listener != null) {
580         listener.progress(this.processed);
581       }
582 
583     } catch (Throwable ex) {
584       log.error(
585           "Fout bij verwerking bericht met id "
586               + ber.getId()
587               + ", melding: "
588               + ex.getLocalizedMessage());
589       log.debug("Fout bij verwerking bericht met id " + ber.getId(), ex);
590       try {
591         conn.rollback();
592       } catch (SQLException e) {
593         log.debug("Rollback exception", e);
594       }
595       ber.setOpmerking(loadLog.toString());
596       loadLog = null;
597       updateBerichtException(ber, ex);
598     } finally {
599       if (loadLog == null) {
600         loadLog = new StringBuilder(ber.getOpmerking());
601       }
602       loadLog.append("\n\n");
603       loadLog.append(
604           String.format(
605               "%s: einde verwerking, duur: %4.1fs",
606               dateTimeFormat.format(new Date()),
607               (System.currentTimeMillis() - startTime) / 1000.0));
608       ber.setOpmerking(loadLog.toString());
609 
610       if (updateResult) {
611         updateProcessingResult(ber);
612       }
613 
614       if (recentSavepoint != null) {
615         // Set savepoint to null, as it is automatically released after commit the
616         // transaction.
617         recentSavepoint = null;
618       }
619 
620       if (renewConnectionAfterCommit) {
621         log.debug("Vernieuwen van verbinding om cursors in Oracle vrij te geven!");
622         if (!(jdbcConverter instanceof OracleJdbcConverter)) {

1124       boolean useArchiveTable,
1125       StringBuilder loadLog,
1126       GeometryJdbcConverter jdbcConverter,
1127       Connection connection,
1128       DatabaseMetaData metaData)
1129       throws SQLException, ParseException {

1132

1135       tableName = row.getTable();
1136     } else {
1137       tableName = row.getTable() + "_archief";

1139

1141

1143     tableColumnMetadata = getTableColumnMetadata(tableName, jdbcConverter, metaData);
1144

1147         // Wanneer archief tabellen niet zijn aangemaakt negeren
1148         return;
1149       } else {
1150         throw new IllegalStateException("Kan tabel metadata niet vinden voor tabel " + tableName);
1151       }
1152     }
1153

1159     // zodat insert voor elke tabel hetzelfde, reuse preparedstatement

1161     for (Iterator<String> it = row.getColumns().iterator(); it.hasNext(); ) {
1162       String column = it.next();
1163       String stringValue = valuesIt.next();

1171       params.add(param);
1172

1174

1176       if (cm.getTypeName().equals(jdbcConverter.getGeomTypeName())
1177           // in geval we in een ander dan public schema werken, en postgis zit in public
1178           // schema
1179           || cm.getTypeName()
1180               .replace("\"", "")
1181               .equals("public." + jdbcConverter.getGeomTypeName())) {

1186

1188         sql.append(", ");
1189         valuesSql.append(", ");

1191     }
1192

1194

1198       sql.append(")");

1206       sql.append(tableName);
1207       sql.append(" where ");
1208

1211         if (first) {
1212           first = false;
1213         } else {
1214           sql.append(" and ");
1215         }
1216         sql.append(column);
1217         sql.append(" = ? ");
1218

1220         Object obj = getValueAsObject(tableName, column, val, null, jdbcConverter, metaData);
1221         params.add(obj);
1222       }
1223       sql.append(")");
1224     }
1225

1227         checkAndGetPreparedStatement(insertSqlPreparedStatements, sql.toString());
1228     if (stm == null) {
1229       SimonManager.getCounter("b3p.rsgb.insertsql.preparestatement").increase();
1230       stm = connection.prepareStatement(sql.toString());
1231       checkAndAddStatement(insertSqlPreparedStatements, sql.toString(), stm);
1232     } else {
1233       SimonManager.getCounter("b3p.rsgb.insertsql.reusestatement").increase();
1234     }
1235     loadLog.append("\nSQL: ");
1236     loadLog.append(sql);
1237     loadLog.append(", params (");
1238     loadLog.append(params);
1239     loadLog.append(")");
1240     log.trace("insert SQL: " + sql + ", params: " + params);
1241

1243       Object param = params.get(i);

1804   private void doSavePoint(TableRow row, GeometryJdbcConverter jdbcConverter, Connection connection)
1805       throws SQLException {
1806     if (!alsoExecuteSql) {
1807       return;
1808     }
1809     if (jdbcConverter.useSavepoints()) {
1810       if (row.isIgnoreDuplicates()) {
1811         if (recentSavepoint == null) {
1812           recentSavepoint = connection.setSavepoint();
1813           log.trace("Created savepoint with id: " + recentSavepoint.getSavepointId());
1814         } else {
1815           log.trace(
1816               "No need for new savepoint, previous insert caused rollback to recent savepoint with id "
1817                   + recentSavepoint.getSavepointId());
1818         }
1819       } else if (recentSavepoint != null) {
1820         log.trace(
1821             "About to insert non-recoverable row, discarding savepoint with id "
1822                 + recentSavepoint.getSavepointId());
1823         connection.releaseSavepoint(recentSavepoint);
1824         recentSavepoint = null;
1825       }
1826     }
