27    private static final Log LOG = LogFactory.getLog(TopNLRsgbTransformer.class);
28    private final StagingProxy stagingProxy;
29    private final DataSource dataSourceTopNL;
30    private final ProgressUpdateListener listener;
31    private final long[] lpIDs;
32    private Processor processor = null;
33  
34    public TopNLRsgbTransformer(
35        DataSource dataSourceTopNL,
36        StagingProxy stagingProxy,
37        long[] lpIDs,
38        ProgressUpdateListener listener)
39        throws JAXBException, SQLException {
40      this.stagingProxy = stagingProxy;
41      this.dataSourceTopNL = dataSourceTopNL;
42      this.lpIDs = lpIDs;
43      this.listener = listener;
44      this.processor = new Processor(this.dataSourceTopNL);
45    }
46  
47    private void transform(long lpID) throws SQLException {
48      STATUS status;
49      String opmerkingen;
50      LaadProces lp = stagingProxy.getLaadProcesById(lpID);
51  
52      if (TopNLType.isTopNLType(lp.getSoort()) && lp.getStatus() == STATUS.STAGING_OK) {
53        stagingProxy.updateLaadProcesStatus(lp, STATUS.RSGB_TOPNL_WAITING, "Transformatie loopt...");
54        File gml = new File(lp.getBestandNaam());
55        try {
56          processor.importIntoDb(gml.toURI().toURL(), TopNLType.valueOf(lp.getSoort().toUpperCase()));
57          switch (processor.getStatus()) {
58            case OK:
59              status = STATUS.RSGB_TOPNL_OK;
60              opmerkingen = "Geen fouten bij inladen.";
61              break;
62            case NOK:
63            default:
64              status = STATUS.RSGB_TOPNL_NOK;
65              opmerkingen =
66                  "Er zijn herstelbare fouten bij inladen opgetreden. Waarschijnlijk is niet alle data geladen van bestand "
67                      + gml;
68          }
69          stagingProxy.updateLaadProcesStatus(lp, status, opmerkingen);
70        } catch (JDOMException | MalformedURLException ex) {
71          LOG.debug("Error loading gml file", ex);
72          opmerkingen = "Laden van bestand " + gml + " is mislukt: " + ex.getLocalizedMessage();
73          status = STATUS.RSGB_TOPNL_NOK;
74          stagingProxy.updateLaadProcesStatus(lp, status, opmerkingen);
75        }
76      } else {
77        LOG.warn(
78            "LaadProces "
79                + lp.getId()
80                + " van soort "
81                + lp.getSoort()
82                + " met status: "
83                + lp.getStatus()
84                + " is overgeslagen.");
85      }
86    }

100   @Override
101   public void run() {
102     try {
103       init();
104       int count = 0;
105       if (listener != null) {
106         listener.total(lpIDs.length);
107         listener.progress(count);
108       }
109       for (long id : lpIDs) {
110         this.transform(id);
111         count++;
112         if (listener != null) {
113           listener.progress(count);
114         }
115       }
116     } catch (Exception e) {
117       LOG.error("Fout tijdens verwerken TOPNL laadprocessen", e);
118       if (listener != null) {
119         listener.exception(e);
120       }
121     }
122   }
