3   import java.beans.PropertyDescriptor;
4   import java.sql.ResultSetMetaData;
5   import java.sql.SQLException;
6   import org.apache.commons.dbutils.BeanProcessor;
7   
8   /**
9    * Custom Bean Processor used to map staging columns. Some columns contain 
10   * underscores which BasicRowProcessor does not map.
11   * 
12   * @author Boy de Wit
13   */
14  public class StagingBeanProcessor extends BeanProcessor {
15      
16      @Override
17      protected int[] mapColumnsToProperties(ResultSetMetaData rsmd,
18              PropertyDescriptor[] props) throws SQLException {
19          
20          int[] arr = new int[props.length];
21          arr[0] = -1; // meaningless
22          
23          int idx = 1;
24          for (PropertyDescriptor descr : props) {            
25              for (int i = 1; i <= rsmd.getColumnCount(); i++) {
26                  String name = rsmd.getColumnName(i);
27                  String testName = name.toLowerCase().replaceAll("_", "");                
28                  
29                  if (descr.getName().toLowerCase().equals(testName)) {
30                      arr[idx] = i;
31                      idx++;
32                      
33                      break;
34                  }
35              }
36          }
37          
38          return arr;
39      }    
40  }
