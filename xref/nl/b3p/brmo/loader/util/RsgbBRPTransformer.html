47      @Override
48      public String transformToDbXml(Bericht bericht) throws SAXException, IOException, TransformerConfigurationException, TransformerException {
49          String current = super.transformToDbXml(bericht);
50          StringBuilder loadLog = new StringBuilder();
51  
52          try {
53              Bericht old = staging.getPreviousBericht(bericht, loadLog);
54  
55              if (old != null) {
56                  Document d = merge(old.getDbXml(), current);
57                  String mergedDBXML = print(d);
58                  bericht.setDbXml(mergedDBXML);
59                  current = mergedDBXML;
60              }
61          } catch (SQLException ex) {
62              log.error("Cannot retrieve old bericht: ", ex);
63          } catch (Exception ex) {
64              log.error("Cannot retrieve old bericht: ", ex);
65          }
66  
67          // retrieve old bericht
68          // apply current to old
69          // return modified dbxml
70          return current;
71      }

83      protected static Document merge(String oldFile, String newFile) throws Exception {
84          XPathFactory xPathFactory = XPathFactory.newInstance();
85          XPath xpath = xPathFactory.newXPath();
86          XPathExpression expression = xpath.compile("/root/data");
87  
88          DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
89          docBuilderFactory.setIgnoringElementContentWhitespace(true);
90          DocumentBuilder docBuilder = docBuilderFactory.newDocumentBuilder();
91          Document base = docBuilder.parse(new InputSource(new StringReader(oldFile)));
92  
93          Element old = (Element) expression.evaluate(base, XPathConstants.NODE);
94          if (old == null) {
95              throw new IOException(oldFile + ": expression does not evaluate to node");
96          }
97  
98          Document merge = docBuilder.parse(new InputSource(new StringReader(newFile)));
99          Node newNode = (Node) expression.evaluate(merge, XPathConstants.NODE);
100 
101         /*
102             (1)Voor elke node in merge, kijk of hij bestaat in base
103                 zo nee, importeer
104                 zo ja, kijk of dit het een na diepste niveau is
105                     zo ja,
106                         ga voor elk childnode na of deze bestaat
107                             zo nee, importeer
108                             zo ja, overschrijf waarde
109                     zo nee, recurse in (1)
110          */
111         merge(base, newNode, old, true/*, merge*/);
112 
113         return base;
114     }

116     private static final String GEEN_WAARDE = "geenWaarde";
117     
118     private static void merge(Document base, Node newNode, Element old, boolean first/*, Node merge*/) {
119         while (newNode.hasChildNodes()) {
120             Node newChild = newNode.getFirstChild();
121             newNode.removeChild(newChild);
122             String name = newChild.getNodeName();
123             NodeList nl = old.getElementsByTagName(name);
124             if (nl.getLength() == 0) { // Geen oude data gevonden voor huidige node
125                 newChild = base.importNode(newChild, true);
126                 newChild.setTextContent(newChild.getTextContent());
127                 old.appendChild(newChild);
128             } else {
129                 Element oldItem = (Element) nl.item(0);
130                 if (first) {
131                     merge(base, newChild, oldItem, false);
132                 } else {
133                     String content = newChild.getTextContent();
134                     if (content.equals(GEEN_WAARDE)) {
135                         oldItem.setTextContent("");
136                     } else if(content.equals("")){
137                         //keep old content
138                     }else {
139                         oldItem.setTextContent(sanitizeValue(newChild.getTextContent()));
140                     }
141                 }
142             }
143         }
144     }

146     private static String sanitizeValue(String val){
147         if(val.contains(GEEN_WAARDE)){
148             String newValue = val.replaceAll(GEEN_WAARDE + " ", "");
149             newValue = newValue.replaceAll(GEEN_WAARDE, "");
150             return newValue;
151         }else{
152             return val;
153         }
154     }

156     protected static String print(Document doc) throws Exception {
157         TransformerFactory transformerFactory = TransformerFactory.newInstance();
158         Transformer transformer = transformerFactory.newTransformer();
159         DOMSource source = new DOMSource(doc);
160         StringWriter sw = new StringWriter();
161         Result result = new StreamResult(sw);
162         transformer.transform(source, result);
163         return sw.toString();
164     }
