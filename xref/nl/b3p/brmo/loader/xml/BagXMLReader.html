55      public static final String MUTATIE_PRODUCT = "Mutatie-product";
56      public static final String LVC_PRODUCT = "LVC-product";
57      private static final String DATUM_TIJD_LV = "datumtijdstempelLV";
58      private static final String MUTATIE_DATUMTOT = "MutatiedatumTot";
59      private static final String STAND_PEILDATUM = "StandPeildatum";
60      private static final String STAND_TECHNISCHEDATUM = "StandTechnischeDatum";

64      public static final Map<String,String> lvcProductToObjectType;

66      static {
67          Map<String,String> m = new HashMap();
68          m.put("Ligplaats", "LIG");
69          m.put("Nummeraanduiding", "NUM");
70          m.put("OpenbareRuimte", "OPR");
71          m.put("Pand", "PND");
72          m.put("Verblijfsobject", "VBO");
73          m.put("Standplaats", "STA");
74          m.put("Woonplaats", "WPL");
75          lvcProductToObjectType = Collections.unmodifiableMap(m);
76      }

78      public BagXMLReader(InputStream in)
79              throws XMLStreamException,
80              TransformerConfigurationException,
81              ParserConfigurationException {
82  
83          streamReader = factory.createXMLStreamReader(in);
84  
85          TransformerFactory tf = TransformerFactory.newInstance();
86          transformer = tf.newTransformer();
87  
88          // Vanwege splitsing is repairen van namespaces nodig, anders ontbreekt
89          // de namespace prefix declaratie xmlns:xlink="http://www.w3.org/1999/xlink"
90          // in de output en mislukt de XSL transformatie
91          xmlof = XMLOutputFactory.newInstance();
92          xmlof.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, Boolean.TRUE);
93  
94          DocumentBuilderFactory dbfactory = DocumentBuilderFactory.newInstance();
95          dbfactory.setNamespaceAware(true);
96          dbfactory.setNamespaceAware(true);
97          builder = dbfactory.newDocumentBuilder();
98  
99          init();
100     }

102     @Override
103     public void init() {
104         soort = BrmoFramework.BR_BAG;
105 
106         String technischeDatum = null;
107         String peilDatum = null;
108         String mutatieDatumTot = null;
109         boolean firstStartElement = true;
110         try {
111             while (streamReader.hasNext()) {
112                 if (streamReader.isStartElement()) {
113                     String localName = streamReader.getLocalName();
114 
115                     if (localName.equals(MUTATIE_PRODUCT) || localName.equals(LVC_PRODUCT)) {
116                         break;
117                     }
118                     //tbv bag-fragment met enkel object zonder wrapper
119                     if (lvcProductToObjectType.containsKey(localName) 
120                             && firstStartElement) {
121                         break;
122                     }
123 
124                     if (localName.equals(DATUM_TIJD_LV)) {
125                         setDatumAsString(streamReader.getElementText());
126                     }
127                     if(localName.equals(STAND_TECHNISCHEDATUM)) {
128                         technischeDatum = streamReader.getElementText();
129                     }
130                     if(localName.equals(STAND_PEILDATUM)) {
131                         peilDatum = streamReader.getElementText();
132                     }
133                     if(localName.equals(MUTATIE_DATUMTOT)) {
134                         mutatieDatumTot = streamReader.getElementText();
135                     }
136                     
137                     firstStartElement = false;
138                 }
139                 streamReader.next();
140             }
141         } catch (XMLStreamException ex) {
142             log.error("Error while streaming XML", ex);
143         }
144         if(peilDatum != null) {
145             setDatumAsString(peilDatum, "yyyyMMdd");
146         } else if(technischeDatum != null) {
147             setDatumAsString(technischeDatum, "yyyyMMdd");
148         } else if(mutatieDatumTot != null) {
149             setDatumAsString(mutatieDatumTot, "yyyy-MM-dd");
150         } else {
151             setBestandsDatum(new Date());
152         }
153     }
154 
155     @Override
156     public boolean hasNext() throws SAXException, IOException, TransformerException {
157         if (nextBericht != null) {
158             return true;
159         }
160         try {
161             while (streamReader.hasNext()) {
162                 if (streamReader.isStartElement() && streamReader.getLocalName().equals(MUTATIE_PRODUCT)) {
163                     // parse en check voor "Nieuw" element en niet "Origineel"/"Wijziging"
164                     StringWriter sw = new StringWriter();
165 
166                     // Hiermee wordt de StreamReader het volgende product geforward
167                     transformer.transform(new StAXSource(streamReader), new StAXResult(xmlof.createXMLStreamWriter(sw)));
168 
169                     Document d = builder.parse(new InputSource(new StringReader(sw.toString())));
170                     if (createMutatieBagBericht(sw.toString(), d)) {
171                         return true;
172                     }
173                  } else if(streamReader.isStartElement() && lvcProductToObjectType.containsKey(streamReader.getLocalName())) {
174                     StringWriter sw = new StringWriter();
175                     // Hiermee wordt de StreamReader het volgende product geforward
176                     transformer.transform(new StAXSource(streamReader), new StAXResult(xmlof.createXMLStreamWriter(sw)));
177                     Document d = builder.parse(new InputSource(new StringReader(sw.toString())));
178                     if (createStandBagBericht(sw.toString(), d)) {
179                         return true;
180                     }
181                 } else {
182                     streamReader.next();
183                 }
184             }
185         } catch (XMLStreamException ex) {
186             log.error("Error while streaming XML", ex);
187         }
188         return false;
189     }
190 
191     @Override
192     public BagBericht next() throws TransformerException, XMLStreamException {
193         BagBericht b = nextBericht;
194         nextBericht = null;
195         return b;
196     }

198     private boolean createStandBagBericht(String brXml, Document d) {
199         if (gebruikStandBericht(d.getDocumentElement())) {
200             nextBericht = new BagBericht(brXml, d);
201             Date datBegGeld = getBerichtBeginGeldigheid(d.getDocumentElement());
202             if (datBegGeld != null && getBestandsDatum().before(datBegGeld)) {
203                 //uitzondering: begindatum in toekomst
204                 //sla op als mutatie met datum in de toekomst
205                 //als dit 2x voorkomt voor zelfde object, jammer dan!
206                 nextBericht.setDatum(datBegGeld);
207                 nextBericht.setVolgordeNummer(0);
208             } else {
209                 // Bij een standlevering zit er geen datum in een "Verwerking" element,
210                 // pak peildatum van stand en zet volgordenr op -1
211                 nextBericht.setDatum(getBestandsDatum());
212                 nextBericht.setVolgordeNummer(-1);
213             }
214             return true;
215         }
216         return false;
217     }

219     private boolean createMutatieBagBericht(String brXml, Document d) {
220         NodeList children = d.getDocumentElement().getChildNodes();
221         for (int i = 0; i < children.getLength(); i++) {
222             Node child = children.item(i);
223             // Mutatie-product met Nieuw als child element gevonden
224             // Origineel/Wijziging kunnen overgeslagen worden omdat er altijd een Nieuw
225             // element voor hetzelfde object is. In de BRMO wordt de historie zelf
226             // bijgehouden dus zijn de wijzigingen niet interessant
227             if ("Nieuw".equals(child.getLocalName())) {
228                 NodeList grandchildren = child.getChildNodes();
229                 for (int j = 0; j < grandchildren.getLength(); j++) {
230                     Node grandchild = grandchildren.item(j);
231                     if (grandchild.getNodeType() == Node.ELEMENT_NODE) {
232                         nextBericht = new BagBericht(brXml, d);
233                         Date datBegGeld = getBerichtBeginGeldigheid(grandchild);
234                         if (datBegGeld != null && getBestandsDatum().before(datBegGeld)) {
235                             //uitzondering: begindatum in toekomst
236                             //sla op met datum in de toekomst
237                             nextBericht.setDatum(datBegGeld);
238                         }
239                         return true;
240                     }
241                 }
242             }
243         }
244         return false;
245     }

247     /**
248      * @param n bericht
249      * @return true als record niet inactief en einddatum geldigheid na bestandsdatum
250      */
251     private boolean gebruikStandBericht(Node n) {
252         String soort = n.getLocalName();
253         NodeList children = n.getChildNodes();
254         for (int i = 0; i < children.getLength(); i++) {
255             Node child = children.item(i);
256             if ("tijdvakgeldigheid".equals(child.getLocalName())) {
257                 NodeList grandchildren = child.getChildNodes();
258                 for (int j = 0; j < grandchildren.getLength(); j++) {
259                     Node grandchild = grandchildren.item(j);
260                     if ("einddatumTijdvakGeldigheid".equals(grandchild.getLocalName())) {
261                         String dateString = grandchild.getTextContent().substring(0, 8);
262                         SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
263                         Date eindGeldigheidDatum = null;
264                         try {
265                             eindGeldigheidDatum = sdf.parse(dateString);
266                         } catch (ParseException pe) {
267                             log.error("Fout bij parsen datum \"" + dateString + "\" met formaat yyyyMMdd", pe);
268                         }
269                         if (eindGeldigheidDatum != null 
270                                 && getBestandsDatum().after(eindGeldigheidDatum)) {
271                             return false;
272                         }
273                     }
274                 }
275             }
276             // aanduidingRecordInactief J verwijderen
277             if ("aanduidingRecordInactief".equals(child.getLocalName())) {
278                 if (child.getTextContent().equalsIgnoreCase("J")) {
279                     return false;
280                 }
281             }
282         }
283         return true;
284     }

286     /**
287      * @param n bericht
288      * @return datum begin geldigheid of null indien niet in toekomst
289      */
290     private Date getBerichtBeginGeldigheid(Node n) {
291         NodeList children = n.getChildNodes();
292         SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
293         Date beginGeldigheid = null;
294         for (int i = 0; i < children.getLength(); i++) {
295             Node child = children.item(i);
296             if ("tijdvakgeldigheid".equals(child.getLocalName())) {
297                 NodeList grandchildren = child.getChildNodes();
298                 for (int j = 0; j < grandchildren.getLength(); j++) {
299                     Node grandchild = grandchildren.item(j);
300                     if ("begindatumTijdvakGeldigheid".equals(grandchild.getLocalName())) {
301                         String dateString = grandchild.getTextContent().substring(0, 8);
302                         try {
303                             beginGeldigheid = sdf.parse(dateString);
304                         } catch (ParseException pe) {
305                             log.error("Fout bij parsen datum \"" + dateString + "\" met formaat yyyyMMdd", pe);
306                         }
307                     }
308                  }
309             }
310         }
311         return beginGeldigheid;
312     }
