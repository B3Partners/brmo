53    public static final String MUTATIE_PRODUCT = "Mutatie-product";
54    public static final String LVC_PRODUCT = "LVC-product";
55    private static final String DATUM_TIJD_LV = "datumtijdstempelLV";
56    private static final String MUTATIE_DATUMTOT = "MutatiedatumTot";
57    private static final String STAND_PEILDATUM = "StandPeildatum";
58    private static final String STAND_TECHNISCHEDATUM = "StandTechnischeDatum";

62    public static final Map<String, String> lvcProductToObjectType;

64    static {
65      lvcProductToObjectType =
66          Map.of(
67              "Ligplaats",
68              "LIG",
69              "Nummeraanduiding",
70              "NUM",
71              "OpenbareRuimte",
72              "OPR",
73              "Pand",
74              "PND",

83    public BagXMLReader(InputStream in)
84        throws XMLStreamException, TransformerConfigurationException, ParserConfigurationException {
85  
86      streamReader = factory.createXMLStreamReader(in);
87  
88      TransformerFactory tf = TransformerFactory.newInstance();
89      transformer = tf.newTransformer();
90  
91      // Vanwege splitsing is repairen van namespaces nodig, anders ontbreekt
92      // de namespace prefix declaratie xmlns:xlink="http://www.w3.org/1999/xlink"
93      // in de output en mislukt de XSL transformatie
94      xmlof = XMLOutputFactory.newInstance();
95      xmlof.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, Boolean.TRUE);
96  
97      DocumentBuilderFactory dbfactory = DocumentBuilderFactory.newInstance();
98      dbfactory.setNamespaceAware(true);
99      builder = dbfactory.newDocumentBuilder();
100 
101     init();
102   }

104   @Override
105   public void init() {
106     soort = BrmoFramework.BR_BAG;
107 
108     String technischeDatum = null;
109     String peilDatum = null;
110     String mutatieDatumTot = null;
111     boolean firstStartElement = true;
112     try {
113       while (streamReader.hasNext()) {
114         if (streamReader.isStartElement()) {
115           String localName = streamReader.getLocalName();
116 
117           if (localName.equals(MUTATIE_PRODUCT) || localName.equals(LVC_PRODUCT)) {
118             break;
119           }
120           // tbv bag-fragment met enkel object zonder wrapper
121           if (lvcProductToObjectType.containsKey(localName) && firstStartElement) {
122             break;
123           }
124 
125           if (localName.equals(DATUM_TIJD_LV)) {
126             setDatumAsString(streamReader.getElementText());
127           }
128           if (localName.equals(STAND_TECHNISCHEDATUM)) {
129             technischeDatum = streamReader.getElementText();
130           }
131           if (localName.equals(STAND_PEILDATUM)) {
132             peilDatum = streamReader.getElementText();
133           }
134           if (localName.equals(MUTATIE_DATUMTOT)) {
135             mutatieDatumTot = streamReader.getElementText();
136           }
137 
138           firstStartElement = false;
139         }
140         streamReader.next();
141       }
142     } catch (XMLStreamException ex) {
143       log.error("Error while streaming XML", ex);
144     }
145     if (peilDatum != null) {
146       setDatumAsString(peilDatum, "yyyyMMdd");
147     } else if (technischeDatum != null) {
148       setDatumAsString(technischeDatum, "yyyyMMdd");
149     } else if (mutatieDatumTot != null) {
150       setDatumAsString(mutatieDatumTot, "yyyy-MM-dd");
151     } else {
152       setBestandsDatum(new Date());
153     }
154   }

156   @Override
157   public boolean hasNext() throws SAXException, IOException, TransformerException {
158     if (nextBericht != null) {
159       return true;
160     }
161     try {
162       while (streamReader.hasNext()) {
163         if (streamReader.isStartElement() && streamReader.getLocalName().equals(MUTATIE_PRODUCT)) {
164           // parse en check voor "Nieuw" element en niet "Origineel"/"Wijziging"
165           StringWriter sw = new StringWriter();
166 
167           // Hiermee wordt de StreamReader het volgende product geforward
168           transformer.transform(
169               new StAXSource(streamReader), new StAXResult(xmlof.createXMLStreamWriter(sw)));
170 
171           Document d = builder.parse(new InputSource(new StringReader(sw.toString())));
172           if (createMutatieBagBericht(sw.toString(), d)) {
173             return true;
174           }
175         } else if (streamReader.isStartElement()
176             && lvcProductToObjectType.containsKey(streamReader.getLocalName())) {
177           StringWriter sw = new StringWriter();
178           // Hiermee wordt de StreamReader het volgende product geforward
179           transformer.transform(
180               new StAXSource(streamReader), new StAXResult(xmlof.createXMLStreamWriter(sw)));
181           Document d = builder.parse(new InputSource(new StringReader(sw.toString())));
182           if (createStandBagBericht(sw.toString(), d)) {
183             return true;
184           }
185         } else {
186           streamReader.next();
187         }
188       }
189     } catch (XMLStreamException ex) {
190       log.error("Error while streaming XML", ex);
191     }
192     return false;
193   }

195   @Override
196   public BagBericht next() throws TransformerException, XMLStreamException {
197     BagBericht b = nextBericht;
198     nextBericht = null;
199     return b;
200   }

202   private boolean createStandBagBericht(String brXml, Document d) {
203     if (gebruikStandBericht(d.getDocumentElement())) {
204       nextBericht = new BagBericht(brXml, d);
205       Date datBegGeld = getBerichtBeginGeldigheid(d.getDocumentElement());
206       if (datBegGeld != null && getBestandsDatum().before(datBegGeld)) {
207         // uitzondering: begindatum in toekomst
208         // sla op als mutatie met datum in de toekomst
209         // als dit 2x voorkomt voor zelfde object, jammer dan!
210         nextBericht.setDatum(datBegGeld);
211         nextBericht.setVolgordeNummer(0);
212       } else {
213         // Bij een standlevering zit er geen datum in een "Verwerking" element,
214         // pak peildatum van stand en zet volgordenr op -1
215         nextBericht.setDatum(getBestandsDatum());
216         nextBericht.setVolgordeNummer(-1);
217       }
218       return true;
219     }
220     return false;
221   }

223   private boolean createMutatieBagBericht(String brXml, Document d) {
224     NodeList children = d.getDocumentElement().getChildNodes();
225     for (int i = 0; i < children.getLength(); i++) {
226       Node child = children.item(i);
227       // Mutatie-product met Nieuw als child element gevonden
228       // Origineel/Wijziging kunnen overgeslagen worden omdat er altijd een Nieuw
229       // element voor hetzelfde object is. In de BRMO wordt de historie zelf
230       // bijgehouden dus zijn de wijzigingen niet interessant
231       if ("Nieuw".equals(child.getLocalName())) {
232         NodeList grandchildren = child.getChildNodes();
233         for (int j = 0; j < grandchildren.getLength(); j++) {
234           Node grandchild = grandchildren.item(j);
235           if (grandchild.getNodeType() == Node.ELEMENT_NODE) {
236             nextBericht = new BagBericht(brXml, d);
237             Date datBegGeld = getBerichtBeginGeldigheid(grandchild);
238             if (datBegGeld != null && getBestandsDatum().before(datBegGeld)) {
239               // uitzondering: begindatum in toekomst
240               // sla op met datum in de toekomst
241               nextBericht.setDatum(datBegGeld);
242             }
243             return true;
244           }
245         }
246       }
247     }
248     return false;
249   }

251   /**
252    * @param n bericht
253    * @return true als record niet inactief en einddatum geldigheid na bestandsdatum
254    */
255   private boolean gebruikStandBericht(Node n) {
256     String soort = n.getLocalName();
257     NodeList children = n.getChildNodes();
258     for (int i = 0; i < children.getLength(); i++) {
259       Node child = children.item(i);
260       if ("tijdvakgeldigheid".equals(child.getLocalName())) {
261         NodeList grandchildren = child.getChildNodes();
262         for (int j = 0; j < grandchildren.getLength(); j++) {
263           Node grandchild = grandchildren.item(j);
264           if ("einddatumTijdvakGeldigheid".equals(grandchild.getLocalName())) {
265             String dateString = grandchild.getTextContent().substring(0, 8);
266             SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
267             Date eindGeldigheidDatum = null;
268             try {
269               eindGeldigheidDatum = sdf.parse(dateString);
270             } catch (ParseException pe) {
271               log.error("Fout bij parsen datum \"" + dateString + "\" met formaat yyyyMMdd", pe);
272             }
273             if (eindGeldigheidDatum != null && getBestandsDatum().after(eindGeldigheidDatum)) {
274               return false;
275             }
276           }
277         }
278       }
279       // aanduidingRecordInactief J verwijderen
280       if ("aanduidingRecordInactief".equals(child.getLocalName())) {
281         if (child.getTextContent().equalsIgnoreCase("J")) {
282           return false;
283         }
284       }
285     }
286     return true;
287   }

289   /**
290    * @param n bericht
291    * @return datum begin geldigheid of null indien niet in toekomst
292    */
293   private Date getBerichtBeginGeldigheid(Node n) {
294     NodeList children = n.getChildNodes();
295     SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
296     Date beginGeldigheid = null;
297     for (int i = 0; i < children.getLength(); i++) {
298       Node child = children.item(i);
299       if ("tijdvakgeldigheid".equals(child.getLocalName())) {
300         NodeList grandchildren = child.getChildNodes();
301         for (int j = 0; j < grandchildren.getLength(); j++) {
302           Node grandchild = grandchildren.item(j);
303           if ("begindatumTijdvakGeldigheid".equals(grandchild.getLocalName())) {
304             String dateString = grandchild.getTextContent().substring(0, 8);
305             try {
306               beginGeldigheid = sdf.parse(dateString);
307             } catch (ParseException pe) {
308               log.error("Fout bij parsen datum \"" + dateString + "\" met formaat yyyyMMdd", pe);
309             }
310           }
311         }
312       }
313     }
314     return beginGeldigheid;
315   }
