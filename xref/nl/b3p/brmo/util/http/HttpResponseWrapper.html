8   package nl.b3p.brmo.util.http;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  
13  /**
14   * A wrapper around a HTTP response to provide access to the response status, headers and body.
15   *
16   * @author Matthijs Laan
17   */
18  public interface HttpResponseWrapper {
19      /**
20       * @return The response status code.
21       * @throws IOException
22       */
23      int getStatusCode() throws IOException;
24  
25      /**
26       * @param header The name of the header value to return. Only single header values are expected.
27       * @return
28       */
29      String getHeader(String header);
30  
31      /**
32       * The input stream for the response body. Closing this stream should release all resources for this request.
33       */
34      InputStream getResponseBody() throws IOException;
35  }
