8   package nl.b3p.brmo.nhr.loader;
9   
10  import java.io.ByteArrayInputStream;
11  import java.io.ByteArrayOutputStream;
12  import java.time.LocalDateTime;
13  import java.util.HashMap;
14  import java.util.List;
15  import java.util.UUID;
16  import javax.xml.bind.JAXBContext;
17  import javax.xml.bind.JAXBElement;
18  import javax.xml.bind.Marshaller;
19  import javax.xml.ws.BindingProvider;
20  import nl.b3p.brmo.loader.BrmoFramework;
21  import nl.kvk.schemas.schemas.hrip.dataservice._2015._02.Dataservice;
22  import nl.kvk.schemas.schemas.hrip.dataservice._2015._02.InschrijvingRequestType;
23  import nl.kvk.schemas.schemas.hrip.dataservice._2015._02.InschrijvingResponseType;
24  import nl.kvk.schemas.schemas.hrip.dataservice._2015._02.MeldingType;
25  import nl.kvk.schemas.schemas.hrip.dataservice._2015._02.ObjectFactory;
26  import org.apache.cxf.ws.addressing.AddressingProperties;
27  import org.apache.cxf.ws.addressing.AttributedURIType;
28  import org.apache.cxf.ws.addressing.JAXWSAConstants;
29  
30  public class NHRLoader {
31    // JAXB context used to serialize NHR responses for the BRMO.
32    private static JAXBContext jaxbContext;
33    private static ObjectFactory factory = new ObjectFactory();
34  
35    /** Send a single NHR request. */
36    public static void sendSingleRequest(
37        Dataservice dataservice, BrmoFramework brmoFramework, String kvkNummer, String rsin)
38        throws Exception {
39      if (jaxbContext == null) {
40        jaxbContext = JAXBContext.newInstance(InschrijvingResponseType.class);
41      }
42  
43      InschrijvingRequestType request = new InschrijvingRequestType();
44      String description;
45      if (kvkNummer != null) {
46        description = String.format("KVK nummer %s", kvkNummer);
47        request.setKvkNummer(kvkNummer);
48      } else if (rsin != null) {
49        description = String.format("KVK RSIN %s", rsin);
50        request.setRsin(rsin);
51      } else {
52        throw new IllegalArgumentException("KVK nummer en RSIN zijn beiden null");
53      }
54  
55      // Generate the UUID manually, as CXF generates urn:uuid:{uuid}, which is explicitly
56      // disallowed.
57      BindingProvider bindingProvider = (BindingProvider) dataservice;
58      AddressingProperties properties =
59          (AddressingProperties)
60              bindingProvider.getRequestContext().get(JAXWSAConstants.CLIENT_ADDRESSING_PROPERTIES);
61      AttributedURIType msgid = new AttributedURIType();
62      msgid.setValue("uuid:" + UUID.randomUUID().toString());
63      properties.setMessageID(msgid);
64  
65      // Send the actual request.
66      InschrijvingResponseType response = dataservice.ophalenInschrijving(request);
67  
68      List<MeldingType> errors = response.getMeldingen().getFout();
69      if (!errors.isEmpty()) {
70        HashMap<String, String> errorMap = new HashMap<>();
71        for (MeldingType melding : errors) {
72          errorMap.put(melding.getCode(), melding.getOmschrijving());
73        }
74  
75        throw new NHRException(errorMap);
76      }
77  
78      // Serialize the XML into the format expected by BRMO.
79      JAXBElement<InschrijvingResponseType> wrappedResponse =
80          factory.createOphalenInschrijvingResponse(response);
81      Marshaller marshaller = jaxbContext.createMarshaller();
82  
83      ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
84      marshaller.marshal(wrappedResponse, outputStream);
85  
86      brmoFramework.loadFromStream(
87          BrmoFramework.BR_NHR,
88          new ByteArrayInputStream(outputStream.toByteArray()),
89          String.format("NHR %s %s", LocalDateTime.now(), description),
90          (Long) null);
91    }
92  }
