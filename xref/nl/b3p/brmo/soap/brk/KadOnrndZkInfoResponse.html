347         sql.append("    kad_perceel.aand_soort_grootte,");

349         sql.append("    kad_onrrnd_zk.lr_bedrag,");
350         sql.append("    kad_perceel.begrenzing_perceel,");
351         sql.append("    kad_perceel.grootte_perceel,");
352         sql.append("    kad_onrrnd_zk.ks_koopjaar,");

354         sql.append("    kad_perceel.omschr_deelperceel ");

358     private static StringBuilder createFromSQL() {
359         StringBuilder sql = new StringBuilder();
360         sql.append("    kad_onrrnd_zk ");
361         sql.append("LEFT OUTER JOIN ");
362         sql.append("    kad_perceel ");
363         sql.append("ON ");
364         sql.append("    ( ");
365         sql.append("        kad_onrrnd_zk.kad_identif = kad_perceel.sc_kad_identif) ");
366         sql.append("LEFT OUTER JOIN ");
367         sql.append("    app_re ");
368         sql.append("ON ");
369         sql.append("    ( ");
370         sql.append("        kad_onrrnd_zk.kad_identif = app_re.sc_kad_identif) ");
371         return sql;
372     }
