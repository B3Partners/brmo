39      private static StringBuilder createFullColumnsSQL() {
40          StringBuilder sql = new StringBuilder();
41          sql.append("    zak_recht.indic_betrokken_in_splitsing,");
42          sql.append("    zak_recht.ar_noemer,");
43          sql.append("    zak_recht.ar_teller,");
44          sql.append("    zak_recht.fk_3avr_aand,");
45          sql.append("    zak_recht.fk_8pes_sc_identif, ");
46          sql.append("    aard_recht_verkort.omschr ");
47          return sql;
48      }
49  
50      private static StringBuilder createFromSQL() {
51          StringBuilder sql = new StringBuilder();
52          sql.append("    zak_recht ");
53          sql.append("INNER JOIN ");
54          sql.append("    aard_recht_verkort ");
55          sql.append("ON ");
56          sql.append("    ( ");
57          sql.append("        zak_recht.fk_3avr_aand = aard_recht_verkort.aand) ");
58          return sql;
59      }
60  
61      private static StringBuilder createWhereSQL() {
62          StringBuilder sql = new StringBuilder();
63          sql.append("    zak_recht.fk_7koz_kad_identif = ? ");
64          // alleen rechten gekoppeld aan subjecten tonen
65          sql.append("AND    fk_8pes_sc_identif is not null ");
66          return sql;
67      }

69      public static RechtenResponse getRechtenByKoz(Long kozId,
70              Map<String, Object> searchContext) throws Exception {
71  
72          DataSource ds = ConfigUtil.getDataSourceRsgb();
73          PreparedStatement stm = null;
74          Connection connRsgb = null;
75          ResultSet rs = null;
76          try {
77              connRsgb = ds.getConnection();
78  
79              StringBuilder sql = new StringBuilder();
80              sql.append("SELECT ");
81              sql.append(createFullColumnsSQL());
82              sql.append("FROM ");
83              sql.append(createFromSQL());
84              sql.append("WHERE ");
85              sql.append(createWhereSQL());
86              LOG.trace(sql);
87              LOG.trace(kozId);
88              stm = connRsgb.prepareStatement(sql.toString());
89              stm.setObject(1, kozId);
90              rs = stm.executeQuery();
91  
92              RechtenResponse rr = null;
93              List<ZakelijkRechtResponse> zrl = null;
94              ZakelijkRechtResponse zkRecht = null;
95              while (rs.next()) {
96                  if (rr == null) {
97                      rr = new RechtenResponse();
98                      zrl = new ArrayList<ZakelijkRechtResponse>();
99                      rr.setZakelijkRecht(zrl);
100                 }
101                 zkRecht = new ZakelijkRechtResponse();
102 
103 //                zkRecht.setAardVerkregenRecht(rs.getString("fk_3avr_aand"));
104                 zkRecht.setAardVerkregenRecht(rs.getString("omschr"));
105                 zkRecht.setIndicatieBetrokkenInSplitsing(
106                         rs.getString("indic_betrokken_in_splitsing")
107                         .equalsIgnoreCase("nee") ? false : true);
108                 zkRecht.setNoemer(rs.getInt("ar_noemer"));
109                 zkRecht.setTeller(rs.getInt("ar_teller"));
110 
111                 Boolean st = (Boolean) searchContext.get(BrkInfo.SUBJECTSTOEVOEGEN);
112                 if (st != null && st) {
113                     String pid = rs.getString("fk_8pes_sc_identif");
114                     NatuurlijkPersoonResponse np
115                             = NatuurlijkPersoonResponse.getRecordById(pid, searchContext);
116                     zkRecht.setNatuurlijkPersoon(np);
117                     if (np == null) {
118                         NietNatuurlijkPersoonResponse nnp
119                                 = NietNatuurlijkPersoonResponse.getRecordById(pid, searchContext);
120                         zkRecht.setNietNatuurlijkPersoon(nnp);
121                     }
122                 }
123                 zrl.add(zkRecht);
124 
125             }
126 
127             return rr;
128         } finally {
129             if (rs != null) {
130                 rs.close();
131             }
132             if (stm != null) {
133                 stm.close();
134             }
135             if (connRsgb != null) {
136                 connRsgb.close();
137             }
138         }
139     }
