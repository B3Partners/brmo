65      public static ArrayList<Long> findKozIDs(Map<String, Object> searchContext) throws Exception {
66  
67          DataSource ds = getDataSourceRsgb();
68          PreparedStatement stm = null;
69          Connection connRsgb = null;
70          ResultSet rs = null;
71          try {
72              connRsgb = ds.getConnection();
73              String dbType = getDbType(connRsgb);
74  
75              StringBuilder sql = createSelectSQL(searchContext, dbType);
76  
77              stm = connRsgb.prepareStatement(sql.toString());
78              stm = addParamsSQL(stm, searchContext, dbType);
79              rs = stm.executeQuery();
80  
81              ArrayList<Long> ids = new ArrayList<Long>();
82              while (rs.next()) {
83                  Long id = rs.getLong("identificatie");
84                  ids.add(id);
85              }
86  
87              return ids;
88          } finally {
89              if (rs != null) {
90                  rs.close();
91              }
92              if (stm != null) {
93                  stm.close();
94              }
95              if (connRsgb != null) {
96                  connRsgb.close();
97              }
98          }
99      }

141     private static StringBuilder createSelectSQL(
142             Map<String, Object> searchContext, String dbType) throws ParseException {
143 
144         Integer maxrows = (Integer) searchContext.get(MAXAANTALRESULTATEN);
145         if (maxrows == null) {
146             maxrows = DBMAXRESULTS;
147             searchContext.put(MAXAANTALRESULTATEN, DBMAXRESULTS);
148         } else if (maxrows > DBMAXRESULTS) {
149             searchContext.put(MAXAANTALRESULTATEN, DBMAXRESULTS);
150         }
151         // één meer ophalen zodat je kunt weten of er meer zijn
152         maxrows += 1;
153 
154         StringBuilder sql = new StringBuilder();
155         switch (dbType) {
156             case DB_POSTGRES:
157                 sql.append("SELECT ");
158                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
159                 sql.append("FROM ");
160                 sql.append(createFromSQL());
161                 sql.append("WHERE ");
162                 sql.append(createWhereSQL(searchContext, dbType));
163                 sql.append("LIMIT ");
164                 sql.append(maxrows);
165                 return sql;
166             case DB_ORACLE:
167                 sql.append("SELECT ");
168                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
169                 sql.append("FROM ");
170                 sql.append(createFromSQL());
171                 sql.append("WHERE ");
172                 sql.append(createWhereSQL(searchContext, dbType));
173                 StringBuilder tempSql = new StringBuilder();
174                 tempSql.append("SELECT * FROM ( ");
175                 tempSql.append(sql);
176                 tempSql.append(" ) WHERE ROWNUM <= ");
177                 tempSql.append(maxrows);
178                 sql = tempSql;
179                 return sql;
180             case DB_MSSQL:
181                 sql.append("SELECT ");
182                 sql.append("TOP ");
183                 sql.append("( ");
184                 sql.append(maxrows);
185                 sql.append(") ");
186                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
187                 sql.append("FROM ");
188                 sql.append(createFromSQL());
189                 sql.append("WHERE ");
190                 sql.append(createWhereSQL(searchContext, dbType));
191                 return sql;
192             default:
193                 throw new UnsupportedOperationException("Unknown database!");
194         }
195     }

198             Map<String, Object> searchContext, String dbType) throws ParseException {
199 
200         StringBuilder sql = new StringBuilder();

207         return sql;

288     private static StringBuilder createWhereSQL(
289             Map<String, Object> searchContext, String dbType) throws ParseException {
290         StringBuilder sql = new StringBuilder();
291 
292         boolean first = true;
293         String condition;
294         condition = "    kad_onrrnd_zk.kad_identif = ? "; //1 Long
295         first = addTerm(first, searchContext.get(IDENTIFICATIE), sql, condition);
296 
297         condition = "   (    nat_prs.nm_geslachtsnaam like ? "
298                 + "    OR niet_nat_prs.naam like ?)"; //2 string
299         first = addTerm(first, searchContext.get(SUBJECTNAAM), sql, condition);
300 
301         //3+4 wkt en bufferlengte
302         first = addGeoTerm(first, sql, searchContext, dbType);
303 
304         condition = "app_re.ka_appartementsindex like ? "; //5 string
305         first = addTerm(first, searchContext.get(APPREVOLGNUMMER), sql, condition);
306 
307         condition = "   (    app_re.ka_kad_gemeentecode like ? "
308                 + "    OR kad_perceel.ka_kad_gemeentecode like ?) "; //6 string
309         first = addTerm(first, searchContext.get(GEMEENTECODE), sql, condition);
310 
311         condition = "   (    app_re.ka_perceelnummer like ? "
312                 + "    OR kad_perceel.ka_perceelnummer like ?) "; //7 string
313         first = addTerm(first, searchContext.get(PERCEELNUMMER), sql, condition);
314 
315         condition = "   (    app_re.ka_sectie like ? "
316                 + "    OR kad_perceel.ka_sectie like ?) "; //8 string
317         first = addTerm(first, searchContext.get(SECTIE), sql, condition);
318 
319         condition = "wnplts.naam like ? "; //9 string
320         first = addTerm(first, searchContext.get(WOONPLAATS), sql, condition);
321 
322         condition = "gem_openb_rmte.naam_openb_rmte like ? "; //10 string 
323         first = addTerm(first, searchContext.get(STRAATNAAM), sql, condition);
324 
325         condition = "addresseerb_obj_aand.huinummer = ? "; //11 Integer
326         first = addTerm(first, searchContext.get(HUISNUMMER), sql, condition);
327 
328         condition = "addresseerb_obj_aand.postcode like ? "; //12 string
329         addTerm(first, searchContext.get(POSTCODE), sql, condition);
330 
331         return sql;
332     }

334     private static boolean addTerm(boolean first, Object o, StringBuilder sql, String condition) {
335         if (o != null) {
336             if (!first) {
337                 sql.append("AND ");
338             }
339             sql.append(condition);
340             first = false;
341         }
342         return first;

345     private static boolean addGeoTerm(boolean first, StringBuilder sql,
346             Map<String, Object> searchContext, String dbType) throws ParseException {
347 
348         String zg = (String) searchContext.get(ZOEKGEBIED);
349         if (zg == null) {
350             return first;
351         }
352         WKTReader r = new WKTReader();
353         //test om injectie te voorkomen, exception indien niet geldig
354         Geometry g = r.read(zg);
355         Integer bl = (Integer) searchContext.get(BUFFERLENGTE);
356 
357         String condition;
358         switch (dbType) {
359             case DB_POSTGRES:
360                 if (bl != null) {
361                     condition = "ST_Intersects(kad_perceel.begrenzing_perceel, "
362                             + "ST_Buffer(ST_GeomFromEWKT('SRID=28992;"
363                             + zg
364                             + "'), "
365                             + bl
366                             + ") ) ";
367                 } else {
368                     condition = "ST_Intersects(kad_perceel.begrenzing_perceel, "
369                             + "ST_GeomFromEWKT('SRID=28992;"
370                             + zg
371                             + "') ) ";
372                 }
373                 break;
374             case DB_ORACLE:
375                 if (bl != null) {
376                     condition = "SDO_GEOM.RELATE(kad_perceel.begrenzing_perceel, 'ANYINTERACT', "
377                             + "SDO_GEOM.SDO_BUFFER(SDO_GEOMETRY(SDO_UTIL.TO_WKTGEOMETRY("
378                             + zg
379                             + "), 28992), 2, "
380                             + bl
381                             + "), 0.005 ) ";
382                 } else {
383                     condition = "SDO_GEOM.RELATE(kad_perceel.begrenzing_perceel, 'ANYINTERACT', "
384                             + "SDO_GEOMETRY(SDO_UTIL.TO_WKTGEOMETRY("
385                             + zg
386                             + "), 28992), 0.005 ) ";
387                 }
388                 break;
389             case DB_MSSQL:
390                 if (bl != null) {
391                     condition = "kad_perceel.begrenzing_perceel.STIntersects( "
392                             + "STGeomFromText("
393                             + zg
394                             + ",28992).STBuffer("
395                             + bl
396                             + ") ) ";
397                 } else {
398                     condition = "kad_perceel.begrenzing_perceel.STIntersects( "
399                             + "STGeomFromText("
400                             + zg
401                             + ",28992) ) ";
402                     break;
403                 }
404             default:
405                 throw new UnsupportedOperationException("Unknown database!");
406         }
407         if (!first) {
408             sql.append("AND ");
409         }
410         sql.append(condition);
411         first = false;
412         return first;
413     }

415     private static PreparedStatement addParamsSQL(PreparedStatement stm,
416             Map<String, Object> searchContext, String dbType) throws SQLException {
417 
418         int index = 1;
419         index = addParam(index, searchContext.get(IDENTIFICATIE), stm, dbType);
420         index = addParam(index, searchContext.get(SUBJECTNAAM), stm, dbType);
421         index = addParam(index, searchContext.get(SUBJECTNAAM), stm, dbType);
422         //bufferlengte en zoekgebied direct in de Where!
423         index = addParam(index, searchContext.get(APPREVOLGNUMMER), stm, dbType);
424         index = addParam(index, searchContext.get(GEMEENTECODE), stm, dbType);
425         index = addParam(index, searchContext.get(GEMEENTECODE), stm, dbType);
426         index = addParam(index, searchContext.get(PERCEELNUMMER), stm, dbType);
427         index = addParam(index, searchContext.get(PERCEELNUMMER), stm, dbType);
428         index = addParam(index, searchContext.get(SECTIE), stm, dbType);
429         index = addParam(index, searchContext.get(SECTIE), stm, dbType);
430         index = addParam(index, searchContext.get(WOONPLAATS), stm, dbType);
431         index = addParam(index, searchContext.get(STRAATNAAM), stm, dbType);
432         index = addParam(index, searchContext.get(HUISNUMMER), stm, dbType);
433         addParam(index, searchContext.get(POSTCODE), stm, dbType);
434 
435         return stm;
436     }

438     private static int addParam(int index, Object o, PreparedStatement stm,
439             String dbType) throws SQLException {
440         if (o != null) {
441             if (o instanceof String) {
442                 stm.setString(index++, "%" + ((String) o) + "%");
443             } else {
444                 stm.setObject(index++, o);
445             }
446         }
447         return index;
448     }

466     public static String getDbType(Connection connRsgb) throws SQLException {
467         String databaseProductName = connRsgb.getMetaData().getDatabaseProductName();
468         if (databaseProductName.contains("PostgreSQL")) {
469             return DB_POSTGRES;
470         } else if (databaseProductName.contains("Oracle")) {
471             return DB_ORACLE;
472         } else if (databaseProductName.contains("Microsoft")) {
473             return DB_MSSQL;
474         } else {
475             throw new UnsupportedOperationException("Unknown database: " + connRsgb.getMetaData().getDatabaseProductName());
476         }
477     }

479     public static Map<String, Object> createSearchContext(BrkInfoRequest request) {
480         Map<String, Object> searchContext = new HashMap<String, Object>();
481         if (request == null) {
482             return searchContext;
483         }
484         if (request.getSubjectNaam() != null) {
485             searchContext.put(BrkInfo.SUBJECTNAAM, request.getSubjectNaam());
486         }
487         if (request.getZoekgebied() != null) {
488             searchContext.put(BrkInfo.ZOEKGEBIED, request.getZoekgebied());
489         }
490         if (request.getBufferAfstand() != null) {
491             searchContext.put(BrkInfo.BUFFERLENGTE, request.getBufferAfstand());
492         }
493 
494         KadOnrndZkInfoRequest koz = request.getKadOnrndZk();
495         if (koz != null) {
496             if (koz.getIdentificatie() != null) {
497                 searchContext.put(BrkInfo.IDENTIFICATIE, Long.parseLong(koz.getIdentificatie()));
498             }
499             if (koz.getAppReVolgnummer() != null) {
500                 searchContext.put(BrkInfo.APPREVOLGNUMMER, koz.getAppReVolgnummer());
501             }
502             if (koz.getGemeentecode() != null) {
503                 searchContext.put(BrkInfo.GEMEENTECODE, koz.getGemeentecode());
504             }
505             if (koz.getPerceelnummer() != null) {
506                 searchContext.put(BrkInfo.PERCEELNUMMER, koz.getPerceelnummer());
507             }
508             if (koz.getSectie() != null) {
509                 searchContext.put(BrkInfo.SECTIE, koz.getSectie());
510             }
511         }
512         PerceelAdresInfoRequest pa = request.getPerceelAdres();
513         if (pa != null) {
514             if (pa.getWoonplaatsNaam() != null) {
515                 searchContext.put(BrkInfo.WOONPLAATS, pa.getWoonplaatsNaam());
516             }
517             if (pa.getHuisnummer() != null) {
518                 searchContext.put(BrkInfo.HUISNUMMER, pa.getHuisnummer());
519             }
520             if (pa.getPostcode() != null) {
521                 searchContext.put(BrkInfo.POSTCODE, pa.getPostcode());
522             }
523             if (pa.getStraatNaam() != null) {
524                 searchContext.put(BrkInfo.STRAATNAAM, pa.getStraatNaam());
525             }
526         }
527         if (!searchContext.isEmpty()) {
528             if (request.getMaxAantalResultaten() != null) {
529                 searchContext.put(BrkInfo.MAXAANTALRESULTATEN, request.getMaxAantalResultaten());
530             }
531             if (request.getAdressenToevoegen() != null) {
532                 searchContext.put(BrkInfo.ADRESSENTOEVOEGEN, request.getAdressenToevoegen());
533             }
534             if (request.getGevoeligeInfoOphalen() != null) {
535                 searchContext.put(BrkInfo.GEVOELIGEINFOOPHALEN, request.getGevoeligeInfoOphalen());
536             }
537             if (request.getSubjectsToevoegen() != null) {
538                 searchContext.put(BrkInfo.SUBJECTSTOEVOEGEN, request.getSubjectsToevoegen());
539             }
540         }
541         return searchContext;
542     }

544     public static BrkInfoResponse createResponse(List<Long> ids,
545             Map<String, Object> searchContext) throws Exception {
546         BrkInfoResponse result = new BrkInfoResponse();
547 
548         Boolean at = (Boolean) searchContext.get(BrkInfo.SUBJECTSTOEVOEGEN);
549         if (at != null && at) {
550             at = (Boolean) searchContext.get(BrkInfo.GEVOELIGEINFOOPHALEN);
551             if (at != null && at) {
552                 result.setBevatGevoeligeInfo(true);
553             }
554         }
555         result.setTimestamp(new Date());
556 
557         for (Long id : ids) {
558             KadOnrndZkInfoResponse koz
559                     = KadOnrndZkInfoResponse.getRecord(id, searchContext);
560             result.addKadOnrndZk(koz);
561         }
562 
563         return result;
564     }
