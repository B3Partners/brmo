98              }
99              if (stm != null) {
100                 stm.close();
101             }
102             if (connRsgb != null) {

146     private static StringBuilder createSelectSQL(
147             Map<String, Object> searchContext, String dbType) throws ParseException {
148 
149         Integer maxrows = (Integer) searchContext.get(MAXAANTALRESULTATEN);
150         if (maxrows == null) {
151             maxrows = DBMAXRESULTS;
152             searchContext.put(MAXAANTALRESULTATEN, DBMAXRESULTS);
153         } else if (maxrows > DBMAXRESULTS) {
154             searchContext.put(MAXAANTALRESULTATEN, DBMAXRESULTS);
155         }
156         // één meer ophalen zodat je kunt weten of er meer zijn
157         maxrows += 1;
158 
159         StringBuilder sql = new StringBuilder();
160         switch (dbType) {
161             case DB_POSTGRES:
162                 sql.append("SELECT ");
163                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
164                 sql.append("FROM ");
165                 sql.append(createFromSQL());
166                 sql.append("WHERE ");
167                 sql.append(createWhereSQL(searchContext, dbType));
168                 sql.append("LIMIT ");
169                 sql.append(maxrows);
170                 return sql;
171             case DB_ORACLE:
172                 sql.append("SELECT ");
173                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
174                 sql.append("FROM ");
175                 sql.append(createFromSQL());
176                 sql.append("WHERE ");
177                 sql.append(createWhereSQL(searchContext, dbType));
178                 StringBuilder tempSql = new StringBuilder();
179                 tempSql.append("SELECT * FROM ( ");
180                 tempSql.append(sql);
181                 tempSql.append(" ) WHERE ROWNUM <= ");
182                 tempSql.append(maxrows);
183                 sql = tempSql;
184                 return sql;
185             case DB_MSSQL:
186                 sql.append("SELECT ");
187                 sql.append("TOP ");
188                 sql.append("( ");
189                 sql.append(maxrows);
190                 sql.append(") ");
191                 sql.append("    DISTINCT kad_onrrnd_zk.kad_identif AS identificatie ");
192                 sql.append("FROM ");
193                 sql.append(createFromSQL());
194                 sql.append("WHERE ");
195                 sql.append(createWhereSQL(searchContext, dbType));
196                 return sql;
197             default:
198                 throw new UnsupportedOperationException("Unknown database!");
199         }
200     }

293     private static StringBuilder createWhereSQL(
294             Map<String, Object> searchContext, String dbType) throws ParseException {
295         StringBuilder sql = new StringBuilder();
296 
297         boolean first = true;
298         String condition;
299         condition = "    kad_onrrnd_zk.kad_identif = ? "; //1 Long
300         first = addTerm(first, searchContext.get(IDENTIFICATIE), sql, condition);
301 
302         condition = "   (    nat_prs.nm_geslachtsnaam like ? "
303                 + "    OR niet_nat_prs.naam like ?)"; //2 string
304         first = addTerm(first, searchContext.get(SUBJECTNAAM), sql, condition);
305 
306         //3+4 wkt en bufferlengte
307         first = addGeoTerm(first, sql, searchContext, dbType);
308 
309         condition = "app_re.ka_appartementsindex like ? "; //5 string
310         first = addTerm(first, searchContext.get(APPREVOLGNUMMER), sql, condition);
311 
312         condition = "   (    app_re.ka_kad_gemeentecode like ? "
313                 + "    OR kad_perceel.ka_kad_gemeentecode like ?) "; //6 string
314         first = addTerm(first, searchContext.get(GEMEENTECODE), sql, condition);
315 
316         condition = "   (    app_re.ka_perceelnummer like ? "
317                 + "    OR kad_perceel.ka_perceelnummer like ?) "; //7 string
318         first = addTerm(first, searchContext.get(PERCEELNUMMER), sql, condition);
319 
320         condition = "   (    app_re.ka_sectie like ? "
321                 + "    OR kad_perceel.ka_sectie like ?) "; //8 string
322         first = addTerm(first, searchContext.get(SECTIE), sql, condition);
323 
324         condition = "wnplts.naam like ? "; //9 string
325         first = addTerm(first, searchContext.get(WOONPLAATS), sql, condition);
326 
327         condition = "gem_openb_rmte.naam_openb_rmte like ? "; //10 string 
328         first = addTerm(first, searchContext.get(STRAATNAAM), sql, condition);
329 
330         condition = "addresseerb_obj_aand.huinummer = ? "; //11 Integer
331         first = addTerm(first, searchContext.get(HUISNUMMER), sql, condition);
332 
333         condition = "addresseerb_obj_aand.postcode like ? "; //12 string
334         addTerm(first, searchContext.get(POSTCODE), sql, condition);
335 
336         return sql;
337     }

350     private static boolean addGeoTerm(boolean first, StringBuilder sql,
351             Map<String, Object> searchContext, String dbType) throws ParseException {
352 
353         String zg = (String) searchContext.get(ZOEKGEBIED);
354         if (zg == null) {
355             return first;
356         }
357         WKTReader r = new WKTReader();
358         //test om injectie te voorkomen, exception indien niet geldig
359         Geometry g = r.read(zg);
360         Integer bl = (Integer) searchContext.get(BUFFERLENGTE);
361 
362         String condition;
363         switch (dbType) {
364             case DB_POSTGRES:
365                 if (bl != null) {
366                     condition = "ST_Intersects(kad_perceel.begrenzing_perceel, "
367                             + "ST_Buffer(ST_GeomFromEWKT('SRID=28992;"
368                             + zg
369                             + "'), "
370                             + bl
371                             + ") ) ";
372                 } else {
373                     condition = "ST_Intersects(kad_perceel.begrenzing_perceel, "
374                             + "ST_GeomFromEWKT('SRID=28992;"
375                             + zg
376                             + "') ) ";
377                 }
378                 break;
379             case DB_ORACLE:
380                 if (bl != null) {
381                     condition = "SDO_GEOM.RELATE(kad_perceel.begrenzing_perceel, 'ANYINTERACT', "
382                             + "SDO_GEOM.SDO_BUFFER(SDO_GEOMETRY(('"
383                             + zg
384                             + "'), 28992), 2, "
385                             + bl
386                             + "), 0.005 )  = 'TRUE' ";
387                 } else {
388                     condition = "SDO_GEOM.RELATE(kad_perceel.begrenzing_perceel, 'ANYINTERACT', "
389                             + "SDO_GEOMETRY(('"
390                             + zg
391                             + "'), 28992), 0.005 ) = 'TRUE' ";
392                 }
393                 break;
394             case DB_MSSQL:
395                 if (bl != null) {
396                     condition = "kad_perceel.begrenzing_perceel.STIntersects( "
397                             + "STGeomFromText('"
398                             + zg
399                             + "',28992).STBuffer("
400                             + bl
401                             + ") ) ";
402                 } else {
403                     condition = "kad_perceel.begrenzing_perceel.STIntersects( "
404                             + "STGeomFromText('"
405                             + zg
406                             + "',28992) ) ";
407                     break;
408                 }
409             default:
410                 throw new UnsupportedOperationException("Unknown database!");
411         }
412         if (!first) {
413             sql.append("AND ");
414         }
415         sql.append(condition);
416         first = false;
417         return first;
418     }

443     private static int addParam(int index, Object o, PreparedStatement stm,
444             String dbType) throws SQLException {
445         if (o != null) {
446             if (o instanceof String) {
447                 stm.setString(index++, "%" + ((String) o) + "%");
448             } else {
449                 stm.setObject(index++, o);

455     public static String getDbType(Connection connRsgb) throws SQLException {
456         String databaseProductName = connRsgb.getMetaData().getDatabaseProductName();
457         if (databaseProductName.contains("PostgreSQL")) {
458             return DB_POSTGRES;
459         } else if (databaseProductName.contains("Oracle")) {
460             return DB_ORACLE;
461         } else if (databaseProductName.contains("Microsoft")) {
462             return DB_MSSQL;
463         } else {
464             throw new UnsupportedOperationException("Unknown database: " + connRsgb.getMetaData().getDatabaseProductName());
465         }

468     public static Map<String, Object> createSearchContext(BrkInfoRequest request) {
469         Map<String, Object> searchContext = new HashMap<>();
470         if (request == null) {
471             return searchContext;
472         }
473         if (request.getSubjectNaam() != null) {
474             searchContext.put(BrkInfo.SUBJECTNAAM, request.getSubjectNaam());
475         }
476         if (request.getZoekgebied() != null) {
477             searchContext.put(BrkInfo.ZOEKGEBIED, request.getZoekgebied());
478         }
479         if (request.getBufferAfstand() != null) {
480             searchContext.put(BrkInfo.BUFFERLENGTE, request.getBufferAfstand());
481         }
482 
483         KadOnrndZkInfoRequest koz = request.getKadOnrndZk();
484         if (koz != null) {
485             if (koz.getIdentificatie() != null) {
486                 searchContext.put(BrkInfo.IDENTIFICATIE, Long.parseLong(koz.getIdentificatie()));
487             }
488             if (koz.getAppReVolgnummer() != null) {
489                 searchContext.put(BrkInfo.APPREVOLGNUMMER, koz.getAppReVolgnummer());
490             }
491             if (koz.getGemeentecode() != null) {
492                 searchContext.put(BrkInfo.GEMEENTECODE, koz.getGemeentecode());
493             }
494             if (koz.getPerceelnummer() != null) {
495                 searchContext.put(BrkInfo.PERCEELNUMMER, koz.getPerceelnummer());
496             }
497             if (koz.getSectie() != null) {
498                 searchContext.put(BrkInfo.SECTIE, koz.getSectie());
499             }
500         }
501         PerceelAdresInfoRequest pa = request.getPerceelAdres();
502         if (pa != null) {
503             if (pa.getWoonplaatsNaam() != null) {
504                 searchContext.put(BrkInfo.WOONPLAATS, pa.getWoonplaatsNaam());
505             }
506             if (pa.getHuisnummer() != null) {
507                 searchContext.put(BrkInfo.HUISNUMMER, pa.getHuisnummer());
508             }
509             if (pa.getPostcode() != null) {
510                 searchContext.put(BrkInfo.POSTCODE, pa.getPostcode());
511             }
512             if (pa.getStraatNaam() != null) {
513                 searchContext.put(BrkInfo.STRAATNAAM, pa.getStraatNaam());
514             }
515         }
516         if (!searchContext.isEmpty()) {
517             if (request.getMaxAantalResultaten() != null) {
518                 searchContext.put(BrkInfo.MAXAANTALRESULTATEN, request.getMaxAantalResultaten());
519             }
520             if (request.getAdressenToevoegen() != null) {
521                 searchContext.put(BrkInfo.ADRESSENTOEVOEGEN, request.getAdressenToevoegen());
522             }
523             if (request.getGevoeligeInfoOphalen() != null) {
524                 searchContext.put(BrkInfo.GEVOELIGEINFOOPHALEN, request.getGevoeligeInfoOphalen());
525             }
526             if (request.getSubjectsToevoegen() != null) {
527                 searchContext.put(BrkInfo.SUBJECTSTOEVOEGEN, request.getSubjectsToevoegen());
528             }
529         }
530         return searchContext;
531     }
