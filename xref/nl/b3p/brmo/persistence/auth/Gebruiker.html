17  package nl.b3p.brmo.persistence.auth;
18  
19  import java.io.Serializable;
20  import java.util.HashSet;
21  import java.util.Set;
22  import javax.persistence.Entity;
23  import javax.persistence.Id;
24  import javax.persistence.JoinColumn;
25  import javax.persistence.JoinTable;
26  import javax.persistence.ManyToMany;
27  import javax.persistence.Table;
28  
29  /**
30   * @author Matthijs Laan
31   */
32  @Entity
33  @Table(name = "gebruiker_")
34  public class Gebruiker implements Serializable {
35  
36    @Id private String gebruikersnaam;
37  
38    private String wachtwoord;
39  
40    @ManyToMany
41    @JoinTable(
42        name = "gebruiker_groepen",
43        joinColumns = @JoinColumn(name = "gebruikersnaam"),
44        inverseJoinColumns = @JoinColumn(name = "groep_"))
45    private Set<Groep> groepen = new HashSet<>();
46  
47    public String getWachtwoord() {
48      return wachtwoord;
49    }
50  
51    /**
52     * Stel nieuw wachtwoord in. Het versleutelen van het wachtwoord naar een hash dient al uitgevoerd
53     * te zijn en moet conform de geconfigureerde {@code CredentialHandler} van de security realm
54     * gedaan zijn om te kunnen werken.
55     *
56     * @param wachtwoord een versleuteld/gehashed wachtwoord
57     */
58    public void setWachtwoord(String wachtwoord) {
59      this.wachtwoord = wachtwoord;
60    }
61  
62    public Set<Groep> getGroepen() {
63      return groepen;
64    }
65  
66    public void setGroepen(Set<Groep> groepen) {
67      this.groepen = groepen;
68    }
69  
70    public String getGebruikersnaam() {
71      return gebruikersnaam;
72    }
73  
74    public void setGebruikersnaam(String gebruikersnaam) {
75      this.gebruikersnaam = gebruikersnaam;
76    }
77  }
