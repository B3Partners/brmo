4   package nl.b3p.brmo.persistence.staging;
5   
6   import javax.persistence.Entity;
7   
8   /**
9    *
10   * @author mprins
11   */
12  @Entity
13  public class BerichtTransformatieProces extends AutomatischProces {
14  
15      private static final String BLOCK_ON_MISSINGNUMBERSFOUND = "blokkeer_transformatie";
16  
17      public void setBlockOnMissingNumbers(boolean block) {
18          this.getConfig().put(BLOCK_ON_MISSINGNUMBERSFOUND, new ClobElement(Boolean.toString(block)));
19      }
20  
21      public boolean getBlockOnMissingNumbers() {
22          String s = ClobElement.nullSafeGet(this.getConfig().get(BLOCK_ON_MISSINGNUMBERSFOUND));
23          return (s == null ? false : Boolean.parseBoolean(s));
24      }
25  }
