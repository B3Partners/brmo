4   package nl.b3p.brmo.persistence.staging;
5   
6   import javax.persistence.Entity;
7   
8   /**
9    * Mail rapportage proces.
10   *
11   * @author mprins
12   */
13  @Entity
14  public class MailRapportageProces extends AutomatischProces {
15  
16      public static final String DELIM = ",";
17      /**
18       * de sleutel {@value EMAIL}.
19       */
20      public static final String EMAIL = "email";
21      /**
22       * de sleutel {@value PIDS}.
23       */
24      public static final String PIDS = "pIDS";
25  
26      /**
27       * de sleutel {@value FOR_STATUS}. Geldige waarden komen uit de verzameling
28       * van {@link AutomatischProces#status }
29       */
30      public static final String FOR_STATUS = "forStatus";
31  
32      /**
33       * haalt de lijst van email adressen op.
34       *
35       * @return string met adressen
36       */
37      public String getMailAdressen() {
38          return ClobElement.nullSafeGet(this.getConfig().get(EMAIL));
39      }
40  
41      /**
42       * haalt de lijst van email adressen op.
43       *
44       * @return array van string met adressen
45       */
46      public String[] getMailAdressenArray() {
47          final String adreslijst = this.getMailAdressen();
48          if (adreslijst != null) {
49              return adreslijst.split(DELIM);
50          } else {
51              return null;
52          }
53      }
54  
55      /**
56       * wordt gebruikt om opslag van mailadressen te normaliseren.
57       *
58       * @param adressen een lijst adressen
59       */
60      public void setMailAdressen(String... adressen) {
61          StringBuilder sb = new StringBuilder();
62          for (String adres : adressen) {
63              sb.append(adres.trim()).append(DELIM);
64          }
65          sb.setLength(sb.length() - 1);
66          this.getConfig().put(EMAIL, new ClobElement(sb.toString()));
67      }
68  
69      /**
70       * voor stripes formulier input...
71       *
72       * @param adres een (lijst) adres(sen)
73       */
74      public void setMailAdressen(String adres) {
75          if (adres == null) {
76              this.getConfig().put(EMAIL, null);
77          } else if (adres.contains(DELIM)) {
78              this.setMailAdressen(adres.split(DELIM));
79          } else {
80              this.getConfig().put(EMAIL, new ClobElement(adres.trim()));
81          }
82      }
83  
84      public void setForStatus(ProcessingStatus status) {
85          if (status == null) {
86              this.getConfig().put(FOR_STATUS, null);
87          } else {
88              this.getConfig().put(FOR_STATUS, new ClobElement(status.name()));
89          }
90      }
91  
92      public ProcessingStatus getForStatus() {
93          if (this.getConfig().get(FOR_STATUS) == null) {
94              return null;
95          } else {
96              return ProcessingStatus.valueOf(this.getConfig().get(FOR_STATUS).getValue());
97          }
98      }
99  }
