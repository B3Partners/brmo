4   package nl.b3p.brmo.stufbg204.util;
5   
6   import java.text.SimpleDateFormat;
7   import java.util.Date;
8   import javax.xml.bind.JAXBContext;
9   import javax.xml.bind.JAXBException;
10  import nl.egem.stuf.stuf0204.FoutBericht;
11  import nl.egem.stuf.stuf0204.Stuurgegevens;
12  import nl.egem.stuf.stuf0204.Systeem;
13  
14  /**
15   * Utility methodes voor StUF BG 204.
16   *
17   * @author mprins
18   */
19  public final class StUFbg204Util {
20      
21      public static final SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddkkmmssSSS");
22      private static JAXBContext jaxbContext;
23  
24      private StUFbg204Util() {
25      }
26      
27      public static FoutBericht maakFout(String errorcode) {
28          return maakFout(errorcode, null);
29      }
30      
31      public static FoutBericht maakFout(String errorcode, Exception e) {
32          final FoutBericht fout = new FoutBericht();
33          Stuurgegevens.Fout f = new Stuurgegevens.Fout();
34          f.setCrossRefNummer(errorcode);
35          Stuurgegevens s = new Stuurgegevens();
36          s.setBerichtsoort("Fo01");
37          s.setFout(f);
38          
39          fout.setStuurgegevens(s);
40          FoutBericht.Body  b = new FoutBericht.Body();
41          b.setCode(errorcode);
42          if(e != null){
43              b.setOmschrijving(e.getLocalizedMessage());
44          }
45          fout.setBody(b);
46          
47  
48          return fout;
49      }
50      
51      public static Stuurgegevens maakStuurgegevens(Stuurgegevens vraagStuurgegevens){
52          final Stuurgegevens sg = vraagStuurgegevens;
53          Systeem s = new Systeem();
54          s.setApplicatie("BRMO");
55          s.setOrganisatie("B3Partners B.V.");
56          sg.setZender(s);
57          sg.setVersieStUF("0204");
58          sg.setTijdstipBericht(sdf.format(new Date()));
59          
60          return sg;
61      }
62          
63      public static JAXBContext getStufJaxbContext() throws JAXBException{
64          if(jaxbContext == null){
65              jaxbContext = JAXBContext.newInstance("nl.egem.stuf.sector.bg._0204:nl.egem.stuf.stuf0204");
66          }
67          return jaxbContext;
68      }
69  }
