88      /**
89       *
90       * 4.2 Stuurgegevens voor kennisgevingberichten. In paragraaf 2.3 zijn vier
91       * varianten binnen een kennisgevingbericht onderkend:
92       *
93       * <h3>Toevoeging</h3>
94       * Bij een toevoeging is in het zendende systeem een occurrence toegevoegd,
95       * omdat is vastgesteld dat in de werkelijkheid een voor het zendende
96       * systeem relevant object bestaat.
97       *
98       * <h3>Wijziging</h3>
99       * Bij een wijziging is in het zendende systeem een occurrence gewijzigd,
100      * omdat is vastgesteld dat er in de werkelijkheid eigenschappen (gegevens)
101      * zijn veranderd van het object waar naar die occurrence verwijst.
102      *
103      * <h3>Verwijdering</h3>
104      * Bij een verwijdering is in het zendende systeem een occurrence
105      * verwijderd, omdat is vastgesteld dat in de werkelijkheid het object
106      * waarnaar de occurrence verwijst, niet meer bestaat of niet meer relevant
107      * is voor het zendende systeem.
108      *
109      * <h3>Correctie</h3>
110      * Bij een correctie is in het zendende systeem een occurrence gewijzigd,
111      * omdat is vastgesteld dat de vastgelegde waarden niet correct waren. Bij
112      * een correctie is het object in de werkelijkheid waarnaar de occurrence
113      * verwijst, zelf niet gewijzigd. Deze verschillende varianten worden als
114      * volgt gecodeerd in het stuurgegeven mutatiesoort:
115      * <ul>
116      * <li>‘T’: Toevoeging
117      * <li>‘W’: Wijziging
118      * <li>‘V’: Verwijdering
119      * <li>‘C’: Correctie
120      * </ul>
121      * Het is aan het ontvangende systeem om te interpreteren in hoeverre een
122      * kennisgeving relevant is en of de kennisgeving het gevolg is van het
123      * ontstaan of verdwijnen van een object c.q. van het relevant worden of
124      * niet meer relevant zijn van het object voor het zendende systeem. Het
125      * kennisgevingbericht bevat geen aanduiding van de gebeurtenis die
126      * aanleiding gaf tot de wijziging van de gegevens (zie paragraaf 2.3).
127      *
128      * <br/>
129      * Naast de mutatiesoort is in een kennisgevingbericht ook relevant hoe het
130      * ontvangende systeem geacht wordt te reageren op het bericht. Een
131      * kennisgevingbericht kan puur informatief bedoeld zijn: het ontvangende
132      * systeem mag zelf beslissen of de kennisgeving al dan niet wordt verwerkt
133      * in de eigen gegevens. Daarnaast kan het verplicht zijn voor het
134      * ontvangende systeem om de gegevens over te nemen. Of een
135      * kennisgevingbericht informatief is of verplicht over te nemen geeft het
136      * stuurgegeven indicator overname aan met 'I' (informatief) respectievelijk
137      * 'V' (verplicht). Aanvullende afspraken over de omgang met deze rubriek
138      * kunnen worden vastgelegd in het sectormodel.
139      *
140      * @param kennisgeving te verwerken kennisgevingsbericht
141      * @return een bevestiging van de kennisgeving
142      */
143     public BevestigingsBericht ontvangKennisgeving(KennisgevingsBericht kennisgeving) {
144         LOG.debug("Er is een kennisgeving ontvangen van soort: " + kennisgeving.getStuurgegevens().getBerichtsoort()
145                 + " en mutatiesoort " + kennisgeving.getStuurgegevens().getKennisgeving().getMutatiesoort());
146         BevestigingsBericht b = new BevestigingsBericht();
147         b.setStuurgegevens(StUFbg204Util.maakStuurgegevens(kennisgeving.getStuurgegevens()));
148 
149         switch (kennisgeving.getStuurgegevens().getKennisgeving().getMutatiesoort()) {
150             case T:
151                 saveBericht(kennisgeving, kennisgeving.getStuurgegevens().getKennisgeving().getTijdstipMutatie());
152                 break;
153             case W:
154             case C:
155                 // dit werkt -nog- niet omdat dit 2 PRS nodes bevat, een was en een wordt...
156                 saveBericht(kennisgeving, kennisgeving.getStuurgegevens().getKennisgeving().getTijdstipMutatie());
157                 break;
158             case V:
159             default:
160                 LOG.warn("Onbekende mutatiesoort wordt niet verwerkt.");
161                 break;
162         }
163         return b;
164     }

166     private void saveBericht(Object body, String datum) {
167         try {
168             DataSource ds = ConfigUtil.getDataSourceStaging();
169             BrmoFramework brmo = new BrmoFramework(ds, null);
170             InputStream in = getXml(body);
171 
172             Date d = StUFbg204Util.sdf.parse(datum);
173             String bestand_naam = "StUF-BG upload op " + StUFbg204Util.sdf.format(new Date());
174 
175             brmo.loadFromStream(BrmoFramework.BR_BRP, in, bestand_naam, d, null);
176             brmo.closeBrmoFramework();
177         } catch (BrmoException ex) {
178             LOG.error("Fout tijdens laden van StUF-BG bericht", ex);
179         } catch (JAXBException | ParseException ex) {
180             LOG.error("Fout tijdens parsen van bericht", ex);
181         }
182 
183     }

185     private InputStream getXml(Object o) throws JAXBException {
186 
187         try {
188             // maak van POJO een inputstream
189             Marshaller jaxbMarshaller = StUFbg204Util.getStufJaxbContext().createMarshaller();
190             ByteArrayOutputStream baos = new ByteArrayOutputStream();
191             jaxbMarshaller.marshal(o, baos);
192             InputStream in = new ByteArrayInputStream(baos.toByteArray());
193 
194             // maak er een document van
195             DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
196             factory.setNamespaceAware(true);
197             DocumentBuilder builder = factory.newDocumentBuilder();
198 
199             Document doc = builder.parse(in);
200             List<SimpleEntry<String, String>> prefixes = new ArrayList<>();
201             getPrefixesRecursive(doc.getDocumentElement(), prefixes);
202             //haal de body eruit met xpath
203             XPathFactory xPathfactory = XPathFactory.newInstance();
204             XPath xpath = xPathfactory.newXPath();
205             XPathExpression expr = xpath.compile("//*[local-name() = 'body']/*");
206             NodeList nodelist = (NodeList) expr.evaluate(doc, XPathConstants.NODESET);
207             
208             Document newDoc = builder.newDocument();
209             Node root = newDoc.createElement("root");
210 
211             for (int i = 0; i < nodelist.getLength(); i++) {
212                 Node n = nodelist.item(i);
213                 Node newNode = newDoc.importNode(n, true);
214                 root.appendChild(newNode);
215             }
216 
217             // Vertaal xml naar inputstream voor verwerking in brmo framework
218             ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
219             Transformer t = TransformerFactory.newInstance().newTransformer();
220             t.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
221             t.transform(new DOMSource(root), new StreamResult(outputStream));
222             InputStream is = new ByteArrayInputStream(outputStream.toByteArray());
223 
224             return is;
225         } catch (ParserConfigurationException | SAXException | IOException | XPathExpressionException | TransformerException ex) {
226             LOG.error("Cannot parse body", ex);
227         }
228         return null;
229     }

231     private static final String XMLNAMESPACE = "xmlns";

233     public static void getPrefixesRecursive(Element element, List<SimpleEntry<String, String>> prefixes) {
234         getPrefixes(element, prefixes);
235         Node parent = element.getParentNode();
236         if (parent instanceof Element) {
237             getPrefixesRecursive((Element) parent, prefixes);
238         }
239     }

241     /**
242      * Get all prefixes defined on this element for the specified namespace.
243      *
244      * @param element dom element
245      * @param prefixes lijst van namespace prefixes
246      */
247     public static void getPrefixes(Element element, List<SimpleEntry<String, String>> prefixes) {
248         NamedNodeMap atts = element.getAttributes();
249         for (int i = 0; i < atts.getLength(); i++) {
250             Node node = atts.item(i);
251             String name = node.getNodeName();
252             if (name != null && (XMLNAMESPACE.equals(name) || name.startsWith(XMLNAMESPACE + ":"))) {
253                 SimpleEntry s = new SimpleEntry(name, node.getNodeValue());
254                 prefixes.add(s);
255             }
256         }
257     }
