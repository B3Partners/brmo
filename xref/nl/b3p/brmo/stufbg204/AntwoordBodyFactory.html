306     private static void calculateNoValue(Object element){
307         Class c = element.getClass();
308         Object val = null;
309         try {
310             Method method = element.getClass().getDeclaredMethod("getValue");
311             val = method.invoke(element);
312         } catch (SecurityException | NoSuchMethodException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
313             LOG.error("Cannot get value for element " + c.toString(), e);
314         }
315         if (val == null) {
316             try {
317                 Method method = element.getClass().getDeclaredMethod("setNoValue", NoValue.class);
318                 val = method.invoke(element, NoValue.GEEN_WAARDE);
319             } catch (SecurityException | NoSuchMethodException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
320                 LOG.error("Cannot set NoValue for element " + c.toString(), e);
321             }
322         }
323     }
