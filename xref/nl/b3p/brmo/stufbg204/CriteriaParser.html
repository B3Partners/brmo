94      public String getCriteria(VraagBericht bericht) throws IllegalArgumentException,UnsupportedOperationException {
95          String q;
96          String entiteitType = bericht.getStuurgegevens().getEntiteittype();
97          Body b = bericht.getBody();
98          switch (entiteitType) {
99              case "PRS": {
100                 q = getPRSCriterias(b);
101                 break;
102             }
103             default:
104                 throw new IllegalArgumentException("Entiteitstype niet ondersteund: " + entiteitType);
105         }
106         if (q != null) {
107             q = "WHERE " + q;
108         }
109         return q;
110     }
111 
112     private String getPRSCriterias(Body b) throws IllegalArgumentException,UnsupportedOperationException {
113         String q;
114         List<PRSVraag> prs = b.getPRS();
115 
116         String from = getPRSCriteria(prs.get(0), true);
117         String toCheck = getPRSCriteria(prs.get(1), true); // maak even een tweede filter aan met dezelfde operator, om te checken of er op hetzelfde gecheckt wordt == equals 
118         String to = getPRSCriteria(prs.get(1), false);
119         if (from == null && to == null) {
120             return null;
121         }
122         if (from.equals(toCheck)) {
123             q = from.replace(">", "=");
124         } else {
125             q = from + " AND " + to;
126         }
127 
128         return q;
129     }
