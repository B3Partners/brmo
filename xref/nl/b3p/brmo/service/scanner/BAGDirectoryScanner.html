6   package nl.b3p.brmo.service.scanner;
7   
8   import java.io.File;
9   import java.io.IOException;
10  import java.util.Arrays;
11  import java.util.Calendar;
12  import java.util.Date;
13  import javax.persistence.Transient;
14  import nl.b3p.brmo.loader.BrmoFramework;
15  import nl.b3p.brmo.loader.util.BrmoDuplicaatLaadprocesException;
16  import nl.b3p.brmo.loader.util.BrmoException;
17  import nl.b3p.brmo.loader.util.BrmoLeegBestandException;
18  import nl.b3p.brmo.persistence.staging.AutomatischProces;
19  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.ERROR;
20  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.PROCESSING;
21  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.WAITING;
22  import nl.b3p.brmo.persistence.staging.BAGScannerProces;
23  import nl.b3p.brmo.service.util.ConfigUtil;
24  import org.apache.commons.io.FileUtils;
25  import org.apache.commons.io.comparator.NameFileComparator;
26  import org.apache.commons.logging.Log;
27  import org.apache.commons.logging.LogFactory;
28  import org.stripesstuff.stripersist.Stripersist;
29  
30  /**
31   *
32   * @author mprins
33   */
34  public class BAGDirectoryScanner extends AbstractExecutableProces {
35  
36      private static final Log LOG = LogFactory.getLog(BAGDirectoryScanner.class);
37  
38      private final BAGScannerProces config;
39  
40      @Transient
41      private ProgressUpdateListener listener;
42  
43      public BAGDirectoryScanner(BAGScannerProces config) {
44          this.config = config;
45      }
46  
47      /**
48       *
49       * @throws BrmoException als de directory niet lees/blader/schrijfbaar is
50       */
51      @Override
52      public void execute() throws BrmoException {
53          this.execute(new ProgressUpdateListener() {
54              @Override
55              public void total(long total) {
56              }
57  
58              @Override
59              public void progress(long progress) {
60              }
61  
62              @Override
63              public void exception(Throwable t) {
64                  LOG.error(t);
65              }
66  
67              @Override
68              public void updateStatus(String status) {
69              }
70  
71              @Override
72              public void addLog(String log) {
73              }
74          });
75      }
76  
77      @Override
78      public void execute(ProgressUpdateListener listener) {
79          this.listener = listener;
80          config.setStatus(PROCESSING);
81          config.setLastrun(new Date());
82          Stripersist.getEntityManager().merge(config);
83          Stripersist.getEntityManager().flush();
84  
85          StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
86          String oldLog = config.getLogfile();
87          if (oldLog != null) {
88              if (oldLog.length() > OLD_LOG_LENGTH) {
89                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
90              } else {
91                  sb.append(oldLog).append(AutomatischProces.LOG_NEWLINE);
92              }
93          }
94  
95          config.setStatus(PROCESSING);
96          String msg = String.format("De BAG scanner met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
97          LOG.info(msg);
98          listener.updateStatus(msg);
99          listener.addLog(msg);
100         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
101         this.active = true;
102 
103         // validatie van de directories, kunnen we lezen/bladeren en evt. schrijven?
104         final File scanDirectory = new File(this.config.getScanDirectory());
105         if (!scanDirectory.isDirectory() || !scanDirectory.canExecute()) {
106             config.setStatus(ERROR);
107             config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
108             this.active = false;
109             msg = String.format("De scan directory '%s' is geen executable directory", scanDirectory);
110             LOG.info(msg);
111             listener.updateStatus(msg);
112             listener.addLog(msg);
113             this.listener.exception(new BrmoException(msg));
114             return;
115         }
116         final String aDir = this.config.getArchiefDirectory();
117         final boolean isArchiving = (aDir != null);
118         File archiefDirectory = null;
119         if (isArchiving) {
120             archiefDirectory = new File(aDir);
121             archiefDirectory.mkdirs();
122             if (!archiefDirectory.isDirectory() || !archiefDirectory.canWrite()) {
123                 config.setStatus(ERROR);
124                 config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
125                 this.active = false;
126                 msg = String.format("De archief directory '%s' is geen beschrijfbare directory", archiefDirectory);
127                 LOG.error(msg);
128                 listener.updateStatus(msg);
129                 listener.addLog(msg);
130                 this.listener.exception(new BrmoException(msg));
131                 return;
132             }
133             if (!scanDirectory.canWrite()) {
134                 config.setStatus(ERROR);
135                 config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
136                 this.active = false;
137                 msg = String.format(String.format("De scan directory '%s' is geen beschrijfbare directory", scanDirectory));
138                 LOG.error(msg);
139                 listener.updateStatus(msg);
140                 listener.addLog(msg);
141                 this.listener.exception(new BrmoException(msg));
142                 return;
143             }
144         }
145 
146         File files[] = scanDirectory.listFiles();
147         Arrays.sort(files, NameFileComparator.NAME_COMPARATOR);
148 
149         int filterAlVerwerkt = 0;
150         int aantalGeladen = 0;
151         int progress = 0;
152 
153         if (files.length < 1) {
154             msg = String.format("Geen bestanden gevonden in scandirectory: %s.", scanDirectory);
155             LOG.info(msg);
156             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
157             listener.updateStatus(msg);
158             listener.addLog(msg);
159         }
160 
161         listener.total(files.length);
162         for (File f : files) {
163             if (f.isDirectory()) {
164                 continue;
165             }
166             msg = String.format("Bestand %s is gevonden in %s.", f, scanDirectory);
167             LOG.info(msg);
168             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
169             listener.updateStatus(msg);
170             listener.addLog(msg);
171             BrmoFramework brmo = null;
172             try {
173                 if (this.isDuplicaatLaadProces(f, BrmoFramework.BR_BAG)) {
174                     msg = String.format("Bestand %s is een duplicaat en wordt overgeslagen.", f);
175                     LOG.info(msg);
176                     sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
177                     listener.updateStatus(msg);
178                     listener.addLog(msg);
179                     filterAlVerwerkt++;
180                 } else {
181                     brmo = new BrmoFramework(ConfigUtil.getDataSourceStaging(), null);
182                     brmo.loadFromFile(BrmoFramework.BR_BAG, getBestandsNaam(f), config.getId());
183                     msg = String.format("Bestand %s is geladen.", f);
184                     LOG.info(msg);
185                     sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
186                     listener.updateStatus(msg);
187                     listener.addLog(msg);
188                     aantalGeladen++;
189                 }
190             } catch (BrmoDuplicaatLaadprocesException duplicaat) {
191                 LOG.info(duplicaat.getLocalizedMessage());
192                 sb.append(duplicaat.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
193                 filterAlVerwerkt++;
194             } catch (BrmoLeegBestandException leegEx) {
195                 // log message maar ga door met verwerking, om een "leeg" bestand bericht/laadproces over te slaan
196                 LOG.warn(leegEx.getLocalizedMessage());
197                 sb.append(leegEx.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
198             } catch (BrmoException ex) {
199                 LOG.error(ex.getLocalizedMessage());
200                 sb.append(ex.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
201             } finally {
202                 if (brmo != null) {
203                     brmo.closeBrmoFramework();
204                 }
205             }
206 
207             if (isArchiving) {
208                 // verplaats naar archief
209                 try {
210                     FileUtils.copyFileToDirectory(f, archiefDirectory);
211                     boolean succes = FileUtils.deleteQuietly(f);
212                     if (succes) {
213                         msg = String.format("  Bestand %s is naar archief %s verplaatst.", f, archiefDirectory);
214                     } else {
215                         msg = String.format("  Bestand %s is naar archief %s verplaatst, maar origineel kon niet worden verwijderd.", f, archiefDirectory);
216                     }
217                 } catch (IOException e) {
218                     msg = String.format("  Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).", f, archiefDirectory, e.getLocalizedMessage());
219                     LOG.error(msg);
220                     listener.updateStatus(msg);
221                     listener.addLog(msg);
222                 }
223                 LOG.info(msg);
224                 listener.updateStatus(msg);
225                 listener.addLog(msg);
226                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
227             }
228             listener.progress(++progress);
229         }
230 
231         listener.addLog("\n\n**** resultaat ****");
232         listener.addLog("Aantal bestanden die al waren geladen: " + filterAlVerwerkt);
233         listener.addLog("Aantal bestanden geladen: " + aantalGeladen + "\n");
234 
235         msg = "Aantal bestanden die al waren verwerkt: " + filterAlVerwerkt + ", aantal bestanden geladen: " + aantalGeladen;
236         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
237         LOG.info(msg);
238         config.setSamenvatting(msg);
239 
240         msg = String.format("De BAG scanner met ID %d is afgerond op %tc.", config.getId(), Calendar.getInstance());
241         LOG.info(msg);
242         listener.updateStatus(msg);
243         listener.addLog(msg);
244         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
245 
246         this.active = false;
247         config.setStatus(WAITING);
248         config.setLogfile(sb.toString());
249 
250         Stripersist.getEntityManager().merge(config);
251     }
252 }
