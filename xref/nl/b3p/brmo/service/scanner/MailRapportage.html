38  /**
39   * Mailt rapporten.
40   *
41   * @author mprins
42   */
43  public class MailRapportage extends AbstractExecutableProces {
44  
45      private static final Log log = LogFactory.getLog(MailRapportage.class);
46      private MailRapportageProces config;
47  
48      public MailRapportage(MailRapportageProces config) {
49          this.config = config;
50      }
51  
52      /**
53       * De inhoud van dit bericht bestaat uit de samenvattingen en de status van
54       * de geselecteerde processen, indien niets geselecteerd rapportage van
55       * alles behalve het eigen proces.
56       *
57       * @throws BrmoException als er een fout optreed in het opzoeken van de
58       * mailserver (indi) of het versturen van de mail
59       */
60      @Override
61      public void execute() throws BrmoException {
62          this.active = true;
63          StringBuilder sb = new StringBuilder();
64          config.setStatus(PROCESSING);
65          String logMsg = String.format("De mail rapportage met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
66          log.info(logMsg);
67          sb.append(logMsg).append(LOG_NEWLINE);
68  
69          try {
70              Context init = new InitialContext();
71              Context env = (Context) init.lookup("java:comp/env");
72              Session session = (Session) env.lookup("mail/session");
73              Message msg = new MimeMessage(session);
74              msg.setFrom(/* 'mail.from' property uit jndi session object */);
75              msg.setSentDate(new Date());
76              // subject
77              String sForStatus = (config.getForStatus() == null ? "alle" : config.getForStatus().toString());
78              String sPIDS = (config.getConfig().get(MailRapportageProces.PIDS) == null ? "alle" : config.getConfig().get(MailRapportageProces.PIDS).getValue());
79              String label = (ClobElement.nullSafeGet(config.getConfig().get("label")) == null ? "" : config.getConfig().get("label").getValue());
80              String subject = String.format("BRMO rapport: %d (%s) voor status: %s (taken: %s)", config.getId(), label, sForStatus, sPIDS);
81              msg.setSubject(subject);
82  
83              // ontvangers
84              String[] adressen = this.config.getMailAdressenArray();
85              InternetAddress[] aanAdr = new InternetAddress[adressen.length];
86              for (int i = 0; i < adressen.length; i++) {
87                  aanAdr[i] = new InternetAddress(adressen[i]);
88              }
89              msg.setRecipients(Message.RecipientType.TO, aanAdr);
90  
91              StringBuilder mailText = new StringBuilder();
92              List<AutomatischProces> processen = this.getProcessen();
93              // bericht mailText samenstellen
94              for (AutomatischProces p : processen) {
95                  logMsg = String.format("Rapportage van taak %d met status: %s", p.getId(), p.getStatus());
96                  log.info(logMsg);
97                  sb.append(logMsg).append(LOG_NEWLINE);
98  
99                  mailText.append("Rapport van taak: ").append(p.getId()).append(" ").append(label)
100                         .append(", type: ").append(p.getClass().getSimpleName()).append(LOG_NEWLINE);
101                 mailText.append("Status van de taak: ").append(p.getStatus()).append(LOG_NEWLINE);
102                 mailText.append("Samenvatting: ").append(LOG_NEWLINE);
103                 mailText.append(p.getSamenvatting()).append(LOG_NEWLINE);
104                 mailText.append("---------------------------------").append(LOG_NEWLINE);
105             }
106             if (processen.isEmpty()) {
107                 mailText.append("Er waren geen automatische processen die voldeden aan de opgegeven voorwaarden: ").append(LOG_NEWLINE);
108                 mailText.append(" - Status: ").append(sForStatus).append(LOG_NEWLINE);
109                 mailText.append(" - Taken: ").append(sPIDS).append(LOG_NEWLINE);
110             }
111             log.debug(mailText.toString());
112             msg.setText(mailText.toString());
113             Transport.send(msg);
114 
115             logMsg = String.format("Klaar met taak %d op %tc", config.getId(), Calendar.getInstance());
116             log.info(logMsg);
117             sb.append(logMsg).append(LOG_NEWLINE);
118             config.setStatus(WAITING);
119             config.updateSamenvattingEnLogfile(sb.toString());
120             config.setLastrun(new Date());
121         } catch (MessagingException ex) {
122             log.error(ex);
123             config.setStatus(ERROR);
124             throw new BrmoException(ex);
125         } catch (NamingException ex) {
126             log.error(ex);
127             config.setStatus(ERROR);
128             throw new BrmoException(ex);
129         } finally {
130             this.active = false;
131         }
132     }
133 
134     /**
135      * Haalt de lijst met processen op uit de database.
136      *
137      * @return lijst met te rapporten processen
138      * @todo Eventueel andere mail rapportage processen ook uitsluiten?
139      */
140     private List<AutomatischProces> getProcessen() {
141         // unit test in nl.b3p.brmo.persistence.staging.MailRapportageProcesTest#testRapportageLijst()
142         List<Predicate> predicates = new ArrayList<Predicate>();
143 
144         final EntityManager em = Stripersist.getEntityManager();
145         CriteriaBuilder cb = em.getCriteriaBuilder();
146         CriteriaQuery<AutomatischProces> cq = cb.createQuery(AutomatischProces.class);
147         Root<AutomatischProces> from = cq.from(AutomatischProces.class);
148 
149         // where status=... filter
150         if (this.config.getForStatus() != null) {
151             Predicate where = cb.equal(from.get("status"), this.config.getForStatus());
152             predicates.add(where);
153         }
154         // processen in... filter
155         String pids = ClobElement.nullSafeGet(this.config.getConfig().get(MailRapportageProces.PIDS));
156         if (pids != null) {
157             List<Long> pidLijst = new ArrayList<Long>();
158             Matcher match = (Pattern.compile("[0-9]+")).matcher(pids);
159             while (match.find()) {
160                 pidLijst.add(Long.valueOf(match.group()));
161             }
162             Predicate in = from.get("id").in(pidLijst);
163             predicates.add(in);
164         }
165         // TODO eventueel filter op dtype
166         // niet het eigen proces rapporteren
167         Predicate notIn = from.get("id").in(this.config.getId()).not();
168         predicates.add(notIn);
169 
170         cq.where(cb.and(predicates.toArray(new Predicate[predicates.size()])));
171         // sorteer op status, daarna op datum zodat errors bovenaan staan
172         cq.orderBy(cb.asc(from.get("status")));
173         cq.orderBy(cb.asc(from.get("lastrun")));
174 
175         return em.createQuery(cq).getResultList();
176     }

184         }
185     }
