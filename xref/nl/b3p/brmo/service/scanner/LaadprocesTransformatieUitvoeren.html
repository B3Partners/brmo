6   import java.util.ArrayList;
7   import java.util.Arrays;
8   import java.util.Calendar;
9   import java.util.Date;
10  import java.util.List;
11  import javax.persistence.Transient;
12  import javax.sql.DataSource;
13  import nl.b3p.brmo.loader.BrmoFramework;
14  import nl.b3p.brmo.loader.RsgbProxy;
15  import nl.b3p.brmo.loader.util.BrmoException;
16  import nl.b3p.brmo.persistence.staging.AutomatischProces;
17  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.ERROR;
18  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.PROCESSING;
19  import static nl.b3p.brmo.persistence.staging.AutomatischProces.ProcessingStatus.WAITING;
20  import nl.b3p.brmo.persistence.staging.LaadprocesTransformatieProces;
21  import nl.b3p.brmo.service.util.ConfigUtil;
22  import org.apache.commons.lang3.ArrayUtils;
23  import org.apache.commons.lang3.exception.ExceptionUtils;
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  import org.stripesstuff.stripersist.Stripersist;

28  /**
29   * Transformeert laadprocessn, zoals bijvoorbeeld BGTLight GML.
30   *
31   * @author mprins
32   */
33  public class LaadprocesTransformatieUitvoeren extends AbstractExecutableProces {
34  
35      private static final Log LOG = LogFactory.getLog(LaadprocesTransformatieUitvoeren.class);
36  
37      private final LaadprocesTransformatieProces config;
38  
39      @Transient
40      private ProgressUpdateListener listener;
41  
42      private boolean transformErrorOccured = false;
43  
44      public LaadprocesTransformatieUitvoeren(LaadprocesTransformatieProces config) {
45          this.config = config;
46      }
47  
48      @Override
49      public void execute() throws BrmoException {
50          this.execute(new ProgressUpdateListener() {
51              @Override
52              public void total(long total) {
53              }
54  
55              @Override
56              public void progress(long progress) {
57              }
58  
59              @Override
60              public void exception(Throwable t) {
61                  // afvangen van de fout conditie zodat de status van het proces goed kan worden gezet
62                  transformErrorOccured = true;
63                  LOG.error(t);
64              }
65  
66              @Override
67              public void updateStatus(String status) {
68              }
69  
70              @Override
71              public void addLog(String log) {
72              }
73          });
74      }
75  
76      @Override
77      public void execute(final ProgressUpdateListener listener) {
78          this.listener = listener;
79          listener.updateStatus("Initialiseren...");
80          listener.addLog(String.format("Initialiseren... %tc", new Date()));
81          this.config.setStatus(PROCESSING);
82          this.config.setLastrun(new Date());
83          Stripersist.getEntityManager().merge(config);
84          Stripersist.getEntityManager().flush();
85  
86          final StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
87          String oldLog = config.getLogfile();
88          if (oldLog != null) {
89              if (oldLog.length() > OLD_LOG_LENGTH) {
90                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
91              } else {
92                  sb.append(oldLog);
93              }
94          }
95  
96          String msg = String.format("Het laadproces transformatie proces met ID %d is gestart op %tc.",
97                  config.getId(), Calendar.getInstance());
98          LOG.info(msg);
99          listener.addLog(msg);
100         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
101 
102         BrmoFramework brmo = null;
103         try {
104             DataSource dataSourceStaging = ConfigUtil.getDataSourceStaging();
105             DataSource dataSourceRsgbBgt = null;
106             DataSource dataSourceTopNL = null;
107             if (LaadprocesTransformatieProces.LaadprocesSoorten.BR_TOPNL.getSoort().equalsIgnoreCase(this.config.getSoort())) {
108                 dataSourceTopNL = ConfigUtil.getDataSourceTopNL();
109             }
110             if (LaadprocesTransformatieProces.LaadprocesSoorten.BR_BGTLIGHT.getSoort().equalsIgnoreCase(this.config.getSoort())) {
111                 dataSourceRsgbBgt = ConfigUtil.getDataSourceRsgbBgt();
112             }
113             brmo = new BrmoFramework(dataSourceStaging, null, dataSourceRsgbBgt, dataSourceTopNL);
114             long[] lpIds = null;
115             
116             if(LaadprocesTransformatieProces.LaadprocesSoorten.BR_TOPNL.getSoort().equalsIgnoreCase(this.config.getSoort())){
117                 List<Long> ids = new ArrayList<>();
118                 for (String type : TopNLDirectoryScanner.subdirectoryNames) {
119                     ids.addAll(Arrays.asList(brmo.getLaadProcessenIds("bestand_datum", "ASC", type, "STAGING_OK")));
120                 }
121                 lpIds = ArrayUtils.toPrimitive(ids.toArray(new Long[ids.size()]));
122             }else{
123                 lpIds = ArrayUtils.toPrimitive(brmo.getLaadProcessenIds("bestand_datum", "ASC", config.getSoort(), "STAGING_OK"));
124             }
125             brmo.setOrderBerichten(!config.alsStandTransformeren());
126             if (lpIds == null || lpIds.length == 0) {
127                 msg = "Er zijn geen laadprocessen van soort " + config.getSoort() + " en status: 'STAGING_OK' om te transformeren.";
128                 LOG.info(msg);
129                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
130                 listener.addLog(msg);
131             } else {
132                 listener.updateStatus("Transformeren...");
133                 Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_LAADPROCES, lpIds,
134                         new nl.b3p.brmo.loader.ProgressUpdateListener() {
135                     @Override
136                     public void total(long total) {
137                         sb.append("Totaal te transformeren: ")
138                                 .append(total).append(AutomatischProces.LOG_NEWLINE);
139                         listener.total(total);
140                     }
141 
142                     @Override
143                     public void progress(long progress) {
144                         listener.progress(progress);
145                     }
146 
147                     @Override
148                     public void exception(Throwable t) {
149                         sb.append("Fout tijdens laadproces transformeren: ")
150                                 .append(t.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
151                         listener.exception(t);
152                     }
153                 }
154                 );
155                 t.join();
156             }
157 
158             if (transformErrorOccured) {
159                 this.config.setStatus(ERROR);
160                 msg = "Handmatige transformatie vanuit de laadprocessen pagina is noodzakelijk.";
161                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
162                 LOG.warn(msg);
163                 msg = String.format("Laadproces transformatie proces met ID %d is niet succesvol afgerond op %tc.", config.getId(), Calendar.getInstance());
164                 LOG.error(msg);
165                 listener.updateStatus("Er is een fout opgetreden.");
166             } else {
167                 this.config.setStatus(WAITING);
168                 msg = String.format("Laadproces transformatie proces met ID %d is succesvol afgerond op %tc.", config.getId(), Calendar.getInstance());
169                 LOG.info(msg);
170                 listener.updateStatus("Transformatie afgerond.");
171             }
172 
173             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
174             this.listener.addLog(msg);
175             this.config.setSamenvatting(msg);
176             this.config.setLogfile(sb.toString());
177         } catch (BrmoException | InterruptedException e) {
178             LOG.error("Fout bij transformeren laadprocessen naar RSGB", e);
179             String m = "Fout bij transformeren laadprocessen naar RSGB: " + ExceptionUtils.getMessage(e);
180             if (e.getCause() != null) {
181                 m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(e);
182             }
183             this.config.setLogfile(sb.toString());
184             this.config.updateSamenvattingEnLogfile(m);
185             this.config.setStatus(ERROR);
186         } finally {
187             if (brmo != null) {
188                 brmo.closeBrmoFramework();
189             }
190             Stripersist.getEntityManager().merge(this.config);
191             Stripersist.getEntityManager().flush();
192             Stripersist.getEntityManager().getTransaction().commit();
193             Stripersist.getEntityManager().clear();
194         }
195     }
