62                  return new BAGDirectoryScanner((BAGScannerProces) config);
63              case BRKScannerProces:
64                  return new BRKDirectoryScanner((BRKScannerProces) config);
65              case MailRapportageProces:
66                  return new MailRapportage((MailRapportageProces) config);
67              case GDS2OphaalProces:
68                  return new GDS2OphalenProces((GDS2OphaalProces) config);
69              case BerichtTransformatieProces:
70                  return new BerichtTransformatieUitvoeren((BerichtTransformatieProces) config);
71              case BerichtDoorstuurProces:
72                  return new BerichtDoorsturenProces((BerichtDoorstuurProces)config);
73              case WebMirrorBAGScannerProces:
74                  return new WebMirrorBAGDirectoryScanner((WebMirrorBAGScannerProces) config);
75              case BGTLightOphaalProces:
76                  return new BGTLightOphalenProces((BGTLightOphaalProces) config);
77              case BGTLightScannerProces:
78                  return new BGTLightDirectoryScanner((BGTLightScannerProces) config);
79              case LaadprocesTransformatieProces:
80                  return new LaadprocesTransformatieUitvoeren((LaadprocesTransformatieProces) config);
81              case MaterializedViewRefresh:
82                  return new MaterializedViewRefreshUitvoeren((MaterializedViewRefresh) config);
83              case BerichtstatusRapportProces:

88                  throw new IllegalArgumentException(imple.name() + " is is geen ondersteund proces...");
89          }

126     /**
127      * bepaal of het bestand een duplicaat is op basis van de bestandsnaam en
128      * soort.
129      *
130      * De flow in
131      * <pre>
132      * loadFromFile(bericht)
133      * 	→ stagingProxy.loadBr(InputStream stream, String type, String fileName,...)
134      * 	→ snapshot reader van de input stream parsed het bericht in een BrkSnapshotXMLReader of BagMutatieXMLReader die bericht voor bericht uitgelezen kunnen worden
135      * 	→ bepaal of laadproces bestaat stagingProxy.laadProcesExists(filenaam/datum)
136      * 	→ laadproces in database maken stagingProxy.writeLaadProces(bestand_naam/bestand_datum/soort/gebied/opmerking/status/status_datum/contact_email)
137      * 	→ uitlezen xml bericht als
138      * 	→ !stagingProxy.berichtExists(laadprocesid/object_ref/datum/volgordenummer)
139      * 	→ stagingProxy.writeBericht(b)
140      * </pre>
141      *
142      * @param input een input bestand
143      * @param soort het type registratie, bijvoorbeeld
144      * {@value nl.b3p.brmo.loader.BrmoFramework#BR_BRK}
145      *
146      * @return {@code true} als het bestand een duplicaat betreft, anders
147      * {@code false}
148      *
149      */
150     protected boolean isDuplicaatLaadProces(File input, String soort) {
151         log.debug("Controle voor duplicaat laadproces, soort: '" + soort + "', bestand: " + input.getName());
152         final String name = getBestandsNaam(input);
153         EntityManager em = Stripersist.getEntityManager();
154         CriteriaBuilder criteriaBuilder = em.getCriteriaBuilder();
155         CriteriaQuery<LaadProces> criteriaQuery = criteriaBuilder.createQuery(LaadProces.class);
156         Root<LaadProces> from = criteriaQuery.from(LaadProces.class);
157         CriteriaQuery<LaadProces> select = criteriaQuery.select(from);
158         Predicate _bestand_naam = criteriaBuilder.equal(from.get("bestand_naam"), name);
159         Predicate _soort = criteriaBuilder.equal(from.get("soort"), soort);
160         criteriaQuery.where(criteriaBuilder.and(_bestand_naam, _soort));
161         TypedQuery<LaadProces> typedQuery = em.createQuery(select);
162 
163         return !typedQuery.getResultList().isEmpty();
164     }

166     /**

172      */
173     protected String getBestandsNaam(File f) {
174         return f.getAbsolutePath();
175     }

178         return new Date(f.lastModified());
