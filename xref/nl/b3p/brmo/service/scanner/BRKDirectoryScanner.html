46    private static final Log log = LogFactory.getLog(BRKDirectoryScanner.class);
47  
48    private final BRKScannerProces config;
49  
50    @Transient private ProgressUpdateListener listener;
51  
52    public BRKDirectoryScanner(BRKScannerProces config) {
53      this.config = config;
54    }

56    /**
57     * Leest de xml berichten uit de directory in in de database, indien geconfigureerd worden
58     * bestanden gearchiveerd.
59     *
60     * @throws BrmoException als de directory niet lees/blader/schrijfbaar is
61     */
62    @Override
63    public void execute() throws BrmoException {
64      this.execute(
65          new ProgressUpdateListener() {
66            @Override
67            public void total(long total) {}
68  
69            @Override
70            public void progress(long progress) {}
71  
72            @Override
73            public void exception(Throwable t) {
74              log.error(t);
75            }
76  
77            @Override
78            public void updateStatus(String status) {}
79  
80            @Override
81            public void addLog(String log) {}
82          });
83    }
84  
85    @Override
86    public void execute(ProgressUpdateListener listener) {
87      final EntityManager em = Stripersist.getEntityManager();
88      this.listener = listener;
89      config.setStatus(PROCESSING);
90      StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
91      String oldLog = config.getLogfile();
92      if (oldLog != null) {
93        if (oldLog.length() > OLD_LOG_LENGTH) {
94          sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
95        } else {
96          sb.append(oldLog);
97        }
98      }
99  
100     String msg =
101         String.format(
102             "De BRK scanner met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
103     log.info(msg);
104     listener.addLog(msg);
105     sb.append(msg);
106 
107     // validatie van de directories, kunnen we lezen/bladeren en evt. schrijven?
108     final File scanDirectory = new File(this.config.getScanDirectory());
109     if (!scanDirectory.isDirectory() || !scanDirectory.canExecute()) {
110       config.setStatus(ERROR);
111       msg = String.format("De scan directory '%s' is geen executable directory", scanDirectory);
112       config.setLogfile(msg);
113       config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
114       this.listener.exception(new BrmoException(msg));
115       return;
116     }
117     // validatie archief directory
118     final String aDir = this.config.getArchiefDirectory();
119     File archiefDirectory = null;
120     if (aDir != null) {
121       archiefDirectory = new File(aDir);
122       archiefDirectory.mkdirs();
123       if (!archiefDirectory.isDirectory() || !archiefDirectory.canWrite()) {
124         config.setStatus(ERROR);
125         msg =
126             String.format(
127                 "De archief directory '%s' is geen beschrijfbare directory, er worden geen bestanden gearchiveerd.",
128                 archiefDirectory);
129         sb.append(msg);
130         config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
131         log.error(msg);
132         archiefDirectory = null;
133         this.listener.exception(new BrmoException(msg));
134       }
135       if (!scanDirectory.canWrite()) {
136         config.setStatus(ERROR);
137         msg =
138             String.format(
139                 "De scan directory '%s' is geen beschrijfbare directory, er worden geen bestanden gearchiveerd.",
140                 scanDirectory);
141         sb.append(msg);
142         config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
143         log.error(msg);
144         archiefDirectory = null;
145         this.listener.exception(new BrmoException(msg));
146       }
147     }
148     config.setLogfile(sb.toString());
149 
150     File files[] =
151         scanDirectory.listFiles((File dir, String name) -> name.toLowerCase().endsWith(".xml"));
152     Arrays.sort(files, NameFileComparator.NAME_COMPARATOR);
153 
154     processXMLFiles(files, scanDirectory, archiefDirectory, em);
155   }
156 
157   /**
158    * verwerk een bestandenlijst.
159    *
160    * @param files array met xml bestanden
161    * @param scanDirectory te scannen directory
162    * @param archiefDirectory waar de gescande files worden neergezet als de archief optie actief is
163    * @param em de te gebruiken EntityManager
164    */
165   private void processXMLFiles(
166       File[] files, File scanDirectory, File archiefDirectory, EntityManager em) {
167     StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE + config.getLogfile());
168     String msg;
169     int filterAlVerwerkt = 0;
170     int aantalGeladen = 0;
171     int progress = 0;
172     SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
173 
174     listener.total(files.length);
175     for (File f : files) {
176       if (f.isDirectory()) {
177         continue;
178       }
179       msg = String.format("Bestand %s is gevonden in %s.", f, scanDirectory);
180       log.info(msg);
181       listener.addLog(msg);
182       sb.append(AutomatischProces.LOG_NEWLINE).append(msg).append(AutomatischProces.LOG_NEWLINE);
183       if (this.isDuplicaatLaadProces(f, BrmoFramework.BR_BRK)) {
184         msg = String.format("  Bestand %s is een duplicaat en wordt overgeslagen.", f);
185         listener.addLog(msg);
186         log.info(msg);
187         sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
188         filterAlVerwerkt++;
189       } else {
190         try {
191           LaadProces lp = new LaadProces();
192           lp.setBestand_naam(getBestandsNaam(f));
193           lp.setBestand_datum(getBestandsDatum(f));
194           lp.setSoort(BrmoFramework.BR_BRK);
195           lp.setStatus(LaadProces.STATUS.STAGING_OK);
196           lp.setOpmerking(
197               String.format(
198                   "Bestand geladen van %s op %s", f.getAbsolutePath(), sdf.format(new Date())));
199           lp.setAutomatischProces(
200               Stripersist.getEntityManager().find(AutomatischProces.class, config.getId()));
201 
202           Bericht b = new Bericht();
203           b.setLaadprocesid(lp);
204           b.setDatum(lp.getBestand_datum());
205           b.setSoort(BrmoFramework.BR_BRK);
206           b.setStatus_datum(new Date());
207           String fileContent = FileUtils.readFileToString(f, StandardCharsets.UTF_8);
208           fileContent = fileContent.replaceAll("\u0000.*", "");
209           b.setBr_orgineel_xml(fileContent);
210 
211           try {
212             BrkSnapshotXMLReader reader =
213                 new BrkSnapshotXMLReader(
214                     new ByteArrayInputStream(
215                         b.getBr_orgineel_xml().getBytes(StandardCharsets.UTF_8)));
216             BrkBericht bericht = reader.next();
217 
218             if (bericht.getDatum() != null) {
219               b.setDatum(bericht.getDatum());
220             }
221             b.setBr_xml(bericht.getBrXml());
222             b.setVolgordenummer(bericht.getVolgordeNummer());
223             lp.setBestand_naam_hersteld(
224                 bericht.getRestoredFileName(lp.getBestand_datum(), bericht.getVolgordeNummer()));
225 
226             // Als objectRef niet opgehaald kan worden,dan kan het
227             //  bericht niet verwerkt worden.
228             String objectRef = bericht.getObjectRef();
229             if (objectRef != null && !objectRef.isEmpty()) {
230               b.setObject_ref(bericht.getObjectRef());
231               b.setStatus(Bericht.STATUS.STAGING_OK);
232               b.setOpmerking("Klaar voor verwerking.");
233             } else {
234               b.setStatus(Bericht.STATUS.STAGING_NOK);
235               b.setOpmerking(
236                   "Object Ref niet gevonden in bericht-xml, neem contact op met leverancier.");
237             }
238           } catch (XMLStreamException | TransformerException e) {
239             b.setStatus(Bericht.STATUS.STAGING_NOK);
240             StringWriter sw = new StringWriter();
241             e.printStackTrace(new PrintWriter(sw));
242             msg = String.format("Fout bij parsen BRK bericht (bestand %s): %s ", f, sw.toString());
243             b.setOpmerking(msg);
244             log.warn(msg);
245           }
246           try {
247             em.persist(lp);
248             em.persist(b);
249             em.merge(this.config);
250             em.flush();
251             em.getTransaction().commit();
252             em.clear();
253             aantalGeladen++;
254             msg =
255                 String.format(
256                     "  Bestand %s is geladen en heeft status: %s. (Leverde bericht id: %s, met status: %s.)",
257                     f, lp.getStatus(), b.getId(), b.getStatus());
258           } catch (PersistenceException pe) {
259             // insert van bericht of lp is mislukt, waarschijnlijk duplicaat bericht
260             // een laadproces maken met duplicaat status en de xml van het bericht in de
261             // opmerking stoppen
262             if (!em.getTransaction().isActive()) {
263               em.getTransaction().begin();
264             }
265             em.getTransaction().rollback();
266             em.getTransaction().begin();
267             lp.setId(null);
268             log.warn(
269                 "Opslaan van bericht uit laadproces "
270                     + lp.getBestand_naam()
271                     + " is mislukt. Object referentie is: "
272                     + b.getObject_ref(),
273                 pe);
274             log.warn("Duplicaat bericht: " + b.getObject_ref() + ":" + b.getBr_orgineel_xml());
275             lp.setOpmerking(
276                 lp.getOpmerking()
277                     + ": Fout, duplicaat bericht. Berichtinhoud: \n\n"
278                     + b.getBr_xml());
279             b = null;
280             lp.setStatus(LaadProces.STATUS.STAGING_DUPLICAAT);
281             lp.setAutomatischProces(em.find(AutomatischProces.class, this.config.getId()));
282             em.merge(this.config);
283             em.persist(lp);
284             em.flush();
285             em.getTransaction().commit();
286             em.clear();
287           }
288           log.info(msg);
289           this.listener.addLog(msg);
290           sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
291 
292         } catch (IOException io) {
293           // mogelijk bij openen en lezen xml bestand
294           log.error(io.getLocalizedMessage());
295           sb.append(io.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
296           listener.exception(io);
297         }
298       }
299       listener.progress(++progress);
300 
301       if (archiefDirectory != null) {
302         // verplaats naar archief
303         try {
304           FileUtils.copyFileToDirectory(f, archiefDirectory);
305           boolean succes = FileUtils.deleteQuietly(f);
306           if (succes) {
307             msg = String.format("  Bestand %s is naar archief %s verplaatst.", f, archiefDirectory);
308           } else {
309             msg =
310                 String.format(
311                     "  Bestand %s is naar archief %s verplaatst, maar origineel kon niet verwijderd worden.",
312                     f, archiefDirectory);
313           }
314         } catch (IOException e) {
315           msg =
316               String.format(
317                   "  Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).",
318                   f, archiefDirectory, e.getLocalizedMessage());
319           log.error(msg);
320         }
321       }
322       log.info(msg);
323       this.listener.addLog(msg);
324       sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
325     }
326     msg = String.format("Klaar met run op %tc", Calendar.getInstance());
327     log.info(msg);
328     listener.updateStatus(msg);
329     listener.addLog(msg);
330     sb.append(msg);
331 
332     listener.addLog("\n\n**** resultaat ****");
333     listener.addLog("\nAantal bestanden die al waren geladen: " + filterAlVerwerkt);
334     listener.addLog("\nAantal bestanden geladen: " + aantalGeladen + "\n");
335 
336     config.setStatus(WAITING);
337     config.setLogfile(sb.toString());
338     config.setLastrun(new Date());
339     config.updateSamenvattingEnLogfile(
340         "Aantal bestanden die al waren verwerkt: "
341             + filterAlVerwerkt
342             + AutomatischProces.LOG_NEWLINE
343             + "Aantal bestanden geladen: "
344             + aantalGeladen
345             + AutomatischProces.LOG_NEWLINE);
346     em.merge(config);
347   }
