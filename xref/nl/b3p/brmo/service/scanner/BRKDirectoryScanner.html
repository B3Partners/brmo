55      /**
56       * Leest de xml berichten uit de directory in in de database, indien
57       * geconfigureerd worden bestanden gearchiveerd.
58       *
59       * @throws BrmoException als de directory niet lees/blader/schrijfbaar is
60       *
61       */
62      @Override
63      public void execute() throws BrmoException {
64          this.execute(new ProgressUpdateListener() {
65              @Override
66              public void total(long total) {
67              }
68  
69              @Override
70              public void progress(long progress) {
71              }
72  
73              @Override
74              public void exception(Throwable t) {
75                  log.error(t);
76              }
77  
78              @Override
79              public void updateStatus(String status) {
80              }
81  
82              @Override
83              public void addLog(String log) {
84              }
85          });
86      }

88      @Override
89      public void execute(ProgressUpdateListener listener) {
90          this.listener = listener;
91          config.setStatus(PROCESSING);
92          StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
93          String oldLog = config.getLogfile();
94          if (oldLog!=null) {
95              if (oldLog.length() > OLD_LOG_LENGTH) {
96                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
97              } else {
98                  sb.append(oldLog);
99              }
100         }
101 
102         String msg = String.format("De BRK scanner met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
103         log.info(msg);
104         listener.addLog(msg);
105         sb.append(msg);
106 
107         // validatie van de directories, kunnen we lezen/bladeren en evt. schrijven?
108         final File scanDirectory = new File(this.config.getScanDirectory());
109         if (!scanDirectory.isDirectory() || !scanDirectory.canExecute()) {
110             config.setStatus(ERROR);
111             msg = String.format("De scan directory '%s' is geen executable directory", scanDirectory);
112             config.setLogfile(msg);
113             config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
114             this.listener.exception(new BrmoException(msg));
115             return;
116         }
117         // validatie archief directory
118         final String aDir = this.config.getArchiefDirectory();
119         File archiefDirectory = null;
120         if (aDir != null) {
121             archiefDirectory = new File(aDir);
122             archiefDirectory.mkdirs();
123             if (!archiefDirectory.isDirectory() || !archiefDirectory.canWrite()) {
124                 archiefDirectory = null;
125                 config.setStatus(ERROR);
126                 msg = String.format("De archief directory '%s' is geen beschrijfbare directory, er worden geen bestanden gearchiveerd.",
127                         archiefDirectory);
128                 sb.append(msg);
129                 config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
130                 log.error(msg);
131                 this.listener.exception(new BrmoException(msg));
132             }
133             if (!scanDirectory.canWrite()) {
134                 archiefDirectory = null;
135                 config.setStatus(ERROR);
136                 msg = String.format("De scan directory '%s' is geen beschrijfbare directory, er worden geen bestanden gearchiveerd.",
137                         scanDirectory);
138                 sb.append(msg);
139                 config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
140                 log.error(msg);
141                 this.listener.exception(new BrmoException(msg));
142             }
143         }
144         config.setLogfile(sb.toString());
145 
146         File files[] = scanDirectory.listFiles(new FilenameFilter() {
147             @Override
148             public boolean accept(File dir, String name) {
149                 return name.toLowerCase().endsWith(".xml");
150             }
151         });
152         Arrays.sort(files, NameFileComparator.NAME_COMPARATOR);
153 
154         processXMLFiles(files, scanDirectory, archiefDirectory);
155 
156         Stripersist.getEntityManager().flush();
157         Stripersist.getEntityManager().getTransaction().commit();
158         Stripersist.getEntityManager().clear();
159     }

161     /**
162      * verwerk een bestandenlijst.
163      *
164      * @param files array met xml bestanden
165      *
166      * @param scanDirectory
167      * @param archiefDirectory
168      */
169     private void processXMLFiles(File[] files, File scanDirectory, File archiefDirectory) {
170         StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE + config.getLogfile());
171         String msg;
172         int filterAlVerwerkt = 0;
173         int aantalGeladen = 0;
174         int progress = 0;
175         SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
176         int commitPageSize = this.getCommitPageSize();
177 
178         listener.total(files.length);
179         for (File f : files) {
180             if (f.isDirectory()) {
181                 continue;
182             }
183             msg = String.format("Bestand %s is gevonden in %s.", f, scanDirectory);
184             log.info(msg);
185             listener.addLog(msg);
186             sb.append(AutomatischProces.LOG_NEWLINE).append(msg).append(AutomatischProces.LOG_NEWLINE);
187             if (this.isDuplicaatLaadProces(f, BrmoFramework.BR_BRK)) {
188                 msg = String.format("  Bestand %s is een duplicaat en wordt overgeslagen.", f);
189                 listener.addLog(msg);
190                 log.info(msg);
191                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
192                 filterAlVerwerkt++;
193             } else {
194                 try {
195                     LaadProces lp = new LaadProces();
196                     lp.setBestand_naam(getBestandsNaam(f));
197                     lp.setBestand_datum(getBestandsDatum(f));
198                     lp.setSoort(BrmoFramework.BR_BRK);
199                     lp.setStatus(LaadProces.STATUS.STAGING_OK);
200                     lp.setOpmerking(String.format("Bestand geladen van %s op %s", f.getAbsolutePath(), sdf.format(new Date())));
201                     lp.setAutomatischProces(Stripersist.getEntityManager().find(AutomatischProces.class, config.getId()));
202 
203                     Bericht b = new Bericht();
204                     b.setLaadprocesid(lp);
205                     b.setDatum(lp.getBestand_datum());
206                     b.setSoort(BrmoFramework.BR_BRK);
207                     b.setStatus_datum(new Date());
208                     String fileContent = FileUtils.readFileToString(f, "UTF-8");
209                     fileContent = fileContent.replaceAll("\u0000.*", "");
210                     b.setBr_orgineel_xml(fileContent);
211 
212                     try {
213                         BrkSnapshotXMLReader reader = new BrkSnapshotXMLReader(
214                                 new ByteArrayInputStream(b.getBr_orgineel_xml().getBytes("UTF-8")));
215                         BrkBericht bericht = reader.next();
216 
217                         if (bericht.getDatum() != null) {
218                             b.setDatum(bericht.getDatum());
219                         }
220                         b.setBr_xml(bericht.getBrXml());
221                         b.setVolgordenummer(bericht.getVolgordeNummer());
222 
223                         //Als objectRef niet opgehaald kan worden,dan kan het
224                         //  bericht niet verwerkt worden.
225                         String objectRef = bericht.getObjectRef();
226                         if (objectRef != null && !objectRef.isEmpty()) {
227                             b.setObject_ref(bericht.getObjectRef());
228                             b.setStatus(Bericht.STATUS.STAGING_OK);
229                             b.setOpmerking("Klaar voor verwerking.");
230                         } else {
231                             b.setStatus(Bericht.STATUS.STAGING_NOK);
232                             b.setOpmerking("Object Ref niet gevonden in bericht-xml, neem contact op met leverancier.");
233                         }
234                     } catch (Exception e) {
235                         b.setStatus(Bericht.STATUS.STAGING_NOK);
236                         StringWriter sw = new StringWriter();
237                         e.printStackTrace(new PrintWriter(sw));
238                         msg = String.format("Fout bij parsen BRK bericht (bestand %s): %s ", f, sw.toString());
239                         b.setOpmerking(msg);
240                         log.warn(msg);
241                     }
242 
243                     Stripersist.getEntityManager().persist(lp);
244                     Stripersist.getEntityManager().persist(b);
245                     Stripersist.getEntityManager().merge(this.config);
246 
247                     aantalGeladen++;
248                     msg = String.format("  Bestand %s is geladen en heeft status: %s. (Leverde bericht id: %s, met status: %s.)", f, lp.getStatus(), b.getId(), b.getStatus());
249                     log.info(msg);
250                     this.listener.addLog(msg);
251                     sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
252 
253                     if (aantalGeladen % commitPageSize == 0) {
254                         log.info("Tussentijds opslaan van berichten, 'commitPageSize' is bereikt");
255                         Stripersist.getEntityManager().flush();
256                         Stripersist.getEntityManager().getTransaction().commit();
257                         Stripersist.getEntityManager().clear();
258                     }
259 
260                 } catch (IOException io) {
261                     // mogelijk bij openen en lezen xml bestand
262                     log.error(io.getLocalizedMessage());
263                     sb.append(io.getLocalizedMessage()).append(AutomatischProces.LOG_NEWLINE);
264                     listener.exception(io);
265                 }
266             }
267             listener.progress(++progress);
268 
269             if (archiefDirectory != null) {
270                 // verplaats naar archief
271                 try {
272                     FileUtils.copyFileToDirectory(f, archiefDirectory);
273                     boolean succes = FileUtils.deleteQuietly(f);
274                     if (succes) {
275                         msg = String.format("  Bestand %s is naar archief %s verplaatst.", f, archiefDirectory);
276                     } else {
277                         msg = String.format("  Bestand %s is naar archief %s verplaatst, maar origineel kon niet verwijderd worden.", f, archiefDirectory); 
278                     }
279                 } catch (IOException e) {
280                     msg = String.format("  Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).", f, archiefDirectory, e.getLocalizedMessage());
281                     log.error(msg);
282                 }
283             }
284             log.info(msg);
285             this.listener.addLog(msg);
286             sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
287         }
288         msg = String.format("Klaar met run op %tc", Calendar.getInstance());
289         log.info(msg);
290         listener.updateStatus(msg);
291         listener.addLog(msg);
292         sb.append(msg);
293 
294         listener.addLog("\n\n**** resultaat ****");
295         listener.addLog("\nAantal bestanden die al waren geladen: " + filterAlVerwerkt);
296         listener.addLog("\nAantal bestanden geladen: " + aantalGeladen + "\n");
297 
298         config.setStatus(WAITING);
299         config.setLogfile(sb.toString());
300         config.setLastrun(new Date());
301         config.updateSamenvattingEnLogfile("Aantal bestanden die al waren verwerkt: "
302                 + filterAlVerwerkt + AutomatischProces.LOG_NEWLINE
303                 + "Aantal bestanden geladen: " + aantalGeladen + AutomatischProces.LOG_NEWLINE);
304         Stripersist.getEntityManager().merge(config);
305     }
306 
307     private int getCommitPageSize() {
308         int commitPageSize;
309         try {
310             String s = ClobElement.nullSafeGet(config.getConfig().get("commitPageSize"));
311             commitPageSize = Integer.parseInt(s);
312             if (commitPageSize < 1 || commitPageSize > defaultCommitPageSize) {
313                 commitPageSize = defaultCommitPageSize;
314             }
315         } catch (NumberFormatException nfe) {
316             commitPageSize = defaultCommitPageSize;
317         }
318 
319         log.debug("Instellen van commit page size op: " + commitPageSize);
320         return commitPageSize;
321     }
