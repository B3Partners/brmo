142   @Override
143   public void execute(ProgressUpdateListener listener) {
144     this.l = listener;
145 
146     try {
147       l.updateStatus("Initialiseren...");
148       final Date startDate = new Date();
149       l.addLog(
150           String.format("Initialiseren GDS2 ophalen proces %d... %tc", config.getId(), startDate));
151       this.config.updateSamenvattingEnLogfile(
152           String.format("Het GDS2 ophalen proces is gestart op %tc.", startDate));
153       this.config.setStatus(AutomatischProces.ProcessingStatus.PROCESSING);
154       this.config.setLastrun(startDate);
155       Stripersist.getEntityManager().merge(this.config);
156       Stripersist.getEntityManager().flush();
157 
158       BestandenlijstOpvragenRequest request = new BestandenlijstOpvragenRequest();
159       BestandenlijstOpvragenType verzoek = new BestandenlijstOpvragenType();
160       AfgifteSelectieCriteriaType criteria = new AfgifteSelectieCriteriaType();
161       request.setVerzoek(verzoek);
162       verzoek.setAfgifteSelectieCriteria(criteria);
163       criteria.setBestandKenmerken(new BestandKenmerkenType());
164 
165       String contractnummer =
166           ClobElement.nullSafeGet(this.config.getConfig().get("gds2_contractnummer"));
167       if (contractnummer != null) {
168         log.debug("GDS2 contractnummer is: " + contractnummer);
169         criteria.getBestandKenmerken().setContractnummer(contractnummer);
170       }
171       String artikelnummer =
172           ClobElement.nullSafeGet(this.config.getConfig().get("gds2_artikelnummer"));
173       if (artikelnummer != null) {
174         log.debug("GDS2 artikelnummer is: " + artikelnummer);
175         l.addLog("GDS2 artikelnummer is: " + artikelnummer);
176         criteria.getBestandKenmerken().setArtikelnummer(artikelnummer);
177       }
178 
179       String alGerapporteerd =
180           ClobElement.nullSafeGet(this.config.getConfig().get("gds2_al_gerapporteerde_afgiftes"));
181       if (!"true".equals(alGerapporteerd)) {
182         criteria.setNogNietGerapporteerd(true);
183       }
184 
185       // datum tijd parsen/instellen
186       GregorianCalendar vanaf;
187       GregorianCalendar tot =
188           getDatumTijd(ClobElement.nullSafeGet(this.config.getConfig().get("totdatum")));
189       String sVanaf = ClobElement.nullSafeGet(this.config.getConfig().get("vanafdatum"));
190       if (tot != null && ("-1".equals(sVanaf) | "-2".equals(sVanaf) | "-3".equals(sVanaf))) {
191         vanaf =
192             getDatumTijd(
193                 ClobElement.nullSafeGet(this.config.getConfig().get("totdatum")),
194                 Integer.parseInt(sVanaf));
195       } else {
196         vanaf = getDatumTijd(sVanaf);
197       }
198 
199       List<AfgifteType> afgiftes = new ArrayList<>();
200       BestandenlijstOpvragenResponse response;
201       int hoogsteKlantAfgifteNummer = -1;
202 
203       Gds2AfgifteServiceV20170401 gds2 = initGDS2();
204       l.updateStatus("Uitvoeren SOAP request naar Kadaster...");
205 
206       boolean afgifteNummerGebaseerdOphalen = false;
207       if (vanaf == null && tot == null) {
208         //  instellen laagste en hoogste afgiftenummer voor dit verzoek, mag alleen als er
209         // geen periode is gedefinieerd
210         KlantAfgiftenummerReeksType afgiftenummers = new KlantAfgiftenummerReeksType();
211         String nr =
212             ClobElement.nullSafeGet(this.config.getConfig().get("klantafgiftenummer_vanaf"));
213         if (StringUtils.isNotBlank(nr)) {
214           BigInteger afgifteNrVanaf = new BigInteger(nr);
215           afgiftenummers.setKlantAfgiftenummerVanaf(afgifteNrVanaf);
216           log.info("Ophalen vanaf klant afgifte nummer: " + afgifteNrVanaf);
217           l.addLog("Ophalen vanaf klant afgifte nummer: " + afgifteNrVanaf);
218 
219           nr = ClobElement.nullSafeGet(this.config.getConfig().get("klantafgiftenummer_totenmet"));
220           if (StringUtils.isNotBlank(nr)) {
221             BigInteger afgifteNrTM = new BigInteger(nr);
222             afgiftenummers.setKlantAfgiftenummerTotmet(afgifteNrTM);
223             log.info("Ophalen tot en met klant afgifte nummer: " + afgifteNrTM);
224             l.addLog("Ophalen tot en met klant afgifte nummer: " + afgifteNrTM);
225           }
226           criteria.setKlantAfgiftenummerReeks(afgiftenummers);
227           afgifteNummerGebaseerdOphalen = true;
228         }
229       }
230 
231       if (afgifteNummerGebaseerdOphalen) {
232         // <editor-fold> afgiftenummer-based ophalen
233         l.addLog(
234             "GDS2 verzoek voor afgiftenummers vanaf: "
235                 + criteria.getKlantAfgiftenummerReeks().getKlantAfgiftenummerVanaf());
236         l.addLog(
237             "GDS2 verzoek voor afgiftenummers tot-en-met: "
238                 + criteria.getKlantAfgiftenummerReeks().getKlantAfgiftenummerTotmet());
239         response =
240             retryBestandenLijstOpvragen(
241                 gds2, request, BESTANDENLIJST_ATTEMPTS, BESTANDENLIJST_RETRY_WAIT);
242         hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
243         boolean hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
244         int aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
245         if (aantalInAntwoord > 0) {
246           afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
247         }
248         l.addLog("Aantal in antwoord: " + aantalInAntwoord);
249         l.addLog("Meer afgiftes beschikbaar: " + hasMore);
250 
251         while (hasMore) {
252           criteria
253               .getKlantAfgiftenummerReeks()
254               .setKlantAfgiftenummerVanaf(
255                   // vanaf ophogen met aantal opgehaald
256                   criteria
257                       .getKlantAfgiftenummerReeks()
258                       .getKlantAfgiftenummerVanaf()
259                       .add(BigInteger.valueOf(aantalInAntwoord)));
260           l.addLog(
261               "GDS2 verzoek voor afgiftenummers vanaf: "
262                   + criteria.getKlantAfgiftenummerReeks().getKlantAfgiftenummerVanaf());
263           l.addLog(
264               "GDS2 verzoek voor afgiftenummers tot-en-met: "
265                   + criteria.getKlantAfgiftenummerReeks().getKlantAfgiftenummerTotmet());
266           response =
267               retryBestandenLijstOpvragen(
268                   gds2, request, BESTANDENLIJST_ATTEMPTS, BESTANDENLIJST_RETRY_WAIT);
269           hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
270           aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
271           hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
272           l.addLog("Aantal in antwoord: " + aantalInAntwoord);
273           l.addLog("Meer afgiftes beschikbaar: " + hasMore);
274 
275           if (aantalInAntwoord > 0) {
276             afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
277           }
278         }
279         // </editor-fold> afgiftenummer-based ophalen
280       } else {
281         // <editor-fold> time-based ophalen
282         GregorianCalendar currentMoment = null;
283         boolean parseblePeriod = false;
284         int loopType = Calendar.DAY_OF_MONTH;
285         int loopMax = 180;
286         int loopNum = 0;
287         boolean reducePeriod = false;
288         boolean increasePeriod = false;
289 
290         if (vanaf != null && tot != null && vanaf.before(tot)) {
291           parseblePeriod = true;
292           currentMoment = vanaf;
293         }
294 
295         boolean morePeriods2Process = false;
296         do /* while morePeriods2Process is true */ {
297           l.addLog("\n*** start periode ***");
298           // zet periode in criteria indien gewenst
299           if (parseblePeriod) {
300             // check of de periodeduur verkleind moet worden
301             if (reducePeriod) {
302               switch (loopType) {
303                 case Calendar.DAY_OF_MONTH:
304                   currentMoment.add(loopType, -1);
305                   loopType = Calendar.HOUR_OF_DAY;
306                   l.addLog("* Verklein loop periode naar uur");
307                   break;
308                 case Calendar.HOUR_OF_DAY:
309                   currentMoment.add(loopType, -1);
310                   loopType = Calendar.MINUTE;
311                   l.addLog("* Verklein loop periode naar minuut");
312                   break;
313                 case Calendar.MINUTE:
314                 default:
315                   /*
316                    * Hier kom je alleen als binnen een minuut meer
317                    * dan 2000 berichten zijn aangamaakt en het
318                    * vinkje ook "al rapporteerde berichten
319                    * ophalen" staat aan.
320                    */
321                   l.addLog("Niet alle gevraagde berichten zijn opgehaald");
322               }
323               reducePeriod = false;
324             }
325 
326             // check of de periodeduur vergroot moet worden
327             if (increasePeriod) {
328               switch (loopType) {
329                 case Calendar.HOUR_OF_DAY:
330                   loopType = Calendar.DAY_OF_MONTH;
331                   l.addLog("* Vergroot loop periode naar dag");
332                   break;
333                 case Calendar.MINUTE:
334                   loopType = Calendar.HOUR_OF_DAY;
335                   l.addLog("* Vergroot loop periode naar uur");
336                   break;
337                 case Calendar.DAY_OF_MONTH:
338                 default:
339                   // not possible
340               }
341               increasePeriod = false;
342             }
343 
344             FilterDatumTijdType d = new FilterDatumTijdType();
345             d.setDatumTijdVanaf(getXMLDatumTijd(currentMoment));
346             l.addLog(String.format("Datum vanaf: %tc", currentMoment.getTime()));
347             currentMoment.add(loopType, 1);
348             d.setDatumTijdTotmet(getXMLDatumTijd(currentMoment));
349             l.addLog(String.format("Datum tot: %tc", currentMoment.getTime()));
350             criteria.setPeriode(d);
351 
352             switch (loopType) {
353               case Calendar.HOUR_OF_DAY:
354                 // 0-23
355                 if (currentMoment.get(loopType) == 0) {
356                   increasePeriod = true;
357                 }
358                 break;
359               case Calendar.MINUTE:
360                 // 0-59
361                 if (currentMoment.get(loopType) == 0) {
362                   increasePeriod = true;
363                 }
364                 break;
365               case Calendar.DAY_OF_MONTH:
366               default:
367                 // alleen dagen tellen, uur en minuut altijd helemaal
368                 loopNum++;
369             }
370 
371             // bereken of einde van periode bereikt is
372             morePeriods2Process = currentMoment.before(tot) && loopNum < loopMax;
373           }
374 
375           verzoek.setAfgifteSelectieCriteria(criteria);
376           response =
377               retryBestandenLijstOpvragen(
378                   gds2, request, BESTANDENLIJST_ATTEMPTS, BESTANDENLIJST_RETRY_WAIT);
379           hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
380 
381           int aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
382           l.addLog("Aantal in antwoord: " + aantalInAntwoord);
383           // opletten; in de xsd staat een default value van 'J' voor
384           // meerAfgiftesbeschikbaar
385           boolean hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
386           l.addLog("Meer afgiftes beschikbaar: " + hasMore);
387 
388           /*
389            * Als "al gerapporteerde berichten" moeten worden opgehaald en
390            * er zitten dan 2000 berichten in het antwoord dan heeft het
391            * geen zin om meer keer de berichten op te halen, je krijgt
392            * telkens dezelfde.
393            */
394           if (hasMore && "true".equals(alGerapporteerd)) {
395             reducePeriod = true;
396             morePeriods2Process = true;
397             increasePeriod = false;
398             // als er geen parsable periode is
399             // (geen periode ingevuld en alGerapporteerd is true
400             // dan moet morePeriods2Process false worden om een
401             // eindloze while loop te voorkomen
402             if (!parseblePeriod) {
403               morePeriods2Process = false;
404             } else {
405               continue;
406             }
407           }
408 
409           if (aantalInAntwoord > 0) {
410             afgiftes.addAll(response.getAntwoord().getBestandenLijst().getAfgifte());
411           }
412 
413           /*
414            * Indicatie nog niet gerapporteerd: Met deze indicatie wordt
415            * aangegeven of uitsluitend de nog niet gerapporteerde
416            * bestanden moeten worden opgenomen in de lijst, of dat alle
417            * beschikbare bestanden worden genoemd. Niet gerapporteerd
418            * betekent in dit geval ‘niet eerder opgevraagd in deze
419            * bestandenlijst’. Als deze indicator wordt gebruikt, dan
420            * worden na terugmelding van de bestandenlijst de bijbehorende
421            * bestanden gemarkeerd als zijnde ‘gerapporteerd’ in het
422            * systeem van GDS.
423            */
424           int moreCount = 0;
425           String dontGetMoreConfig =
426               ClobElement.nullSafeGet(
427                   this.config.getConfig().get("gds2_niet_gerapporteerde_afgiftes_niet_ophalen"));
428 
429           while (hasMore && !"true".equals(dontGetMoreConfig)) {
430             l.updateStatus(
431                 "Uitvoeren SOAP request naar Kadaster voor meer afgiftes..." + moreCount++);
432             criteria.setNogNietGerapporteerd(true);
433             response =
434                 retryBestandenLijstOpvragen(
435                     gds2, request, BESTANDENLIJST_ATTEMPTS, BESTANDENLIJST_RETRY_WAIT);
436             hoogsteKlantAfgifteNummer = response.getAntwoord().getKlantAfgiftenummerMax();
437 
438             List<AfgifteType> afgifteLijst =
439                 response.getAntwoord().getBestandenLijst().getAfgifte();
440             for (AfgifteType t : afgiftes) {
441               // lijst urls naar aparte logfile loggen
442               if (t.getDigikoppelingExternalDatareferences() != null
443                   && t.getDigikoppelingExternalDatareferences().getDataReference() != null) {
444                 for (DataReference dr :
445                     t.getDigikoppelingExternalDatareferences().getDataReference()) {
446                   log.info(
447                       "GDS2url te downloaden: "
448                           + dr.getTransport().getLocation().getSenderUrl().getValue());
449                   break;
450                 }
451               }
452             }
453             afgiftes.addAll(afgifteLijst);
454             aantalInAntwoord = response.getAntwoord().getAfgifteAantalInLijst();
455             l.addLog("Aantal in antwoord: " + aantalInAntwoord);
456             hasMore = response.getAntwoord().isMeerAfgiftesbeschikbaar();
457             l.addLog("Nog meer afgiftes beschikbaar: " + hasMore);
458           }
459 
460         } while (morePeriods2Process);
461         // </editor-fold> time-based ophalen
462       }
463       l.total(afgiftes.size());
464       l.addLog("Totaal aantal op te halen berichten: " + afgiftes.size());
465       l.addLog("Hoogste klant afgifte nummer: " + hoogsteKlantAfgifteNummer);
466 
467       this.config
468           .getConfig()
469           .put("hoogste_afgiftenummer", new ClobElement("" + hoogsteKlantAfgifteNummer));
470 
471       final String soort =
472           this.config.getConfig().getOrDefault("gds2_br_soort", new ClobElement("brk")).getValue();
473       switch (soort) {
474         case "brk2":
475           verwerkAfgiftes(afgiftes, getCertificaatBaseURL(response.getAntwoord()), soort);
476           break;
477         case "brk":
478           verwerkAfgiftes(afgiftes, getCertificaatBaseURL(response.getAntwoord()), soort);
479           break;
480         case "bag":
481           // bag heeft geen contractnummer
482           verwerkAfgiftes(afgiftes, getAnoniemBaseURL(response.getAntwoord()), soort);
483           break;
484         default:
485           throw new BrmoException("Onbekende basisregistratie soort: " + soort);
486       }
487 
488     } catch (Exception e) {
489       log.error("Fout bij ophalen van GDS2 berichten", e);
490       this.config.setLastrun(new Date());
491       this.config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
492       this.config.setStatus(AutomatischProces.ProcessingStatus.ERROR);
493       String m = "Fout bij ophalen van GDS2 berichten: " + ExceptionUtils.getMessage(e);
494       if (e.getCause() != null) {
495         m += ", oorzaak: " + ExceptionUtils.getRootCauseMessage(e);
496       }
497       l.addLog(m);
498       l.exception(e);
499     } finally {
500       if (Stripersist.getEntityManager().getTransaction().getRollbackOnly()) {
501         // XXX bij rollback only wordt status niet naar ERROR gezet vanwege
502         // rollback, zou in aparte transactie moeten
503         Stripersist.getEntityManager().getTransaction().rollback();
504       } else {
505         Stripersist.getEntityManager().merge(this.config);
506         Stripersist.getEntityManager().getTransaction().commit();
507       }
508       l.updateStatus("Uitvoeren SOAP request naar Kadaster afgerond");
509     }
510   }

541     l.updateStatus(msg);
542     l.addLog(msg);
543 
544     InputStream input;
545     int attempt = 0;
546     while (true) {
547       try {
548         URLConnection connection = new URL(url).openConnection();

585     }

627               l.addLog(msg);
628               log.info(msg);
629             } finally {
630               brmo.updateLaadProcesMeta(
631                   brmo.getLaadProcesIdByFileName(localLpName),
632                   a.getKlantAfgiftenummer(),

634                   a.getBestandKenmerken().getArtikelnummer(),
635                   a.getBestandKenmerken().getContractnummer(),

641             innerentry = innerzip.getNextEntry();
642           }
643         } else {
644           log.warn("Overslaan van onbekend bestand in bag mutaties: " + entry.getName());
645         }
646       }
647       entry = zip.getNextEntry();
648     }
649     zip.close();
650     brmo.closeBrmoFramework();

653   private Bericht laadAfgifte(AfgifteType a, String url, String brkSoort) throws Exception {
654     EntityManager em = Stripersist.getEntityManager();
655     String msg = "Downloaden " + url;
656     l.updateStatus(msg);
657     l.addLog(msg);
658 
659     ByteArrayOutputStream bos = new ByteArrayOutputStream();
660 
661     int attempt = 0;
662     while (true) {
663       try {
664         URLConnection connection = new URL(url).openConnection();
665         if (connection instanceof HttpsURLConnection) {
666           ((HttpsURLConnection) connection).setSSLSocketFactory(context.getSocketFactory());
667         }
668         InputStream input = (InputStream) connection.getContent();
669         IOUtils.copy(input, bos);
670         break;
671       } catch (Exception e) {
672         attempt++;
673         if (attempt == AFGIFTE_DOWNLOAD_ATTEMPTS) {
674           l.addLog(
675               "Fout bij laatste poging downloaden afgifte: "
676                   + e.getClass().getName()
677                   + ": "
678                   + e.getMessage());
679           throw e;
680         } else {
681           l.addLog(
682               "Fout bij poging "
683                   + attempt
684                   + " om afgifte te downloaden: "
685                   + e.getClass().getName()
686                   + ": "
687                   + e.getMessage());
688           Thread.sleep(AFGIFTE_DOWNLOAD_RETRY_WAIT);
689           l.addLog(
690               "Uitvoeren poging " + (attempt + 1) + " om afgifte " + url + " te downloaden...");
691         }
692       }
693     }
694 
695     LaadProces lp = new LaadProces();
696     lp.setBestand_naam(getLaadprocesBestandsnaam(a));
697     lp.setKlantafgiftenummer(a.getKlantAfgiftenummer());
698     lp.setContractafgiftenummer(a.getContractAfgiftenummer());
699     lp.setArtikelnummer(a.getBestandKenmerken().getArtikelnummer());
700     lp.setContractnummer(a.getBestandKenmerken().getContractnummer());
701     lp.setAfgifteid(a.getAfgifteID());
702     lp.setAfgiftereferentie(a.getAfgiftereferentie());
703     lp.setBestandsreferentie(a.getBestand().getBestandsreferentie());
704     lp.setBeschikbaar_tot(a.getBeschikbaarTot().toGregorianCalendar().getTime());
705     lp.setBestand_naam_hersteld(a.getBestand().getBestandsnaam());
706 
707     if (a.getDigikoppelingExternalDatareferences() != null
708         && a.getDigikoppelingExternalDatareferences().getDataReference() != null) {
709       for (DataReference dr : a.getDigikoppelingExternalDatareferences().getDataReference()) {
710         lp.setBestand_datum(
711             dr.getLifetime().getCreationTime().getValue().toGregorianCalendar().getTime());
712         break;
713       }
714     }
715     lp.setSoort(brkSoort);
716     lp.setStatus(LaadProces.STATUS.STAGING_OK);
717     lp.setStatus_datum(new Date());
718     SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
719     lp.setOpmerking("GDS2 download van " + url + " op " + sdf.format(new Date()));
720     lp.setAutomatischProces(em.find(AutomatischProces.class, config.getId()));
721 
722     Bericht b = new Bericht();
723     b.setLaadprocesid(lp);
724     b.setDatum(lp.getBestand_datum());
725     b.setSoort(brkSoort);
726     b.setStatus_datum(new Date());
727 
728     ZipInputStream zip = new ZipInputStream(new ByteArrayInputStream(bos.toByteArray()));
729     ZipEntry entry = zip.getNextEntry();
730     while (entry != null && !entry.getName().toLowerCase().endsWith(".xml")) {
731       msg = "Overslaan zip entry geen XML: " + entry.getName();
732       l.addLog(msg);
733       entry = zip.getNextEntry();
734     }
735     if (entry == null) {
736       msg = "Geen geschikt XML bestand gevonden in zip bestand!";
737       l.addLog(msg);
738       return null;
739     }
740     b.setBr_orgineel_xml(IOUtils.toString(zip, StandardCharsets.UTF_8));
741 
742     if (log.isDebugEnabled()) {
743       // dump xml bestand naar /tmp/ voordat de xml wordt geparsed
744       String fName = File.createTempFile(lp.getBestand_naam(), ".xml").getAbsolutePath();
745       log.debug("Dump xml bericht naar: " + fName);
746       IOUtils.write(b.getBr_orgineel_xml(), new FileWriter(fName, StandardCharsets.UTF_8));
747     }
748     try {
749       String brXML;
750       Date brDatum;
751       Integer brVolgordeNummer;
752       String brObjectRef;
753       if (BrmoFramework.BR_BRK.equals(brkSoort)) {
754         BrkSnapshotXMLReader reader =
755             new BrkSnapshotXMLReader(
756                 new ByteArrayInputStream(b.getBr_orgineel_xml().getBytes(StandardCharsets.UTF_8)));
757         BrkBericht parsedBericht = reader.next();
758         brXML = parsedBericht.getBrXml();
759         brDatum = parsedBericht.getDatum();
760         brVolgordeNummer = parsedBericht.getVolgordeNummer();
761         brObjectRef = parsedBericht.getObjectRef();
762       } else {
763         Brk2SnapshotXMLReader reader2 =
764             new Brk2SnapshotXMLReader(
765                 new ByteArrayInputStream(b.getBr_orgineel_xml().getBytes(StandardCharsets.UTF_8)));
766         Brk2Bericht parsedBericht = reader2.next();
767         brXML = parsedBericht.getBrXml();
768         brDatum = parsedBericht.getDatum();
769         brVolgordeNummer = parsedBericht.getVolgordeNummer();
770         brObjectRef = parsedBericht.getObjectRef();
771       }
772 
773       if (null != brDatum) {
774         b.setDatum(brDatum);
775       }
776       b.setBr_xml(brXML);
777       b.setVolgordenummer(brVolgordeNummer);
778 
779       // Als objectRef niet opgehaald kan worden,dan kan het
780       // bericht niet verwerkt worden.
781       if (null != brObjectRef && !brObjectRef.isBlank()) {
782         b.setObject_ref(brObjectRef);
783         b.setStatus(Bericht.STATUS.STAGING_OK);
784         b.setOpmerking("Klaar voor verwerking.");
785       } else {
786         b.setStatus(Bericht.STATUS.STAGING_NOK);
787         b.setOpmerking("Object Ref niet gevonden in bericht-xml, neem contact op met leverancier.");
788       }
789     } catch (Exception e) {
790       b.setStatus(Bericht.STATUS.STAGING_NOK);
791       StringWriter sw = new StringWriter();
792       e.printStackTrace(new PrintWriter(sw));
793       b.setOpmerking("Fout bij parsen BRK bericht: " + sw);
794     }
795     try {
796       em.persist(lp);
797       em.persist(b);
798       em.merge(this.config);
799       em.flush();
800       em.getTransaction().commit();
801       em.clear();
802     } catch (PersistenceException pe) {
803       if (!em.getTransaction().isActive()) {
804         em.getTransaction().begin();
805       }
806       em.getTransaction().rollback();
807       em.getTransaction().begin();
808       lp.setId(null);
809       log.warn("Opslaan van bericht uit laadproces " + lp.getBestand_naam() + " is mislukt.", pe);
810       log.warn(
811           "Duplicaat bericht: "
812               + b.getObject_ref()
813               + ":"
814               + b.getBr_orgineel_xml()
815               + "("
816               + b.getBr_xml()
817               + ")");
818       lp.setOpmerking(
819           lp.getOpmerking() + ": Fout, duplicaat bericht. Berichtinhoud: \n\n" + b.getBr_xml());
820       b = null;
821       lp.setStatus(LaadProces.STATUS.STAGING_DUPLICAAT);
822       lp.setAutomatischProces(em.find(AutomatischProces.class, this.config.getId()));
823       em.merge(this.config);
824       em.persist(lp);
825       em.flush();
826       em.getTransaction().commit();
827       em.clear();
828     }
829     return b;
830   }

832   /**
833    * Post de xml naar de geconfigureerde url.
834    *
835    * @param proces proces waarvoor doorgetuurd wordt
836    * @param l update listener
837    * @param b door te sturen bericht
838    * @param endpoint url waarnaartoe wordt gepost
839    * @return {@code true} als succesvol doorgestuurd
840    */
841   public static boolean doorsturenBericht(
842       AutomatischProces proces, ProgressUpdateListener l, Bericht b, String endpoint) {
843     String msg;
844     SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
845     try {
846       if (endpoint == null || endpoint.length() < 1) {
847         return false;
848       }
849       URL url = new URL(endpoint);
850       HttpURLConnection conn = (HttpURLConnection) url.openConnection();
851       conn.setDoOutput(true);
852       conn.setRequestMethod("POST");
853       conn.setRequestProperty("Content-Length", "" + b.getBr_orgineel_xml().length());
854       conn.setRequestProperty("Content-Type", "application/octet-stream");
855       IOUtils.copy(
856           IOUtils.toInputStream(b.getBr_orgineel_xml(), StandardCharsets.UTF_8),
857           conn.getOutputStream());
858       conn.disconnect();
859       if ((conn.getResponseCode() == HttpURLConnection.HTTP_OK)
860           || (conn.getResponseCode() == HttpURLConnection.HTTP_CREATED)
861           || (conn.getResponseCode() == HttpURLConnection.HTTP_ACCEPTED)) {
862         msg =
863             "Bericht "
864                 + b.getObject_ref()
865                 + " op "
866                 + sdf.format(new Date())
867                 + " doorgestuurd naar "
868                 + endpoint;
869         b.setStatus(Bericht.STATUS.STAGING_FORWARDED);
870       } else {
871         msg =
872             String.format(
873                 "HTTP foutcode bij doorsturen bericht %s op %s naar endpoint %s: %s",
874                 b.getObject_ref(),
875                 sdf.format(new Date()),
876                 endpoint,
877                 conn.getResponseCode() + ": " + conn.getResponseMessage());
878         b.setStatus(Bericht.STATUS.STAGING_NOK);
879       }
880       b.setOpmerking(msg);
881       l.addLog(msg);
882 
883       return b.getStatus() == Bericht.STATUS.STAGING_FORWARDED;
884     } catch (Exception e) {
885       msg =
886           String.format(
887               "Fout bij doorsturen bericht op %s naar endpoint %s: %s",
888               sdf.format(new Date()), endpoint, e.getClass() + ": " + e.getMessage());
889       b.setOpmerking(msg);
890       b.setStatus(Bericht.STATUS.STAGING_NOK);
891       l.addLog(msg);
892       log.error(msg, e);
893       return false;
894     }
895   }

897   /**
898    * verwerk de afgiftes in de response en stuur eventueel door.
899    *
900    * @param afgiftes lijst afgiftes
901    * @throws Exception if any
902    */
903   private void verwerkAfgiftes(List<AfgifteType> afgiftes, BaseURLType baseUrl, final String soort)
904       throws Exception {
905     int filterAlVerwerkt = 0;
906     int aantalGeladen = 0;
907     int aantalDoorgestuurd = 0;
908     int progress = 0;
909     List<Long> geladenBerichtIds = new ArrayList<>();
910     String doorsturenUrl =
911         ClobElement.nullSafeGet(this.config.getConfig().get("delivery_endpoint"));
912 
913     for (AfgifteType a : afgiftes) {
914       String url = getAfgifteURL(a, baseUrl);
915 
916       if (null != url && !url.isBlank()) {
917         if (isAfgifteAlGeladen(a)) {
918           filterAlVerwerkt++;
919         } else if (soort.equalsIgnoreCase(BrmoFramework.BR_BAG)) {
920           laadBagAfgifte(a, url);
921           aantalGeladen++;
922         } else {
923           Bericht b = laadAfgifte(a, url, soort);
924           if (b != null && doorsturenUrl != null) {
925             geladenBerichtIds.add(b.getId());
926           }
927           aantalGeladen++;
928         }
929       }
930       l.progress(++progress);
931     }
932 
933     Bericht b;
934     if (doorsturenUrl != null && !geladenBerichtIds.isEmpty()) {
935       l.addLog("Doorsturen van de opgehaalde berichten, in totaal " + geladenBerichtIds.size());
936       l.total(geladenBerichtIds.size());
937       progress = 0;
938 
939       for (Long bId : geladenBerichtIds) {
940         b = Stripersist.getEntityManager().find(Bericht.class, bId);
941         l.updateStatus("Doorsturen bericht: " + b.getObject_ref());
942         doorsturenBericht(this.config, l, b, doorsturenUrl);
943         Stripersist.getEntityManager().merge(b);
944         Stripersist.getEntityManager().merge(this.config);
945         Stripersist.getEntityManager().flush();
946         Stripersist.getEntityManager().getTransaction().commit();
947         Stripersist.getEntityManager().clear();
948         aantalDoorgestuurd++;
949         l.progress(++progress);
950       }
951     }
952 
953     l.addLog("\n\n**** resultaat ****\n");
954     l.addLog("Aantal afgiftes die al waren verwerkt: " + filterAlVerwerkt);
955     l.addLog("\nAantal afgiftes geladen: " + aantalGeladen);
956     l.addLog("\nAantal afgiftes doorgestuurd: " + aantalDoorgestuurd + "\n");
957 
958     l.addLog(
959         String.format(
960             "Het GDS2 ophalen proces met ID %d is afgerond op %tc",
961             config.getId(), Calendar.getInstance()));
962     this.config.updateSamenvattingEnLogfile(
963         String.format(
964             "Het GDS2 ophalen proces, gestart op %tc, is afgerond op %tc. "
965                 + LOG_NEWLINE
966                 + "Aantal afgiftes die al waren verwerkt: %d"
967                 + LOG_NEWLINE
968                 + "Aantal afgiftes geladen: %d"
969                 + LOG_NEWLINE
970                 + "Aantal afgiftes doorgestuurd: %d"
971                 + LOG_NEWLINE
972                 + "Hoogste klantafgiftenummer: %s",
973             this.config.getLastrun(),
974             Calendar.getInstance(),
975             filterAlVerwerkt,
976             aantalGeladen,
977             aantalDoorgestuurd,
978             ClobElement.nullSafeGet(this.config.getConfig().get("hoogste_afgiftenummer"))));
979 
980     this.config.setStatus(AutomatischProces.ProcessingStatus.WAITING);
981     this.config.setLastrun(new Date());
982     Stripersist.getEntityManager().merge(this.config);
983     // om geheugen problemen te vookomen bij runs met veel downloads en berichten
984     // een flush/commit/clear forceren
985     Stripersist.getEntityManager().flush();
986     Stripersist.getEntityManager().getTransaction().commit();
987     Stripersist.getEntityManager().clear();
988   }

990   private Gds2AfgifteServiceV20170401 initGDS2() throws Exception {
991     Gds2AfgifteServiceV20170401 gds2 =
992         new Gds2AfgifteServiceV20170401Service().getAGds2AfgifteServiceV20170401();
993     BindingProvider bp = (BindingProvider) gds2;
994     Map<String, Object> ctxt = bp.getRequestContext();
995 
996     // soap berichten logger inhaken (actief met TRACE level)
997     List<Handler> handlerChain = bp.getBinding().getHandlerChain();
998     handlerChain.add(new LogMessageHandler());
999     bp.getBinding().setHandlerChain(handlerChain);
1000 
1001     String endpoint = (String) ctxt.get(BindingProvider.ENDPOINT_ADDRESS_PROPERTY);
1002     l.addLog("Kadaster endpoint: " + endpoint);
1003     l.updateStatus("Laden keys...");
1004 
1005     l.addLog("Initializing KeyManagerFactory");
1006     KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
1007     KeyStore ks = KeyStore.getInstance("jks");
1008     final char[] thePassword = "changeit".toCharArray();
1009     PrivateKey privateKey =
1010         getPrivateKeyFromPEM(this.config.getConfig().get("gds2_privkey").getValue());
1011     Certificate certificate =
1012         getCertificateFromPEM(this.config.getConfig().get("gds2_pubkey").getValue());
1013     ks.load(null);
1014     ks.setKeyEntry("thekey", privateKey, thePassword, new Certificate[] {certificate});
1015     kmf.init(ks, thePassword);
1016 
1017     l.updateStatus("Opzetten SSL context...");
1018     context = createSslContext(kmf);
1019 
1020     Client client = ClientProxy.getClient(gds2);
1021     HTTPConduit http = (HTTPConduit) client.getConduit();
1022     TLSClientParameters tlsClientParameters = new TLSClientParameters();
1023     tlsClientParameters.setSSLSocketFactory(context.getSocketFactory());
1024     http.setTlsClientParameters(tlsClientParameters);
1025 
1026     return gds2;
1027   }

1029   private SSLContext createSslContext(KeyManagerFactory kmf)
1030       throws KeyStoreException,
1031           IOException,
1032           CertificateException,
1033           NoSuchProviderException,
1034           NoSuchAlgorithmException,
1035           KeyManagementException {
1036     final SSLContext sslContext = SSLContext.getInstance("TLS", "SunJSSE");
1037 
1038     l.addLog("Loading PKIX Overheid keystore");
1039     KeyStore ks = KeyStore.getInstance("jks");
1040     ks.load(
1041         nl.b3p.gds2.GDS2Util.class.getResourceAsStream("/pkioverheid.jks"),
1042         "changeit".toCharArray());
1043 
1044     l.addLog("Initializing default TrustManagerFactory");
1045     final TrustManagerFactory javaDefaultTrustManager =
1046         TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
1047     javaDefaultTrustManager.init((KeyStore) null);
1048     X509TrustManager defaultX509TrustManager = null;
1049     for (TrustManager t : javaDefaultTrustManager.getTrustManagers()) {
1050       if (t instanceof X509TrustManager) {
1051         defaultX509TrustManager = (X509TrustManager) t;
1052         break;
1053       }
1054     }
1055     l.addLog("Initializing PKIX TrustManagerFactory");
1056     final TrustManagerFactory customCaTrustManager = TrustManagerFactory.getInstance("PKIX");
1057     customCaTrustManager.init(ks);
1058 
1059     l.addLog("Initializing SSLContext");
1060     sslContext.init(
1061         kmf.getKeyManagers(),
1062         new TrustManager[] {
1063           new TrustManagerDelegate(
1064               // customCaTrustManager is PKIX dus altijd X.509 (RFC3280)
1065               (X509TrustManager) customCaTrustManager.getTrustManagers()[0],
1066               defaultX509TrustManager)
1067         },
1068         null);
1069     return sslContext;
1070   }
