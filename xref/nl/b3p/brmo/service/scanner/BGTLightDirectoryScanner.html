71      @Override
72      public void execute(ProgressUpdateListener listener) {
73          this.listener = listener;
74          config.setStatus(PROCESSING);
75          StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE);
76          String oldLog = config.getLogfile();
77          if (oldLog != null) {
78              if (oldLog.length() > OLD_LOG_LENGTH) {
79                  sb.append(oldLog.substring(oldLog.length() - OLD_LOG_LENGTH / 10));
80              } else {
81                  sb.append(oldLog);
82              }
83          }
84  
85          String msg = String.format("De BGT Light scanner met ID %d is gestart op %tc.", config.getId(), Calendar.getInstance());
86          LOG.info(msg);
87          listener.addLog(msg);
88          sb.append(msg);
89  
90          // validatie van de directories, kunnen we lezen/bladeren en evt. schrijven?
91          final File scanDirectory = new File(this.config.getScanDirectory());
92          if (!scanDirectory.isDirectory() || !scanDirectory.canExecute()) {
93              config.setStatus(ERROR);
94              msg = String.format("De scan directory '%s' is geen executable directory", scanDirectory);
95              config.setLogfile(msg);
96              config.setSamenvatting("Er is een fout opgetreden, details staan in de logs");
97              this.listener.exception(new BrmoException(msg));
98              return;
99          }
100 
101         config.setLogfile(sb.toString());
102 
103         File files[] = scanDirectory.listFiles(new FilenameFilter() {
104             @Override
105             public boolean accept(File dir, String name) {
106                 return name.toLowerCase().endsWith(".zip");
107             }
108         });
109         Arrays.sort(files, NameFileComparator.NAME_COMPARATOR);
110 
111         processZipFiles(files, scanDirectory);
112 
113         Stripersist.getEntityManager().flush();
114         Stripersist.getEntityManager().getTransaction().commit();
115         Stripersist.getEntityManager().clear();
116     }
117 
118     /**
119      * verwerk een bestandenlijst.
120      *
121      * @param files array met xml bestanden
122      *
123      * @param scanDirectory
124      * @param archiefDirectory
125      */
126     private void processZipFiles(File[] files, File scanDirectory) {
127         StringBuilder sb = new StringBuilder(AutomatischProces.LOG_NEWLINE + config.getLogfile());
128         String msg;
129         int filterAlVerwerkt = 0;
130         int aantalGeladen = 0;
131         int progress = 0;
132         SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
133         int commitPageSize = this.getCommitPageSize();
134 
135         listener.total(files.length);
136         for (File f : files) {
137             if (f.isDirectory()) {
138                 continue;
139             }
140             msg = String.format("Bestand %s is gevonden in %s.", f, scanDirectory);
141             LOG.info(msg);
142             listener.addLog(msg);
143             sb.append(AutomatischProces.LOG_NEWLINE).append(msg).append(AutomatischProces.LOG_NEWLINE);
144             if (this.isDuplicaatLaadProces(f, BrmoFramework.BR_BGTLIGHT)) {
145                 msg = String.format("  Bestand %s is een duplicaat en wordt overgeslagen.", f);
146                 listener.addLog(msg);
147                 LOG.info(msg);
148                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
149                 filterAlVerwerkt++;
150             } else {
151                 LaadProcestaging/LaadProces.html#LaadProces">LaadProces lp = new LaadProces();
152                 lp.setBestand_naam(getBestandsNaam(f));
153                 lp.setBestand_datum(getBestandsDatum(f));
154                 lp.setSoort(BrmoFramework.BR_BGTLIGHT);
155                 lp.setStatus(LaadProces.STATUS.STAGING_OK);
156                 lp.setOpmerking(String.format("Bestand geladen van %s op %s", f.getAbsolutePath(), sdf.format(new Date())));
157                 lp.setAutomatischProces(Stripersist.getEntityManager().find(AutomatischProces.class, config.getId()));
158                 Stripersist.getEntityManager().persist(lp);
159                 Stripersist.getEntityManager().merge(this.config);
160 
161                 aantalGeladen++;
162                 msg = String.format("  Bestand %s is geladen en heeft status: %s.", f, lp.getStatus());
163                 LOG.info(msg);
164                 this.listener.addLog(msg);
165                 sb.append(msg).append(AutomatischProces.LOG_NEWLINE);
166                 if (aantalGeladen % commitPageSize == 0) {
167                     LOG.debug("Tussentijds opslaan van berichten, 'commitPageSize' is bereikt");
168                     Stripersist.getEntityManager().flush();
169                     Stripersist.getEntityManager().getTransaction().commit();
170                     Stripersist.getEntityManager().clear();
171                 }
172             }
173             listener.progress(++progress);
174         }
175         msg = String.format("Klaar met run op %tc", Calendar.getInstance());
176         LOG.info(msg);
177         listener.updateStatus(msg);
178         listener.addLog(msg);
179         sb.append(msg);
180 
181         listener.addLog("\n\n**** resultaat ****");
182         listener.addLog("\nAantal bestanden die al waren geladen: " + filterAlVerwerkt);
183         listener.addLog("\nAantal bestanden geladen: " + aantalGeladen + "\n");
184 
185         config.setStatus(WAITING);
186         config.setLogfile(sb.toString());
187         config.setLastrun(new Date());
188         config.updateSamenvattingEnLogfile("Aantal bestanden die al waren verwerkt: "
189                 + filterAlVerwerkt + AutomatischProces.LOG_NEWLINE
190                 + "Aantal bestanden geladen: " + aantalGeladen + AutomatischProces.LOG_NEWLINE);
191         Stripersist.getEntityManager().merge(config);
192     }

194     private int getCommitPageSize() {
195         int commitPageSize;
196         try {
197             String s = ClobElement.nullSafeGet(config.getConfig().get("commitPageSize"));
198             commitPageSize = Integer.parseInt(s);
199             if (commitPageSize < 1 || commitPageSize > defaultCommitPageSize) {
200                 commitPageSize = defaultCommitPageSize;
201             }
202         } catch (NumberFormatException nfe) {
203             commitPageSize = defaultCommitPageSize;
204         }
205 
206         LOG.debug("Instellen van commit page size op: " + commitPageSize);
207         return commitPageSize;
208     }
