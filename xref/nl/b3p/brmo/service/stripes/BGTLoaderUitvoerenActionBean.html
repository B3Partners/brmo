25  public class BGTLoaderUitvoerenActionBean implements ActionBean, ProgressUpdateListener {
26      private static final Log LOG = LogFactory.getLog(BGTLoaderUitvoerenActionBean.class);
27      private static final String JSP = "/WEB-INF/jsp/beheer/bgtloaderuitvoeren.jsp";
28      private String exceptionStacktrace;
29      private String status;
30      private StringBuilder log = new StringBuilder();
31      private ActionBeanContext context;
32      @Validate(converter = EntityTypeConverter.class)
33      private BGTLoaderProces proces;
34      private Date start;
35      private Date update;
36      private boolean complete;
37  
38      @Before
39      public void before() {
40          this.start = new Date();
41      }
42  
43      @After
44      public void completed() {
45          this.complete = true;
46          this.update = new Date();
47      }
48  
49      @DefaultHandler
50      @WaitPage(path = JSP, delay = 1000, refresh = 1000)
51      public Resolution execute() {
52          if (proces == null) {
53              getContext().getMessages().add(new SimpleMessage("Proces ongeldig!"));
54              completed();
55              return new ForwardResolution(JSP);
56          }
57  
58          // opnieuw laden van config omdat de waitpage de entity detached
59          this.update = new Date();
60          proces = (BGTLoaderProces) Stripersist.getEntityManager().find(AutomatischProces.class, proces.getId());
61          final ProcesExecutable _proces = AbstractExecutableProces.getProces(proces);
62          try {
63              _proces.execute(this);
64          } finally {
65              completed();
66              Stripersist.getEntityManager().merge(proces);
67              Stripersist.getEntityManager().getTransaction().commit();
68          }
69          return new ForwardResolution(JSP);
70      }
71  
72      @Override
73      public void total(long total) {
74      }
75  
76      @Override
77      public void progress(long progress) {
78          this.update = new Date();
79      }
80  
81      @Override
82      public void exception(Throwable t) {
83          this.exceptionStacktrace = t.getLocalizedMessage();
84      }
85  
86      @Override
87      public void updateStatus(String status) {
88          this.status = status;
89      }
90  
91      @Override
92      public void addLog(String line) {
93          this.log.append(line).append("\n");
94          LOG.info(log);
95      }
