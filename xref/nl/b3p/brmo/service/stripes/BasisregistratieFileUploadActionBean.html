34  /**
35   * Action voor het handmatig uploaden van bestanden met basisregistratiegegevens
36   * via een formulier.
37   */
38  @StrictBinding
39  public class BasisregistratieFileUploadActionBean implements ActionBean {
40      private static final int ZIP_HEADER = 0x504b0304;
41  
42      private static final Log log = LogFactory.getLog(BasisregistratieFileUploadActionBean.class);
43  
44      private ActionBeanContext context;
45  
46      @Validate(required=true)
47      private String basisregistratie;
48  
49      @Validate(required=true)
50      private FileBean bestand;
51  
52      @DefaultHandler
53      @DontValidate
54      public Resolution form() {
55          return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
56      }
57  
58      public Resolution upload() throws BrmoException {
59          DataSource ds = ConfigUtil.getDataSourceStaging();
60          BrmoFramework brmo = new BrmoFramework(ds, null);
61          ZipInputStream zip = null;
62          ZipEntry entry = null;
63          int errors = 0;
64          int empty = 0;
65          int duplicate = 0;
66          final int ERROR_LIMIT = 5;
67          try {
68              final MutableLong theTotal = new MutableLong(0);
69              int extractedFiles = 0;
70              final ProgressUpdateListener totalAdder = new ProgressUpdateListener() {
71  
72                  @Override
73                  public void total(long total) {
74                      theTotal.setValue(theTotal.getValue() + total);
75                  }
76  
77                  @Override
78                  public void progress(long progress) {
79                  }
80  
81                  @Override
82                  public void exception(Throwable t) {
83                  }
84              };
85  
86              // Check of bestand begint met ZIP header
87              InputStream in = new BufferedInputStream(bestand.getInputStream());
88              in.mark(4);
89              int header = new DataInputStream(in).readInt();
90              in.reset();
91  
92              if(header == ZIP_HEADER) {
93                  // Pak .xml bestanden in ZIP uit en verwerk deze
94  
95                  zip = new ZipInputStream(in);
96                  try {
97                      entry = zip.getNextEntry();
98                      while(entry != null) {
99                          if(!entry.getName().toLowerCase().endsWith(".xml")) {
100                             log.warn("Overslaan zip entry geen XML: " + entry.getName());
101                         } else {
102                             log.debug("Lezen XML bestand uit zip: " + entry.getName());
103                             try {
104                                 brmo.loadFromStream(basisregistratie, CloseShieldInputStream.wrap(zip), bestand.getFileName() + "/" + entry.getName(), totalAdder, null);
105                             } catch(BrmoLeegBestandException e) {
106                                 log.info("Negeer ZIP entry " + entry.getName() + ": " + e.getMessage());
107                                 empty++;
108                             } catch(BrmoDuplicaatLaadprocesException e) {
109                                 log.info("Negeer ZIP entry " + entry.getName() + ": duplicaat laadproces");
110                                 duplicate++;
111                             } catch(BrmoException e) {
112                                 errors++;
113                                 log.error("BrmoException bij laden ZIP entry " + entry.getName() + (errors > ERROR_LIMIT ? "; afbreken" : "; doorgaan met volgende"), e);
114                                 if(errors > ERROR_LIMIT) {
115                                     throw new BrmoException("Maximum aantal fouten (" + ERROR_LIMIT + ") bereikt bij inladen ZIP bestanden, inladen afgebroken. Controleer brmo-service.log", e);
116                                 }
117                             }
118                             extractedFiles++;
119                         }
120                         entry = zip.getNextEntry();
121                     }
122                 } finally {
123                     zip.close();
124                 }
125             } else {
126                 // Geen ZIP, direct BR XML laden
127                 brmo.loadFromStream(basisregistratie, in, bestand.getFileName(), totalAdder, null);
128             }
129 
130             List<String> warnings = new ArrayList<>();
131             if(errors > 0) {
132                 warnings.add("let op: " + errors + " bericht" + (errors > 1 ? "en" : "") + " niet ingeladen wegens fouten");
133             }
134             if(empty > 0) {
135                 warnings.add("let op: " + empty + (empty > 1 ? " lege berichten" : " leeg bericht") + " overgeslagen");
136             }
137             if(duplicate > 0) {
138                 warnings.add("let op: " + duplicate + (duplicate > 1 ? " duplicate berichten" : " duplicaat bericht") + " overgeslagen");
139             }
140 
141             String warning = warnings.isEmpty() ? "" : ", " + String.join(", ", warnings.toArray(new String[] {})) + ", zie log voor bestandsnamen uit ZIP en details!";
142 
143             getContext().getMessages().add(new SimpleMessage("Bestand " + bestand.getFileName() + " is ingelezen, " + theTotal.getValue() + " berichten" + (extractedFiles > 0 ? " (uit " + extractedFiles + " uitgepakte XML bestanden)" : "") + warning));
144             log.info(String.format("Stored %s data from file \"%s\" uploaded via form", basisregistratie, bestand.getFileName()));
145         } catch(Exception ex) {
146             String msg = "Fout bij inlezen bestand " + (zip != null && entry != null ? entry.getName() + " uit ZIP bestand " + bestand.getFileName() : bestand.getFileName());
147             log.error(msg, ex);
148 
149             getContext().getMessages().add(new SimpleMessage(msg + ": " + ExceptionUtils.getRootCauseMessage(ex)));
150         } finally {
151             try {
152                 bestand.delete();
153             } catch(IOException e) {
154                 // ignore
155             }
156             brmo.closeBrmoFramework();
157         }
158 
159         return new ForwardResolution("/WEB-INF/jsp/bestand/form.jsp");
160     }
