4   package nl.b3p.brmo.service.proxy;
5   
6   import java.io.File;
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.text.SimpleDateFormat;
10  import java.util.Date;
11  import java.util.zip.GZIPInputStream;
12  import javax.servlet.ServletException;
13  import javax.servlet.http.HttpServlet;
14  import javax.servlet.http.HttpServletRequest;
15  import javax.servlet.http.HttpServletResponse;
16  import org.apache.commons.io.FileUtils;
17  import org.apache.commons.logging.Log;
18  import org.apache.commons.logging.LogFactory;
19  
20  /**
21   * Endpoint servlet welke geposte bestanden opslaat in een directory conform de configuratie. Deze
22   * servlet ondersteund gzip compressie. Voorbeeld:<br>
23   * {@code curl -X POST -H "Content-Type: application/xml"
24   * http://localhost:8037/brmo-proxyservice/post/brk2 -d @'pom.xml'} <br>
25   * of met gzip compressie:<br>
26   * {@code curl -v -s --trace-ascii http_trace.log --data-binary @src/test/resources/web.xml.gz -H
27   * "Content-Type: text/xml" -H "Content-Encoding: gzip" -X POST
28   * http://localhost:8037/brmo-proxyservice/post/brk2} <br>
29   * of met certificaat authenticatie en wget:<br>
30   * {@code wget --post-file="test.xml" --append-output=logs/wget.log --ca-cert=./ca.pem
31   * --certificate=./client.pem --private-key=./key.pem
32   * https://somehost.nl/brmo-proxyservice/post/brk2 -O - >> output.log}
33   *
34   * @author mprins
35   */
36  public class BerichtEndpointFileServlet extends HttpServlet {
37  
38    private static final Log log = LogFactory.getLog(BerichtEndpointFileServlet.class);
39  
40    private static final String SAVE_DIR = "save_dir";
41    private static final String MAX_UPLOAD_SIZE = "max_upload_size";
42    private File saveDir;
43    private int maxUploadSize;
44  
45    @Override
46    public void init() throws ServletException {
47      String _saveDir = this.getInitParameter(SAVE_DIR);
48      if (_saveDir == null || _saveDir.isEmpty()) {
49        throw new ServletException("De parameter '" + SAVE_DIR + "' is benodigd.");
50      }
51      this.saveDir = new File(_saveDir);
52      try {
53        FileUtils.forceMkdir(saveDir);
54        log.info("Initialisatie met directory: " + this.saveDir);
55      } catch (IOException ex) {
56        throw new ServletException(ex);
57      }
58      if (!this.saveDir.canWrite()) {
59        throw new ServletException("De directory '" + _saveDir + "' is niet schrijfbaar.");
60      }
61      try {
62        this.maxUploadSize = Integer.parseInt(this.getInitParameter(MAX_UPLOAD_SIZE)) * 1024;
63      } catch (NumberFormatException nfe) {
64        log.warn(nfe);
65        this.maxUploadSize = 25 * 1024 * 1024;
66        log.warn("De maximale upload size is ingesteld op 25 MB (default).");
67      }
68    }
69  
70    /**
71     * Handles the HTTP {@code POST}.
72     *
73     * @param request servlet request
74     * @param response servlet response
75     * @throws ServletException if a servlet-specific error occurs
76     * @throws IOException if an I/O error occurs
77     */
78    @Override
79    protected void doPost(HttpServletRequest request, HttpServletResponse response)
80        throws ServletException, IOException {
81      if (request.getContentLength() > this.maxUploadSize) {
82        throw new ServletException(
83            "De 'max_upload_size' van " + this.maxUploadSize + " is overschreden.");
84      }
85      InputStream in = request.getInputStream();
86      if ("gzip".equals(request.getHeader("Content-Encoding"))) {
87        in = new GZIPInputStream(in);
88      }
89      File _tmpfile = File.createTempFile(this.getFileName(), ".xml", FileUtils.getTempDirectory());
90      FileUtils.copyInputStreamToFile(in, _tmpfile);
91      FileUtils.moveToDirectory(_tmpfile, this.saveDir, true);
92      log.info(
93          String.format(
94              "Aangeboden bestand '%s' opgeslagen in directory: %s.",
95              _tmpfile.getName(), this.saveDir));
96      response.setStatus(HttpServletResponse.SC_OK);
97    }
98  
99    /**
100    * "verzint" de bestandsnaam.
101    *
102    * @return de bestandsnaam {@code post-op-yyyy-MM-dd-HH_mm_ss_SSS}
103    */
104   private String getFileName() {
105     final SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd_HH-mm-ss-SSS");
106     return "post-op_" + sdf.format(new Date()) + "_";
107   }
108 }
