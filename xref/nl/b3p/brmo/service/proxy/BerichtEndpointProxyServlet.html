4   package nl.b3p.brmo.service.proxy;
5   
6   import java.io.IOException;
7   import java.net.HttpURLConnection;
8   import java.net.MalformedURLException;
9   import java.net.URL;
10  import javax.servlet.ServletException;
11  import javax.servlet.http.HttpServlet;
12  import javax.servlet.http.HttpServletRequest;
13  import javax.servlet.http.HttpServletResponse;
14  import org.apache.commons.io.IOUtils;
15  import org.apache.commons.logging.Log;
16  import org.apache.commons.logging.LogFactory;
17  
18  /**
19   * Endpoint servlet welke geposte bestanden doorstuurt naar de BRMO. Voorbeeld:
20   * {@code curl -X POST -H "Content-Type: application/xml" http://localhost:8084/brmo-proxyservice/post/brk -d  @'/testbestanden/BRK_KLEIN_SNAPSHOT.xml'}
21   *
22   * @author mprins
23   */
24  public class BerichtEndpointProxyServlet extends HttpServlet {
25  
26      private static final Log log = LogFactory.getLog(BerichtEndpointProxyServlet.class);
27  
28      private static final String PROXY_FOR_URL = "proxy_for_url";
29      private URL proxyForUrl;
30      private static final String MAX_UPLOAD_SIZE = "max_upload_size";
31      private int maxUploadSize;
32  
33      @Override
34      public void init() throws ServletException {
35          try {
36              this.maxUploadSize = Integer.parseInt(this.getInitParameter(MAX_UPLOAD_SIZE)) * 1024;
37          } catch (NumberFormatException nfe) {
38              this.maxUploadSize = 25 * 1024 * 1024;
39              log.warn("De maximale upload size is ingesteld op 25 MB (default).");
40          }
41          try {
42              this.proxyForUrl = new URL(this.getInitParameter(PROXY_FOR_URL));
43          } catch (MalformedURLException ex) {
44              throw new ServletException(ex);
45          }
46      }
47  
48      /**
49       * Handles the HTTP {@code POST} method.
50       *
51       * @param request servlet request
52       * @param response servlet response
53       * @throws ServletException if a servlet-specific error occurs
54       * @throws IOException if an I/O error occurs
55       */
56      @Override
57      protected void doPost(HttpServletRequest request, HttpServletResponse response)
58              throws ServletException, IOException {
59  
60          if (request.getContentLength() > this.maxUploadSize) {
61              throw new ServletException("De 'max_upload_size' is overschreden.");
62          }
63          HttpURLConnection conn = (HttpURLConnection) this.proxyForUrl.openConnection();
64          conn.setDoOutput(true);
65          conn.setUseCaches(false);
66          conn.setRequestMethod("POST");
67          conn.setRequestProperty("Content-Length", "" + request.getContentLength());
68          conn.setRequestProperty("Content-Type", "application/octet-stream");
69          int copied = IOUtils.copy(request.getInputStream(), conn.getOutputStream());
70          conn.disconnect();
71  
72          log.info(String.format("BRMO response status: %d: %s (%d bytes).",
73                  conn.getResponseCode(), conn.getResponseMessage(), copied));
74  
75          response.setStatus(HttpServletResponse.SC_OK);
76      }
77  }
