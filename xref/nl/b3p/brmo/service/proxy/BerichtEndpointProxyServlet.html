4   package nl.b3p.brmo.service.proxy;
5   
6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.net.HttpURLConnection;
9   import java.net.MalformedURLException;
10  import java.net.URL;
11  import java.util.zip.GZIPInputStream;
12  import javax.servlet.ServletException;
13  import javax.servlet.http.HttpServlet;
14  import javax.servlet.http.HttpServletRequest;
15  import javax.servlet.http.HttpServletResponse;
16  import org.apache.commons.io.IOUtils;
17  import org.apache.commons.logging.Log;
18  import org.apache.commons.logging.LogFactory;
19  
20  /**
21   * Endpoint servlet welke geposte bestanden doorstuurt naar de BRMO. Voorbeeld: {@code curl -X POST
22   * -H "Content-Type: application/xml" http://localhost:8084/brmo-proxyservice/post/brk
23   * -d @'/testbestanden/BRK_KLEIN_SNAPSHOT.xml'}
24   *
25   * @author mprins
26   */
27  public class BerichtEndpointProxyServlet extends HttpServlet {
28  
29    private static final Log log = LogFactory.getLog(BerichtEndpointProxyServlet.class);
30  
31    private static final String PROXY_FOR_URL = "proxy_for_url";
32    private URL proxyForUrl;
33    private static final String MAX_UPLOAD_SIZE = "max_upload_size";
34    private int maxUploadSize;
35  
36    @Override
37    public void init() throws ServletException {
38      try {
39        this.maxUploadSize = Integer.parseInt(this.getInitParameter(MAX_UPLOAD_SIZE)) * 1024 * 1024;
40        log.info("De maximale upload size is ingesteld op " + this.maxUploadSize + " MB.");
41      } catch (NumberFormatException nfe) {
42        this.maxUploadSize = 25 * 1024 * 1024;
43        log.warn("De maximale upload size is ingesteld op 25 MB (default).");
44      }
45      try {
46        this.proxyForUrl = new URL(this.getInitParameter(PROXY_FOR_URL));
47      } catch (MalformedURLException ex) {
48        throw new ServletException(ex);
49      }
50    }
51  
52    /**
53     * Handles the HTTP {@code POST} method.
54     *
55     * @param request servlet request
56     * @param response servlet response
57     * @throws ServletException if a servlet-specific error occurs
58     * @throws IOException if an I/O error occurs
59     */
60    @Override
61    protected void doPost(HttpServletRequest request, HttpServletResponse response)
62        throws ServletException, IOException {
63  
64      if (request.getContentLength() > this.maxUploadSize) {
65        throw new ServletException("De 'max_upload_size' is overschreden.");
66      }
67      HttpURLConnection conn = (HttpURLConnection) this.proxyForUrl.openConnection();
68      conn.setDoOutput(true);
69      conn.setUseCaches(false);
70      conn.setRequestMethod("POST");
71      conn.setRequestProperty("Content-Length", "" + request.getContentLength());
72      conn.setRequestProperty("Content-Type", "application/octet-stream");
73      InputStream in = request.getInputStream();
74      if ("gzip".equals(request.getHeader("Content-Encoding"))) {
75        in = new GZIPInputStream(in);
76      }
77      int copied = IOUtils.copy(in, conn.getOutputStream());
78      conn.disconnect();
79  
80      log.info(
81          String.format(
82              "BRMO response status: %d: %s (%d bytes).",
83              conn.getResponseCode(), conn.getResponseMessage(), copied));
84  
85      response.setStatus(HttpServletResponse.SC_OK);
86    }
87  }
