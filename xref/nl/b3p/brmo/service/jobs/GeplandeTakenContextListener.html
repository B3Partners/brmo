4   package nl.b3p.brmo.service.jobs;
5   
6   import javax.servlet.ServletContextEvent;
7   import javax.servlet.ServletContextListener;
8   import static nl.b3p.brmo.service.jobs.GeplandeTakenInit.QUARTZ_FACTORY_KEY;
9   import static nl.b3p.brmo.service.jobs.GeplandeTakenInit.SCHEDULER_NAME;
10  import org.quartz.impl.StdSchedulerFactory;
11  
12  /**
13   *
14   * @author mprins
15   */
16  public class GeplandeTakenContextListener implements ServletContextListener {
17  
18      /**
19       * Doet niets, setup van de StdSchedulerFactory doen we in een servlet
20       * vanwege initialisatie van stripes die eerst gebeurt in het stripes
21       * filter.
22       *
23       * @param sce de servlet context
24       * @see GeplandeTakenInit
25       */
26      @Override
27      public void contextInitialized(ServletContextEvent sce) {
28      }
29  
30      @Override
31      public void contextDestroyed(ServletContextEvent sce) {
32          try {
33              StdSchedulerFactory f = (StdSchedulerFactory) sce.getServletContext().
34                      getAttribute(QUARTZ_FACTORY_KEY);
35              sce.getServletContext().removeAttribute(QUARTZ_FACTORY_KEY);
36              if (f != null) {
37                  f.getScheduler(SCHEDULER_NAME).shutdown();
38              }
39          } catch (Exception ex) {
40              // sce.getServletContext().log(ex);
41          }
42      }
43  }
