19      public static void updateSequence(final String sequenceName, final long nextVal, final DataSource ds) throws SQLException {
20          try (Connection conn = ds.getConnection()) {
21              conn.setAutoCommit(true);
22              final GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(conn);
23              LOG.debug("update sequence: " + sequenceName);
24              // "update" gebruiken omdat we bij oracle stored procedure benaderen?? hoeft niet meer sinds 18c
25              // Object o = new QueryRunner(geomToJdbc.isPmdKnownBroken()).update(conn, geomToJdbc.getUpdateSequenceSQL(sequenceName, nextVal));
26              int rowsUpdated = new QueryRunner(geomToJdbc.isPmdKnownBroken()).execute(conn, geomToJdbc.getUpdateSequenceSQL(sequenceName, nextVal));
27              LOG.trace("updated sequence: " + sequenceName + ", " + rowsUpdated + " rows updated met SQL: " + geomToJdbc.getUpdateSequenceSQL(sequenceName, nextVal));
28          }
