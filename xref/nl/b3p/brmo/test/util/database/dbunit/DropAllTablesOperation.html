6   import java.sql.SQLException;
7   import java.sql.Statement;
8   import java.util.Arrays;
9   import java.util.Collections;
10  import java.util.List;
11  import org.apache.commons.logging.Log;
12  import org.apache.commons.logging.LogFactory;
13  import org.dbunit.DatabaseUnitException;
14  import org.dbunit.database.DatabaseConfig;
15  import org.dbunit.database.DatabaseSequenceFilter;
16  import org.dbunit.database.IDatabaseConnection;
17  import org.dbunit.dataset.FilteredDataSet;
18  import org.dbunit.dataset.IDataSet;
19  import org.dbunit.dataset.filter.ITableFilter;
20  import org.dbunit.operation.DatabaseOperation;
21  
22  public class DropAllTablesOperation extends DatabaseOperation {
23    private static final Log LOG = LogFactory.getLog(DropAllTablesOperation.class);
24    private final String DROP_PGSQL = "drop table %s cascade";
25    private final String DROP_ORACLE = "drop table %s cascade purge";
26  
27    /**
28     * Executes this operation on the specified database using the specified dataset contents.
29     *
30     * @param connection the database connection.
31     * @param dataSet the dataset to be used by this operation.
32     */
33    @Override
34    public void execute(IDatabaseConnection connection, IDataSet dataSet)
35        throws DatabaseUnitException, SQLException {
36      // get ordered list of tables
37      ITableFilter filter = new DatabaseSequenceFilter(connection);
38      final String dropStmt =
39          connection
40                  .getConfig()
41                  .getProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY)
42                  .getClass()
43                  .getName()
44                  .contains("Oracle")
45              ? DROP_ORACLE
46              : DROP_PGSQL;
47      IDataSet dataset;
48      if (null == dataSet) {
49        dataset = new FilteredDataSet(filter, connection.createDataSet());
50      } else {
51        dataset = new FilteredDataSet(filter, dataSet);
52      }
53      LOG.debug("to be dropped table names: " + Arrays.toString(dataset.getTableNames()));
54      String[] tableNames = dataset.getTableNames();
55      List<String> reversedTableNames = Arrays.asList(tableNames);
56      Collections.reverse(reversedTableNames);
57      LOG.debug("to be dropped table names in order: " + Arrays.toString(dataset.getTableNames()));
58      try (Statement stmt = connection.getConnection().createStatement()) {
59        for (String table : reversedTableNames) {
60          stmt.addBatch(String.format(dropStmt, table));
61        }
62        stmt.executeBatch();
63      }
64    }
65  }
