4   package nl.b3p.brmo.web.geotools;
5   
6   import javax.servlet.ServletContextEvent;
7   import javax.servlet.ServletContextListener;
8   import org.geotools.factory.GeoTools;
9   import org.geotools.util.logging.Logging;
10  
11  /**
12   *
13   * @author mprins
14   */
15  public class GeotoolsContextListener implements ServletContextListener {
16  
17      @Override
18      public void contextInitialized(ServletContextEvent sce) {
19          try {
20              GeoTools.init();
21              Logging.ALL.setLoggerFactory("org.geotools.util.logging.Log4JLoggerFactory");
22          } catch (ClassNotFoundException | IllegalArgumentException ex) {
23              // ignore
24          }
25      }
26  
27      @Override
28      public void contextDestroyed(ServletContextEvent sce) {
29          org.geotools.util.WeakCollectionCleaner.DEFAULT.exit();
30      }
31  
32  }
