131     private static void printHelp() {
132         HelpFormatter formatter = new HelpFormatter();
133         formatter.setOptionComparator(new Comparator<Option>() {
134             @Override
135             public int compare(Option lhs, Option rhs) {
136                 List[] lists = new List[]{modeOpts, dbOpts};
137                 for (List l : lists) {
138                     int lhsIndex = l.indexOf(lhs);
139                     if (lhsIndex != -1) {
140                         return Integer.compare(lhsIndex, l.indexOf(rhs));
141                     }
142                 }
143                 return lhs.getArgName().compareTo(rhs.getArgName());
144             }
145         });
146         final int W = 100;
147         StringWriter sw = new StringWriter();
148         PrintWriter pw = new PrintWriter(sw);
149         formatter.printUsage(pw, 80, "java -jar brmo-commandline.jar --<actie> --dbprops <db-props>");
150         pw.print("\nActies:\n");
151         formatter.printOptions(pw, W, modeOptions, 2, 2);
152         pw.print("Configuratie:\n");
153         formatter.printOptions(pw, W, dbOptions, 2, 2);
154 
155         System.err.println(sw.toString());
156     }

158     public static void main(String[] args) {
159         Options options = buildOptions();
160         CommandLine cl = null;
161         try {
162             CommandLineParser parser = new DefaultParser();
163             cl = parser.parse(options, args);
164         } catch (ParseException e) {
165             LOG.fatal(e);
166             printHelp();
167             System.exit(EX_USAGE.code);
168         }
169 
170         int exitcode = 0;
171         BasicDataSource dsStaging = null;
172         BasicDataSource dsRsgb = null;
173 
174         try {
175             dbProps.load(new FileInputStream(cl.getOptionValue("dbprops")));
176             switch (dbProps.getProperty("dbtype")) {
177                 case "oracle":
178                     Class.forName("oracle.jdbc.OracleDriver");
179                     break;
180                 case "postgis":
181                     Class.forName("org.postgresql.Driver");
182                     break;
183                 default:
184                     throw new IllegalArgumentException("Het database type " + dbProps.getProperty("dbtype") + " wordt niet ondersteund of is niet opgegeven.");
185             }
186 
187             dsStaging = new BasicDataSource();
188             LOG.info("Verbinding maken met Staging database... ");
189             dsStaging.setUrl(dbProps.getProperty("staging.url"));
190             dsStaging.setUsername(dbProps.getProperty("staging.user"));
191             dsStaging.setPassword(dbProps.getProperty("staging.password"));
192             dsStaging.setConnectionProperties(dbProps.getProperty("staging.options", ""));
193 
194             // alleen rsgb verbinding maken als nodig
195             if (cl.hasOption("torsgb") || cl.hasOption("versieinfo")) {
196                 LOG.info("Verbinding maken met RSGB database... ");
197                 dsRsgb = new BasicDataSource();
198                 dsRsgb.setUrl(dbProps.getProperty("rsgb.url"));
199                 dsRsgb.setUsername(dbProps.getProperty("rsgb.user"));
200                 dsRsgb.setPassword(dbProps.getProperty("rsgb.password"));
201                 dsRsgb.setConnectionProperties(dbProps.getProperty("rsgb.options", ""));
202             }
203 
204             // staging-only commando's
205             if (cl.hasOption("list")) {
206                 exitcode = list(dsStaging, cl.getOptionValue("l", "text"));
207             } else if (cl.hasOption("berichtstatus")) {
208                 exitcode = berichtStatus(dsStaging, cl.getOptionValue("berichtstatus", "text"));
209             } else if (cl.hasOption("jobstatus")) {
210                 exitcode = jobStatus(dsStaging, cl.getOptionValue("jobstatus", "text"));
211             } else if (cl.hasOption("load")) {
212                 // omdat we 2 verplichte argumenten hebben en 1 optionele die als String[]
213                 // worden doorgegeven een stream gebruike om eea aan mekaar te plakken
214                 exitcode = load(dsStaging, Stream.of(cl.getOptionValues("load"), cl.getArgs())
215                         .flatMap(Stream::of).toArray(String[]::new));
216             } else if (cl.hasOption("loaddir")) {
217                 exitcode = loaddir(dsStaging,
218                         Stream.of(cl.getOptionValues("loaddir"), cl.getArgs())
219                         .flatMap(Stream::of).toArray(String[]::new));
220             } else if (cl.hasOption("delete")) {
221                 exitcode = delete(dsStaging, cl.getOptionValue("delete"));
222             } else if (cl.hasOption("exportgds")) {
223                 exitcode = getMutations(dsStaging, cl.getOptionValues("exportgds"));
224             } else if(cl.hasOption("afgiftelijst")){
225                 String [] files = cl.getOptionValues("afgiftelijst");//cl.getOptionValue("afgiftelijst")
226                 exitcode = checkAfgiftelijst(dsStaging, files[0], files[1]);
227             }
228             // ----------------
229             // rsgb commando's
230             else if (cl.hasOption("torsgb")) {
231                 exitcode = toRsgb(dsStaging, dsRsgb, cl.getOptionValue("berichtstatus", "ignore"));
232             } // ----------------
233             // alle schema's / databases
234             else if (cl.hasOption("versieinfo")) {
235                 exitcode = versieInfo(dsStaging, dsRsgb, cl.getOptionValue("versieinfo", "text"));
236             }
237         } catch (BrmoException | InterruptedException ex) {
238             LOG.error("Fout tijdens uitvoeren met argumenten: " + Arrays.toString(args), ex);
239             System.err.println(ex.getLocalizedMessage());
240             exitcode = 1;
241         } catch (ClassNotFoundException ex) {
242             LOG.error("Database driver is niet gevonden.", ex);
243             System.err.println(ex.getLocalizedMessage());
244             exitcode = EX_DATAERR.code;
245         } catch (IOException ex) {
246             LOG.error("Het laden van het configuratie bestand is mislukt.", ex);
247             System.err.println(ex.getLocalizedMessage());
248             exitcode = EX_NOINPUT.code;
249         }
250         try {
251             if (dsStaging != null) {
252                 dsStaging.close();
253             }
254             if (dsRsgb != null) {
255                 dsRsgb.close();
256             }
257         } catch (SQLException ex) {
258             LOG.debug("Mogelijke fout tijdens afsluiten database verbindingen.", ex);
259         }
260 
261         System.exit(exitcode);
262     }

264     private static int toRsgb(DataSource dataSourceStaging, DataSource dataSourceRsgb, String errorState) throws BrmoException, InterruptedException {
265         LOG.info("Start staging naar rsgb transformatie.");
266         BrmoFramework brmo = new BrmoFramework(dataSourceStaging, dataSourceRsgb);
267         brmo.setOrderBerichten(true);
268         brmo.setErrorState(errorState);
269         Thread t = brmo.toRsgb();

272         brmo.closeBrmoFramework();
273         return 0;

424     private static int getMutations(DataSource ds, String... opts) {
425         LOG.info("Start export GDS2 berichten naar " + opts[0]);
426         Connection con = null;
427         String dir = opts[0];
428         int exitcode = 0;
429         try {
430             LOG.info("Ophalen automatisch proces(sen) waarmee GDS2 berichten zijn geladen.");
431             con = ds.getConnection();
432             Statement stmt = con.createStatement();
433             String autoProcessen = "select id FROM automatisch_proces where dtype = 'GDS2OphaalProces'";
434             ResultSet rs = stmt.executeQuery(autoProcessen);
435             while (rs.next()) {
436                 Long id = rs.getLong(1);
437                 LOG.info("Ophalen laadprocessen voor automatisch proces: " + id);
438                 String processen = "select id,bestand_naam from laadproces where automatisch_proces = " + id;
439                 Statement laadprocesStmt = con.createStatement();
440                 ResultSet lpRs = laadprocesStmt.executeQuery(processen);
441                 while (lpRs.next()) {
442                     Long lpId = lpRs.getLong(1);
443                     String bestandsNaam = lpRs.getString(2);
444                     String berichten = "select id, br_orgineel_xml, object_ref from bericht where laadprocesid = " + lpId;
445                     Statement berichtStmt = con.createStatement();
446                     ResultSet berRs = berichtStmt.executeQuery(berichten);
447                     while (berRs.next()) {
448                         LOG.info("Bericht " + id + " - " + bestandsNaam);
449                         String xml = berRs.getString("br_orgineel_xml");
450                         writeXML(xml, bestandsNaam, dir);
451                     }
452                 }
453             }
454         } catch (SQLException ex) {
455             LOG.error("Fout bij ophalen berichten/laadprocessen/automatische processen.", ex);
456             exitcode = EX_UNAVAILABLE.code;
457         } finally {
458             if (con != null) {
459                 try {
460                     con.close();
461                 } catch (SQLException ex) {
462                     LOG.warn("Database verbinding sluiten is mislukt.", ex);
463                 }
464             }
465         }
466         return exitcode;
467     }

469     private static void writeXML(String xml, String filename, String directory) {
470         ZipOutputStream out = null;
471         try {
472             filename = Paths.get(filename + ".zip").normalize().toString();
473             final File f = new File(directory, filename );
474             out = new ZipOutputStream(new FileOutputStream(f));
475             ZipEntry e = new ZipEntry(filename);
476             out.putNextEntry(e);
477             byte[] data = xml.getBytes();
478             out.write(data, 0, data.length);
479             out.closeEntry();
480             out.close();
481         } catch (IOException ex) {
482             LOG.error("Schrijven van bestand " + filename + " in " + directory + " is mislukt.", ex);
483         } finally {
484             if (out != null) {
485                 try {
486                     out.close();
487                 } catch (IOException ex) {
488                     LOG.warn("Sluiten .zip file is mislukt.", ex);
489                 }
490             }
491         }
492     }

503     private static int load(DataSource ds, String... opts) throws BrmoException {
504         String fileName = opts[0];
505         String brType = opts[1];
506         String archiefDir = opts.length == 3 ? opts[2] : null;

513         brmo.closeBrmoFramework();
514         LOG.info(String.format("Klaar met laden van bestand: %s, type %s", fileName, brType));
515         archiveerBestand(fileName, archiefDir);
516         return 0;

519     /**
520      * verwerk xml en zip bestanden uit directory naar staging db.
521      *
522      * @param ds datasource
523      * @param opts array met {directory, type BR, optionele archief directory}
524      * @return {@code 0} als succesvol,{@code EX_DATAERR} in geval van een
525      * waarschuwing
526      * @throws BrmoException in geval van niet herstalebare laadfout
527      */
528     private static int loaddir(DataSource ds, String... opts) throws BrmoException {
529         String scanDir = opts[0];
530         String brType = opts[1];
531         String archiefDir = opts.length == 3 ? opts[2] : null;
532 
533         int exitcode = 0;
534         if (!scanDir.endsWith(File.separator)) {
535             scanDir += File.separator;
536         }
537         LOG.info(String.format("Begin laden van directory: %s, type %s", scanDir, brType));
538 
539         File dir = new File(scanDir);
540         if (dir.isDirectory()) {
541             boolean withWarnings = false;
542             String[] fNames = dir.list((File f, String name) -> {
543                 return (name.endsWith(".xml") || name.endsWith(".XML") || name.endsWith(".zip") || name.endsWith(".ZIP"));
544             });
545             BrmoFramework brmo = new BrmoFramework(ds, null);
546             brmo.setOrderBerichten(true);
547             brmo.setErrorState("ignore");
548 
549             for (String fName : fNames) {
550                 try {
551                     LOG.debug(String.format("Begin laden van bestand: %s, type %s", fName, brType));
552                     brmo.loadFromFile(brType, scanDir + fName, null);
553                     LOG.info(String.format("Klaar met laden van bestand: %s, type %s", fName, brType));
554                 } catch (BrmoDuplicaatLaadprocesException dup) {
555                     LOG.warn(String.format("Laden duplicaat bestand %s overgeslagen. Oorzaak: %s", fName, dup.getLocalizedMessage()));
556                     withWarnings = true;
557                 } catch (BrmoLeegBestandException leeg) {
558                     LOG.warn(String.format("Laden 'leeg' bestand %s overgeslagen. Oorzaak: %s", fName, leeg.getLocalizedMessage()));
559                     withWarnings = true;
560                 } finally {
561                     brmo.closeBrmoFramework();
562                 }
563                 archiveerBestand(scanDir + fName, archiefDir);
564             }
565             if (withWarnings) {
566                 exitcode = EX_DATAERR.code;
567             }
568         } else {
569             throw new BrmoException("Opgegeven directory " + scanDir + " is geen directory.");
570         }
571         LOG.info(String.format("Klaar met laden van directory: %s, type %s", scanDir, brType));
572         return exitcode;
573     }
574 
575     /**
576      * verplaats bestand naar archief directory.
577      *
578      * @param fileName te verplaatsen bestandsnaam (volledig pad)
579      * @param archiefDir doel directory (in geval {@code null} dat wordt
580      * verplaatsen overgeslagen)
581      */
582     private static void archiveerBestand(final String fileName, final String archiefDir) {
583         if (archiefDir != null) {
584             File archiefDirectory = new File(archiefDir);
585             LOG.debug(String.format("Archiveren %s naar %s.", fileName, archiefDir));
586             try {
587                 FileUtils.moveFileToDirectory(new File(fileName), archiefDirectory, true);
588                 LOG.debug(String.format("Bestand %s is naar archief %s verplaatst.", fileName, archiefDirectory));
589             } catch (IOException e) {
590                 LOG.error(String.format("Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).", fileName, archiefDirectory, e.getLocalizedMessage()));
591             }
592         }
593     }
