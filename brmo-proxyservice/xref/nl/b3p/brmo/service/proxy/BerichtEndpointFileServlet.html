4   package nl.b3p.brmo.service.proxy;

6   import java.io.File;
7   import java.io.IOException;
8   import java.text.SimpleDateFormat;
9   import java.util.Date;
10  import javax.servlet.ServletException;
11  import javax.servlet.http.HttpServlet;
12  import javax.servlet.http.HttpServletRequest;
13  import javax.servlet.http.HttpServletResponse;
14  import org.apache.commons.io.FileUtils;
15  import org.apache.commons.logging.Log;
16  import org.apache.commons.logging.LogFactory;
17  
18  /**
19   * Endpoint servlet welke geposte bestanden opslaat in een directory conform de
20   * configuratie. Voorbeeld:
21   * {@code curl -X POST -H "Content-Type: application/xml" http://localhost:8037/brmo-proxyservice/post/brk  -d  @'pom.xml'}
22   * of met certificaat authenticatie en wget:
23   * {@code wget --post-file="test.xml" --append-output=logs/wget.log --ca-cert=./ca.pem --certificate=./client.pem --private-key=./key.pem https://somehost.nl/brmo-proxyservice/post/brk -O - >> output.log}
24   *
25   * @author mprins
26   */
27  public class BerichtEndpointFileServlet extends HttpServlet {
28  
29      private static final Log log = LogFactory.getLog(BerichtEndpointFileServlet.class);
30  
31      private static final String SAVE_DIR = "save_dir";
32      private File saveDir;
33      private static final String MAX_UPLOAD_SIZE = "max_upload_size";
34      private int maxUploadSize;
35  
36      @Override
37      public void init() throws ServletException {
38          String _saveDir = this.getInitParameter(SAVE_DIR);
39          if (_saveDir == null && !_saveDir.isEmpty()) {
40              throw new ServletException("De parameter '" + SAVE_DIR + "' is benodigd.");
41          }
42          this.saveDir = new File(_saveDir);
43          try {
44              FileUtils.forceMkdir(saveDir);
45              log.info("Initialisatie met directory: " + this.saveDir);
46          } catch (IOException ex) {
47              throw new ServletException(ex);
48          }
49          if (!this.saveDir.canWrite()) {
50              throw new ServletException("De directory '" + _saveDir + "' is niet schrijfbaar.");
51          }
52          try {
53              this.maxUploadSize = Integer.parseInt(this.getInitParameter(MAX_UPLOAD_SIZE)) * 1024;
54          } catch (NumberFormatException nfe) {
55              this.maxUploadSize = 25 * 1024 * 1024;
56              log.warn("De maximale upload size is ingesteld op 25 MB (default).");
57          }
58      }
59  
60      /**
61       * Handles the HTTP {@code POST}.
62       *
63       * @param request servlet request
64       * @param response servlet response
65       * @throws ServletException if a servlet-specific error occurs
66       * @throws IOException if an I/O error occurs
67       */
68      @Override
69      protected void doPost(HttpServletRequest request, HttpServletResponse response)
70              throws ServletException, IOException {
71          if (request.getContentLength() > this.maxUploadSize) {
72              throw new ServletException("De 'max_upload_size' van " + this.maxUploadSize + " is overschreden.");
73          }
74  
75          File _tmpfile = File.createTempFile(this.getFileName(), ".xml", FileUtils.getTempDirectory());
76          FileUtils.copyInputStreamToFile(request.getInputStream(), _tmpfile);
77          FileUtils.moveToDirectory(_tmpfile, this.saveDir, true);
78          log.info(String.format("Aangeboden bestand '%s' opgeslagen in directory: %s.",
79                  _tmpfile.getName(), this.saveDir));
80          response.setStatus(HttpServletResponse.SC_OK);
81      }
82  
83      /**
84       * "verzint" de bestandsnaam.
85       *
86       * @return de bestandsnaam {@code post-op-yyyy-MM-dd-HH_mm_ss_SSS}
87       */
88      private String getFileName() {
89          final SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd_HH-mm-ss-SSS");
90          return "post-op_" + sdf.format(new Date()) + "_";
91      }
92  }
