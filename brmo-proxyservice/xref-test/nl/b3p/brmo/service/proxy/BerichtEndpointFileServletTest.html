30  public class BerichtEndpointFileServletTest {
31  
32      private String saveDir = "/bericht-post-test-";
33      private ServletUnitClient client;
34      private ServletRunner sr;
35  
36      @BeforeEach
37      public void setUp() throws Exception {
38          HttpUnitOptions.setDefaultCharacterSet("UTF-8");
39          HttpUnitOptions.setLoggingHttpHeaders(true);
40          HttpUnitOptions.setPostIncludesCharset(true);
41  
42          saveDir = System.getProperty("java.io.tmpdir", "/tmp") + saveDir + new BigInteger(130, new SecureRandom()).toString(32);
43          sr = new ServletRunner();
44          Hashtable<String, String> servletParams = new Hashtable<>();
45          servletParams.put("save_dir", saveDir);
46          sr.registerServlet("/post/brk", BerichtEndpointFileServlet.class.getName(), servletParams);
47          client = sr.newClient();
48      }
49  
50      @AfterEach
51      public void tearDown() throws Exception {
52          sr.shutDown();
53          FileUtils.cleanDirectory(new File(saveDir));
54          FileUtils.deleteQuietly(new File(saveDir));
55      }
56  
57      @Test
58      public void testPostBRK() throws Exception {
59          WebRequest p = new PostMethodWebRequest(
60                  "http://localhost:8080/post/brk",
61                  BerichtEndpointFileServletTest.class.getResourceAsStream("/web.xml"),
62                  "text/xml; charset=utf-8'"
63          );
64          WebResponse response = client.getResponse(p);
65          assertNotNull(response, "No response received");
66          assertEquals(200, response.getResponseCode(), "Response not OK");
67  
68          // servlet scrijft naar saveDir, /tmp/brk/post-op_<yyyy-MM-dd_HH-mm-ss-SSS>_<randomuniek>.xml
69          File expected = new File(BerichtEndpointFileServletTest.class.getResource("/web.xml").getFile());
70          File actual = Files.newDirectoryStream(new File(saveDir).toPath(), "*.{xml}").iterator().next().toFile();
71  
72          assertTrue(actual.getName().startsWith("post-op"), "Bestand begint niet met 'post-op");
73          assertTrue(FileUtils.contentEquals(
74                  expected,
75                  actual
76          ), "File inhoud niet gelijk");
77      }
