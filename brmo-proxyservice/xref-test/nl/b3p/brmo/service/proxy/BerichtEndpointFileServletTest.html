31  class BerichtEndpointFileServletTest {
32  
33    private String saveDir = "/bericht-post-test-";
34    private ServletUnitClient client;
35    private ServletRunner sr;
36  
37    @BeforeEach
38    @SuppressModernizer
39    void setUp() {
40      HttpUnitOptions.setDefaultCharacterSet("UTF-8");
41      HttpUnitOptions.setLoggingHttpHeaders(true);
42      HttpUnitOptions.setPostIncludesCharset(true);
43  
44      saveDir =
45          System.getProperty("java.io.tmpdir", "/tmp")
46              + saveDir
47              + new BigInteger(130, new SecureRandom()).toString(32);
48      sr = new ServletRunner();
49      // modernizer zeurt over gebruik Hashtable ipv HashMap
50      Hashtable<String, String> servletParams = new Hashtable<>();
51      servletParams.put("save_dir", saveDir);
52      sr.registerServlet("/post/brk2", BerichtEndpointFileServlet.class.getName(), servletParams);
53      client = sr.newClient();
54    }
55  
56    @AfterEach
57    void tearDown() throws Exception {
58      sr.shutDown();
59      FileUtils.deleteQuietly(new File(saveDir));
60    }
61  
62    @Test
63    void testPostBRK() throws Exception {
64      WebRequest p =
65          new PostMethodWebRequest(
66              "http://localhost:8080/post/brk2",
67              BerichtEndpointFileServletTest.class.getResourceAsStream("/test.xml"),
68              "text/xml; charset=utf-8'");
69      WebResponse response = client.sendRequest(p);
70      assertAll(
71          () -> assertNotNull(response, "No response received"),
72          () -> assertEquals(200, response.getResponseCode(), "Response not OK"),
73          // side effect: servlet schrijft naar saveDir
74          () ->
75              assertTrue(Files.exists(new File(saveDir).toPath()), "save directory does not exist"));
76  
77      File expected =
78          new File(BerichtEndpointFileServletTest.class.getResource("/test.xml").getFile());
79      assertNotNull(expected, "Expected file not found");
80      // servlet schrijft naar saveDir,
81      // /tmp/brk/post-op_<yyyy-MM-dd_HH-mm-ss-SSS>_<randomuniek>.xml
82      File actual =
83          Files.newDirectoryStream(new File(saveDir).toPath(), "*.{xml}").iterator().next().toFile();
84  
85      assertTrue(actual.getName().startsWith("post-op"), "Bestand begint niet met 'post-op");
86      assertTrue(FileUtils.contentEquals(expected, actual), "File inhoud niet gelijk");
87    }

89    /**
90     * met de hand testen tegen draaiende servlet (met en zonder crompressie):
91     *
92     * <ul>
93     *   <li>{@code curl -v -s --trace-ascii http_trace.log
94     *       --data-binary @'brmo-loader/src/test/resources/verminderenstukdelen/MUTKX01-ASN00V2937-Bericht1.xml.gz'
95     *       -H "Content-Type: text/xml" -H "Content-Encoding: gzip" -X POST
96     *       http://localhost:8037/brmo-proxyservice/post/brk }
97     *   <li>{@code curl -v -s --trace-ascii http_trace.log
98     *       -d @'brmo-loader/src/test/resources/verminderenstukdelen/MUTKX01-ASN00V2937-Bericht1.xml'
99     *       -H "Content-Type: application/xml" -X POST
100    *       http://localhost:8037/brmo-proxyservice/post/brk }
101    * </ul>
102    *
103    * @throws Exception if any
104    */
105   @Test
106   void testPostGzippedBRK() throws Exception {
107     WebRequest p =
108         new PostMethodWebRequest(
109             "http://localhost:8080/post/brk2",
110             BerichtEndpointFileServletTest.class.getResourceAsStream("/test.xml.gz"),
111             "text/xml; charset=utf-8");
112     p.setHeaderField("Content-Encoding", "gzip");
113     WebResponse response = client.sendRequest(p);
114 
115     assertAll(
116         () -> assertNotNull(response, "No response received"),
117         () -> assertEquals(200, response.getResponseCode(), "Response not OK"),
118         // side effect: servlet schrijft naar saveDir
119         () ->
120             assertTrue(Files.exists(new File(saveDir).toPath()), "save directory does not exist"));
121 
122     File expected =
123         new File(BerichtEndpointFileServletTest.class.getResource("/test.xml").getFile());
124     assertNotNull(expected, "Expected file not found");
125     // servlet schrijft naar saveDir,
126     // /tmp/brk/post-op_<yyyy-MM-dd_HH-mm-ss-SSS>_<randomuniek>.xml
127     File actual =
128         Files.newDirectoryStream(new File(saveDir).toPath(), "*.{xml}").iterator().next().toFile();
129     assertTrue(actual.getName().startsWith("post-op"), "Bestand begint niet met 'post-op");
130     assertTrue(FileUtils.contentEquals(expected, actual), "File inhoud niet gelijk");
131   }
