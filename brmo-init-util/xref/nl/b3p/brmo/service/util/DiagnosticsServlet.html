6   import java.io.IOException;
7   import java.sql.DatabaseMetaData;
8   import javax.servlet.Servlet;
9   import javax.servlet.ServletConfig;
10  import javax.servlet.ServletException;
11  import javax.servlet.ServletRequest;
12  import javax.servlet.ServletResponse;
13  import javax.sql.DataSource;
14  import nl.b3p.brmo.service.util.ConfigUtil;
15  import org.apache.commons.logging.Log;
16  import org.apache.commons.logging.LogFactory;
17  
18  /**
19   *
20   * @author mprins
21   */
22  public class DiagnosticsServlet implements Servlet {
23  
24      private static final Log LOG = LogFactory.getLog(DiagnosticsServlet.class);
25  
26      @Override
27      public void init(ServletConfig config) throws ServletException {
28  
29          try {
30              // lookup db connectie en log info
31              DataSource rsgb = ConfigUtil.getDataSourceRsgb();
32              DatabaseMetaData metadata = rsgb.getConnection().getMetaData();
33              LOG.info(String.format("\nDatabase en driver informatie\n\n  Database product: %s\n  Database version: %s\n  Database major:   %s\n  Database minor:   %s\n\n  DBdriver product: %s\n  DBdriver version: %s\n  DBdriver major:   %s\n  DBdriver minor:   %s",
34                      metadata.getDatabaseProductName(),
35                      metadata.getDatabaseProductVersion().replace('\n', ' '),
36                      metadata.getDatabaseMajorVersion(),
37                      metadata.getDatabaseMinorVersion(),
38                      metadata.getDriverName(),
39                      metadata.getDriverVersion(),
40                      metadata.getDriverMajorVersion(),
41                      metadata.getDriverMinorVersion()
42              ));
43  
44              // foute oracle drivers loggen
45              if (metadata.getDriverName().startsWith("Oracle")) {
46                  if (metadata.getDriverVersion().startsWith("12.1.0.1")) {
47                      LOG.error("De geïnstalleerde JDBC driver heeft bekende problemen met prepared statements, de applicatie zal niet goed werken. Issue #322");
48                      LOG.info("zie: https://github.com/B3Partners/brmo/wiki/Welke-jdbc-driver voor meer informatie");
49                  }
50  
51                  // 11.2.0.3
52                  if (metadata.getDriverVersion().startsWith("11.2.0.3")) {
53                      LOG.error("De geïnstalleerde JDBC driver heeft bekende problemen met type conversies, de applicatie zal mogelijk niet goed werken. Issue #322");
54                      LOG.info("zie: https://github.com/B3Partners/brmo/wiki/Welke-jdbc-driver voor meer informatie");
55                  }
56              }
57              metadata.getConnection().close();
58          } catch (Exception ex) {
59              LOG.error(ex);
60          }
61  
62      }
63  
64      @Override
65      public ServletConfig getServletConfig() {
66          throw new UnsupportedOperationException("Not supported yet.");
67      }
68  
69      @Override
70      public void service(ServletRequest req, ServletResponse res) throws ServletException, IOException {
71          throw new UnsupportedOperationException("Not supported yet.");
72      }
73  
74      @Override
75      public String getServletInfo() {
76          return "Diagnostics info logging servlet";
77      }
78  
79      @Override
80      public void destroy() {
81          // nothing
82      }
83  
84  }
