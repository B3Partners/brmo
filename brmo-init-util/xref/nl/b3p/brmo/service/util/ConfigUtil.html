22      private static final String JDBC_NAME_RSGB = "jdbc/brmo/rsgb";

31      public static Integer MAX_UPLOAD_SIZE;
32      public static String TEMP_FOLDER;
33  
34      @Override
35      public void init(ServletConfig config) throws ServletException {
36          String tempSize = config.getInitParameter("max_upload_size");
37          if (tempSize != null && !tempSize.isEmpty()) {
38              MAX_UPLOAD_SIZE = new Integer(tempSize) * 1024;
39          } else {
40              MAX_UPLOAD_SIZE = 500 * 1024;
41          }
42          
43          String tempFolder = config.getInitParameter("temp_folder");
44          if (tempFolder != null && !tempFolder.isEmpty()) {          
45              TEMP_FOLDER = tempFolder;
46          } else {
47              TEMP_FOLDER = System.getProperty("java.io.tmpdir");
48          }        
49      }
50  
51      public static DataSource getDataSourceStaging() throws BrmoException {
52          try {
53              if (datasourceStaging == null) {
54                  InitialContext ic = new InitialContext();
55                  Context xmlContext = (Context) ic.lookup(JNDI_NAME);
56                  datasourceStaging = (DataSource) xmlContext.lookup(JDBC_NAME_STAGING);
57              }
58          } catch (Exception ex) {
59              log.error("Fout verbinden naar staging db. ", ex);
60              throw new BrmoException(ex);
61          }
62  
63          return datasourceStaging;
64      }
65
