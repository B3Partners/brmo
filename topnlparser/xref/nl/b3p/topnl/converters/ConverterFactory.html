23      private final Converter top250nlConverter = new Top250NLConverter();
24      private final Converter top100nlConverter = new Top100NLConverter();
25      private final Converter top50nlConverter = new Top50NLConverter();
26      private final Converter top10nlConverter = new Top10NLConverter();
27      private JAXBContext context250nl = null;
28      private JAXBContext context100nl = null;
29      private JAXBContext context50nl = null;
30      private JAXBContext context10nl = null;
31      
32      public ConverterFactory() throws JAXBException{
33          
34          context250nl = JAXBContext.newInstance("nl.b3p.topnl.top250nl");
35          context100nl = JAXBContext.newInstance("nl.b3p.topnl.top100nl");
36          context50nl = JAXBContext.newInstance("nl.b3p.topnl.top50nl");
37          context10nl = JAXBContext.newInstance("nl.b3p.topnl.top10nl");
38      }
39      
40      public Converter getConverter(TopNLType type){
41          switch(type){
42              case TOP250NL:
43                  return top250nlConverter;
44              case TOP100NL:
45                  return top100nlConverter;
46              case TOP50NL:
47                  return top50nlConverter;
48              case TOP10NL:
49                  return top10nlConverter;
50              default:
51                  throw new IllegalArgumentException("Converter not yet implemented: " + type.getType());
52                  
53          }
54      }
55      
56      public JAXBContext getContext(TopNLType type) {
57          switch (type) {
58              case TOP10NL:
59                  return context10nl;
60              case TOP50NL:
61                  return context50nl;
62              case TOP100NL:
63                  return context100nl;
64              case TOP250NL:
65                  return context250nl;
66              default:
67                  throw new IllegalArgumentException("TopNL type " + type.getType() + " not (yet) supported.");
68          }
69      }
70  
71  }
