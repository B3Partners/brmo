17  package nl.b3p.topnl;
18  
19  import java.io.IOException;
20  import java.net.URL;
21  import org.jdom2.Document;
22  import org.jdom2.Element;
23  import org.jdom2.JDOMException;
24  import org.jdom2.input.SAXBuilder;
25  
26  /**
27   *
28   * @author Meine Toonen meinetoonen@b3partners.nl
29   */
30  public class TopNLTypeFactory {
31  
32      private static final String TOP250NLNAMESPACE = "http://register.geostandaarden.nl/gmlapplicatieschema/top250nl/1.2.1";
33      private static final String TOP100NLNAMESPACE = "http://register.geostandaarden.nl/gmlapplicatieschema/top100nl/1.1.0";
34      private static final String TOP50NLNAMESPACE = "http://www.kadaster.nl/top50nl/1.1";
35      private static final String TOP10NLNAMESPACE = "http://register.geostandaarden.nl/gmlapplicatieschema/top10nl/1.2.0";
36  
37      public static TopNLType getTopNLType(URL is) throws JDOMException, IOException {
38          Document inputXml = new SAXBuilder().build(is);
39          return getTopNLType(inputXml);       
40      }
41      
42      public static TopNLType getTopNLType(String is) throws JDOMException, IOException {
43          Document inputXml = new SAXBuilder().build(is);
44          return getTopNLType(inputXml);
45      }
46  
47      private static TopNLType getTopNLType(Document inputXml){
48           if (!inputXml.hasRootElement()) {
49              throw new IllegalArgumentException("Document contains no root element");
50          }
51          Element rootElem = inputXml.getRootElement();
52          String currentNamespace = rootElem.getNamespace().getURI();
53          switch (currentNamespace) {
54              case TOP250NLNAMESPACE:
55                  return TopNLType.TOP250NL;
56              case TOP100NLNAMESPACE:
57                  return TopNLType.TOP100NL;
58              case TOP50NLNAMESPACE:
59                  return TopNLType.TOP50NL;
60              case TOP10NLNAMESPACE:
61                  return TopNLType.TOP10NL;
62              default:
63                  throw new IllegalArgumentException("Type not recognized: " + currentNamespace);
64          }
65      }
66  }
