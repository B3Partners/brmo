58      public Database(DataSource ds) throws SQLException {
59          this.dataSource = ds;
60          if (ds != null) {
61              gjc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(dataSource.getConnection());
62              run = new QueryRunner(dataSource, gjc.isPmdKnownBroken());
63          }
64      }

66      public void save(TopNLEntity entity) throws ParseException, SQLException {
67  
68          try {
69  
70              if (entity instanceof Hoogte) {
71                  saveHoogte(entity);
72              } else if (entity instanceof FunctioneelGebied) {
73                  saveFunctioneelGebied(entity);
74              } else if (entity instanceof Gebouw) {
75                  saveGebouw(entity);
76              } else if (entity instanceof GeografischGebied) {
77                  saveGeografischGebied(entity);
78              } else if (entity instanceof Inrichtingselement) {
79                  saveInrichtingselement(entity);
80              } else if (entity instanceof Plaats) {
81                  savePlaats(entity);
82              } else if (entity instanceof RegistratiefGebied) {
83                  saveRegistratiefGebied(entity);
84              } else if (entity instanceof Relief) {
85                  saveRelief(entity);
86              } else if (entity instanceof Spoorbaandeel) {
87                  saveSpoorbaandeel(entity);
88              } else if (entity instanceof Terrein) {
89                  saveTerrein(entity);
90              } else if (entity instanceof Waterdeel) {
91                  saveWaterdeel(entity);
92              } else if (entity instanceof Wegdeel) {
93                  saveWegdeel(entity);
94              } else if (entity instanceof PlanTopografie) {
95                  savePlanTopografie(entity);
96              } else{
97                  throw new IllegalArgumentException("Type of entity not (yet) implemented.");
98              }
99  
100         } catch (SQLException e) {
101             log.error("Error inserting entity: ", e);
102             throw e;
103         }
104     }

106     private Gebouw saveGebouw(TopNLEntity entity) throws SQLException, ParseException {
107         Gebouwref="../../../nl/b3p/topnl/entities/Gebouw.html#Gebouw">Gebouw h = (Gebouw) entity;
108 
109         ResultSetHandler<Gebouw> handler = new BeanHandler(Gebouw.class, new BasicRowProcessor(new DbUtilsGeometryColumnConverter(gjc)));
110         Object nativeGeom = gjc.convertToNativeGeometryObject(h.getGeometrie());
111         Object[] args = getVarargs(entity,
112                 h.getTypeGebouw(),
113                 h.getStatus(),
114                 h.getFysiekVoorkomen(),
115                 h.getHoogteklasse(),
116                 h.getHoogte(),
117                 h.getSoortnaam(),
118                 h.getNaam(),
119                 h.getNaamFries(),
120                 nativeGeom);
121         Gebouw inserted = run.insert("INSERT INTO " + h.getTopnltype() + ".gebouw (" + getTopNLEntityColumns() + ",typegebouw,status,fysiekvoorkomen,hoogteklasse,hoogte,soortnaam,naam,naamfries,geometrie) VALUES (" + getTopNLEntityReplacementChars() + ",?,?,?,?,?,?,?,?,?)",
122                 handler,
123                 args);
124 
125         return inserted;
126     }

391     private Object[] getVarargs(TopNLEntity entity, Object... specificArgs) {
392         Object[] genericArgs = {
393             entity.getIdentificatie(),
394             entity.getTopnltype(),
395             entity.getBrontype(),
396             entity.getBronactualiteit() != null ? new java.sql.Date(entity.getBronactualiteit().getTime()) : null,
397             entity.getBronbeschrijving(),
398             entity.getBronnauwkeurigheid(),
399             entity.getObjectBeginTijd() != null ? new java.sql.Date(entity.getObjectBeginTijd().getTime()) : null,
400             entity.getObjectEindTijd() != null ? new java.sql.Date(entity.getObjectEindTijd().getTime()) : null,
401             entity.getVisualisatieCode()
402         };
403         int numGeneric = genericArgs.length;
404         Object[] completeArgs = new Object[numGeneric + specificArgs.length];
405         System.arraycopy(genericArgs, 0, completeArgs, 0, numGeneric);
406 
407         for (int i = 0; i < specificArgs.length; i++) {
408             Object specificArg = specificArgs[i];
409             completeArgs[numGeneric + i] = specificArg;
410         }
411 
412         return completeArgs;
413     }
