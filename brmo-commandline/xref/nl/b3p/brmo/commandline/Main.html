129     private static void printHelp() {
130         HelpFormatter formatter = new HelpFormatter();
131         formatter.setOptionComparator(new Comparator<Option>() {
132             @Override
133             public int compare(Option lhs, Option rhs) {
134                 List[] lists = new List[]{modeOpts, dbOpts};
135                 for (List l : lists) {
136                     int lhsIndex = l.indexOf(lhs);
137                     if (lhsIndex != -1) {
138                         return new Integer(lhsIndex).compareTo(l.indexOf(rhs));
139                     }
140                 }
141                 return lhs.getArgName().compareTo(rhs.getArgName());
142             }
143         });
144         final int W = 100;
145         StringWriter sw = new StringWriter();
146         PrintWriter pw = new PrintWriter(sw);
147         formatter.printUsage(pw, 80, "java -jar brmo-commandline.jar --<actie> --dbprops <db-props>");
148         formatter.printUsage(pw, 80, "[Oracle] java -cp \"brmo-commandline.jar;ojdbc8.jar;lib/*\" nl.b3p.brmo.commandline.Main --<actie> --dbprops <db-props>");
149         pw.print("\n                [Omdat de Oracle jdbc driver niet gedistribueerd mag worden"
150                 + "\n                 dient deze zelf op de commando regel te worden opgegeven.]\n");
151         pw.print("\nActies:\n");
152         formatter.printOptions(pw, W, modeOptions, 2, 2);
153         pw.print("Configuratie:\n");
154         formatter.printOptions(pw, W, dbOptions, 2, 2);
155 
156         System.err.println(sw.toString());
157     }

159     public static void main(String... args) {
160         Options options = buildOptions();
161         CommandLine cl = null;
162         try {
163             CommandLineParser parser = new DefaultParser();
164             cl = parser.parse(options, args);
165         } catch (ParseException e) {
166             LOG.fatal(e);
167             printHelp();
168             System.exit(EX_USAGE.code);
169         }
170 
171         int exitcode = 0;
172         BasicDataSource dsStaging = null;
173         BasicDataSource dsRsgb = null;
174         BasicDataSource dsRsgbbgt = null;
175 
176         try {
177             dbProps.load(new FileInputStream(cl.getOptionValue("dbprops")));
178             switch (dbProps.getProperty("dbtype")) {
179                 case "oracle":
180                     Class.forName("oracle.jdbc.OracleDriver");
181                     break;
182                 case "postgis":
183                     Class.forName("org.postgresql.Driver");
184                     break;
185                 case "jtds-sqlserver":
186                     Class.forName("net.sourceforge.jtds.jdbc.Driver");
187                     break;
188                 default:
189                     throw new IllegalArgumentException("Het database type " + dbProps.getProperty("dbtype") + " wordt niet ondersteund of is niet opgegeven.");
190             }
191 
192             dsStaging = new BasicDataSource();
193             LOG.info("Verbinding maken met Staging database... ");
194             dsStaging.setUrl(dbProps.getProperty("staging.url"));
195             dsStaging.setUsername(dbProps.getProperty("staging.user"));
196             dsStaging.setPassword(dbProps.getProperty("staging.password"));
197             dsStaging.setConnectionProperties(dbProps.getProperty("staging.options", ""));
198 
199             // alleen rsgb verbinding maken als nodig
200             if (cl.hasOption("torsgb") || cl.hasOption("versieinfo")) {
201                 LOG.info("Verbinding maken met RSGB database... ");
202                 dsRsgb = new BasicDataSource();
203                 dsRsgb.setUrl(dbProps.getProperty("rsgb.url"));
204                 dsRsgb.setUsername(dbProps.getProperty("rsgb.user"));
205                 dsRsgb.setPassword(dbProps.getProperty("rsgb.password"));
206                 dsRsgb.setConnectionProperties(dbProps.getProperty("rsgb.options", ""));
207             }
208 
209             // alleen rsgbbgt verbinding maken als nodig
210             if (cl.hasOption("versieinfo") || cl.hasOption("torsgbbgt")) {
211                 LOG.info("Verbinding maken met RSGB BGT database... ");
212                 dsRsgbbgt = new BasicDataSource();
213                 dsRsgbbgt.setUrl(dbProps.getProperty("rsgbbgt.url"));
214                 dsRsgbbgt.setUsername(dbProps.getProperty("rsgbbgt.user"));
215                 dsRsgbbgt.setPassword(dbProps.getProperty("rsgbbgt.password"));
216                 dsRsgbbgt.setConnectionProperties(dbProps.getProperty("rsgbbgt.options", ""));
217             }
218 
219             // staging-only commando's
220             if (cl.hasOption("list")) {
221                 exitcode = list(dsStaging, cl.getOptionValue("l", "text"));
222             } else if (cl.hasOption("berichtstatus")) {
223                 exitcode = berichtStatus(dsStaging, cl.getOptionValue("berichtstatus", "text"));
224             } else if (cl.hasOption("jobstatus")) {
225                 exitcode = jobStatus(dsStaging, cl.getOptionValue("jobstatus", "text"));
226             } else if (cl.hasOption("load")) {
227                 // omdat we 2 verplichte argumenten hebben en 1 optionele die als String[]
228                 // worden doorgegeven een stream gebruike om eea aan mekaar te plakken
229                 exitcode = load(dsStaging, Stream.of(cl.getOptionValues("load"), cl.getArgs())
230                         .flatMap(Stream::of).toArray(String[]::new));
231             } else if (cl.hasOption("loaddir")) {
232                 exitcode = loaddir(dsStaging,
233                         Stream.of(cl.getOptionValues("loaddir"), cl.getArgs())
234                         .flatMap(Stream::of).toArray(String[]::new));
235             } else if (cl.hasOption("delete")) {
236                 exitcode = delete(dsStaging, cl.getOptionValue("delete"));
237             } else if (cl.hasOption("exportgds")) {
238                 exitcode = getMutations(dsStaging, cl.getOptionValues("exportgds"));
239             } // ----------------
240             // rsgb commando's
241             else if (cl.hasOption("torsgb")) {
242                 exitcode = toRsgb(dsStaging, dsRsgb, cl.getOptionValue("berichtstatus", "ignore"));
243             } // ----------------
244             // rsgbbgt commando's
245             else if (cl.hasOption("torsgbbgt")) {
246                 exitcode = toRsgbBgt(dsStaging, dsRsgbbgt, cl.getOptionValue("torsgbbgt", "false"));
247             } // ----------------
248             // alle schema's / databases
249             else if (cl.hasOption("versieinfo")) {
250                 exitcode = versieInfo(dsStaging, dsRsgb, dsRsgbbgt, cl.getOptionValue("versieinfo", "text"));
251             }
252         } catch (BrmoException | InterruptedException ex) {
253             LOG.error("Fout tijdens uitvoeren met argumenten: " + Arrays.toString(args), ex);
254             System.err.println(ex.getLocalizedMessage());
255             exitcode = 1;
256         } catch (ClassNotFoundException ex) {
257             LOG.error("Database driver is niet gevonden.", ex);
258             System.err.println(ex.getLocalizedMessage());
259             exitcode = EX_DATAERR.code;
260         } catch (IOException ex) {
261             LOG.error("Het laden van het configuratie bestand is mislukt.", ex);
262             System.err.println(ex.getLocalizedMessage());
263             exitcode = EX_NOINPUT.code;
264         }
265         try {
266             if (dsStaging != null) {
267                 dsStaging.close();
268             }
269             if (dsRsgb != null) {
270                 dsRsgb.close();
271             }
272             if (dsRsgbbgt != null) {
273                 dsRsgbbgt.close();
274             }
275         } catch (SQLException ex) {
276             LOG.debug("Mogelijke fout tijdens afsluiten database verbindingen.", ex);
277         }
278 
279         System.exit(exitcode);
280     }

282     private static int toRsgb(DataSource dataSourceStaging, DataSource dataSourceRsgb, String errorState) throws BrmoException, InterruptedException {
283         LOG.info("Start staging naar rsgb transformatie.");
284         BrmoFramework brmo = new BrmoFramework(dataSourceStaging, dataSourceRsgb);
285         brmo.setOrderBerichten(true);
286         brmo.setErrorState(errorState);
287         Thread t = brmo.toRsgb();
288         t.join();
289         LOG.info("Klaar met staging naar rsgb transformatie.");
290         brmo.closeBrmoFramework();
291         return 0;
292     }

294     private static int toRsgbBgt(BasicDataSource dsStaging, BasicDataSource dsRsgbbgt, String loadingUpdate) throws BrmoException, InterruptedException {
295         LOG.info("Start staging naar rsgbbgt transformatie.");
296         try {
297             GeoTools.init();
298             Logging.ALL.setLoggerFactory("org.geotools.util.logging.Log4JLoggerFactory");
299 
300             // Geotools maakt een eigen database verbinding via jndi
301             System.setProperty(Context.INITIAL_CONTEXT_FACTORY, "org.apache.naming.java.javaURLContextFactory");
302             System.setProperty(Context.URL_PKG_PREFIXES, "org.apache.naming");
303             InitialContext ic = new InitialContext();
304             ic.createSubcontext("java:");
305             ic.createSubcontext("java:comp");
306             ic.createSubcontext("java:comp/env");
307             ic.createSubcontext("java:comp/env/jdbc");
308             ic.createSubcontext("java:comp/env/jdbc/brmo");
309             ic.bind("java:comp/env/jdbc/brmo/rsgbbgt", dsRsgbbgt);
310         } catch (ClassNotFoundException | IllegalArgumentException | NamingException ex) {
311             LOG.debug("Er is iets misgegaan bij de GeoTools of JNDI initialisatie", ex);
312         }
313 
314         BrmoFramework brmo = new BrmoFramework(dsStaging, null, dsRsgbbgt);
315         // bepaal welke bgtlight LP's er zijn om te transformeren
316         final long[] lpIds = ArrayUtils.toPrimitive(brmo.getLaadProcessenIds("bestand_datum", "ASC", BrmoFramework.BR_BGTLIGHT, Bericht.STATUS.STAGING_OK.name()));
317         brmo.setOrderBerichten(loadingUpdate.equalsIgnoreCase("true"));
318         Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_LAADPROCES, lpIds, null);
319         t.join();
320         LOG.info("Klaar met staging naar rsgbbgt transformatie.");
321         brmo.closeBrmoFramework();
322         return 0;
323     }

452     private static int getMutations(DataSource ds, String... opts) {
453         LOG.info("Start export GDS2 berichten naar " + opts[0]);
454         Connection con = null;
455         String dir = opts[0];
456         int exitcode = 0;
457         try {
458             LOG.info("Ophalen automatisch proces(sen) waarmee GDS2 berichten zijn geladen.");
459             con = ds.getConnection();
460             Statement stmt = con.createStatement();
461             String autoProcessen = "select id FROM automatisch_proces where dtype = 'GDS2OphaalProces'";
462             ResultSet rs = stmt.executeQuery(autoProcessen);
463             while (rs.next()) {
464                 Long id = rs.getLong(1);
465                 LOG.info("Ophalen laadprocessen voor automatisch proces: " + id);
466                 String processen = "select id,bestand_naam from laadproces where automatisch_proces = " + id;
467                 Statement laadprocesStmt = con.createStatement();
468                 ResultSet lpRs = laadprocesStmt.executeQuery(processen);
469                 while (lpRs.next()) {
470                     Long lpId = lpRs.getLong(1);
471                     String bestandsNaam = lpRs.getString(2);
472                     String berichten = "select id, br_orgineel_xml, object_ref from bericht where laadprocesid = " + lpId;
473                     Statement berichtStmt = con.createStatement();
474                     ResultSet berRs = berichtStmt.executeQuery(berichten);
475                     while (berRs.next()) {
476                         LOG.info("Bericht " + id + " - " + bestandsNaam);
477                         String xml = berRs.getString("br_orgineel_xml");
478                         writeXML(xml, bestandsNaam, dir);
479                     }
480                 }
481             }
482         } catch (SQLException ex) {
483             LOG.error("Fout bij ophalen berichten/laadprocessen/automatische processen.", ex);
484             exitcode = EX_UNAVAILABLE.code;
485         } finally {
486             if (con != null) {
487                 try {
488                     con.close();
489                 } catch (SQLException ex) {
490                     LOG.warn("Database verbinding sluiten is mislukt.", ex);
491                 }
492             }
493         }
494         return exitcode;
495     }

497     private static void writeXML(String xml, String filename, String directory) {
498         ZipOutputStream out = null;
499         try {
500             final File f = new File(directory + "/" + filename + ".zip");
501             out = new ZipOutputStream(new FileOutputStream(f));
502             ZipEntry e = new ZipEntry(filename);
503             out.putNextEntry(e);
504             byte[] data = xml.getBytes();
505             out.write(data, 0, data.length);
506             out.closeEntry();
507             out.close();
508         } catch (IOException ex) {
509             LOG.error("Schrijven van bestand " + filename + " in " + directory + " is mislukt.", ex);
510         } finally {
511             if (out != null) {
512                 try {
513                     out.close();
514                 } catch (IOException ex) {
515                     LOG.warn("Sluiten .zip file is mislukt.", ex);
516                 }
517             }
518         }
519     }

521     /**
522      * verwerk xml of zip bestand naar staging db.
523      *
524      * @param ds datasource
525      * @param opts array met {vollidig pad naar bestand, type BR, optionele
526      * archief directory}
527      * @return {@code 0} als succesvol
528      * @throws BrmoException in geval van niet laadfout
529      */
530     private static int load(DataSource ds, String... opts) throws BrmoException {
531         String fileName = opts[0];
532         String brType = opts[1];
533         String archiefDir = opts.length == 3 ? opts[2] : null;
534 
535         LOG.debug(String.format("Begin laden van bestand: %s, type %s", fileName, brType));
536         BrmoFramework brmo = new BrmoFramework(ds, null);
537         brmo.setOrderBerichten(true);
538         brmo.setErrorState("ignore");
539         brmo.loadFromFile(brType, fileName);
540         brmo.closeBrmoFramework();
541         LOG.info(String.format("Klaar met laden van bestand: %s, type %s", fileName, brType));
542         archiveerBestand(fileName, archiefDir);
543         return 0;
544     }

546     /**
547      * verwerk xml en zip betsanden uit directory naar staging db.
548      *
549      * @param ds datasource
550      * @param opts array met {directory, type BR, optionele archief directory}
551      * @return {@code 0} als succesvol,{@code EX_DATAERR} in geval van een
552      * waarschuwing
553      * @throws BrmoException in geval van niet herstalebare laadfout
554      */
555     private static int loaddir(DataSource ds, String... opts) throws BrmoException {
556         String scanDir = opts[0];
557         String brType = opts[1];
558         String archiefDir = opts.length == 3 ? opts[2] : null;
559 
560         int exitcode = 0;
561         if (!scanDir.endsWith(File.separator)) {
562             scanDir += File.separator;
563         }
564         LOG.info(String.format("Begin laden van directory: %s, type %s", scanDir, brType));
565 
566         File dir = new File(scanDir);
567         if (dir.isDirectory()) {
568             boolean withWarnings = false;
569             String[] fNames = dir.list((File f, String name) -> {
570                 return (name.endsWith(".xml") || name.endsWith(".XML") || name.endsWith(".zip") || name.endsWith(".ZIP"));
571             });
572             BrmoFramework brmo = new BrmoFramework(ds, null);
573             brmo.setOrderBerichten(true);
574             brmo.setErrorState("ignore");
575 
576             for (String fName : fNames) {
577                 try {
578                     LOG.debug(String.format("Begin laden van bestand: %s, type %s", fName, brType));
579                     brmo.loadFromFile(brType, scanDir + fName);
580                     LOG.info(String.format("Klaar met laden van bestand: %s, type %s", fName, brType));
581                 } catch (BrmoDuplicaatLaadprocesException dup) {
582                     LOG.warn(String.format("Laden duplicaat bestand %s overgeslagen. Oorzaak: %s", fName, dup.getLocalizedMessage()));
583                     withWarnings = true;
584                 } catch (BrmoLeegBestandException leeg) {
585                     LOG.warn(String.format("Laden 'leeg' bestand %s overgeslagen. Oorzaak: %s", fName, leeg.getLocalizedMessage()));
586                     withWarnings = true;
587                 } finally {
588                     brmo.closeBrmoFramework();
589                 }
590                 archiveerBestand(scanDir + fName, archiefDir);
591             }
592             if (withWarnings) {
593                 exitcode = EX_DATAERR.code;
594             }
595         } else {
596             throw new BrmoException("Opgegeven directory " + scanDir + " is geen directory.");
597         }
598         LOG.info(String.format("Klaar met laden van directory: %s, type %s", scanDir, brType));
599         return exitcode;
600     }

602     /**
603      * verplaats bestand naar archief directory.
604      *
605      * @param fileName te verplaatsen bestandsnaam (volledig pad)
606      * @param archiefDir doel directory (in geval {@code null} dat wordt
607      * verplaatsen overgeslagen)
608      */
609     private static void archiveerBestand(final String fileName, final String archiefDir) {
610         if (archiefDir != null) {
611             File archiefDirectory = new File(archiefDir);
612             LOG.debug(String.format("Archiveren %s naar %s.", fileName, archiefDir));
613             try {
614                 FileUtils.moveFileToDirectory(new File(fileName), archiefDirectory, true);
615                 LOG.debug(String.format("Bestand %s is naar archief %s verplaatst.", fileName, archiefDirectory));
616             } catch (IOException e) {
617                 LOG.error(String.format("Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).", fileName, archiefDirectory, e.getLocalizedMessage()));
618             }
619         }
620     }
