201     private static void printHelp() {
202         HelpFormatter formatter = new HelpFormatter();
203         formatter.setOptionComparator(
204                 (lhs, rhs) -> {
205                     List[] lists = new List[] {modeOpts, dbOpts};
206                     for (List l : lists) {
207                         int lhsIndex = l.indexOf(lhs);
208                         if (lhsIndex != -1) {
209                             return Integer.compare(lhsIndex, l.indexOf(rhs));
210                         }
211                     }
212                     return lhs.getArgName().compareTo(rhs.getArgName());
213                 });
214         final int W = 100;
215         StringWriter sw = new StringWriter();
216         PrintWriter pw = new PrintWriter(sw);
217         formatter.printUsage(
218                 pw, 80, "java -jar brmo-commandline.jar --<actie> --dbprops <db-props>");
219         pw.print("\nActies:\n");
220         formatter.printOptions(pw, W, modeOptions, 2, 2);
221         pw.print("Configuratie:\n");
222         formatter.printOptions(pw, W, dbOptions, 2, 2);
223 
224         System.err.println(sw.toString());
225     }

227     public static void main(String[] args) {
228         Options options = buildOptions();
229         CommandLine cl = null;
230         try {
231             CommandLineParser parser = new DefaultParser();
232             cl = parser.parse(options, args);
233         } catch (ParseException e) {
234             LOG.fatal(e);
235             printHelp();
236             System.exit(EX_USAGE.code);
237         }
238 
239         int exitcode = 0;
240         BasicDataSource dsStaging = null;
241         BasicDataSource dsRsgb = null;
242         BasicDataSource dsRsgbBrk = null;
243 
244         try {
245             dbProps.load(new FileInputStream(cl.getOptionValue("dbprops")));
246             switch (dbProps.getProperty("dbtype")) {
247                 case "oracle":
248                     Class.forName("oracle.jdbc.OracleDriver");
249                     break;
250                 case "postgis":
251                     Class.forName("org.postgresql.Driver");
252                     break;
253                 default:
254                     throw new IllegalArgumentException(
255                             "Het database type "
256                                     + dbProps.getProperty("dbtype")
257                                     + " wordt niet ondersteund of is niet opgegeven.");
258             }
259 
260             dsStaging = new BasicDataSource();
261             LOG.info("Verbinding maken met Staging database... ");
262             dsStaging.setUrl(dbProps.getProperty("staging.url"));
263             dsStaging.setUsername(dbProps.getProperty("staging.user"));
264             dsStaging.setPassword(dbProps.getProperty("staging.password"));
265             dsStaging.setConnectionProperties(dbProps.getProperty("staging.options", ""));
266 
267             // alleen rsgb verbinding maken als nodig
268             if (cl.hasOption("torsgb") || cl.hasOption("versieinfo")) {
269                 LOG.info("Verbinding maken met RSGB database... ");
270                 dsRsgb = new BasicDataSource();
271                 dsRsgb.setUrl(dbProps.getProperty("rsgb.url"));
272                 dsRsgb.setUsername(dbProps.getProperty("rsgb.user"));
273                 dsRsgb.setPassword(dbProps.getProperty("rsgb.password"));
274                 dsRsgb.setConnectionProperties(dbProps.getProperty("rsgb.options", ""));
275 
276                 LOG.info("Verbinding maken met RSGB BRK database... ");
277                 dsRsgbBrk = new BasicDataSource();
278                 dsRsgbBrk.setUrl(dbProps.getProperty("rsgbbrk.url"));
279                 dsRsgbBrk.setUsername(dbProps.getProperty("rsgbbrk.user"));
280                 dsRsgbBrk.setPassword(dbProps.getProperty("rsgbbrk.password"));
281                 dsRsgbBrk.setConnectionProperties(dbProps.getProperty("rsgbbrk.options", ""));
282             }
283 
284             // staging-only commando's
285             if (cl.hasOption("list")) {
286                 exitcode = list(dsStaging, cl.getOptionValue("l", "text"));
287             } else if (cl.hasOption("berichtstatus")) {
288                 exitcode = berichtStatus(dsStaging, cl.getOptionValue("berichtstatus", "text"));
289             } else if (cl.hasOption("jobstatus")) {
290                 exitcode = jobStatus(dsStaging, cl.getOptionValue("jobstatus", "text"));
291             } else if (cl.hasOption("load")) {
292                 // omdat we 2 verplichte argumenten hebben en 1 optionele die als String[]
293                 // worden doorgegeven een stream gebruike om eea aan mekaar te plakken
294                 exitcode =
295                         load(
296                                 dsStaging,
297                                 Stream.of(cl.getOptionValues("load"), cl.getArgs())
298                                         .flatMap(Stream::of)
299                                         .toArray(String[]::new));
300             } else if (cl.hasOption("loaddir")) {
301                 exitcode =
302                         loaddir(
303                                 dsStaging,
304                                 Stream.of(cl.getOptionValues("loaddir"), cl.getArgs())
305                                         .flatMap(Stream::of)
306                                         .toArray(String[]::new));
307             } else if (cl.hasOption("delete")) {
308                 exitcode = delete(dsStaging, cl.getOptionValue("delete"));
309             } else if (cl.hasOption("exportgds")) {
310                 exitcode = getMutations(dsStaging, cl.getOptionValues("exportgds"));
311             } else if (cl.hasOption("afgiftelijst")) {
312                 String[] files =
313                         cl.getOptionValues("afgiftelijst"); // cl.getOptionValue("afgiftelijst")
314                 exitcode = checkAfgiftelijst(dsStaging, files[0], files[1]);
315             }
316             // ----------------
317             // rsgb commando's
318             else if (cl.hasOption("torsgb")) {
319                 exitcode =
320                         toRsgb(
321                                 dsStaging,
322                                 dsRsgb,
323                                 dsRsgbBrk,
324                                 cl.getOptionValue("berichtstatus", "ignore"));
325             } // ----------------
326             // alle schema's / databases
327             else if (cl.hasOption("versieinfo")) {
328                 exitcode =
329                         versieInfo(
330                                 dsStaging,
331                                 dsRsgb,
332                                 dsRsgbBrk,
333                                 cl.getOptionValue("versieinfo", "text"));
334             }
335         } catch (BrmoException | InterruptedException ex) {
336             LOG.error("Fout tijdens uitvoeren met argumenten: " + Arrays.toString(args), ex);
337             System.err.println(ex.getLocalizedMessage());
338             exitcode = 1;
339         } catch (ClassNotFoundException ex) {
340             LOG.error("Database driver is niet gevonden.", ex);
341             System.err.println(ex.getLocalizedMessage());
342             exitcode = EX_DATAERR.code;
343         } catch (IOException ex) {
344             LOG.error("Het laden van het configuratie bestand is mislukt.", ex);
345             System.err.println(ex.getLocalizedMessage());
346             exitcode = EX_NOINPUT.code;
347         }
348         try {
349             if (dsStaging != null) {
350                 dsStaging.close();
351             }
352             if (dsRsgb != null) {
353                 dsRsgb.close();
354             }
355         } catch (SQLException ex) {
356             LOG.debug("Mogelijke fout tijdens afsluiten database verbindingen.", ex);
357         }
358 
359         System.exit(exitcode);
360     }

362     private static int toRsgb(
363             DataSource dataSourceStaging,
364             DataSource dataSourceRsgb,
365             DataSource dataSourceRsgbBrk,
366             String errorState)
367             throws BrmoException, InterruptedException {
368         LOG.info("Start staging naar rsgb transformatie.");
369         BrmoFramework brmo =
370                 new BrmoFramework(dataSourceStaging, dataSourceRsgb, dataSourceRsgbBrk);
371         brmo.setOrderBerichten(true);
372         brmo.setErrorState(errorState);
373         Thread t = brmo.toRsgb();
374         t.join();
375         LOG.info("Klaar met staging naar rsgb transformatie.");
376         brmo.closeBrmoFramework();
377         return 0;
378     }

478             File f = new File(input);
479

487             System.out.print("Afgifte gecontroleerd:");

491             System.err.println("Error reading afgiftelijst: " + ex.getLocalizedMessage());
492             brmo.closeBrmoFramework();
493             return 1;

561     private static int getMutations(DataSource ds, String... opts) {
562         LOG.info("Start export GDS2 berichten naar " + opts[0]);
563         Connection con = null;
564         String dir = opts[0];
565         int exitcode = 0;
566         try {
567             LOG.info("Ophalen automatisch proces(sen) waarmee GDS2 berichten zijn geladen.");
568             con = ds.getConnection();
569             Statement stmt = con.createStatement();
570             String autoProcessen =
571                     "select id FROM automatisch_proces where dtype = 'GDS2OphaalProces'";
572             ResultSet rs = stmt.executeQuery(autoProcessen);
573             while (rs.next()) {
574                 Long id = rs.getLong(1);
575                 LOG.info("Ophalen laadprocessen voor automatisch proces: " + id);
576                 String processen =
577                         "select id,bestand_naam from laadproces where automatisch_proces = " + id;
578                 Statement laadprocesStmt = con.createStatement();
579                 ResultSet lpRs = laadprocesStmt.executeQuery(processen);
580                 while (lpRs.next()) {
581                     Long lpId = lpRs.getLong(1);
582                     String bestandsNaam = lpRs.getString(2);
583                     String berichten =
584                             "select id, br_orgineel_xml, object_ref from bericht where laadprocesid = "
585                                     + lpId;
586                     Statement berichtStmt = con.createStatement();
587                     ResultSet berRs = berichtStmt.executeQuery(berichten);
588                     while (berRs.next()) {
589                         LOG.info("Bericht " + id + " - " + bestandsNaam);
590                         String xml = berRs.getString("br_orgineel_xml");
591                         writeXML(xml, bestandsNaam, dir);
592                     }
593                 }
594             }
595         } catch (SQLException ex) {
596             LOG.error("Fout bij ophalen berichten/laadprocessen/automatische processen.", ex);
597             exitcode = EX_UNAVAILABLE.code;
598         } finally {
599             if (con != null) {
600                 try {
601                     con.close();
602                 } catch (SQLException ex) {
603                     LOG.warn("Database verbinding sluiten is mislukt.", ex);
604                 }
605             }
606         }
607         return exitcode;
608     }

610     private static void writeXML(String xml, String filename, String directory) {
611         ZipOutputStream out = null;
612         try {
613             filename = Paths.get(filename + ".zip").normalize().toString();
614             final File f = new File(directory, filename);
615             out = new ZipOutputStream(new FileOutputStream(f));
616             ZipEntry e = new ZipEntry(filename);
617             out.putNextEntry(e);
618             byte[] data = xml.getBytes();
619             out.write(data, 0, data.length);
620             out.closeEntry();
621             out.close();
622         } catch (IOException ex) {
623             LOG.error(
624                     "Schrijven van bestand " + filename + " in " + directory + " is mislukt.", ex);
625         } finally {
626             if (out != null) {
627                 try {
628                     out.close();
629                 } catch (IOException ex) {
630                     LOG.warn("Sluiten .zip file is mislukt.", ex);
631                 }
632             }
633         }
634     }

636     /**
637      * verwerk xml of zip bestand naar staging db.
638      *
639      * @param ds datasource
640      * @param opts array met {vollidig pad naar bestand, type BR, optionele archief directory}
641      * @return {@code 0} als succesvol
642      * @throws BrmoException in geval van niet laadfout
643      */
644     private static int load(DataSource ds, String... opts) throws BrmoException {
645         String fileName = opts[0];
646         String brType = opts[1];
647         String archiefDir = opts.length == 3 ? opts[2] : null;
648 
649         LOG.debug(String.format("Begin laden van bestand: %s, type %s", fileName, brType));
650         BrmoFramework brmo = new BrmoFramework(ds, null, null);
651         brmo.setOrderBerichten(true);
652         brmo.setErrorState("ignore");
653         brmo.loadFromFile(brType, fileName, null);
654         brmo.closeBrmoFramework();
655         LOG.info(String.format("Klaar met laden van bestand: %s, type %s", fileName, brType));
656         archiveerBestand(fileName, archiefDir);
657         return 0;
658     }

660     /**
661      * verwerk xml en zip bestanden uit directory naar staging db.
662      *
663      * @param ds datasource
664      * @param opts array met {directory, type BR, optionele archief directory}
665      * @return {@code 0} als succesvol,{@code EX_DATAERR} in geval van een waarschuwing
666      * @throws BrmoException in geval van niet herstalebare laadfout
667      */
668     private static int loaddir(DataSource ds, String... opts) throws BrmoException {
669         String scanDir = opts[0];
670         String brType = opts[1];
671         String archiefDir = opts.length == 3 ? opts[2] : null;
672 
673         int exitcode = 0;
674         if (!scanDir.endsWith(File.separator)) {
675             scanDir += File.separator;
676         }
677         LOG.info(String.format("Begin laden van directory: %s, type %s", scanDir, brType));
678 
679         File dir = new File(scanDir);
680         if (dir.isDirectory()) {
681             boolean withWarnings = false;
682             String[] fNames =
683                     dir.list(
684                             (File f, String name) -> {
685                                 return (name.endsWith(".xml")
686                                         || name.endsWith(".XML")
687                                         || name.endsWith(".zip")
688                                         || name.endsWith(".ZIP"));
689                             });
690             BrmoFramework brmo = new BrmoFramework(ds, null, null);
691             brmo.setOrderBerichten(true);
692             brmo.setErrorState("ignore");
693 
694             for (String fName : fNames) {
695                 try {
696                     LOG.debug(String.format("Begin laden van bestand: %s, type %s", fName, brType));
697                     brmo.loadFromFile(brType, scanDir + fName, null);
698                     LOG.info(
699                             String.format(
700                                     "Klaar met laden van bestand: %s, type %s", fName, brType));
701                 } catch (BrmoDuplicaatLaadprocesException dup) {
702                     LOG.warn(
703                             String.format(
704                                     "Laden duplicaat bestand %s overgeslagen. Oorzaak: %s",
705                                     fName, dup.getLocalizedMessage()));
706                     withWarnings = true;
707                 } catch (BrmoLeegBestandException leeg) {
708                     LOG.warn(
709                             String.format(
710                                     "Laden 'leeg' bestand %s overgeslagen. Oorzaak: %s",
711                                     fName, leeg.getLocalizedMessage()));
712                     withWarnings = true;
713                 } finally {
714                     brmo.closeBrmoFramework();
715                 }
716                 archiveerBestand(scanDir + fName, archiefDir);
717             }
718             if (withWarnings) {
719                 exitcode = EX_DATAERR.code;
720             }
721         } else {
722             throw new BrmoException("Opgegeven directory " + scanDir + " is geen directory.");
723         }
724         LOG.info(String.format("Klaar met laden van directory: %s, type %s", scanDir, brType));
725         return exitcode;
726     }

728     /**
729      * verplaats bestand naar archief directory.
730      *
731      * @param fileName te verplaatsen bestandsnaam (volledig pad)
732      * @param archiefDir doel directory (in geval {@code null} dat wordt verplaatsen overgeslagen)
733      */
734     private static void archiveerBestand(final String fileName, final String archiefDir) {
735         if (archiefDir != null) {
736             File archiefDirectory = new File(archiefDir);
737             LOG.debug(String.format("Archiveren %s naar %s.", fileName, archiefDir));
738             try {
739                 FileUtils.moveFileToDirectory(new File(fileName), archiefDirectory, true);
740                 LOG.debug(
741                         String.format(
742                                 "Bestand %s is naar archief %s verplaatst.",
743                                 fileName, archiefDirectory));
744             } catch (IOException e) {
745                 LOG.error(
746                         String.format(
747                                 "Bestand %s is NIET naar archief %s verplaatst, oorzaak: (%s).",
748                                 fileName, archiefDirectory, e.getLocalizedMessage()));
749             }
750         }
751     }
