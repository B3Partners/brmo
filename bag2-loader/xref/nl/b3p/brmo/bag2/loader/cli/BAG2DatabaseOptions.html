8   package nl.b3p.brmo.bag2.loader.cli;
9   
10  import picocli.CommandLine.*;
11  
12  public class BAG2DatabaseOptions {
13    @Option(
14        names = {"-c", "--connection"},
15        paramLabel = "<string>")
16    String connectionString = "jdbc:postgresql:bag?sslmode=disable&reWriteBatchedInserts=true";
17  
18    @Option(names = {"-u", "--user"})
19    String user = "bag";
20  
21    @Option(
22        names = {"-p", "--password"},
23        interactive = true,
24        arity = "0..1")
25    String password = "bag";
26  
27    @Option(names = "--batch-size", paramLabel = "number", hidden = true)
28    Integer batchSize;
29  
30    @Option(
31        names = {"--no-pg-copy"},
32        negatable = true,
33        hidden = true)
34    boolean usePgCopy = true;
35  
36    public String getConnectionString() {
37      return connectionString;
38    }
39  
40    public void setConnectionString(String connectionString) {
41      this.connectionString = connectionString;
42    }
43  
44    public String getUser() {
45      return user;
46    }
47  
48    public void setUser(String user) {
49      this.user = user;
50    }
51  
52    public String getPassword() {
53      return password;
54    }
55  
56    public void setPassword(String password) {
57      this.password = password;
58    }
59  
60    public Integer getBatchSize() {
61      return batchSize;
62    }
63  
64    public void setBatchSize(Integer batchSize) {
65      this.batchSize = batchSize;
66    }
67  
68    public boolean isUsePgCopy() {
69      return usePgCopy;
70    }
71  
72    public void setUsePgCopy(boolean usePgCopy) {
73      this.usePgCopy = usePgCopy;
74    }
75  }
