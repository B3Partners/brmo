70    @Command(name = "download", sortOptions = false)
71    public int download(
72        @Option(names = "--no-delete", negatable = true) boolean noDelete,
73        @Option(names = "--kadaster-user") String kadasterUser,
74        @Option(names = "--kadaster-password") String kadasterPassword,
75        @Option(names = "--url", defaultValue = LVBAG_BESTANDEN_API_URL) String url,
76        @Option(names = "--query-params", defaultValue = "artikelnummers=2529") String queryParams,
77        @Option(names = "--path", defaultValue = "") String downloadPath,
78        @Option(names = "--mirror-base-url") String mirrorBaseUrl,
79        @Option(
80                names = {"-h", "--help"},
81                usageHelp = true)
82            boolean showHelp)
83        throws Exception {
84  
85      log.info(BAG2LoaderUtils.getUserAgent());
86      Instant start = Instant.now();
87  
88      CookieManager kadasterCookieManager = new CookieManager(null, CookiePolicy.ACCEPT_ALL);
89      mijnKadasterLogin(URI.create(url), kadasterUser, kadasterPassword, kadasterCookieManager);
90  
91      JSONArray bestanden = getBagBestanden(url, queryParams, kadasterCookieManager);
92  
93      log.info("Aantal beschikbare bestanden: " + bestanden.length());
94  
95      List<JSONObject> toDownload = new ArrayList<>();
96      for (int i = 0; i < bestanden.length(); i++) {
97        JSONObject bestand = bestanden.getJSONObject(i);
98  
99        File f = Path.of(downloadPath, bestand.getString("naam")).toFile();
100       if (!f.exists() || f.length() != bestand.getLong("grootte")) {
101         toDownload.add(bestand);
102       }
103     }
104 
105     if (!noDelete) {
106       deleteZipFilesNotInBestanden(downloadPath, bestanden);
107     }
108 
109     long totalBytes =
110         toDownload.stream().map(bestand -> bestand.getLong("grootte")).reduce(Long::sum).orElse(0L);
111     log.info(
112         String.format(
113             "Aantal te downloaden naar directory \"%s\": %d bestanden (%s)",
114             Path.of(downloadPath).toAbsolutePath(),
115             toDownload.size(),
116             byteCountToDisplaySize(totalBytes)));
117 
118     int count = 0;
119     long bytesRead = 0;
120     for (JSONObject bestand : toDownload) {
121       url = bestand.getString("url");
122       String name = bestand.getString("naam");
123 
124       try (ResumingInputStream input =
125               new ResumingInputStream(
126                   new HttpStartRangeInputStreamProvider(
127                       URI.create(url),
128                       new Java11HttpClientWrapper(
129                           HttpClient.newBuilder().cookieHandler(kadasterCookieManager))));
130           OutputStream out = new FileOutputStream(Path.of(downloadPath, name).toFile())) {
131 
132         log.info(
133             String.format(
134                 "Bestand %2d/%d (%.1f%%): downloaden %s...",
135                 ++count, toDownload.size(), (100.0 / totalBytes) * bytesRead, name));
136         IOUtils.copyLarge(input, out);
137       }
138       bytesRead += bestand.getLong("grootte");
139     }
140     if (!toDownload.isEmpty()) {
141       String msg = "";
142       if (mirrorBaseUrl != null) {
143         for (int i = 0; i < bestanden.length(); i++) {
144           JSONObject bestand = bestanden.getJSONObject(i);
145           String name = bestand.getString("naam");
146           bestand.put("url", URI.create(mirrorBaseUrl).resolve(name));
147         }
148         File bestandenJSONMirror = Path.of(downloadPath, "bestanden.json").toFile();
149         try (OutputStream out = new FileOutputStream(bestandenJSONMirror)) {
150           IOUtils.write(bestanden.toString(2), out, StandardCharsets.UTF_8);
151           msg =
152               String.format(
153                   ", JSON voor mirror \"%s\" geschreven naar \"%s\"",
154                   mirrorBaseUrl, bestandenJSONMirror);
155         }
156       }
157       log.info("Alle bestanden gedownload in " + formatTimeSince(start) + msg);
158     }
159 
160     return ExitCode.OK;
161   }

163   @Command(name = "apply", sortOptions = false)
164   public int apply(
165       @Mixin BAG2DatabaseOptions dbOptions,
166       @Mixin BAG2ProgressOptions progressOptions,
167       @Option(names = "--kadaster-user") String kadasterUser,
168       @Option(names = "--kadaster-password") String kadasterPassword,
169       @Option(names = "--url", defaultValue = LVBAG_BESTANDEN_API_URL) String url,
170       @Option(names = "--query-params", defaultValue = "artikelnummers=2529") String queryParams,
171       @Option(
172               names = {"-h", "--help"},
173               usageHelp = true)
174           boolean showHelp)
175       throws Exception {
176 
177     log.info(BAG2LoaderUtils.getUserAgent());
178     Instant start = Instant.now();
179 
180     CookieManager kadasterCookieManager = new CookieManager(null, CookiePolicy.ACCEPT_ALL);
181     mijnKadasterLogin(URI.create(url), kadasterUser, kadasterPassword, kadasterCookieManager);
182 
183     JSONArray bestanden = getBagBestanden(url, queryParams, kadasterCookieManager);
184 
185     log.info("Aantal beschikbare bestanden: " + bestanden.length());
186 
187     List<String> urls = new ArrayList<>();
188     for (int i = 0; i < bestanden.length(); i++) {
189       JSONObject bestand = bestanden.getJSONObject(i);
190       urls.add(bestand.getString("url"));
191     }
192 
193     try (BAG2Database db = parent.getBAG2Database(dbOptions)) {
194       BAG2ProgressReporter progressReporter =
195           progressOptions.isConsoleProgressEnabled()
196               ? new BAG2ConsoleProgressReporter()
197               : new BAG2ProgressReporter();
198 
199       parent.applyMutaties(
200           db,
201           dbOptions,
202           new BAG2LoadOptions(),
203           progressReporter,
204           urls.toArray(String[]::new),
205           kadasterCookieManager);
206       log.info("Alle mutatiebestanden verwerkt in " + formatTimeSince(start));
207       return ExitCode.OK;
208     }
209   }

211   private static void mijnKadasterLogin(
212       URI forUri, String username, String password, CookieManager cookieManager)
213       throws IOException, InterruptedException {
214     if (!forUri.getHost().endsWith("kadaster.nl")) {
215       return;
216     }
217     if (username == null || password == null) {
218       throw new IllegalArgumentException("Gebruikersnaam en wachtwoord zijn verplicht");
219     }
220 
221     HttpRequest request =
222         HttpRequest.newBuilder()
223             .uri(URI.create(KADASTER_LOGIN_URL))
224             .header("Content-Type", "application/x-www-form-urlencoded")
225             .POST(
226                 HttpRequest.BodyPublishers.ofString(
227                     String.format(
228                         "user=%s&password=%s",
229                         URLEncoder.encode(username, StandardCharsets.UTF_8),
230                         URLEncoder.encode(password, StandardCharsets.UTF_8))))
231             .build();
232 
233     HttpClient httpClient = HttpClient.newBuilder().cookieHandler(cookieManager).build();
234 
235     HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
236 
237     if (response.statusCode() != 302) {
238       throw new IllegalArgumentException(
239           String.format("Fout bij inloggen op Mijn Kadaster met gebruikersnaam \"%s\"", username));
240     }
241 
242     Map<String, List<String>> cookies =
243         cookieManager.get(URI.create(KADASTER_LOGIN_URL), new HashMap<>());
244 
245     Optional<String> kadasterTicketIdCookie =
246         cookies.getOrDefault("Cookie", List.of()).stream()
247             .filter(c -> c.startsWith("KadasterTicketId="))
248             .findFirst();
249 
250     if (kadasterTicketIdCookie.isEmpty()) {
251       throw new IllegalArgumentException("Geen KadasterTicketId cookie ontvangen na inloggen");
252     }
253   }

255   private static JSONArray getBagBestanden(
256       String url, String queryParams, CookieManager kadasterCookieManager) throws Exception {
257     url = url + "?" + queryParams;
258     log.info("Opvragen bestanden JSON vanaf URL " + url);
259     HttpRequest request = HttpRequest.newBuilder().uri(URI.create(url)).build();
260     HttpClient httpClient = HttpClient.newBuilder().cookieHandler(kadasterCookieManager).build();
261     HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
262 
263     JSONArray bestanden;
264     try {
265       bestanden = new JSONArray(response.body());
266     } catch (Exception e) {
267       throw new Exception("Fout bij parsen BAG bestanden JSON, body: " + response.body(), e);
268     }
269     return bestanden;
270   }

272   private static void deleteZipFilesNotInBestanden(String downloadPath, JSONArray bestanden)
273       throws IOException {
274     final Set<String> names = new HashSet<>();
275     for (int i = 0; i < bestanden.length(); i++) {
276       names.add(bestanden.getJSONObject(i).getString("naam"));
277     }
278     try (Stream<Path> stream = Files.list(Path.of(downloadPath))) {
279       stream
280           .filter(
281               p ->
282                   !Files.isDirectory(p)
283                       && p.getFileName().toString().endsWith(".zip")
284                       && !names.contains(p.getFileName().toString()))
285           .forEach(
286               p -> {
287                 log.info("Verwijderen ZIP bestand niet in bestandenlijst: " + p.getFileName());
288                 try {
289                   Files.delete(p);
290                 } catch (Exception e) {
291                   log.error(
292                       String.format("Fout bij verwijderen ZIP bestand %s: %s", p.getFileName(), e));
293                 }
294               });
295     }
