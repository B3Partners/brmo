8   package nl.b3p.brmo.bag2.loader;
9   
10  import nl.b3p.brmo.bag2.schema.BAG2Object;
11  
12  import javax.xml.stream.Location;
13  
14  public class BAG2ToevoegingMutatie extends BAG2Mutatie {
15      private BAG2Object toevoeging;
16  
17      BAG2ToevoegingMutatie(Location location, BAG2Object toevoeging) {
18          super(location);
19          this.toevoeging = toevoeging;
20      }
21  
22      public BAG2Object getToevoeging() {
23          return toevoeging;
24      }
25  
26      @Override
27      public String toString() {
28          return "BAG2ToevoegingMutatie{" +
29                  "location=" + getLocation() +
30                  ", toevoeging=" + toevoeging +
31                  '}';
32      }
33  }
