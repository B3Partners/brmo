133     public BAG2GMLMutatieGroepStream(InputStream in) throws XMLStreamException {
134         this.cursor = initCursor(buildSMInputFactory().rootElementCursor(in));
135         this.geometryReader = buildGeometryReader();
136     }
137 
138     protected SMInputFactory buildSMInputFactory() {
139         // Using alternative StAX parsers explicitly:
140         // final XMLInputFactory stax = new WstxInputFactory(); // Woodstox
141         // final XMLInputFactory stax = new com.fasterxml.aalto.stax.InputFactoryImpl(); // Aalto
142         final XMLInputFactory xmlInputFactory = XMLInputFactory.newFactory(); // JRE Default, depends on JAR's present or javax.xml.stream.XMLInputFactory property, can be SJSXP
143         log.trace("StAX XMLInputFactory: " + xmlInputFactory.getClass().getName());
144 
145         xmlInputFactory.setProperty(XMLInputFactory.IS_COALESCING, Boolean.TRUE); // Coalesce characters
146         xmlInputFactory.setProperty(XMLInputFactory.SUPPORT_DTD, Boolean.FALSE); // No XML entity expansions or external entities
147 
148         return new SMInputFactory(xmlInputFactory);
149     }
150 
151     protected XmlStreamGeometryReader buildGeometryReader() {
152         GeometryFactory geometryFactory = new GeometryFactory(new PrecisionModel(), 28992, new Force2DCoordinateSequenceFactory());
153         return new XmlStreamGeometryReader(this.cursor.getStreamReader(), geometryFactory);
154     }

156     private SMInputCursor initCursor(SMInputCursor cursor) throws XMLStreamException {
157         QName root = cursor.advance().getQName();
158 
159         if(root.equals(BAG_STAND)) {
160             isMutaties = false;
161         } else if (root.equals(BAG_MUTATIES)) {
162             isMutaties = true;
163         } else {
164             throw new IllegalArgumentException("XML root element moet bagStand of bagMutaties zijn");
165         }
166 
167         SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
168         Date standTechnischeDatum = null;
169         if(isMutaties) {
170             cursor = cursor.childElementCursor().advance();
171             cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_BAG_EXTRACT_MUTATIES, "bagInfo");
172 
173             SMInputCursor bagInfoCursor = cursor.descendantElementCursor().advance();
174             Date mutatieDatumVanaf = null, mutatieDatumTot = null;
175             do {
176                 try {
177                     switch (bagInfoCursor.getLocalName()) {
178                         case "StandTechnischeDatum":
179                             standTechnischeDatum = df.parse(bagInfoCursor.collectDescendantText());
180                             break;
181                         case "MutatiedatumVanaf":
182                             mutatieDatumVanaf = df.parse(bagInfoCursor.collectDescendantText());
183                             break;
184                         case "MutatiedatumTot":
185                             mutatieDatumTot = df.parse(bagInfoCursor.collectDescendantText());
186                             break;
187                     }
188                 } catch(ParseException ignored) {
189                 }
190             } while(bagInfoCursor.getNext() != null);
191             bagInfo = new BagInfo(standTechnischeDatum, mutatieDatumVanaf, mutatieDatumTot);
192 
193             cursor.getNext();
194             cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_MUTATIELEVERING, "mutatieBericht");
195 
196             cursor = cursor.childElementCursor(new QName(NS_MUTATIELEVERING, "mutatieGroep"));
197 
198             hasMutatieGroep = cursor.getNext() != null;
199         } else {
200             cursor = cursor.childElementCursor().advance();
201             cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_BAG_EXTRACT, "bagInfo");
202 
203             SMInputCursor bagInfoCursor = cursor.descendantElementCursor().advance();
204             String gemeenteIdentificatie = null;
205             do {
206                 switch(bagInfoCursor.getLocalName()) {
207                     case "StandTechnischeDatum":
208                         try {
209                             standTechnischeDatum = df.parse(bagInfoCursor.collectDescendantText());
210                         } catch(ParseException ignored) {
211                         }
212                         break;
213                     case "GemeenteIdentificatie":
214                         if (gemeenteIdentificatie != null) {
215                             throw new IllegalArgumentException("Alleen een enkele GemeenteIdentificatie in een GemeenteCollectie wordt ondersteund");
216                         }
217                         gemeenteIdentificatie = bagInfoCursor.collectDescendantText();
218                         break;
219                     case "Gebied-NLD":
220                         // "9999" is code for entire NL area
221                         gemeenteIdentificatie = "9999";
222                         break;
223                 }
224             } while(bagInfoCursor.getNext() != null);
225 
226             bagInfo = new BagInfo(standTechnischeDatum, null, null, gemeenteIdentificatie);
227 
228             cursor.getNext();
229             cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_STANDLEVERING, "standBestand");
230 
231             cursor = cursor.childElementCursor(new QName(NS_STANDLEVERING, "stand"));
232         }
233 
234         return cursor;
235     }

241     @Override
242     public Iterator<BAG2MutatieGroep> iterator() {
243         return new Iterator<>() {
244             SMEvent event = cursor.getCurrEvent();
245 
246             @Override
247             public boolean hasNext() {
248                 if (isMutaties && !hasMutatieGroep) {
249                     return false;
250                 }
251 
252                 if (event != null) {
253                     return true;
254                 }
255                 try {
256                     event = cursor.getNext();
257                     return event != null;
258                 } catch(XMLStreamException e) {
259                     throw new RuntimeException(e);
260                 }
261             }
262 
263             @Override
264             public BAG2MutatieGroep next() {
265                 if (event == null) {
266                     if(!hasNext()) {
267                         throw new IllegalStateException("No more items");
268                     }
269                 }
270                 // Make sure cursor.getNext() is called in a future next() call
271                 event = null;
272 
273                 try {
274                     if (isMutaties) {
275                         if (!hasMutatieGroep) {
276                             throw new IllegalStateException("No items");
277                         }
278 
279                         cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_MUTATIELEVERING, "mutatieGroep");
280                         SMInputCursor mutatieCursor = cursor.childElementCursor().advance();
281                         List<BAG2Mutatie> mutaties = new ArrayList<>();
282                         do {
283                             String mutatieNaam = mutatieCursor.getLocalName();
284 
285                             Location location = mutatieCursor.getCursorLocation();
286                             if (mutatieNaam.equals("wijziging")) {
287                                 SMInputCursor wijziging = mutatieCursor.childElementCursor().advance();
288                                 wijziging.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_MUTATIELEVERING, "was");
289                                 BAG2Object was = parseBAG2ObjectFromBagObjectParentElement(wijziging, NS_BAG_EXTRACT_MUTATIES);
290                                 wijziging.getNext();
291                                 wijziging.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_MUTATIELEVERING, "wordt");
292                                 BAG2Object wordt = parseBAG2ObjectFromBagObjectParentElement(wijziging, NS_BAG_EXTRACT_MUTATIES);
293 
294                                 mutaties.add(new BAG2WijzigingMutatie(location, was, wordt));
295                             } else if (mutatieNaam.equals("toevoeging")) {
296                                 SMInputCursor wijziging = mutatieCursor.childElementCursor().advance();
297                                 wijziging.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_MUTATIELEVERING, "wordt");
298                                 BAG2Object toevoeging = parseBAG2ObjectFromBagObjectParentElement(wijziging, NS_BAG_EXTRACT_MUTATIES);
299                                 mutaties.add(new BAG2ToevoegingMutatie(location, toevoeging));
300 
301                             } else if (mutatieNaam.equals("verwijdering")) {
302                                 throw new IllegalArgumentException("Verwijdering-mutaties mogen niet voorkomen in de BAG2");
303                             } else {
304                                 throw new IllegalArgumentException("Onbekende mutatie: " + mutatieNaam);
305                             }
306                         } while(mutatieCursor.getNext() == SMEvent.START_ELEMENT);
307 
308                         return new BAG2MutatieGroep(mutaties);
309                     } else {
310                         cursor.getStreamReader().require(XMLStreamConstants.START_ELEMENT, NS_STANDLEVERING, "stand");
311                         Location location = cursor.getCursorLocation();
312                         BAG2Object object = parseBAG2ObjectFromBagObjectParentElement(cursor, NS_BAG_EXTRACT);
313                         return new BAG2MutatieGroep(List.of(new BAG2ToevoegingMutatie(location, object)));
314                     }
315                 } catch(Exception e) {
316                     throw new RuntimeException(e);
317                 }
318             }
319         };
320     }

322     private BAG2Object parseBAG2ObjectFromBagObjectParentElement(SMInputCursor bagObjectParentCursor, String bagObjectNamespace) throws XMLStreamException, FactoryException, IOException {
323         SMInputCursor bagObjectCursor = bagObjectParentCursor
324                 .childElementCursor(new QName(bagObjectNamespace, "bagObject")).advance()
325                 .childElementCursor().advance();
326         return parseBAG2Object(bagObjectCursor);
327     }

329     private BAG2Object parseBAG2Object(SMInputCursor bagObjectCursor) throws XMLStreamException, FactoryException, IOException {
330         String name = bagObjectCursor.getLocalName();
331 
332         final BAG2ObjectType objectType = BAG2Schema.getInstance().getObjectTypeByName(name);
333         if (objectType == null) {
334             throw new IllegalArgumentException("Onbekend object type: " + name);
335         }
336 
337         Map<String, Object> attributes = new HashMap<>();
338         SMInputCursor attributeCursor = bagObjectCursor.childElementCursor();
339         while (attributeCursor.getNext() != null) {
340             parseAttribute(attributeCursor, attributes);
341         }
342         return new BAG2Object(objectType, attributes);
343     }

345     private void parseAttribute(SMInputCursor attribute, Map<String, Object> objectAttributes) throws XMLStreamException, FactoryException, IOException {
346         String attributeName = attribute.getLocalName();
347 
348         switch (attributeName) {
349             case "geometrie":
350                 // Position cursor at child element
351                 SMInputCursor geomCursor = attribute.childElementCursor().advance();
352                 // Sometimes there is an element like "punt" which has the actual geometry as child element
353                 if (!geomCursor.getNsUri().equals(NS_GML_32)) {
354                     geomCursor.childElementCursor().advance();
355                 }
356                 Geometry geom = geometryReader.readGeometry();
357                 geom.setSRID(SRID);
358                 objectAttributes.put(attributeName, geom);
359                 break;
360             case "voorkomen":
361                 // Flatten al Voorkomen child attributes, according to the schema the element names do not conflict
362                 parseVoorkomen(attribute, objectAttributes);
363                 break;
364             case "BeschikbaarLV":
365                 // Flatten al Voorkomen/BeschikbaarLV child attributes to the top level, according to the schema the element
366                 // names do not conflict
367                 parseBeschikbaarLV(attribute, objectAttributes);
368                 break;
369             case "heeftAlsNevenadres":
370                 parseNevenadres(attribute, objectAttributes);
371                 break;
372             case "maaktDeelUitVan":
373                 parseMaaktDeelUitVan(attribute, objectAttributes);
374                 break;
375             case "gebruiksdoel":
376                 parseGebruiksdoel(attribute, objectAttributes);
377                 break;
378             default:
379                 // String attribute value as default
380 
381                 // This also works for ligtIn en ligtAan
382                 objectAttributes.put(attributeName, attribute.collectDescendantText().trim());
383                 break;
384         }
385     }

387     private void parseVoorkomen(SMInputCursor attributeCursor, Map<String, Object> objectAttributes) throws XMLStreamException, FactoryException, IOException {
388         attributeCursor = attributeCursor.childElementCursor(new QName(NS_HISTORIE, "Voorkomen")).advance().childElementCursor();
389         while (attributeCursor.getNext() != null) {
390             parseAttribute(attributeCursor, objectAttributes);
391         }
392     }

394     private void parseBeschikbaarLV(SMInputCursor attributeCursor, Map<String, Object> objectAttributes) throws XMLStreamException, FactoryException, IOException {
395         attributeCursor = attributeCursor.childElementCursor();
396         while (attributeCursor.getNext() != null) {
397             parseAttribute(attributeCursor, objectAttributes);
398         }
399     }

401     private void parseNevenadres(SMInputCursor attributeCursor, Map<String, Object> objectAttributes) throws XMLStreamException {
402         Set<String> values = new HashSet<>();
403         attributeCursor = attributeCursor.childElementCursor(new QName(NS_OBJECTEN_REF, "NummeraanduidingRef"));
404         while (attributeCursor.getNext() != null) {
405             values.add(attributeCursor.collectDescendantText().trim());
406         }
407         objectAttributes.put("heeftAlsNevenadres", values);
408     }

412         attributeCursor = attributeCursor.childElementCursor(new QName(NS_OBJECTEN_REF, "PandRef"));
413         while (attributeCursor.getNext() != null) {

417     }
418 
419     private void parseGebruiksdoel(SMInputCursor attributeCursor, Map<String, Object> objectAttributes) throws XMLStreamException {
420         Set<String> values = (Set<String>) objectAttributes.get("gebruiksdoel");
421         if (values == null) {
422             values = new HashSet<>();
423             objectAttributes.put("gebruiksdoel", values);
424         }
425         values.add(attributeCursor.collectDescendantText().trim());
426     }
