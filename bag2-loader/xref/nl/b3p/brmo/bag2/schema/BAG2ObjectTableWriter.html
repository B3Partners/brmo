107     private void deletePreviousVersion(BAG2Object object) throws Exception {
108         BAG2ObjectType objectType = object.getObjectType();
109         Map<ObjectType,QueryBatch> deleteBatches = getProgress().deleteBatches;
110         if (!deleteBatches.containsKey(objectType)) {
111             String args = objectType.getPrimaryKeys().stream()
112                     .map(k -> getSchemaSQLMapper().getColumnNameForObjectType(objectType, k.getName()) + " = ?")
113                     .collect(Collectors.joining(" and "));
114             // Array attributes are deleted because of 'on delete cascade' on the foreign key
115             String sql = String.format("delete from %s where %s", getSchemaSQLMapper().getTableNameForObjectType(objectType, getTablePrefix()), args);
116             // Set batch size to 1 so deletes are executed immediately, for performance deletes could be batched but
117             // they would need to be executed before inserts of updated versions (with the same key), that is more
118             // complicated
119             deleteBatches.put(objectType, new PreparedStatementQueryBatch(getConnection(), sql, 1));
120         }
121         QueryBatch batch = deleteBatches.get(objectType);
122 
123         Object[] params = objectType.getPrimaryKeys().stream().map(pk -> {
124             try {
125                 AttributeColumnMapping mapping = objectType.getAttributeByName(pk.getName());
126                 Object attribute = object.getAttributes().get(pk.getName());
127                 return mapping.toQueryParameter(attribute);
128             } catch (Exception e) {
129                 throw new RuntimeException(e);
130             }
131         }).toArray();
132 
133         boolean executed = batch.addBatch(params);
134         getProgress().updatedCount++;
135 
136         if (executed) {
137             updateProgress();
138         }
139     }

142     protected void addObjectToBatch(SchemaObjectInstance object) throws Exception {
143         // Never write NIET BAG objects to database
144         if (!object.getAttributes().containsKey(TIJDSTIP_NIETBAGLV)) {
145             super.addObjectToBatch(object);
146         }

149     public void write(InputStream bagXml) throws Exception {
150         CountingInputStream counter = new CountingInputStream(bagXml);
151         BAG2GMLMutatieGroepStream bag2Objects = new BAG2GMLMutatieGroepStream(counter);
152         getProgress().bagInfo = bag2Objects.getBagInfo();
153         updateProgress(Stage.LOAD_OBJECTS);
154 
155         try {
156             for (BAG2MutatieGroep mutatieGroep: bag2Objects) {
157                 for (BAG2Mutatie mutatie: mutatieGroep.getMutaties()) {
158                     if (mutatie instanceof BAG2WijzigingMutatie) {
159                         // Don't do an update but a simpler delete and insert of the updated version
160                         // Executed on main thread, but worker thread will not be executing new versions of the record we're
161                         // deleting
162 
163                         // No check for duplicates for wijzigingen: no harm in doing the same wijziging twice and we
164                         // can't tell by only the keys if it is exactly the same wijziging but in the maandmutaties for
165                         // a different gemeente or the same version changed twice
166 
167                         BAG2WijzigingMutatie wijzigingMutatie = (BAG2WijzigingMutatie) mutatie;
168                         deletePreviousVersion(wijzigingMutatie.getWas());
169                         addObjectToBatch(wijzigingMutatie.getWordt());
170                     } else if(mutatie instanceof BAG2ToevoegingMutatie) {
171                         BAG2ToevoegingMutatie toevoegingMutatie = (BAG2ToevoegingMutatie) mutatie;
172 
173                         if (ignoreDuplicates && isDuplicate(toevoegingMutatie.getToevoeging())) {
174                             continue;
175                         }
176 
177                         prepareDatabaseForObject(toevoegingMutatie.getToevoeging());
178                         getProgress().incrementObjectCount();
179                         addObjectToBatch(toevoegingMutatie.getToevoeging());
180                     }
181 
182                     if (getObjectLimit() != null && getProgress().getObjectCount() == getObjectLimit()) {
183                         break;
184                     }
185                 }
186             }
187         } catch(Exception e) {
188             if (isMultithreading()) {
189                 // Make sure worker thread exits
190                 abortWorkerThread();
191             }
192             throw e;
193         }
194     }

196     private boolean isDuplicate(BAG2Object object) {
197         if (keysPerObjectType == null) {
198             throw new IllegalStateException("keysPerObject type must be set to enable ignoring of duplicates");
199         }
200         // Primary keys for all BAG2 objects are always same
201         Pair<Object,Object> keys = Pair.of(
202                 object.getAttributes().get("identificatie"),
203                 object.getAttributes().get("voorkomenidentificatie")
204         );
205         Set<Pair<Object, Object>> seenKeys = keysPerObjectType.computeIfAbsent(object.getObjectType(), k -> new HashSet<>());
206         if (seenKeys.contains(keys)) {
207             if (log.isDebugEnabled()) {
208                 log.debug(String.format("\rIgnoring duplicate %s %s", object.getObjectType().getName(), keys));
209             }
210             return true;
211         }
212         seenKeys.add(keys);
213         return false;
214     }

216     public void complete() throws Exception {
217         super.endOfObjects();
218         for(QueryBatch batch: getProgress().deleteBatches.values()) {
219             batch.executeBatch();
220         }
221         super.complete();
222         super.closeBatches();
223     }
