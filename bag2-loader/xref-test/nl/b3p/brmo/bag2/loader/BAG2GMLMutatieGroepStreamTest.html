71      @Test
72      void testMutaties() throws Exception {
73          ZipInputStream zip = new ZipInputStream(BAG2TestFiles.getTestInputStream("1978MUT15082021-15092021-000002.zip"));
74          zip.getNextEntry();
75          BAG2GMLMutatieGroepStream stream = new BAG2GMLMutatieGroepStream(zip);
76  
77          assertNotNull(stream.getBagInfo());
78          SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
79          assertEquals(df.parse("2021-09-15"), stream.getBagInfo().getStandTechnischeDatum());
80          assertEquals(df.parse("2021-08-15"), stream.getBagInfo().getMutatieDatumVanaf());
81          assertEquals(df.parse("2021-09-15"), stream.getBagInfo().getMutatieDatumTot());
82  
83          Iterator<BAG2MutatieGroep> iterator = stream.iterator();
84          BAG2MutatieGroep mutatieGroep = iterator.next();
85  
86          assertNotNull(mutatieGroep);
87  
88          assertEquals(2, mutatieGroep.getMutaties().size());
89  
90          BAG2Mutatie mutatie = mutatieGroep.getMutaties().get(0);
91          assertInstanceOf(BAG2WijzigingMutatie.class, mutatie);
92          assertNull(((BAG2WijzigingMutatie)mutatie).getWas().getAttributes().get("eindGeldigheid"));
93          assertEquals("2021-08-16", ((BAG2WijzigingMutatie)mutatie).getWordt().getAttributes().get("eindGeldigheid"));
94  
95          mutatie = mutatieGroep.getMutaties().get(1);
96          assertInstanceOf(BAG2ToevoegingMutatie.class, mutatie);
97  
98          BAG2Object obj = ((BAG2ToevoegingMutatie)mutatie).getToevoeging();
99          String identificatie = (String)obj.getAttributes().get("identificatie");
100         assertEquals("1978100007698683", identificatie);
101 
102         mutatieGroep = iterator.next();
103         assertNotNull(mutatieGroep);
104         mutatieGroep = iterator.next();
105         assertNotNull(mutatieGroep);
106         assertTrue(mutatieGroep.isSingleToevoeging());
107 
108         int count = 3;
109         while (iterator.hasNext()) {
110             iterator.next();
111             count++;
112         }
113         assertEquals(847, count);
114     }

116     @Test
117     void testMutatieMultipleWijzigingen() throws Exception {
118         // A mutatieGroep is not always wijziging+toevoeging or only toevoeging. There may be multiple wijziging
119         // elements, test these are parsed correctly
120 
121         try(InputStream in = BAG2TestFiles.getTestInputStream("mut-multiple-wijziging.xml")) {
122             BAG2GMLMutatieGroepStream stream = new BAG2GMLMutatieGroepStream(in);
123 
124             assertNotNull(stream.getBagInfo());
125 
126             Iterator<BAG2MutatieGroep> iterator = stream.iterator();
127             BAG2MutatieGroep mutatieGroep = iterator.next();
128 
129             assertEquals(3, mutatieGroep.getMutaties().size());
130             assertInstanceOf(BAG2WijzigingMutatie.class, mutatieGroep.getMutaties().get(0));
131             assertInstanceOf(BAG2WijzigingMutatie.class, mutatieGroep.getMutaties().get(1));
132             assertInstanceOf(BAG2ToevoegingMutatie.class, mutatieGroep.getMutaties().get(2));
133 
134             mutatieGroep = iterator.next();
135             assertNotNull(mutatieGroep);
136             assertFalse(iterator.hasNext());
137         }

140     @ParameterizedTest(name="[{index}] {0}")
141     @MethodSource
142     void testAllFeatureTypes(String file, String objectName, Object[][] expectedObjects) throws XMLStreamException {
143         BAG2GMLMutatieGroepStream stream = new BAG2GMLMutatieGroepStream(BAG2TestFiles.getTestInputStream(file));
144         assertNotNull(stream.getBagInfo().getStandTechnischeDatum());
145         assertEquals("9999", stream.getBagInfo().getGemeenteIdentificatie());
146         Iterator<BAG2MutatieGroep> iterator = stream.iterator();
147         int objectNum = 1;
148         for(Object expectedObj: expectedObjects) {
149             Object[][] attributes = (Object[][])expectedObj;
150             BAG2MutatieGroep mutatieGroep = iterator.next();
151             assertNotNull(mutatieGroep);
152             BAG2Object object = mutatieGroep.getSingleToevoeging();
153             assertEquals(objectName, object.getObjectType().getName());
154             Map<String,Object> objectAttributes = object.getAttributes();
155             //System.out.println(objectAttributes);
156             Set<String> allKeys = new HashSet<>(objectAttributes.keySet());
157             for(Object[] attribute: attributes) {
158                 String key = (String) attribute[0];
159                 Object value = attribute[1];
160                 assertTrue(objectAttributes.containsKey(key), "Object #" + objectNum + ": Attribute should exist: " + key);
161                 Object objectValue = objectAttributes.get(key);
162                 assertEquals(value, objectValue, "Object #" + objectNum + ": Attribute equals: " + key);
163                 allKeys.remove(key);
164             }
165             assertEquals(0, allKeys.size(), "Object #" + objectNum + ": Unexpected attributes: " + allKeys);
166             objectNum++;
167         }
168     }

170     @Test
171     public void testPandIs2D() throws Exception {
172         // BAG Pand have a Z coordinate, but this is always 0 and not meaningful. Test that the class removes the Z
173         // coordinate. When writing to Oracle Spatial we need to drop the Z or create the index with XYZ. Currently a
174         // XY index is hardcoded in OracleDialect.
175 
176         BAG2GMLMutatieGroepStream stream = new BAG2GMLMutatieGroepStream(BAG2TestFiles.getTestInputStream("stand-pnd.xml"));
177         Iterator<BAG2MutatieGroep> iterator = stream.iterator();
178         BAG2Object object = iterator.next().getSingleToevoeging();
179         assertNotNull(object);
180         Geometry geometry = (Geometry)object.getAttributes().get("geometrie");
181         assertNotNull(geometry);
182         Coordinate coordinate = geometry.getCoordinate();
183         assertNotNull(coordinate);
184         assertTrue(Double.isNaN(coordinate.getZ()));
185     }
