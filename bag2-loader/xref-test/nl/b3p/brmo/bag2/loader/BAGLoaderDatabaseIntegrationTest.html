31  /**
32   * Laadt een BAG stand in de database. Test of tabellen en actueel views bestaan en gevuld zijn.
33   * <i>NB</i> De anader BAG (materalized) views worden in de datamodel module getest; deze test
34   * laat dus een gevulde BAG database achter.
35   *
36   * @author mprins
37   */
38  public class BAGLoaderDatabaseIntegrationTest {
39      private static final String[] BAGTABLES = new String[]{"ligplaats", "ligplaats_nevenadres", "nummeraanduiding", "openbareruimte", "pand", "standplaats", "standplaats_nevenadres", "verblijfsobject", "verblijfsobject_gebruiksdoel", "verblijfsobject_maaktdeeluitvan", "verblijfsobject_nevenadres", "woonplaats"};
40      private static final String[] BAGACTUEELVIEWS = new String[]{"v_ligplaats_actueel", "v_nummeraanduiding_actueel", "v_openbareruimte_actueel", "v_pand_actueel", "v_standplaats_actueel", "v_verblijfsobject_actueel", "v_woonplaats_actueel"};
41      private static String dbUrl = System.getProperty("dburl");
42      private static String dbUser = System.getProperty("dbuser", "rsgb");
43      private static String dbPass = System.getProperty("dbpassword", "rsgb");
44  
45      private BAG2Database bag2Database;
46      private BAG2DatabaseOptions databaseOptions;
47      private BAG2LoadOptions bag2LoadOptions;
48      private IDatabaseConnection bag;
49      private String testFileName;
50      private String tableQualifierPrefix = "";
51  
52      @BeforeAll
53      static void beforeAll() {
54          BAG2LoaderMain.configureLogging(false);
55      }
56  
57      @BeforeEach
58      void setUp() throws Exception {
59          assumeFalse(StringUtils.isEmpty(dbUrl), "skipping integration test: missing database url");
60          URL u = BAGLoaderDatabaseIntegrationTest.class.getResource("/BAGGEM1904L-15102021.zip");
61          assumeFalse(null == u, "skipping integration test: missing testdata");
62          testFileName = u.getFile();
63          bag = new DatabaseConnection(DriverManager.getConnection(dbUrl, dbUser, dbPass));
64  
65          if (dbUrl.contains("postgresql")) {
66              bag.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
67              bag.getConfig().setProperty(DatabaseConfig.FEATURE_QUALIFIED_TABLE_NAMES, true);
68              tableQualifierPrefix = "bag.";
69          }
70          if (dbUrl.contains("oracle")) {
71              bag.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
72              bag.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
73          }
74  
75          bag2LoadOptions = new BAG2LoadOptions();
76          // dit ruim alle BAG tabelle op met een cascade; dat is niet wat je wilt omdat daar tabellen
77          // een views van het RSGB schema bij zitten
78          // bag2LoadOptions.setDropIfExists(true);
79  
80          databaseOptions = new BAG2DatabaseOptions();
81          databaseOptions.setConnectionString(dbUrl);
82          databaseOptions.setUser(dbUser);
83          databaseOptions.setPassword(dbPass);
84          bag2Database = new BAG2Database(databaseOptions);
85      }
86  
87      @AfterEach
88      void cleanup() throws SQLException {
89          // geladen data wordt niet opgeruimd
90          if (null != bag) bag.close();
91      }
92  
93      @Test
94      void testStand() throws SQLException {
95          BAG2LoaderMain loader = new BAG2LoaderMain();
96  
97          try {
98              loader.loadFiles(bag2Database, databaseOptions, bag2LoadOptions, new BAG2ProgressReporter(), new String[]{testFileName}, null);
99          } catch (Exception e) {
100             fail("Laden BAG data is mislukt. " + e.getLocalizedMessage(), e);
101         }
102         // check tables
103         for (String t : BAGTABLES) {
104             // omdat sommige BAG tabellen ook in RSGB schema zitten bag qualifier gebruiken
105             t = tableQualifierPrefix + t;
106             assertTrue(bag.getRowCount(t) > 0, "Onverwacht lege tabel: " + t);
107         }
108         // check views
109         for (String t : BAGACTUEELVIEWS) {
110             assertTrue(bag.getRowCount(t) > 0, "Onverwacht lege view: " + t);
111         }
112     }
113 }
