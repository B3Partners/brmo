46  /**
47   *
48   * Draaien met:
49   * {@code mvn -Dit.test=EigendomInfoIntegrationTest -Dtest.onlyITs=true verify -Ppostgresql > target/postgresql.log}
50   * voor bijvoorbeeld PostgreSQL.

58  @RunWith(Parameterized.class)
59  @Category(JTDSDriverBasedFailures.class)
60  public class EigendomInfoIntegrationTest extends TestUtil {
61  
62      @Parameterized.Parameters(name = "{index}: verwerken bestand: {0}")
63      public static Collection params() {
64          return Arrays.asList(new Object[][]{
65              // {"rBestandsNaam", "objNamespace", "kadId", aantalBronDoc, "sBestandsNaam", "volgordeNummer"},
66              {"/rsgb.xml", "NL.KAD.OnroerendeZaak", "66860489870000", 3, "/staging-flat.xml", "0"}
67          });
68      }

70      private IDatabaseConnection rsgb;
71      private IDatabaseConnection staging;
72  
73      private final Lock sequential = new ReentrantLock(true);

75      /*
76       * test parameters.
77       */
78      private final String rBestandsNaam;
79      private final String objNamespace;
80      private final String objId;
81      private final int aantalBronDoc;
82      private final String sBestandsNaam;
83      private final String volgordeNummer;
84  
85      public EigendomInfoIntegrationTest(final String rBestandsNaam, final String objNamespace, final String objId,
86              final int aantalBronDoc, final String sBestandsNaam, final String volgordeNummer) {
87  
88          this.rBestandsNaam = rBestandsNaam;
89          this.objNamespace = objNamespace;
90          this.objId = objId;
91          this.aantalBronDoc = aantalBronDoc;
92          this.sBestandsNaam = sBestandsNaam;
93          this.volgordeNummer = volgordeNummer;
94      }
95  
96      @Before
97      @Override
98      public void setUp() throws Exception {
99          rsgb = new DatabaseDataSourceConnection(this.dsRsgb);
100         staging = new DatabaseDataSourceConnection(dsStaging);
101         IDataSet rsgbDataSet = new XmlDataSet(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource(rBestandsNaam).toURI())));
102 
103         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
104         fxdb.setCaseSensitiveTableNames(false);
105         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource(sBestandsNaam).toURI())));
106 
107         if (this.isMsSQL) {
108             // TODO kijken of en hoe dit op mssql werkt, vooralsnog problemen met jDTS driver
109             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
110             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
111         } else if (this.isOracle) {
112             rsgb = new DatabaseConnection(dsRsgb.getConnection().unwrap(oracle.jdbc.OracleConnection.class), DBPROPS.getProperty("rsgb.username").toUpperCase());
113             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
114             rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
115 
116             staging = new DatabaseConnection(dsStaging.getConnection().unwrap(oracle.jdbc.OracleConnection.class), DBPROPS.getProperty("staging.username").toUpperCase());
117             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
118             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
119 
120             rsgbDataSet = new XmlDataSet(new FileInputStream(new File(BrkInfoIntegrationTest.class.getResource("/oracle-" + rBestandsNaam.substring(1)).toURI())));
121         } else if (this.isPostgis) {
122             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
123             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
124         }
125 
126         sequential.lock();
127 
128         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
129         DatabaseOperation.CLEAN_INSERT.execute(rsgb, rsgbDataSet);
130 
131         refreshMViews(new String[]{"mb_util_app_re_kad_perceel"}, this.dsRsgb);
132     }
133 
134     @After
135     public void cleanup() throws Exception {
136         CleanUtil.cleanRSGB_BRK(rsgb, true);
137         CleanUtil.cleanSTAGING(staging);
138 
139         rsgb.close();
140         staging.close();
141 
142         sequential.unlock();
143     }

145     /**
146      * testcase voor
147      * {@link EigendomInfo#createEigendomMutatieContext(nl.b3p.brmo.soap.eigendom.EigendomMutatieRequest)}
148      * en {@link EigendomInfo#createEigendomMutatieResponse(java.util.Map)} .
149      *
150      * @throws Exception if any
151      */
152     @Test
153     public void testEigendomMutatie() throws Exception {
154         EigendomMutatieRequest request = new EigendomMutatieRequest();
155         request.setMaxAantalResultaten(10);
156         request.setIdentificatie(objNamespace + ":" + objId);
157         Map<String, Object> context = EigendomInfo.createEigendomMutatieContext(request);
158 
159         EigendomMutatieResponse response = EigendomInfo.createEigendomMutatieResponse(context);
160         List<EigendomMutatie> mutaties = response.getEigendomMutatie();
161 
162         assertFalse(mutaties.isEmpty());
163         EigendomMutatie mutatie = mutaties.get(0);
164         assertEquals(objId, mutatie.getIdentificatienummer());
165 
166         List<Document> documenten = mutatie.getBrondocumenten().getDocument();
167         assertFalse(documenten.isEmpty());
168         assertEquals(aantalBronDoc, documenten.size());
169     }

171     /**
172      * testcase voor
173      * {@link EigendomInfo#createMutatieListContext(nl.b3p.brmo.soap.eigendom.MutatieListRequest)}
174      * en {@link EigendomInfo#createMutatieListResponse(java.util.Map)} .
175      *
176      * @throws Exception if any
177      */
178     @Test
179     public void testMutatieList() throws Exception {
180         final GregorianCalendar gregorianCalendar = new GregorianCalendar();
181         DatatypeFactory datatypeFactory = DatatypeFactory.newInstance();
182         XMLGregorianCalendar now = datatypeFactory.newXMLGregorianCalendar(gregorianCalendar);
183         gregorianCalendar.set(2015, 1, 1);
184         XMLGregorianCalendar then = datatypeFactory.newXMLGregorianCalendar(gregorianCalendar);
185 
186         MutatieListRequest request = new MutatieListRequest();
187         request.setFromDate(then);
188         request.setToDate(now);
189         request.setObjectprefix(objNamespace);
190 
191         Map<String, Object> context = EigendomInfo.createMutatieListContext(request);
192         MutatieListResponse response = EigendomInfo.createMutatieListResponse(context);
193         List<MutatieEntry> mutatieentries = response.getMutatieEntry();
194         assertFalse(mutatieentries.isEmpty());
195 
196         MutatieEntry mutatieentry = mutatieentries.get(0);
197         assertEquals(objNamespace + ":" + objId, mutatieentry.getObjectRef());
198         assertEquals(volgordeNummer, mutatieentry.getVolgnummer());
199     }
