15  @WebService(
16          targetNamespace = "http://brmo.b3p.nl/brk/1.0/soap-brk",
17          wsdlLocation = "WEB-INF/wsdl/brkinfo.wsdl"
18  )
19  @HandlerChain(file = "/handler-chain.xml")
20  public class GetBrkInfoImpl {
21      private static final Log LOG = LogFactory.getLog(GetBrkInfoImpl.class);
22      /**
23       * Web service operation
24       */
25      @WebMethod(operationName = "getBrkInfo")
26      @RequestWrapper(className = "nl.b3p.brmo.soap.brk.getBrkInfo")
27      @ResponseWrapper(className = "nl.b3p.brmo.soap.brk.getBrkInfoResponse")
28      public BrkInfoResponse getBrkInfo(@WebParam(name = "request") BrkInfoRequest request) throws BrkInfoException {
29  
30          Map<String, Object> searchContext = BrkInfo.createSearchContext(request);
31          if (searchContext.isEmpty()) {
32              throw new BrkInfoException("Request not valid", "minimaal één zoekterm vereist!");
33          }
34          
35          ArrayList<Long> ids = null;
36          try {
37              ids = BrkInfo.findKozIDs(searchContext);
38          } catch (Exception ex) {
39              LOG.error(ex);
40              throw new BrkInfoException("Database reported errors", ex.getLocalizedMessage());
41          }
42  
43          if (ids==null || ids.isEmpty()) {
44              throw new BrkInfoException("Database reported", "Geen resultaten");
45          }
46          
47          if (ids.size() > (Integer)searchContext.get(BrkInfo.MAXAANTALRESULTATEN)) {
48              throw new BrkInfoException("Database reported", 
49                      "Meer resultaten dan toegestaan: "
50                              +(Integer)searchContext.get(BrkInfo.MAXAANTALRESULTATEN)
51                              +", pas uw voorwaarden aan!");
52          }
53          
54          BrkInfoResponse result = null;
55          try {
56              result = BrkInfo.createResponse(ids, searchContext);
57          } catch (Exception ex) {
58              LOG.error(ex);
59              throw new BrkInfoException("Database reported errors", ex.getLocalizedMessage());
60          }
61          
62          return result;
63      }
