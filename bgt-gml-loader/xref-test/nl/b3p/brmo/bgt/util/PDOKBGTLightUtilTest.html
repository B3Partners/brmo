21      private static final Log LOG = LogFactory.getLog(PDOKBGTLightUtilTest.class);
22  
23      /**
24       * offline test.
25       */
26      @Test
27      public void calculateGridIdsOfflineTest() {
28          TreeSet<Integer> ids = (TreeSet) PDOKBGTLightUtil.calculateGridIds(
29                  /* dit vlakt overlap met cel 38725 in het testbestand, maar niet met andere cellen */
30                  "POLYGON ((143850 487030, 149760 486760, 149470 490950, 143600 491000, 143850 487030))",
31                  PDOKBGTLightUtilTest.class.getResource("/geojson/tileinfo.json").toString()
32          );
33          assertTrue("Er zit 1 element in de lijst met ids", ids.size() == 1);
34          assertEquals("Het element in de lijst met ids is", 38725, ids.first().intValue());
35      }
36  
37      /**
38       * Online test. deze test faalt als het PKI Overheid moeder certificaten
39       * niet is geinstalleerd; zie:
40       * <a href="https://github.com/B3Partners/brmo/wiki/BGTLightOphaalProces#ssl-fouten">BGTLightOphaalProces
41       * ssl-fouten</a>.
42       */
43      @Test
44      public void calculateGridIdsOnlineTest() {
45          TreeSet<Integer> ids = (TreeSet) PDOKBGTLightUtil.calculateGridIds(
46                  /* dit is de geometrie van grid cel 38725 */
47                  "POLYGON ((146000 488000, 148000 488000, 148000 490000, 146000 490000, 146000 488000))",
48                  "https://www.pdok.nl/download/service/tileinfo.json?dataset=bgt&format=citygml"
49          );
50          assertTrue("Er zitten 9 elementen in de lijst met ids", ids.size() == 9);
51          assertEquals("Het element in de lijst met ids is", 38382, ids.first().intValue());
52      }
