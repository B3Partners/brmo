17  package nl.b3p.brmo.bgt.util;
18  
19  import java.lang.reflect.Method;
20  import org.apache.commons.dbutils.QueryRunner;
21  import org.apache.commons.logging.Log;
22  import org.geotools.util.factory.Hints;
23  import org.geotools.jdbc.JDBCDataStore;
24  
25  /**
26   *
27   * @author matthijsln
28   */
29  public class JDBCDataStoreUtil {
30      public static boolean truncateTable(JDBCDataStore dataStore, String typeName, boolean isOracle, Log log) {
31          log.info("Probeer tabel leeg te maken: " + typeName);
32  
33          try {
34              StringBuffer sql = new StringBuffer("TRUNCATE TABLE ");
35              Method encodeTableName = dataStore.getClass().getDeclaredMethod("encodeTableName", String.class, StringBuffer.class, Hints.class);
36              encodeTableName.setAccessible(true);
37              encodeTableName.invoke(dataStore, typeName, sql, null);
38  
39              if(isOracle) {
40                  sql.append(" PURGE MATERIALIZED VIEW LOG REUSE STORAGE");
41              }
42              log.trace("SQL: " + sql.toString());
43              new QueryRunner(dataStore.getDataSource()).update(sql.toString());
44              log.info("Tabel leeggemaakt");
45              return true;
46          } catch(Exception e) {
47              log.debug("Exception bij TRUNCATE", e);
48              log.warn("Fout bij TRUNCATE, probeer met DELETE FROM: " + e.getClass() + ": " + e.getMessage());
49  
50              try {
51                  StringBuffer sql = new StringBuffer("DELETE FROM ");
52                  Method encodeTableName = dataStore.getClass().getDeclaredMethod("encodeTableName", String.class, StringBuffer.class, Hints.class);
53                  encodeTableName.setAccessible(true);
54                  encodeTableName.invoke(dataStore, typeName, sql, null);
55                  log.trace("SQL: " + sql.toString());
56                  new QueryRunner(dataStore.getDataSource()).update(sql.toString());
57                  log.info("Tabel leeggemaakt met DELETE FROM");
58                  return true;
59              } catch(Exception e2) {
60                  log.error("Fout bij DELETE FROM, tabel kon niet leeggemaakt worden! ", e2);
61                  return false;
62              }
63          }
64      }
65  }
