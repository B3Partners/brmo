24  /**
25   * utilities mbt PDOK BGT GML light formaat en de dwnload website.
26   *
27   * @author mprins
28   */
29  public class PDOKBGTLightUtil {
30  
31      private static final Log LOG = LogFactory.getLog(PDOKBGTLightUtil.class);
32      /**
33       * het featuretype van de {@code tileinfo.json} van de pdok bgt download
34       * site.
35       */
36      public static final String TILEINFO_F_TYPE = "geom:Polygon:srid=28992,extractset:String,id:Integer";
37  
38      /**
39       * Bepaald de lijst met grid ids.
40       *
41       * @param wktGeom aandachtsgebied
42       * @param jsonUrl url naar de pdok bgt gml light index
43       * @return lijst met grid ids
44       *
45       * @see #calculateGridIds(org.locationtech.jts.geom.Geometry,
46       * org.geotools.data.simple.SimpleFeatureCollection)
47       */
48      public static Set<Integer> calculateGridIds(String wktGeom, String jsonUrl) {
49          Geometry geometry = null;
50          SimpleFeatureCollection fc = null;
51          try {
52              org.locationtech.jts.geom.GeometryFactory geometryFactory = JTSFactoryFinder.getGeometryFactory();
53              WKTReader reader = new WKTReader(geometryFactory);
54              geometry = reader.read(wktGeom);
55              geometry.setSRID(28992);
56  
57              URL u = new URL(jsonUrl);
58              JSONParser parser = new JSONParser();
59              Object json = parser.parse(new java.io.InputStreamReader(u.openStream()));
60              final FeatureJSON io = new FeatureJSON();
61              try {
62                  final SimpleFeatureType jsonType = DataUtilities.createType("tileinfo", TILEINFO_F_TYPE);
63                  io.setFeatureType(jsonType);
64              } catch (SchemaException se) {
65                  LOG.warn("Aanmaken van tileinfo featuretype is mislukt", se);
66              }
67              fc = (SimpleFeatureCollection) io.readFeatureCollection(json.toString());
68          } catch (org.locationtech.jts.io.ParseException ex) {
69              LOG.error("Parsen van de WKT selectie geometrie is mislukt", ex);
70          } catch (IOException | org.json.simple.parser.ParseException io) {
71              LOG.error("Ophalen of parsen van de geojson file met tiles is mislukt", io);
72          }
73          return calculateGridIds(geometry, fc);
74      }
75  
76      public static Set<Integer> calculateGridIds(Geometry aoi, SimpleFeatureCollection grid) {
77          final Set<Integer> ids = new TreeSet();
78          SimpleFeature sf;
79          aoi = aoi.buffer(10, 1);
80          try (SimpleFeatureIterator sfi = grid.features()) {
81              while (sfi.hasNext()) {
82                  sf = sfi.next();
83                  if (aoi.intersects((Geometry) sf.getDefaultGeometry())) {
84                      LOG.debug(sf.getAttribute("id") + " Toevoegen aan de lijst van op te halen ids");
85                      ids.add((Integer) (sf.getAttribute("id")));
86                  }
87              }
88          }
89          LOG.debug("De lijst bevat de volgende id's: " + ids);
90          return ids;
91      }
92  
93      private PDOKBGTLightUtil() {
94      }
