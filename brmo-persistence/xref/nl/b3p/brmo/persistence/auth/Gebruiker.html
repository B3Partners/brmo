17  package nl.b3p.brmo.persistence.auth;
18  
19  import java.io.Serializable;
20  import java.io.UnsupportedEncodingException;
21  import java.security.MessageDigest;
22  import java.security.NoSuchAlgorithmException;
23  import java.util.HashSet;
24  import java.util.Set;
25  import javax.persistence.Entity;
26  import javax.persistence.Id;
27  import javax.persistence.JoinColumn;
28  import javax.persistence.JoinTable;
29  import javax.persistence.ManyToMany;
30  import javax.persistence.Table;
31  
32  /**
33   *
34   * @author Matthijs Laan
35   */
36  @Entity
37  @Table(name = "gebruiker_")
38  public class Gebruiker implements Serializable {
39  
40      public static final int MIN_PASSWORD_LENGTH = 8;
41      private static final String DIGEST_ALGORITM = "SHA-1";
42      private static final String DIGEST_CHARSET = "UTF-8";
43  
44      @Id
45      private String gebruikersnaam;
46  
47      private String wachtwoord;
48  
49      @ManyToMany
50      @JoinTable(name = "gebruiker_groepen", joinColumns = @JoinColumn(name = "gebruikersnaam"), inverseJoinColumns = @JoinColumn(name = "groep_"))
51      private Set<Groep> groepen = new HashSet<Groep>();
52  
53      //    @ElementCollection
54      //    @JoinTable(joinColumns=@JoinColumn(name="gebruikersnaam"))
55      //    private Map<String,String> details = new HashMap<String,String>();
56      /**
57       *
58       * @param password wachtwoord
59       * @throws NoSuchAlgorithmException als er geen DIGEST_ALGORITM bestaat
60       * @throws UnsupportedEncodingException als er een onbekendeencodingsfout
61       * optreed
62       */
63      public void changePassword(String password) throws NoSuchAlgorithmException, UnsupportedEncodingException {
64          MessageDigest md = (MessageDigest) MessageDigest.getInstance(DIGEST_ALGORITM);
65          md.update(password.getBytes(DIGEST_CHARSET));
66          byte[] digest = md.digest();
67  
68          /* Converteer byte array naar hex-weergave */
69          StringBuilder sb = new StringBuilder(digest.length * 2);
70          for (int i = 0; i < digest.length; i++) {
71              sb.append(Integer.toHexString(digest[i] >> 4 & 0xf)); /* and mask met 0xf nodig door sign-extenden van bytes... */
72              sb.append(Integer.toHexString(digest[i] & 0xf));
73          }
74          setWachtwoord(sb.toString());
75      }
76  
77      public String getWachtwoord() {
78          return wachtwoord;
79      }
80  
81      public void setWachtwoord(String wachtwoord) {
82          this.wachtwoord = wachtwoord;
83      }
84  
85      public Set<Groep> getGroepen() {
86          return groepen;
87      }
88  
89      public void setGroepen(Set<Groep> groepen) {
90          this.groepen = groepen;
91      }
92  
93      public String getGebruikersnaam() {
94          return gebruikersnaam;
95      }
96  
97      public void setGebruikersnaam(String gebruikersnaam) {
98          this.gebruikersnaam = gebruikersnaam;
99      }
100 }
