4   package nl.b3p.brmo.persistence.staging;
5   
6   import java.util.Collections;
7   import java.util.Set;
8   import java.util.StringTokenizer;
9   import java.util.TreeSet;
10  import javax.persistence.Entity;
11  
12  /**
13   * Proces configuratie voor de BGT light downloader.
14   *
15   * @author mprins
16   */
17  @Entity
18  public class BGTLightOphaalProces extends AutomatischProces {
19  
20      private static final String ARCHIEF_DIRECTORY = "archiefdirectory";
21  
22      /**
23       * Zoekt de geconfigureerde archief directory op.
24       *
25       * @return de naam van de directory voor deze configuratie of null
26       */
27      public String getArchiefDirectory() {
28          return ClobElement.nullSafeGet(this.getConfig().get(ARCHIEF_DIRECTORY));
29      }
30  
31      /**
32       * Stelt de te gebruiken archief directory in.
33       *
34       * @param archiefDirectory te gebruiken archief directory
35       */
36      public void setArchiefDirectory(String archiefDirectory) {
37          if (archiefDirectory == null) {
38              this.getConfig().put(ARCHIEF_DIRECTORY, null);
39          } else {
40              this.getConfig().put(ARCHIEF_DIRECTORY, new ClobElement(archiefDirectory));
41          }
42      }
43  
44      public String getTileInfoUrl() {
45          return ClobElement.nullSafeGet(this.getConfig().get("geojsonurl"));
46      }
47  
48      public String getOphaalgebied() {
49          return ClobElement.nullSafeGet(this.getConfig().get("ophaalgebied"));
50      }
51  
52      public String getOphaalUrl() {
53          return ClobElement.nullSafeGet(this.getConfig().get("ophaalurl"));
54      }
55  
56      public Set<Integer> getGridIds() {
57          TreeSet<Integer> ids = new TreeSet();
58          String cvsId = ClobElement.nullSafeGet(this.getConfig().get("gridids"));
59          if (cvsId != null) {
60              StringTokenizer st = new StringTokenizer(cvsId, ",;", false);
61              while (st.hasMoreTokens()) {
62                  try {
63                      ids.add(Integer.parseInt(st.nextToken().trim()));
64                  } catch (NumberFormatException ignore) {
65                      // LOG.debug("Niet parsable integer.", ignore);
66                  }
67              }
68          }
69          return Collections.unmodifiableSet(ids);
70      }
71  }
