4   package nl.b3p.brmo.persistence.staging;
5   
6   import java.util.ArrayList;
7   import java.util.Collections;
8   import java.util.List;
9   import javax.persistence.Entity;
10  
11  /**
12   *
13   * @author mprins
14   */
15  @Entity
16  public class LaadprocesTransformatieProces extends AutomatischProces {
17  
18      public String getSoort() {
19          return ClobElement.nullSafeGet(this.getConfig().get("laadprocessoort"));
20      }
21  
22      public boolean alsStandTransformeren() {
23          String alsStand = ClobElement.nullSafeGet(this.getConfig().get("versneldtransformeren"));
24          return "true".equals(alsStand);
25      }
26  
27      /**
28       * soorten laadproces waar we iets mee kunnen.
29       */
30      public enum LaadprocesSoorten {
31          BR_TOPNL("topnl");
32  
33          private static final ArrayList<String> soorten = new ArrayList();
34  
35          static {
36              for (LaadprocesSoorten s : values()) {
37                  soorten.add(s.getSoort());
38              }
39          }
40          private final String soort;
41  
42          LaadprocesSoorten(String soort) {
43              this.soort = soort;
44          }
45  
46          public String getSoort() {
47              return soort;
48          }
49  
50          public static List<String> soorten() {
51              return Collections.unmodifiableList(soorten);
52          }
53      }
54  }
