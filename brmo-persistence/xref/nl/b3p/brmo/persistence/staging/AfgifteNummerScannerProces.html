17  package nl.b3p.brmo.persistence.staging;
18  
19  import javax.persistence.Entity;
20  
21  /**
22   *
23   * @author mprins
24   */
25  @Entity
26  public class AfgifteNummerScannerProces extends AutomatischProces {
27  
28      private static final String CONTRACTNUMMER = "contractnummer";
29      private static final String AFGIFTENUMMERTYPE = "afgiftenummertype";
30      public static final String MISSINGNUMBERSFOUND = "ontbrekendenummersgevonden";
31      private static final String INSERT_LAADPROCESSEN = "laadprocessen_toevoegen";
32  
33      /**
34       * Zoekt het geconfigureerde contractnummer op.
35       *
36       * @return contractnummer of null
37       */
38      public String getContracNummer() {
39          return ClobElement.nullSafeGet(this.getConfig().get(CONTRACTNUMMER));
40      }
41  
42      /**
43       * set contractnummer.
44       *
45       * @param contractnummer of null
46       */
47      public void setContracNummer(String contractnummer) {
48          if (contractnummer == null) {
49              this.getConfig().put(CONTRACTNUMMER, null);
50          } else {
51              this.getConfig().put(CONTRACTNUMMER, new ClobElement(contractnummer));
52          }
53      }
54  
55      /**
56       * Zoekt het geconfigureerde afgifte nummer type op.
57       *
58       * @return afgiftenummertype of een lege String
59       */
60      public String getAfgifteNummerType() {
61          return ClobElement.nullSafeGet(this.getConfig().get(AFGIFTENUMMERTYPE));
62      }
63  
64      /**
65       * set afgifte nummer type.
66       *
67       * @param afgiftenummertype of null
68       */
69      public void setAfgifteNummerType(String afgiftenummertype) {
70          if (afgiftenummertype == null) {
71              this.getConfig().put(AFGIFTENUMMERTYPE, null);
72          } else {
73              this.getConfig().put(AFGIFTENUMMERTYPE, new ClobElement(afgiftenummertype));
74          }
75      }
76  
77      public void setOntbrekendeNummersGevonden(boolean gevonden) {
78          this.getConfig().put(MISSINGNUMBERSFOUND, new ClobElement(Boolean.toString(gevonden)));
79      }
80  
81      public boolean getOntbrekendeNummersGevonden() {
82          String s = ClobElement.nullSafeGet(this.getConfig().get(MISSINGNUMBERSFOUND));
83          return (s == null ? false : Boolean.parseBoolean(s));
84      }
85  
86      public void setOntbrekendeAfgiftenummersToevoegen(boolean toevoegen) {
87          this.getConfig().put(INSERT_LAADPROCESSEN, new ClobElement(Boolean.toString(toevoegen)));
88      }
89  
90      public boolean getOntbrekendeAfgiftenummersToevoegen() {
91          String s = ClobElement.nullSafeGet(this.getConfig().get(INSERT_LAADPROCESSEN));
92          return (s == null ? false : Boolean.parseBoolean(s));
93      }
94  }
