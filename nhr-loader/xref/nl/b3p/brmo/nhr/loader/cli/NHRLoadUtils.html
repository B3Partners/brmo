42  public class NHRLoadUtils {
43    public static BrmoFramework getFramework(NHRDatabaseOptions databaseOptions)
44        throws BrmoException {
45      BasicDataSource dsStaging = new BasicDataSource();
46  
47      dsStaging.setUrl(databaseOptions.getConnectionString());
48      dsStaging.setUsername(databaseOptions.getUser());
49      dsStaging.setPassword(databaseOptions.getPassword());
50  
51      BrmoFramework fw = new BrmoFramework(dsStaging, null, null);
52      fw.setOrderBerichten(true);
53      return fw;
54    }

56    private static Properties getCryptoProperties(
57        NHRCertificateOptions certificateOptions, String alias) {
58      Properties props = new Properties();
59  
60      props.setProperty(
61          "org.apache.ws.security.crypto.provider",
62          "org.apache.ws.security.components.crypto.Merlin");
63      props.setProperty(
64          "org.apache.ws.security.crypto.merlin.keystore.file", certificateOptions.getKeystore());
65      props.setProperty(
66          "org.apache.ws.security.crypto.merlin.keystore.password",
67          certificateOptions.getKeystorePassword());
68      props.setProperty("org.apache.ws.security.crypto.merlin.keystore.type", "PKCS12");
69      props.setProperty("org.apache.ws.security.crypto.merlin.keystore.alias", alias);
70  
71      return props;
72    }

74    public static Dataservice getDataservice(
75        String targetLocation, boolean preprod, NHRCertificateOptions certificateOptions)
76        throws Exception {
77      DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
78      documentBuilderFactory.setNamespaceAware(true);
79  
80      DataserviceService dataServiceService = new DataserviceService();
81      Dataservice dataService = dataServiceService.getDataserviceSoap11();
82      Client client = ClientProxy.getClient(dataService);
83      Endpoint endpoint = client.getEndpoint();
84      HTTPConduit http = (HTTPConduit) client.getConduit();
85  
86      // Set up TLS certificates/keys
87      TrustManagerFactory trustManagerFactory = TrustManagerFactory.getInstance("SunX509");
88      KeyStore trustStore = KeyStore.getInstance("PKCS12");
89  
90      try (FileInputStream trustStoreFile = new FileInputStream(certificateOptions.getTruststore())) {
91        trustStore.load(trustStoreFile, certificateOptions.getTruststorePassword().toCharArray());
92      }
93      trustManagerFactory.init(trustStore);
94  
95      KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance("SunX509");
96      KeyStore keyStore = KeyStore.getInstance("PKCS12");
97  
98      try (FileInputStream keyStoreFile = new FileInputStream(certificateOptions.getKeystore())) {
99        keyStore.load(keyStoreFile, certificateOptions.getKeystorePassword().toCharArray());
100     }
101     String alias = certificateOptions.getKeystoreAlias();
102     if (alias == null || alias.isBlank()) {
103       alias = keyStore.aliases().nextElement();
104     }
105     keyManagerFactory.init(keyStore, certificateOptions.getKeystorePassword().toCharArray());
106 
107     SSLContext context = SSLContext.getInstance("TLSv1.2");
108     context.init(keyManagerFactory.getKeyManagers(), trustManagerFactory.getTrustManagers(), null);
109 
110     TLSClientParameters tlsClientParameters = new TLSClientParameters();
111     tlsClientParameters.setKeyManagers(keyManagerFactory.getKeyManagers());
112     tlsClientParameters.setSslContext(context);
113     http.setTlsClientParameters(tlsClientParameters);
114 
115     Map<String, Object> props = new HashMap<>();
116 
117     props.put(
118         WSHandlerConstants.ACTION,
119         WSHandlerConstants.TIMESTAMP + " " + WSHandlerConstants.SIGNATURE);
120     props.put(WSHandlerConstants.INCLUDE_SIGNATURE_TOKEN, "true");
121     props.put(WSHandlerConstants.SIG_KEY_ID, "DirectReference");
122     props.put(WSHandlerConstants.USE_SINGLE_CERTIFICATE, "false");
123 
124     props.put(WSHandlerConstants.SIG_PROP_REF_ID, "signatureProperties");
125     props.put("signatureProperties", getCryptoProperties(certificateOptions, alias));
126 
127     props.put(
128         WSHandlerConstants.SIGNATURE_PARTS,
129         "{}{http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd}Timestamp;"
130             + "{}{http://schemas.xmlsoap.org/soap/envelope/}Body;"
131             + "{}{http://www.w3.org/2005/08/addressing}To;"
132             + "{}{http://www.w3.org/2005/08/addressing}ReplyTo;"
133             + "{}{http://www.w3.org/2005/08/addressing}MessageID;"
134             + "{}{http://www.w3.org/2005/08/addressing}Action");
135 
136     props.put(WSHandlerConstants.USER, alias);
137     props.put(
138         WSHandlerConstants.PW_CALLBACK_REF,
139         (CallbackHandler)
140             callbacks -> {
141               for (Callback callback : callbacks) {
142                 ((WSPasswordCallback) callback)
143                     .setPassword(certificateOptions.getKeystorePassword());
144                 return;
145               }
146             });
147 
148     WSS4JOutInterceptor wssOut = new WSS4JOutInterceptor(props);
149     endpoint.getOutInterceptors().add(wssOut);
150 
151     // Set up necessary WS-Addressing fields.
152     AddressingProperties maps = new AddressingProperties();
153     EndpointReferenceType anonref = new EndpointReferenceType();
154     AttributedURIType anonymous = new AttributedURIType();
155     anonymous.setValue("http://www.w3.org/2005/08/addressing/anonymous");
156     anonref.setAddress(anonymous);
157     maps.setReplyTo(anonref);
158     maps.setFaultTo(anonref);
159 
160     // <wsa:To> needs to point at a predefined value.
161     EndpointReferenceType toval = new EndpointReferenceType();
162     AttributedURIType target = new AttributedURIType();
163     target.setValue(
164         preprod
165             ? "http://es.kvk.nl/kvk-DataservicePP/2015/02"
166             : "http://es.kvk.nl/kvk-Dataservice/2015/02");
167     toval.setAddress(target);
168     maps.setTo(toval);
169 
170     BindingProvider bindingProvider = (BindingProvider) dataService;
171     bindingProvider
172         .getRequestContext()
173         .put(BindingProvider.ENDPOINT_ADDRESS_PROPERTY, targetLocation);
174     bindingProvider.getRequestContext().put(JAXWSAConstants.CLIENT_ADDRESSING_PROPERTIES, maps);
175 
176     return dataService;
177   }
