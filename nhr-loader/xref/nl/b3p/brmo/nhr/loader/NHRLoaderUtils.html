8   package nl.b3p.brmo.nhr.loader;
9   
10  import java.text.MessageFormat;
11  import java.util.ResourceBundle;
12  
13  public class NHRLoaderUtils {
14  
15    public static final String BUNDLE_NAME = "NHRLoader";
16  
17    static final ResourceBundle bundle = ResourceBundle.getBundle(BUNDLE_NAME);
18  
19    public static String getUserAgent() {
20      return String.format(
21          "%s, %s (%s)/%s, %s/%s",
22          getBundleString("app.user-agent"),
23          System.getProperty("os.name"),
24          System.getProperty("os.arch"),
25          System.getProperty("os.version"),
26          System.getProperty("java.vm.name"),
27          System.getProperty("java.vm.version"));
28    }
29  
30    public static ResourceBundle getBundle() {
31      return bundle;
32    }
33  
34    public static String getBundleString(String key) {
35      return bundle.getString(key);
36    }
37  
38    public static String getMessageFormattedString(String key, Object... args) {
39      ResourceBundle bundle = getBundle();
40      return new MessageFormat(bundle.getString(key), bundle.getLocale()).format(args);
41    }
42  
43    public static String getLoaderVersion() {
44      return getBundleString("app.version");
45    }
46  
47    public static String getBrmoVersion() {
48      return getBundleString("brmo.version");
49    }
50  }
