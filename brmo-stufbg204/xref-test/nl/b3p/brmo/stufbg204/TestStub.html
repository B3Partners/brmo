10  import org.junit.jupiter.api.BeforeEach;
11  import org.junit.jupiter.api.TestInfo;

13  import javax.naming.Context;
14  import javax.naming.InitialContext;
15  import javax.naming.NamingException;

17  import java.util.Properties;

19  import static java.lang.System.getProperty;
20  import static org.junit.jupiter.api.Assumptions.assumeFalse;

22  /**
23   *
24   * @author mprins
25   */
26  public abstract class TestStub {
27  
28      private static boolean haveSetupJNDI = false;
29  
30      private static final Log LOG = LogFactory.getLog(TestStub.class);
31      /**
32       * properties uit {@code <DB smaak>.properties} en
33       * {@code local.<DB smaak>.properties}.
34       *
35       * @see #loadDBprop()
36       */
37      protected final Properties DBPROPS = new Properties();
38  
39      /**
40       * {@code true} als we met een Oracle database bezig zijn.
41       */
42      protected boolean isOracle;
43  
44      /**
45       * {@code true} als we met een MS SQL Server database bezig zijn.
46       */
47      protected boolean isMsSQL;
48  
49      /**
50       * {@code true} als we met een Postgis database bezig zijn.
51       */
52      protected boolean isPostgis;
53  
54      protected BasicDataSource dsStaging;
55      protected BasicDataSource dsRsgb;
56  
57      /**
58       * test of de database properties zijn aangegeven, zo niet dan skippen we
59       * alle tests in deze test.
60       */
61      @BeforeAll
62      public static void checkDatabaseIsProvided() {
63          assumeFalse(getProperty("database.properties.file") == null, "Verwacht database omgeving te zijn aangegeven.");
64      }
65  
66      /**
67       * subklassen dienen zelf een setup te hebben; vanwege de overerving gaat
68       * deze methode af na de {@code @Before} methoden van de superklasse, bijv.
69       * {@link #loadDBprop()}.
70       *
71       * @throws Exception if any
72       */
73      @BeforeEach
74      abstract public void setUp() throws Exception;
75  
76      /**
77       * initialize database props using the environment provided file.
78       *
79       * @throws java.io.IOException if loading the property file fails
80       */
81      @BeforeEach
82      public void loadDBprop() throws IOException {
83          // the `database.properties.file`  is set in the pom.xml or using the commandline
84          DBPROPS.load(TestStub.class.getClassLoader()
85                  .getResourceAsStream(System.getProperty("database.properties.file")));
86          try {
87              // see if a local version exists and use that to override
88              DBPROPS.load(TestStub.class.getClassLoader()
89                      .getResourceAsStream("local." + System.getProperty("database.properties.file")));
90          } catch (IOException | NullPointerException e) {
91              // ignore this
92          }
93  
94          isOracle = "oracle".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
95          isMsSQL = "sqlserver".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
96          isPostgis = "postgis".equalsIgnoreCase(DBPROPS.getProperty("dbtype"));
97  
98          try {
99              Class.forName(DBPROPS.getProperty("jdbc.driverClassName"));
100         } catch (ClassNotFoundException ex) {
101             LOG.error("Database driver niet gevonden.", ex);
102         }
103 
104         dsStaging = new BasicDataSource();
105         dsStaging.setUrl(DBPROPS.getProperty("staging.url"));
106         dsStaging.setUsername(DBPROPS.getProperty("staging.username"));
107         dsStaging.setPassword(DBPROPS.getProperty("staging.password"));
108         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
109 
110         dsRsgb = new BasicDataSource();
111         dsRsgb.setUrl(DBPROPS.getProperty("rsgb.url"));
112         dsRsgb.setUsername(DBPROPS.getProperty("rsgb.username"));
113         dsRsgb.setPassword(DBPROPS.getProperty("rsgb.password"));
114         dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
115 
116         setupJNDI();
117     }
118 
119     /**
120      * Log de naam van de test als deze begint.
121      */
122     @BeforeEach
123     public void startTest(TestInfo testInfo) {
124         LOG.info("==== Start test methode: " + testInfo.getDisplayName());
125     }
126 
127     @AfterEach
128     public void endTest(TestInfo testInfo) {
129         LOG.info("==== Einde test methode: " + testInfo.getDisplayName());
130     }
131 
132     /**
133      * setup jndi voor testcases.
134      */
135     private void setupJNDI() {
136         if (!haveSetupJNDI) {
137             try {
138                 System.setProperty(Context.INITIAL_CONTEXT_FACTORY, "org.apache.naming.java.javaURLContextFactory");
139                 System.setProperty(Context.URL_PKG_PREFIXES, "org.apache.naming");
140                 InitialContext ic = new InitialContext();
141                 ic.createSubcontext("java:");
142                 ic.createSubcontext("java:comp");
143                 ic.createSubcontext("java:comp/env");
144                 ic.createSubcontext("java:comp/env/jdbc");
145                 ic.createSubcontext("java:comp/env/jdbc/brmo");
146                 ic.bind("java:comp/env/jdbc/brmo/rsgb", dsRsgb);
147                 ic.bind("java:comp/env/jdbc/brmo/staging", dsStaging);
148                 haveSetupJNDI = true;
149             } catch (NamingException ex) {
150                 LOG.warn("Opzetten van datasource jndi is mislukt", ex);
151             }
152         }
153     }
