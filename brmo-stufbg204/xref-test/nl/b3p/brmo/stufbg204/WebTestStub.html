15  /**
16   * Integratie test utility klasse om online integratie tests te bouwen welke tegen een
17   * brmo-stufbg204 kunnen worden uitgevoerd.
18   *
19   * @author Mark Prins
20   */
21  public abstract class WebTestStub extends TestStub {
22  
23    public static final int HTTP_PORT = 9091;
24    /** the server root url. {@value} */
25    public static final String BASE_TEST_URL = "http://localhost:" + HTTP_PORT + "/brmo-stufbg204/";
26  
27    /** onze test client. */
28    protected static CloseableHttpClient client;
29  
30    protected static HttpHost target;
31    protected static HttpClientContext localContext;
32  
33    /** initialize http client. */
34    @BeforeAll
35    public static void setUpClass() {
36      target = new HttpHost("localhost", HTTP_PORT, "http");
37      AuthCache authCache = new BasicAuthCache();
38      authCache.put(target, new BasicScheme());
39  
40      localContext = HttpClientContext.create();
41      localContext.setAuthCache(authCache);
42  
43      CredentialsProvider credsProvider = new BasicCredentialsProvider();
44      credsProvider.setCredentials(
45          new AuthScope(target.getHostName(), target.getPort()),
46          new UsernamePasswordCredentials("brmo", "brmo"));
47  
48      client =
49          HttpClients.custom()
50              .useSystemProperties()
51              .setUserAgent("brmo integration test")
52              .setRedirectStrategy(new LaxRedirectStrategy())
53              .setDefaultCookieStore(new BasicCookieStore())
54              .setDefaultCredentialsProvider(credsProvider)
55              .build();
56    }
57  
58    /**
59     * close http client connections.
60     *
61     * @throws IOException if any occurs closing the http connection
62     */
63    @AfterAll
64    public static void tearDownClass() throws IOException {
65      client.close();
66    }
67  }
