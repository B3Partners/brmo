17  package nl.b3p.brmo.test.util.database;
18  
19  import nl.b3p.jdbc.util.converter.GeometryJdbcConverter;
20  import nl.b3p.jdbc.util.converter.GeometryJdbcConverterFactory;
21  import org.apache.commons.dbutils.QueryRunner;
22  import org.apache.commons.dbutils.handlers.ColumnListHandler;
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  
26  import javax.sql.DataSource;
27  import java.sql.Connection;
28  import java.sql.SQLException;
29  import java.util.List;
30  
31  public final class ViewUtils {
32      private static final Log LOG = LogFactory.getLog(ViewUtils.class);
33  
34      /**
35       * maak een lijst van de materialized views.
36       *
37       * @param ds (rsgb) database koppeling
38       * @return lijst met namen
39       * @throws SQLException als de query op de databron mislukt
40       */
41      public static List<String> listAllMaterializedViews(final DataSource ds) throws SQLException {
42          List<String> mviews;
43          try (Connection conn = ds.getConnection()) {
44              final GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(conn);
45              mviews = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(conn, geomToJdbc.getMViewsSQL(), new ColumnListHandler<>());
46              LOG.debug("Gevonden materialized views: " + mviews);
47          }
48          return mviews;
49      }
50  
51      /**
52       * refresh alle materialized views in de database.
53       * <i>NB</i> niet duidelijk of de database de views in de juiste volgorde teruggeeft...
54       *
55       * @param ds (rsgb) database koppeling
56       * @throws SQLException bij het benaderen van de database (nb als de refresh mislukt om wat voor reden wordt dat alleen gelogd)
57       */
58      public static void refreshAllMaterializedViews(final DataSource ds) throws SQLException {
59          List<String> mviews = listAllMaterializedViews(ds);
60          if (mviews != null && mviews.size() > 1) {
61              refreshMViews(mviews.toArray(new String[mviews.size()]), ds);
62          }
63      }
64  
65      /**
66       * refresh de gegeven lijst van materialized views in de database.
67       *
68       * @param mviews lijst van namen van views
69       * @param ds     database koppeling
70       * @throws SQLException bij het benaderen van de database (nb als de refresh mislukt om wat voor reden wordt dat alleen gelogd)
71       */
72      public static void refreshMViews(final String[] mviews, final DataSource ds) throws SQLException {
73          try (Connection conn = ds.getConnection()) {
74              conn.setAutoCommit(true);
75              final GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(conn);
76              for (String mview : mviews) {
77                  try {
78                      LOG.debug("Start verversen van materialized view: " + mview);
79                      // "update" gebruiken omdat we bij oracle stored procedure benaderen
80                      Object o = new QueryRunner(geomToJdbc.isPmdKnownBroken()).update(conn, geomToJdbc.getMViewRefreshSQL(mview));
81                      LOG.trace("Klaar met verversen van materialized view: " + mview);
82                  } catch (SQLException sqle) {
83                      LOG.error("Bijwerken van materialized view `" + mview + "` is mislukt. ", sqle);
84                  }
85              }
86          }
87      }
88  
89      private ViewUtils() {
90      }
91  }
