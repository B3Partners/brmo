6   import org.junit.jupiter.api.BeforeEach;
7   import org.junit.jupiter.api.Test;

9   import static org.junit.jupiter.api.Assertions.assertEquals;
10  
11  /**
12   * Testcase voor {@link CronFormatterTag}.
13   *
14   * @author mprins
15   */
16  public class CronFormatterTagTest {
17  
18      private CronFormatterTag tag;
19  
20      @BeforeEach
21      public void setUp() {
22          tag = new CronFormatterTag();
23      }
24  
25      @Test
26      public void testBuildOutputLocale() {
27          tag.setLocale("nl");
28          tag.setCronExpression("0 0 12 1/1 * ? *");
29          assertEquals("om 12:00 elke dag", tag.buildOutput(), "Tag should output 'om 12:00 elke dag'");
30      }
31  
32      @Test
33      public void testBuildOutputUKLocale() {
34          tag.setLocale("uk");
35          tag.setCronExpression("0 0 12 1/1 * ? *");
36          assertEquals("at 12:00 every day", tag.buildOutput(), "Tag should output 'at 12:00 every day'");
37      }
38  
39      @Test
40      public void testBuildOutputNoLocale() {
41          tag.setCronExpression("0 0 12 1/1 * ? *");
42          assertEquals("om 12:00 elke dag", tag.buildOutput(), "Tag should output 'om 12:00 elke dag'");
43      }
44  }
