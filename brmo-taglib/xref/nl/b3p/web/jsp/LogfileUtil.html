17  package nl.b3p.web.jsp;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.nio.file.DirectoryStream;
22  import java.nio.file.Files;
23  import java.nio.file.Path;
24  import java.nio.file.Paths;
25  import java.util.ArrayList;
26  import java.util.Enumeration;
27  import java.util.List;
28  import org.apache.commons.io.FilenameUtils;
29  import org.apache.commons.logging.Log;
30  import org.apache.commons.logging.LogFactory;
31  import org.apache.log4j.Appender;
32  import org.apache.log4j.FileAppender;
33  import org.apache.log4j.Logger;
34  
35  /**
36   *
37   * @author mprins
38   */
39  public final class LogfileUtil {
40      
41      private static final Log LOG = LogFactory.getLog(LogfileUtil.class);
42  
43      /**
44       * opzoeken van log file, de logger heeft de naam 'file' in de log4j
45       * properties.
46       *
47       * @return volledig pad naar actuele logfile 'file', mogelijk null. Het pad
48       * is de string die geconfigureerd is in de log4j properties
49       */
50      public static String getLogfile() {
51          String file = null;
52          Enumeration e = Logger.getRootLogger().getAllAppenders();
53          while (e.hasMoreElements()) {
54              Appender a = (Appender) e.nextElement();
55              if (a instanceof FileAppender && a.getName() != null && a.getName().equals("file")) {
56                  LOG.debug("Gevonden logfile (naam): " + ((FileAppender) a).getFile());
57                  file = ((FileAppender) a).getFile();
58                  break;
59              }
60          }
61          File f = new File(file);
62          return f.getAbsolutePath();
63      }
64  
65      /**
66       * zoek alle logfiles, ook geroteerde, op aan de hand van de basename van de
67       * actuele logfile.
68       *
69       * @return lijst met logfile namen incl. pad
70       * @see #getLogfile()
71       */
72      public static List<String> getLogfileList() {
73          List<String> files = new ArrayList<>();
74          // opzoeken van brmo log files, de logger heeft de naam 'file' in de log4j properties
75          final File f = new File(getLogfile());
76  
77          // filter op basename van actuele logfile zodat geroteerde files ook in de lijst komen
78          DirectoryStream.Filter<Path> filter = (Path p) -> (p.getFileName().toString().startsWith(FilenameUtils.getBaseName(f.getName())));
79  
80          try (DirectoryStream<Path> directoryStream = Files.newDirectoryStream(Paths.get(f.getParent()), filter)) {
81              for (Path path : directoryStream) {
82                  files.add(path.toString());
83              }
84          } catch (IOException ioe) {
85              LOG.warn("Lijst van logfiles ophalen is mislukt.", ioe);
86          }
87          return files;
88      }
89      
90      private LogfileUtil() {
91      }
92  }
