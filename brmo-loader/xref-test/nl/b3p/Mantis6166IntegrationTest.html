81    @Override
82    @BeforeEach
83    public void setUp() throws Exception {
84      dsStaging = new BasicDataSource();
85      dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
86      dsStaging.setUsername(params.getProperty("staging.user"));
87      dsStaging.setPassword(params.getProperty("staging.passwd"));
88      dsStaging.setAccessToUnderlyingConnectionAllowed(true);
89  
90      dsRsgb = new BasicDataSource();
91      dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
92      dsRsgb.setUsername(params.getProperty("rsgb.user"));
93      dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
94      dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
95  
96      rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
97      staging = new DatabaseDataSourceConnection(dsStaging);
98  
99      if (this.isOracle) {
100       rsgb =
101           new DatabaseConnection(
102               OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
103               params.getProperty("rsgb.user").toUpperCase());
104       rsgb.getConfig()
105           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
106       rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
107       staging =
108           new DatabaseConnection(
109               OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
110               params.getProperty("staging.user").toUpperCase());
111       staging
112           .getConfig()
113           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
114       staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
115     } else if (this.isPostgis) {
116       rsgb.getConfig()
117           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
118       staging
119           .getConfig()
120           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
121     } else {
122       fail("Geen ondersteunde database aangegegeven.");
123     }
124 
125     brmo = new BrmoFramework(dsStaging, dsRsgb, null);
126     brmo.setOrderBerichten(true);
127 
128     sequential.lock();
129   }

131   private void loadData(
132       final String bestandsNaam, final long aantalProcessen, final long aantalBerichten)
133       throws Exception {
134     assumeTrue(
135         Mantis6166IntegrationTest.class.getResource(bestandsNaam) != null,
136         "Het bestand met testdata zou moeten bestaan.");
137 
138     IDataSet stagingDataSet =
139         new XmlDataSet(
140             new FileInputStream(
141                 new File(Mantis6166IntegrationTest.class.getResource(bestandsNaam).toURI())));
142     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
143 
144     assumeTrue(aantalBerichten > 0, "Er zijn meer dan 0 berichten in het bestand om te laden");
145     assumeTrue(aantalProcessen > 0, "Er zijn meer dan 0 laadprocessen in het bestand om te laden");
146     assumeTrue(
147         aantalBerichten == brmo.getCountBerichten("brk", "STAGING_OK"),
148         "Er zijn x STAGING_OK berichten om te verwerken");
149     assumeTrue(
150         aantalProcessen == brmo.getCountLaadProcessen("brk", "STAGING_OK"),
151         "Er zijn x STAGING_OK laadprocessen in de database");
152   }

154   @AfterEach
155   public void cleanup() throws Exception {
156     if (brmo != null) {
157       // in geval van niet waar gemaakte assumptions
158       brmo.closeBrmoFramework();
159     }
160 
161     CleanUtil.cleanRSGB_BRK(rsgb, true);
162     rsgb.close();
163     dsRsgb.close();
164 
165     CleanUtil.cleanSTAGING(staging, false);
166     staging.close();
167     dsStaging.close();
168 
169     try {
170       sequential.unlock();
171     } catch (IllegalMonitorStateException e) {
172       // in geval van niet waar gemaakte assumptions
173       LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed");
174     }
175   }

178    * transformeer stand bericht.
179    *
180    * @throws Exception if any
181    */
182   @DisplayName("Stand")
183   @ParameterizedTest(name = "{index}: verwerken bestand: {0}")
184   @MethodSource("argumentsProvider")
185   public void testStand(
186       final String bestandsNaam,
187       final long aantalProcessen,
188       final long aantalBerichten,
189       final long stand,
190       final long verwijder,
191       final long[] mutaties)
192       throws Exception {
193     loadData(bestandsNaam, aantalProcessen, aantalBerichten);
194     Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_IDS, new long[] {stand}, null);
195     t.join();
196 
197     assertEquals(
198         aantalBerichten - 1,
199         brmo.getCountBerichten("brk", "STAGING_OK"),
200         "Mutatie berichten zijn niet getransformeerd");
201     assertEquals(1l, brmo.getCountBerichten("brk", "RSGB_OK"), "Een bericht is OK getransformeerd");
202 
203     ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
204     assertEquals(1, kad_perceel.getRowCount(), "Er is 1 perceel geladen");
205 
206     ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_perceel");
207     assertEquals(1, kad_onrrnd_zk.getRowCount(), "Er is 1 onroerende zaak geladen");
208   }

210   /**
211    * transformeer stand bericht en mutaties.
212    *
213    * @throws Exception if any
214    */
215   @ParameterizedTest(name = "Stand en mutatie #{index}: verwerken bestand: {0}")
216   @MethodSource("argumentsProvider")
217   public void testStandMutatie(
218       final String bestandsNaam,
219       final long aantalProcessen,
220       final long aantalBerichten,
221       final long stand,
222       final long verwijder,
223       final long[] mutaties)
224       throws Exception {
225     loadData(bestandsNaam, aantalProcessen, aantalBerichten);
226     long[] transformIds = new long[mutaties.length + 1];
227     transformIds[0] = stand;
228     for (int i = 1; i < transformIds.length; i++) {
229       transformIds[i] = mutaties[i - 1];
230     }
231     Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_IDS, transformIds, null);
232     t.join();
233 
234     assertEquals(
235         aantalBerichten - 1,
236         brmo.getCountBerichten("brk", "RSGB_OK"),
237         "Alle berichten behalve verwijderen zijn OK getransformeerd");
238     assertEquals(
239         1l,
240         brmo.getCountBerichten("brk", "STAGING_OK"),
241         "Een (verwijder) bericht is niet getransformeerd en heeft STAGING_OK status");
242 
243     ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
244     assertEquals(1, kad_perceel.getRowCount(), "Er is 1 perceel geladen");
245 
246     ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
247     assertEquals(1, kad_onrrnd_zk.getRowCount(), "Er is 1 onroerende zaak geladen");
248 
249     ITable kad_perceel_archief = rsgb.createDataSet().getTable("kad_perceel_archief");
250     assertEquals(
251         transformIds.length - 1,
252         kad_perceel_archief.getRowCount(),
253         "Er zijn (verwerkt aantal -1) percelen gearchiveerd");
254 
255     ITable kad_onrrnd_zk_archief = rsgb.createDataSet().getTable("kad_onrrnd_zk_archief");
256     assertEquals(
257         transformIds.length - 1,
258         kad_onrrnd_zk_archief.getRowCount(),
259         "Er zijn (verwerkt aantal -1) onroerende zaken gearchiveerd");
260   }
