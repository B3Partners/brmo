58      @Override
59      @BeforeEach
60      public void setUp() throws Exception {
61          dsStaging = new BasicDataSource();
62          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
63          dsStaging.setUsername(params.getProperty("staging.user"));
64          dsStaging.setPassword(params.getProperty("staging.passwd"));
65          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
66  
67          dsRsgb = new BasicDataSource();
68          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
69          dsRsgb.setUsername(params.getProperty("rsgb.user"));
70          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
71          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
72  
73          rsgb = new DatabaseDataSourceConnection(dsRsgb);
74          staging = new DatabaseDataSourceConnection(dsStaging);
75  
76          if (this.isMsSQL) {
77              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
78              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
79          } else if (this.isOracle) {
80              rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
81              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
82              rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
83              staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
84              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
85              staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
86          } else if (this.isPostgis) {
87              // we hebben alleen nog postgres over
88              rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
89              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
90          }
91  
92          IDataSet stagingDataSet = new XmlDataSet(new FileInputStream(new File(Mantis6380IntegrationTest.class.getResource("/mantis6380/staging.xml").toURI())));
93          sequential.lock();
94  
95          if (this.isMsSQL) {
96              InsertIdentityOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
97          } else {
98              DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
99          }
100         brmo = new BrmoFramework(dsStaging, dsRsgb);
101 
102         // skip als de bron data er niet is
103         Assumptions.assumeTrue(2L == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
104                 "Er zijn geen 2 STAGING_OK berichten");
105         Assumptions.assumeTrue(2L == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"),
106                 "Er zijn geen 2 STAGING_OK laadproces");
107     }

116 
117         CleanUtil.cleanSTAGING(staging, false);
118         staging.close();
119         dsStaging.close();
120 
121         sequential.unlock();
122     }

124     @Test
125     public void testTransformStand() throws Exception {
126         brmo.setOrderBerichten(false);
127         Thread t = brmo.toRsgb();
128         t.join();
129 
130         assertEquals(
131                 1L, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
132                 "Niet alle berichten zijn OK getransformeerd");
133         assertEquals(
134                 1L, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"), "Niet alle berichten zijn OK getransformeerd");
135         assertEquals(0L, brmo.getCountBerichten(null, null, "brk", "RSGB_NOK"),
136                 "Er zijn berichten met status RSGB_NOK");
137 
138         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
139         assertEquals(1, kad_onrrnd_zk.getRowCount(), "Aantal actuele onroerende zaken is incorrect");
140 
141         ITable kad_perceel = rsgb.createDataSet().getTable("kad_perceel");
142         assertEquals(1, kad_perceel.getRowCount(), "Aantal actuele percelen is incorrect");
143         assertEquals("37640054270000", kad_perceel.getValue(0, "sc_kad_identif").toString(),
144                 "Perceel identif is incorrect");
145 
146         ITable subject = rsgb.createDataSet().getTable("subject");
147         assertEquals(1, subject.getRowCount(), "Aantal subjecten klopt niet");
148 
149         ITable ingeschr_niet_nat_prs = rsgb.createDataSet().getTable("ingeschr_niet_nat_prs");
150         assertEquals(1, ingeschr_niet_nat_prs.getRowCount(), "Aantal ingeschr_niet_nat_prs klopt niet");
151 
152         ITable zak_recht = rsgb.createDataSet().getTable("zak_recht");
153         assertEquals(2, zak_recht.getRowCount(), "Aantal zakelijke rechten klopt niet");
154         // rij 1
155         assertEquals("NL.KAD.Tenaamstelling.AKR1.100000007757634", zak_recht.getValue(0, "kadaster_identif").toString());
156         assertEquals("1", zak_recht.getValue(0, "ar_noemer").toString());
157         assertEquals("1", zak_recht.getValue(0, "ar_teller").toString());
158         // rij 2
159         assertEquals("NL.KAD.ZakelijkRecht.AKR1.100000007757634", zak_recht.getValue(1, "kadaster_identif").toString());
160         assertNull(zak_recht.getValue(1, "ar_noemer"), "noemer is niet null");
161         assertNull(zak_recht.getValue(1, "ar_teller"), "teller is niet null");
162     }

284      * daarna mutatie bericht en test of dat correct is gedaan. Verwacht een
285      * AssertionError omdat het aantal zakelijk rechten dan niet correct is
286      * (het oude zakelijk recht wordt niet opgeruimd).
287      *

289     @Test
290     public void testTransformDeleteStandDaarnaMutatieBerichten() {
291         assertThrows(AssertionError.class, ()->{
292             brmo.setOrderBerichten(false);
293             Thread t = brmo.toRsgb();
294             t.join();
295 
296             assertEquals(
297                     1L, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
298                     "Niet alle berichten zijn OK getransformeerd");
299             assertEquals(
300                     1L, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"), "Niet alle berichten zijn OK getransformeerd");
301             assertEquals(0L, brmo.getCountBerichten(null, null, "brk", "RSGB_NOK"),
302                     "Er zijn berichten met status RSGB_NOK");
303 
304             // delete stand bericht
305             brmo.delete(1L);
306 
307             assertEquals(
308                     1L, brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
