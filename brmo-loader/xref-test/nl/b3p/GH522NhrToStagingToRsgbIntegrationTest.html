279   @DisplayName("NHR to STAGING to RSGB")
280   @ParameterizedTest(name = "case #{index}: bestand: {0}")
281   @MethodSource("argumentsProvider")
282   public void testNhrXMLToStagingToRsgb(String bestandNaam, Map<String, Integer> rowCounts)
283       throws Exception {
284 
285     assumeFalse(
286         null == GH522NhrToStagingToRsgbIntegrationTest.class.getResource(bestandNaam),
287         "Het test bestand moet er zijn.");
288 
289     brmo.loadFromFile(
290         "nhr",
291         GH522NhrToStagingToRsgbIntegrationTest.class.getResource(bestandNaam).getFile(),
292         null);
293     LOG.info("klaar met laden van berichten in staging DB.");
294 
295     // alleen het eerste bericht heeft br_orgineel_xml, de rest niet
296     ITable bericht =
297         staging.createQueryTable("bericht", "select * from bericht where volgordenummer=0");
298     assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
299     LOG.debug("\n\n" + bericht.getValue(0, "br_orgineel_xml") + "\n\n");
300     assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null");
301     Object berichtId = bericht.getValue(0, "id");
302 
303     bericht =
304         staging.createQueryTable(
305             "bericht", "select * from bericht where object_ref like 'nhr.%Vestg%'");
306     assertEquals(
307         rowCounts.get("vestg"),
308         Integer.valueOf(bericht.getRowCount()),
309         "aantal (niet)commerciele vestiging berichten onjuist");
310 
311     LOG.info("Transformeren berichten naar rsgb DB.");
312     Thread t = brmo.toRsgb();
313     t.join();
314 
315     // na de verwerking moet soap payload er ook nog zijn
316     bericht =
317         staging.createQueryTable(
318             "bericht", "select * from bericht where br_orgineel_xml is not null");
319     assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
320     assertNotNull(
321         bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null na transformatie");
322     assertEquals(
323         berichtId,
324         bericht.getValue(0, "id"),
325         "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
326 
327     // check RSGB tabellen voor aantallen
328     for (String table : rowCounts.keySet()) {
329       checkRowCount(table, rowCounts.get(table));
330     }
331   }
332 
333   private void checkRowCount(String table, int aantal) throws SQLException, DataSetException {
334     ITable _tbl;
335     if (table.startsWith("select ")) {
336       List<String> words = Arrays.asList(table.split(" "));
337       _tbl = rsgb.createQueryTable(words.get(words.indexOf(table) + 1), table);
338     } else {
339       _tbl = rsgb.createTable(table);
340     }
341 
342     assertEquals(aantal, _tbl.getRowCount(), "aantal '" + table + "' records klopt niet");
343   }
