111     @BeforeEach
112     @Override
113     public void setUp() throws Exception {
114         BasicDataSource dsStaging = new BasicDataSource();
115         dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
116         dsStaging.setUsername(params.getProperty("staging.user"));
117         dsStaging.setPassword(params.getProperty("staging.passwd"));
118         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
119         dsStaging.setConnectionProperties(params.getProperty("staging.options", ""));
120 
121         BasicDataSource dsRsgb = new BasicDataSource();
122         dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
123         dsRsgb.setUsername(params.getProperty("rsgb.user"));
124         dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
125         dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
126         dsRsgb.setConnectionProperties(params.getProperty("rsgb.options", ""));
127 
128         staging = new DatabaseDataSourceConnection(dsStaging);
129         rsgb = new DatabaseDataSourceConnection(dsRsgb);
130 
131         if (this.isMsSQL) {
132             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
133             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
134         } else if (this.isOracle) {
135             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
136             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
137             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
138 
139             rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
140             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
141             rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
142         } else if (this.isPostgis) {
143             // we hebben alleen nog postgres over
144             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
145             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
146         }
147 
148         brmo = new BrmoFramework(dsStaging, dsRsgb);
149         brmo.setOrderBerichten(true);
150 
151         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
152         fxdb.setCaseSensitiveTableNames(false);
153         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(NhrToStagingToRsgbIntegrationTest.class.getResource("/staging-empty-flat.xml").toURI())));
154 
155         sequential.lock();
156 
157         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
158 
159         Assumptions.assumeTrue(0L == brmo.getCountBerichten(null, null, "nhr", "STAGING_OK"),
160                 "Er zijn geen STAGING_OK berichten");
161         Assumptions.assumeTrue(0L == brmo.getCountLaadProcessen(null, null, "nhr", "STAGING_OK"),
162                 "Er zijn geen STAGING_OK laadprocessen");
163     }

168 
169         CleanUtil.cleanSTAGING(staging, false);
170         staging.close();
171 
172         CleanUtil.cleanRSGB_NHR(rsgb);
173         rsgb.close();
174 
175         sequential.unlock();
176     }

178     @DisplayName("NHR to STAGING to RSGB")
179     @ParameterizedTest(name = "case #{index}: bestand: {0}")
180     @MethodSource("argumentsProvider")
181     public void testNhrXMLToStagingToRsgb(String bestandNaam, long aantalBerichten, long aantalProcessen,
182                                           long aantalPrs, long aantalSubj, long aantalNiet_nat_prs, long aantalNat_prs, String vestgID,
183                                           long aantalVestg_activiteit, long kvkNummer, String[] sbiCodes, int aantalFunctionarissen)
184             throws Exception {
185         assumeFalse(null == NhrToStagingToRsgbIntegrationTest.class.getResource(bestandNaam), "Het test bestand moet er zijn.");
186 
187         brmo.loadFromFile(BESTANDTYPE, NhrToStagingToRsgbIntegrationTest.class.getResource(bestandNaam).getFile(), null);
188         LOG.info("klaar met laden van berichten in staging DB.");
189 
190         List<Bericht> berichten = brmo.listBerichten();
191         List<LaadProces> processen = brmo.listLaadProcessen();
192         assertNotNull(berichten, "De verzameling berichten bestaat niet.");
193         assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
194         assertNotNull(processen, "De verzameling processen bestaat niet.");
195         assertEquals(aantalProcessen, processen.size(), "Het aantal processen is niet als verwacht.");
196 
197         // alleen het eerste bericht heeft br_orgineel_xml, de rest niet
198         ITable bericht = staging.createQueryTable("bericht", "select * from bericht where volgordenummer=0");
199         assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
200         LOG.debug("\n\n" + bericht.getValue(0, "br_orgineel_xml") + "\n\n");
201         assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null");
202         Object berichtId = bericht.getValue(0, "id");
203 
204         LOG.info("Transformeren berichten naar rsgb DB.");
205         Thread t = brmo.toRsgb();
206         t.join();
207 
208         // na de verwerking moet soap payload er ook nog zijn
209         bericht = staging.createQueryTable("bericht", "select * from bericht where br_orgineel_xml is not null");
210         assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
211         assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null na transformatie");
212         assertEquals(berichtId, bericht.getValue(0, "id"),
213                 "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
214 
215         assertEquals(aantalBerichten, brmo.getCountBerichten(null, null, "nhr", "RSGB_OK"),
216                 "Niet alle berichten zijn OK getransformeerd");
217         berichten = brmo.listBerichten();
218         for (Bericht b : berichten) {
219             assertNotNull(b, "Bericht is 'null'");
220             assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
221         }
222 
223         ITable prs = rsgb.createDataSet().getTable("prs");
224         ITable niet_nat_prs = rsgb.createDataSet().getTable("niet_nat_prs");
225         ITable nat_prs = rsgb.createDataSet().getTable("nat_prs");
226         ITable subject = rsgb.createDataSet().getTable("subject");
227 
228         assertEquals(aantalPrs, prs.getRowCount(), "Het aantal 'prs' records klopt niet");
229         assertEquals(aantalNiet_nat_prs, niet_nat_prs.getRowCount(),
230                 "Het aantal 'niet_nat_prs' records klopt niet");
231         assertEquals(aantalNat_prs, nat_prs.getRowCount(), "Het aantal 'nat_prs' records klopt niet");
232         assertEquals(aantalSubj, subject.getRowCount(), "het aantal 'subject' records klopt niet");
233 
234         boolean foundKvk = false;
235         for (int i = 0; i < subject.getRowCount(); i++) {
236             if (subject.getValue(i, "identif").toString().contains("nhr.maatschAct.kvk")) {
237                 assertEquals(kvkNummer + "", subject.getValue(i, "kvk_nummer") + "", "KVK nummer klopt niet");
238                 foundKvk = true;
239             }
240         }
241         if (!foundKvk) {
242             fail("KVK nummer maatschappelijke activiteit klopt niet, verwacht te vinden: " + kvkNummer);
243         }
244 
245         if (vestgID != null) {
246             ITable vestg = rsgb.createDataSet().getTable("vestg");
247             assertEquals(vestgID, vestg.getValue(0, "sc_identif"),
248                     "De 'sc_identif' van hoofdvestiging klopt niet");
249             assertEquals(sbiCodes[0], vestg.getValue(0, "fk_sa_sbi_activiteit_sbi_code"),
250                     "De sbi code van (hoofd)vestiging klopt niet");
251             assertEquals("Ja", vestg.getValue(0, "sa_indic_hoofdactiviteit"),
252                     "De 'sa_indic_hoofdactiviteit' van (hoofd)vestiging klopt niet");
253         }
254         // TODO check aantal vestigingen, zie ook: #522, nu hooguit 1 hoofdVestg...
255         // assertEquals("het aantal 'vestg' records klopt niet", aantalVestg, vestg.getRowCount());
256 
257         ITable vestg_activiteit = rsgb.createDataSet().getTable("vestg_activiteit");
258         assertEquals(aantalVestg_activiteit, vestg_activiteit.getRowCount(),
259                 "Het aantal 'vestg_activiteit' records klopt niet");
260 
261         ITable sbi_activiteit = rsgb.createDataSet().getTable("sbi_activiteit");
262         ITable herkomst_metadata = rsgb.createDataSet().getTable("herkomst_metadata");
263         for (String sbiCode : sbiCodes) {
264             boolean foundSbiCode = false;
265             boolean foundSbiCodeMeta = false;
266             for (int i = 0; i < sbi_activiteit.getRowCount(); i++) {
267                 if (sbiCode.equals(sbi_activiteit.getValue(i, "sbi_code").toString())) {
268                     LOG.debug("SBI code " + sbiCode + " gevonden in sbi_activiteit");
269                     foundSbiCode = true;
270                     break;
271                 }
272             }
273             if (!foundSbiCode) {
274                 fail("SBI code niet gevonden in sbi_activiteit, verwacht te vinden: " + sbiCode);
275             }
276             for (int i = 0; i < herkomst_metadata.getRowCount(); i++) {
277                 if (sbiCode.equals(herkomst_metadata.getValue(i, "waarde").toString())) {
278                     LOG.debug("SBI code " + sbiCode + " gevonden in herkomst_metadata");
279                     foundSbiCodeMeta = true;
280                     break;
281                 }
282             }
283             if (!foundSbiCodeMeta) {
284                 fail("SBI code niet gevonden in herkomst_metadata, verwacht te vinden: " + sbiCode);
285             }
286         }
287 
288         ITable functionaris = rsgb.createDataSet().getTable("functionaris");
289         assertEquals(aantalFunctionarissen, functionaris.getRowCount(),
290                 "Het aantal 'functionaris' records klopt niet");
291         if (kvkNummer == 41177576) {
292             // gebruik een gesorteerde weergaven van functionaris tabel omdat de volgorde van de rijen anders kan zijn terwijl je dan toch dezelfde data hebt geladen
293             // er is een Gevolmachtigde Directeur met beperkte volmacht in de laatste rij
294             functionaris = rsgb.createQueryTable("functionaris", "select * from functionaris order by functie ASC");
295             assertEquals("Gevolmachtigde", functionaris.getValue(aantalFunctionarissen - 1, "functie"));
296             assertEquals("Directeur", functionaris.getValue(aantalFunctionarissen - 1, "functionaristypering"));
297             assertEquals("B", functionaris.getValue(aantalFunctionarissen - 1, "volledig_beperkt_volmacht"));
298         }
