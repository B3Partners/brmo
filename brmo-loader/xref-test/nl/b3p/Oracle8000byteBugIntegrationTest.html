81      private final Lock sequential = new ReentrantLock();

83      @Override
84      @BeforeEach
85      public void setUp() throws Exception {
86          assumeTrue(this.isOracle, "Deze test is alleen voor Oracle.");
87  
88          BasicDataSource dsStaging = new BasicDataSource();
89          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
90          dsStaging.setUsername(params.getProperty("staging.user"));
91          dsStaging.setPassword(params.getProperty("staging.passwd"));
92          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
93  
94          BasicDataSource dsRsgb = new BasicDataSource();
95          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
96          dsRsgb.setUsername(params.getProperty("rsgb.user"));
97          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
98          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
99  
100         staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
101         staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
102         staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
103 
104         rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
105         rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
106         rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
107 
108         brmo = new BrmoFramework(dsStaging, dsRsgb);
109 
110         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
111         fxdb.setCaseSensitiveTableNames(false);
112         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(Oracle8000byteBugIntegrationTest.class.getResource("/staging-empty-flat.xml").toURI())));
113 
114         sequential.lock();
115 
116         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
117         assumeTrue(0L == brmo.getCountBerichten(null, null, "brk", "STAGING_OK"),
118                 "Er zijn geen STAGING_OK berichten");
119         assumeTrue(0L == brmo.getCountLaadProcessen(null, null, "brk", "STAGING_OK"),
120                 "Er zijn geen STAGING_OK laadprocessen");
121     }

123     @AfterEach
124     public void cleanup() throws Exception {
125         if (brmo != null) {
126             brmo.closeBrmoFramework();
127         }
128 
129         if (rsgb != null) {
130             CleanUtil.cleanRSGB_BRK(rsgb, true);
131             rsgb.close();
132         }
133 
134         if (staging != null) {
135             CleanUtil.cleanSTAGING(staging, false);
136             staging.close();
137         }
138 
139         try {
140             sequential.unlock();
141         } catch (IllegalMonitorStateException e) {
142             // in geval van niet waar gemaakte assumptions
143             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed");
144         }
145     }

147     @ParameterizedTest(name = "testBrkXMLToStaging {index}: bestand: {1}")
148     @MethodSource("argumentsProvider")
149     public void testBrkXMLToStaging(String bestandType, String bestandNaam, int brxml_bytes, long aantalBerichten, long aantalProcessen, String datumEersteMutatie) throws Exception {
150         assumeFalse(null == Oracle8000byteBugIntegrationTest.class.getResource(bestandNaam), "Het test bestand moet er zijn.");
151 
152         brmo.loadFromFile(bestandType, Oracle8000byteBugIntegrationTest.class.getResource(bestandNaam).getFile(), null);
153         LOG.debug("klaar met laden van berichten in staging DB.");
154 
155         List<Bericht> berichten = brmo.listBerichten();
156         List<LaadProces> processen = brmo.listLaadProcessen();
157         assertNotNull(berichten, "De verzameling berichten bestaat niet.");
158         assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
159         assertNotNull(processen, "De verzameling processen bestaat niet.");
160         assertEquals(aantalProcessen, processen.size(), "Het aantal processen is niet als verwacht.");
161 
162         ITable bericht = staging.createDataSet().getTable("bericht");
163         assertEquals(aantalBerichten, bericht.getRowCount(), "Het aantal berichten klopt niet");
164         assertEquals(datumEersteMutatie, bericht.getValue(0, "datum").toString(), "De datum klopt niet");
165 
166         assertEquals(brxml_bytes, bericht.getValue(0, "br_xml").toString().getBytes().length,
167                 "incorrect aantal bytes");
168         assertNull(bericht.getValue(0, "db_xml"), "DB xml is niet null (want nog niet getransformeerd)");
169         assertFalse(bericht.getValue(0, "br_xml").toString().contains("? x m l"), "BR xml is vernachelt");
170         assertFalse(bericht.getValue(0, "br_xml").toString().startsWith(" < ? x m l"),
171                 "BR xml is vernachelt");
172 
173         LOG.debug("Transformeren berichten naar rsgb DB.");
174         Thread t = brmo.toRsgb();
175         t.join();
176 
177         assertEquals(aantalBerichten, brmo.getCountBerichten(null, null, "brk", "RSGB_OK"),
178                 "Niet alle berichten zijn OK getransformeerd");
179 
180         berichten = brmo.listBerichten();
181         for (Bericht b : berichten) {
182             assertNotNull(b, "Bericht is 'null'");
183             assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
184             // test toepassing van de 8000byte HACK in de stagingproxy
185             int dflt = b.getDbXml().getBytes().length;
186             int utf8 = b.getDbXml().getBytes(StandardCharsets.UTF_8).length;
187             assertNotEquals(8000, dflt, "db xml is 8000 bytes");
188             assertNotEquals(8000, utf8, "db xml in UTF8 is 8000 bytes");
189 
190             dflt = b.getBrXml().getBytes().length;
191             utf8 = b.getBrXml().getBytes(StandardCharsets.UTF_8).length;
192             assertNotEquals(8000, dflt, "br xml is 8000 bytes");
193             assertNotEquals(8000, utf8, "br xml in UTF8 is 8000 bytes");
194         }
195 
196         bericht = staging.createDataSet().getTable("bericht");
197         assertEquals(aantalBerichten, bericht.getRowCount(), "Het aantal berichten");
198         assertEquals(brxml_bytes, bericht.getValue(0, "br_xml").toString().getBytes().length, "8000 bytes");
199         assertFalse(bericht.getValue(0, "br_xml").toString().contains("? x m l"), "BR xml is vernachelt");
200         assertFalse(bericht.getValue(0, "br_xml").toString().startsWith(" < ? x m l"),
201                 "BR xml is vernachelt");
202 
203         ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
204         assertEquals(aantalBerichten, kad_onrrnd_zk.getRowCount(), "Het aantal onroerende zaken klopt niet.");
205     }
