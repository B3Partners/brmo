72  @Category(JTDSDriverBasedFailures.class)

74  
75      private final static Log LOG = LogFactory.getLog(Top250NLIntegrationTest.class);
76  
77      @BeforeClass
78      public static void setUpClass() {
79      }
80  
81      @AfterClass
82      public static void tearDownClass() {
83      }
84  
85      private final Lock sequential = new ReentrantLock();
86      private IDatabaseConnection staging;
87      private IDatabaseConnection topnl;
88      private BrmoFramework brmo;
89  
90      @Before
91      public void setUp() throws Exception {
92          BasicDataSource dsStaging = new BasicDataSource();
93          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
94          dsStaging.setUsername(params.getProperty("staging.user"));
95          dsStaging.setPassword(params.getProperty("staging.passwd"));
96          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
97          staging = new DatabaseDataSourceConnection(dsStaging);
98  
99          BasicDataSource dsTopnl = new BasicDataSource();
100         dsTopnl.setUrl(params.getProperty("topnl.jdbc.url"));
101         dsTopnl.setUsername(params.getProperty("topnl.user"));
102         dsTopnl.setPassword(params.getProperty("topnl.passwd"));
103         dsTopnl.setAccessToUnderlyingConnectionAllowed(true);
104         topnl = new DatabaseDataSourceConnection(dsTopnl);
105 
106         brmo = new BrmoFramework(dsStaging, null, dsTopnl);
107 
108         if (this.isMsSQL) {
109             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
110             topnl.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
111         } else if (this.isOracle) {
112             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
113             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
114             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
115 
116             topnl = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsTopnl.getConnection()), params.getProperty("rsgb.user").toUpperCase());
117             topnl.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
118             topnl.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
119         } else if (this.isPostgis) {
120             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
121             topnl.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
122         }
123         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
124         fxdb.setCaseSensitiveTableNames(false);
125         // lege bericht en laadproces tabellen
126         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(Top250NLIntegrationTest.class.getResource("/staging-empty-flat.xml").toURI())));
127 
128         sequential.lock();
129 
130         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
131 
132         assumeTrue("Er zijn STAGING_OK berichten", 0l == brmo.getCountBerichten(null, null, BR_BAG, "STAGING_OK"));
133         assumeTrue("Er zijn STAGING_OK laadprocessen", 0l == brmo.getCountLaadProcessen(null, null, BR_BAG, "STAGING_OK"));
134     }
135 
136     @After
137     public void cleanup() throws Exception {
138         brmo.closeBrmoFramework();
139 
140         //   CleanUtil.cleanSTAGING(staging);
141         staging.close();
142 
143         topnl.close();
144         try {
145             sequential.unlock();
146         } catch (IllegalMonitorStateException e) {
147             // in geval van niet waar gemaakte assumptions
148             LOG.debug("unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
149         }
150     }

152     @Test
153     public void loadInStagingenProcess() throws BrmoException {
154         brmo.loadFromFile(BR_TOPNL, Top250NLIntegrationTest.class.getResource("/topnl/top250nl.gml").getFile());
155         final List<LaadProces> lpList = brmo.listLaadProcessen();
156         assertEquals("Er is 1 laadproces.", 1, lpList.size());
157         assertEquals("Het gebied is Nederland.", "Nederland", lpList.get(0).getGebied());
158     }
159 
160     @Test
161     public void process() throws BrmoException, InterruptedException {
162         brmo.loadFromFile(BR_TOPNL, Top250NLIntegrationTest.class.getResource("/topnl/top250nl.gml").getFile());
163         final List<LaadProces> lpList = brmo.listLaadProcessen();
164         assertEquals("Er is 1 laadproces.", 1, lpList.size());
165 
166         List<Long> ids = new ArrayList<>();
167         for (String type : new String[]{"Top10NL", "Top50NL", "Top100NL", "Top250NL", "topnl"}) {
168             ids.addAll(Arrays.asList(brmo.getLaadProcessenIds("bestand_datum", "ASC", type, "STAGING_OK")));
169         }
170         long[] lpIds = ArrayUtils.toPrimitive(ids.toArray(new Long[ids.size()]));
171 
172         Thread t = brmo.toRsgb(RsgbProxy.BerichtSelectMode.BY_LAADPROCES, lpIds, (ProgressUpdateListener) null);
173         t.join();
174 
175         assertEquals("Alles OK", LaadProces.STATUS.RSGB_TOPNL_OK, lpList.get(0).getStatus());
176 
177     }
