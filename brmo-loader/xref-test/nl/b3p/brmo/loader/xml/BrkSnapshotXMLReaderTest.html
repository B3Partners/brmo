48      /**
49       * Test next() methode met klein mutatie bestand.
50       *
51       * @throws Exception if any
52       */
53      @Test
54      public void mutTestSmallXML() throws Exception {
55          BrkSnapshotXMLReader bReader;
56          BrkBericht brk = null;
57          
58          bReader = new BrkSnapshotXMLReader(BrkSnapshotXMLReader.class.getResourceAsStream(mutSmallXml));
59          assertTrue(bReader.hasNext());
60          int total = 0;
61          while (bReader.hasNext()) {
62              brk = bReader.next();
63              System.out.println(
64                      "datum: " + brk.getDatum()
65                      + "\tobj ref: " + brk.getObjectRef()
66                      + "\tvolgordenummer: " + brk.getVolgordeNummer()
67              );
68              total++;
69          }
70          assertEquals(mutSmallXmlNieuwCount, total);
71          assertNotNull(brk);
72          assertEquals("NL.KAD.OnroerendeZaak:53860293910012", brk.getObjectRef());
73          assertEquals(new SimpleDateFormat("yyyy-MM-dd").parse("2009-11-30"),
74                  brk.getDatum());
75          assertEquals((Integer)1, brk.getVolgordeNummer());
76      }

78      /**
79       * Test next() methode met groter mutatie bestand en vergelijk parsen van
80       * file met en zonder extra regeleinden.
81       *
82       * @throws Exception if any
83       */
84      @Test
85      public void mutTestLargeXML() throws Exception {
86          BrkSnapshotXMLReader bReader;
87          BrkBericht brk = null;
88          BrkBericht brkReformatted = null;
89  
90          int total = 0;
91          bReader = new BrkSnapshotXMLReader(BrkSnapshotXMLReader.class.getResourceAsStream(mutXML));
92          assertTrue(bReader.hasNext());
93          while (bReader.hasNext()) {
94              brk = bReader.next();
95              total++;
96          }
97          assertEquals(mutXmlNieuwCount, total);
98          assertNotNull(brk);
99          
100         bReader = new BrkSnapshotXMLReader(BrkSnapshotXMLReader.class.getResourceAsStream(mutXMLReformatted));
101         assertTrue(bReader.hasNext());
102         total = 0;
103         while (bReader.hasNext()) {
104             brkReformatted = bReader.next();
105             total++;
106         }
107         assertEquals(mutXmlNieuwCount, total);
108         assertNotNull(brkReformatted);
109         // vergelijk resultaat van de twee readers
110         assertEquals(
111                 brk.getObjectRef(), brkReformatted.getObjectRef(),
112                 "van beide laatste mutaties moet ObjectRef hetzelfde zijn");
113         assertEquals(
114                 brk.getDatum(), brkReformatted.getDatum(),
115                 "van beide laatste mutaties moet datum van laatste hetzelfde zijn");
116         assertEquals(
117                 brk.getVolgordeNummer(), brkReformatted.getVolgordeNummer(),
118                 "van beide laatste mutaties moet VolgordeNummer hetzelfde zijn");
119     }
120     /**
121      * Test next() methode met xml bestanden in een zip met mutaties.
122      *
123      * @throws Exception if any
124      */
125     @Test
126     public void mutTestZipFile() throws Exception {
127         ZipInputStream zis = new ZipInputStream(
128                 BrkSnapshotXMLReader.class.getResourceAsStream(mutZipName));
129         int total = 0;
130         try {
131             ZipEntry entry = zis.getNextEntry();
132             while (entry != null) {
133                 BrkSnapshotXMLReader brkreader = new BrkSnapshotXMLReader(CloseShieldInputStream.wrap(zis));
134 
135                 assertEquals(mutXmlsInZip[0], entry.getName(), "De bestandsnaam moet kloppen.");
136                 while (brkreader.hasNext()) {
137                     BrkBericht brk = brkreader.next();
138                     total++;
139                 }
140                 entry = zis.getNextEntry();
141             }
142         } finally {
143             zis.close();
144         }
145         assertEquals(mutZipNameNieuwCount, total, "Verwacht dat het aantal mutaties gelijk is.");
146     }
