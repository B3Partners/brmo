20  @RunWith(Parameterized.class)
21  public class BGTLightFileReaderTest {
22  
23      private final SimpleDateFormat fmt = new SimpleDateFormat("yyyy-MM-dd");
24      /**
25       * test parameter.
26       */
27      private final String bestandNaam;
28      /**
29       * test parameter.
30       */
31      private final String datum;
32      /**
33       * test expectation.
34       */
35      private final String gebied;
36  
37      @Parameters
38      public static Collection params() {
39          return Arrays.asList(new Object[][]{
40              // {"filename","date", "gebied"},
41              {File.separator + "bgt-gml-loader" + File.separator + "data" + File.separator + "38468_0-20160429.zip", "2016-04-29", "grid 38468 aggrlevel 0"},
42              {"38468_0-20160429.zip", "2016-04-29", "grid 38468 aggrlevel 0"}
43          });
44      }
45  
46      public BGTLightFileReaderTest(String bestandNaam, String datum, String gebied) {
47          this.bestandNaam = bestandNaam;
48          this.datum = datum;
49          this.gebied = gebied;
50      }
51  
52      @Test(expected = NullPointerException.class)
53      public void testNullBestandsNaam() throws Exception {
54          BGTLightFileReader r = new BGTLightFileReader(null);
55      }
56  
57      // TODO omdat deze bestanden niet bestaan...
58      @Test(expected = java.io.FileNotFoundException.class)
59      public void testBGTLightXMLReader() throws Exception {
60          BGTLightFileReader r = new BGTLightFileReader(bestandNaam);
61          r.setDatumAsString(datum);
62  
63          assertEquals("Naam moet identiek zijn", bestandNaam, r.getBestandsNaam());
64          assertEquals("Gebied moet overeen komen", gebied, r.getGebied());
65          assertEquals("Datum moet overeen komen", fmt.parse(datum), r.getBestandsDatum());
66      }
67  }
