35  /**
36   * Testcase voor GbavXMLReader.
37   *
38   * @author Mark Prins
39   */
40  public class GbavXMLReaderTest {
41  
42      private static final Log LOG = LogFactory.getLog(GbavXMLReaderTest.class);
43  
44      @Test
45      public void berichtTest() throws Exception {
46          GbavXMLReader r;
47          GbavBericht b = null;
48          assumeNotNull("testdata moet aanwezig zijn", GbavXMLReaderTest.class.getResourceAsStream("gbav-voorbeeld.xml"));
49          r = new GbavXMLReader(GbavXMLReader.class.getResourceAsStream("gbav-voorbeeld.xml"));
50          assertTrue(r.hasNext());
51          int total = 0;
52          List<String> objectRefs = new ArrayList();
53          while (r.hasNext()) {
54              b = r.next();
55              objectRefs.add(b.getObjectRef());
56              LOG.debug(String.format("Bericht #%d van %tF %2$tT, object ref %s",
57                      b.getVolgordeNummer(),
58                      b.getDatum(),
59                      b.getObjectRef()
60              ));
61              total++;
62              assertEquals("Soort komt niet overeen", BrmoFramework.BR_GBAV, b.getSoort());
63              assertEquals("Volgordenummer komt niet overeen", 0, b.getVolgordeNummer().intValue());
64              assertEquals("Datum komt niet overeen", new Date(2018 - 1900, 1, 1), b.getDatum());
65          }
66  
67          assertEquals(1, total);
68          assertArrayEquals(new String[]{GbavXMLReader.PREFIX + "b3e513383d0454de798b29c6900643130e3b2d6b",}, objectRefs.toArray(new String[]{}));
69      }
70  
71      @Test
72      public void meerderePersonenTest() throws Exception {
73          GbavXMLReader r;
74          GbavBericht b = null;
75  
76          assumeNotNull("testdata moet aanwezig zijn", GbavXMLReaderTest.class.getResourceAsStream("persoonslijst.xml"));
77          r = new GbavXMLReader(GbavXMLReaderTest.class.getResourceAsStream("persoonslijst.xml"));
78          assertTrue(r.hasNext());
79          int total = 0;
80          List<String> objectRefs = new ArrayList();
81          while (r.hasNext()) {
82              b = r.next();
83              objectRefs.add(b.getObjectRef());
84              LOG.debug(String.format("Bericht #%d van %tF %2$tT, object ref %s",
85                      b.getVolgordeNummer(),
86                      b.getDatum(),
87                      b.getObjectRef()
88              ));
89              assertEquals("Soort komt niet overeen", BrmoFramework.BR_GBAV, b.getSoort());
90              assertEquals("Volgordenummer komt niet overeen", total, b.getVolgordeNummer().intValue());
91              total++;
92          }
93  
94          assertEquals(4, total);
95          assertArrayEquals(new String[]{
96                          GbavXMLReader.PREFIX + "a1672c578792e79cadd113ae5ccfd3a867dde805",
97                          GbavXMLReader.PREFIX + "200ba4cd285de672506eea37179f25e988511188",
98                          GbavXMLReader.PREFIX + "b8600b644b291dd67909c54ac87b46e6c1a38a1f",
99                          GbavXMLReader.PREFIX + "9d0ca6e9a6d67bf098ffa88d2df6281805b5b0a0"
100                 },
101                 objectRefs.toArray(new String[]{})
102         );
103     }
