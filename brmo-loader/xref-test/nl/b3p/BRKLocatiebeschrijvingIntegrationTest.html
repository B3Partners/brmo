74    @BeforeEach
75    @Override
76    public void setUp() throws Exception {
77      dsStaging = new BasicDataSource();
78      dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
79      dsStaging.setUsername(params.getProperty("staging.user"));
80      dsStaging.setPassword(params.getProperty("staging.passwd"));
81      dsStaging.setAccessToUnderlyingConnectionAllowed(true);
82  
83      dsRsgb = new BasicDataSource();
84      dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
85      dsRsgb.setUsername(params.getProperty("rsgb.user"));
86      dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
87      dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
88  
89      brmo = new BrmoFramework(dsStaging, dsRsgb, null);
90      brmo.setOrderBerichten(true);
91  
92      staging = new DatabaseDataSourceConnection(dsStaging);
93      rsgb = new DatabaseDataSourceConnection(dsRsgb, params.getProperty("rsgb.schema"));
94  
95      if (this.isOracle) {
96        staging =
97            new DatabaseConnection(
98                OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
99                params.getProperty("staging.user").toUpperCase());
100       staging
101           .getConfig()
102           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
103       staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
104 
105       rsgb =
106           new DatabaseConnection(
107               OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()),
108               params.getProperty("rsgb.user").toUpperCase());
109       rsgb.getConfig()
110           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
111       rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
112     } else if (this.isPostgis) {
113       // we hebben alleen nog postgres over
114       staging
115           .getConfig()
116           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
117       rsgb.getConfig()
118           .setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
119     }
120     sequential.lock();
121   }

139   @ParameterizedTest(name = "Locatie beschrijving #{index}: bestand: {0}")
140   @MethodSource("argumentsProvider")
141   public void testLocatieBeschrijving(
142       String bestandNaam, long aantalBerichten, String lo_loc__omschr) throws Exception {
143     IDataSet stagingDataSet =
144         new XmlDataSet(
145             new FileInputStream(
146                 new File(
147                     BRKLocatiebeschrijvingIntegrationTest.class.getResource(bestandNaam).toURI())));
148 
149     DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
150 
151     assumeTrue(
152         aantalBerichten == brmo.getCountBerichten("brk", "STAGING_OK"),
153         "Het aantal STAGING_OK berichten is anders dan verwacht");
154 
155     List<Bericht> berichten = brmo.listBerichten();
156     assertNotNull(berichten, "De verzameling berichten bestaat niet.");
157     assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
158 
159     LOG.debug("Transformeren berichten naar rsgb DB.");
160     Thread t = brmo.toRsgb();
161     t.join();
162 
163     assertEquals(
164         aantalBerichten,
165         brmo.getCountBerichten("brk", "RSGB_OK"),
166         "Niet alle berichten zijn OK getransformeerd");
167 
168     ITable kad_onrrnd_zk = rsgb.createDataSet().getTable("kad_onrrnd_zk");
169     assertEquals(
170         aantalBerichten,
171         kad_onrrnd_zk.getRowCount(),
172         "Het aantal onroerend zaak records komt niet overeen");
173     assertEquals(
174         lo_loc__omschr,
175         kad_onrrnd_zk.getValue(0, "lo_loc__omschr"),
176         "Beschrijving in record komt niet overeen");
177   }
