50   * voor bijvoorbeeld MS SQL.

78      @BeforeEach
79      @Override
80      public void setUp() throws Exception {
81          BasicDataSource dsStaging = new BasicDataSource();
82          dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
83          dsStaging.setUsername(params.getProperty("staging.user"));
84          dsStaging.setPassword(params.getProperty("staging.passwd"));
85          dsStaging.setAccessToUnderlyingConnectionAllowed(true);
86          dsStaging.setConnectionProperties(params.getProperty("staging.options", ""));
87  
88          BasicDataSource dsRsgb = new BasicDataSource();
89          dsRsgb.setUrl(params.getProperty("rsgb.jdbc.url"));
90          dsRsgb.setUsername(params.getProperty("rsgb.user"));
91          dsRsgb.setPassword(params.getProperty("rsgb.passwd"));
92          dsRsgb.setAccessToUnderlyingConnectionAllowed(true);
93          dsRsgb.setConnectionProperties(params.getProperty("rsgb.options", ""));
94  
95          staging = new DatabaseDataSourceConnection(dsStaging);
96          rsgb = new DatabaseDataSourceConnection(dsRsgb);
97  
98          if (this.isMsSQL) {
99              staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
100             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new MsSqlDataTypeFactory());
101         } else if (this.isOracle) {
102             staging = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()), params.getProperty("staging.user").toUpperCase());
103             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
104             staging.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
105 
106             rsgb = new DatabaseConnection(OracleConnectionUnwrapper.unwrap(dsRsgb.getConnection()), params.getProperty("rsgb.user").toUpperCase());
107             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new Oracle10DataTypeFactory());
108             rsgb.getConfig().setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
109         } else if (this.isPostgis) {
110             // we hebben alleen nog postgres over
111             staging.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
112             rsgb.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY, new PostgresqlDataTypeFactory());
113         }
114 
115         brmo = new BrmoFramework(dsStaging, dsRsgb);
116         brmo.setOrderBerichten(true);
117 
118         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
119         fxdb.setCaseSensitiveTableNames(false);
120         IDataSet stagingDataSet = fxdb.build(new FileInputStream(new File(Mantis15059BedrijfAlsFunctionarisIntegrationTest.class.getResource("/staging-empty-flat.xml").toURI())));
121 
122         sequential.lock();
123 
124         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
125 
126         assumeTrue(0L == brmo.getCountBerichten(null, null, BESTANDTYPE, "STAGING_OK"),
127                 "Er zijn geen STAGING_OK berichten");
128         assumeTrue(0L == brmo.getCountLaadProcessen(null, null, BESTANDTYPE, "STAGING_OK"),
129                 "Er zijn geen STAGING_OK laadprocessen");
130     }
131

137         staging.close();
138 
139         CleanUtil.cleanRSGB_NHR(rsgb);
140         rsgb.close();
141 
142         sequential.unlock();
143     }

145     @ParameterizedTest(name = "{index}: bestand: {0}")
146     @MethodSource("argumentsProvider")
147     public void testNhrXMLToStagingToRsgb(String bestandNaam, long aantalBerichten, long aantalProcessen,
148                                           long aantalPrs, long aantalSubj, long aantalNiet_nat_prs, long aantalNat_prs,
149                                           String vestgID, long aantalVestg, long aantalVestg_activiteit,
150                                           long kvkNummer, String[] sbiCodes, int aantalFunctionarissen) throws Exception {
151         assumeFalse(null == Mantis15059BedrijfAlsFunctionarisIntegrationTest.class.getResource(bestandNaam), "Het test bestand moet er zijn.");
152 
153         brmo.loadFromFile(BESTANDTYPE, Mantis15059BedrijfAlsFunctionarisIntegrationTest.class.getResource(bestandNaam).getFile(), null);
154         LOG.info("klaar met laden van berichten in staging DB.");
155 
156         List<Bericht> berichten = brmo.listBerichten();
157         List<LaadProces> processen = brmo.listLaadProcessen();
158         assertNotNull(berichten, "De verzameling berichten bestaat niet.");
159         assertEquals(aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
160         assertNotNull(processen, "De verzameling processen bestaat niet.");
161         assertEquals(aantalProcessen, processen.size(), "Het aantal processen is niet als verwacht.");
162 
163         // alleen het eerste bericht heeft br_orgineel_xml, de rest niet
164         ITable bericht = staging.createQueryTable("bericht", "select * from bericht where volgordenummer=0");
165         assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
166         LOG.debug("\n\n" + bericht.getValue(0, "br_orgineel_xml") + "\n\n");
167         assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null");
168         Object berichtId = bericht.getValue(0, "id");
169 
170         LOG.info("Transformeren berichten naar rsgb DB.");
171         Thread t = brmo.toRsgb();
172         t.join();
173 
174         // na de verwerking moet soap payload er ook nog zijn
175         bericht = staging.createQueryTable("bericht", "select * from bericht where br_orgineel_xml is not null");
176         assertEquals(1, bericht.getRowCount(), "Er zijn meer of minder dan 1 rij");
177         assertNotNull(bericht.getValue(0, "br_orgineel_xml"), "BR origineel xml is null na transformatie");
178         assertEquals(berichtId, bericht.getValue(0, "id"),
179                 "bericht met br_orgineel_xml moet hetzelfde id hebben na transformatie");
180 
181         assertEquals(aantalBerichten, brmo.getCountBerichten(null, null, BESTANDTYPE, "RSGB_OK"),
182                 "Niet alle berichten zijn OK getransformeerd");
183         berichten = brmo.listBerichten();
184         for (Bericht b : berichten) {
185             assertNotNull(b, "Bericht is 'null'");
186             assertNotNull(b.getDbXml(), "'db-xml' van bericht is 'null'");
187         }
188 
189         ITable prs = rsgb.createDataSet().getTable("prs");
190         ITable niet_nat_prs = rsgb.createDataSet().getTable("niet_nat_prs");
191         ITable nat_prs = rsgb.createDataSet().getTable("nat_prs");
192         ITable subject = rsgb.createDataSet().getTable("subject");
193 
194         assertEquals(aantalPrs, prs.getRowCount(), "Het aantal 'prs' records klopt niet");
195         assertEquals(aantalNiet_nat_prs, niet_nat_prs.getRowCount(),
196                 "Het aantal 'niet_nat_prs' records klopt niet");
197         assertEquals(aantalNat_prs, nat_prs.getRowCount(), "Het aantal 'nat_prs' records klopt niet");
198         assertEquals(aantalSubj, subject.getRowCount(), "het aantal 'subject' records klopt niet");
199 
200         boolean foundKvk = false;
201         for (int i = 0; i < subject.getRowCount(); i++) {
202             if (subject.getValue(i, "identif").toString().contains("nhr.maatschAct.kvk")) {
203                 assertEquals(kvkNummer + "", subject.getValue(i, "kvk_nummer") + "", "KVK nummer klopt niet");
204                 foundKvk = true;
205             }
206         }
207         if (!foundKvk) {
208             fail("KVK nummer maatschappelijke activiteit klopt niet, verwacht te vinden: " + kvkNummer);
209         }
210 
211         if (vestgID != null) {
212             ITable vestg = rsgb.createDataSet().getTable("vestg");
213             assertEquals(vestgID, vestg.getValue(0, "sc_identif"),
214                     "De 'sc_identif' van hoofdvestiging klopt niet");
215             assertEquals(sbiCodes[0], vestg.getValue(0, "fk_sa_sbi_activiteit_sbi_code"),
216                     "De sbi code van (hoofd)vestiging klopt niet");
217             assertEquals("Ja", vestg.getValue(0, "sa_indic_hoofdactiviteit"),
218                     "De 'sa_indic_hoofdactiviteit' van (hoofd)vestiging klopt niet");
219         }
220 
221         ITable vestg = rsgb.createDataSet().getTable("vestg");
222         assertEquals(aantalVestg, vestg.getRowCount(), "het aantal 'vestg' records klopt niet");
223 
224         ITable vestg_activiteit = rsgb.createDataSet().getTable("vestg_activiteit");
225         assertEquals(aantalVestg_activiteit, vestg_activiteit.getRowCount(),
226                 "Het aantal 'vestg_activiteit' records klopt niet");
227 
228         ITable sbi_activiteit = rsgb.createDataSet().getTable("sbi_activiteit");
229         ITable herkomst_metadata = rsgb.createDataSet().getTable("herkomst_metadata");
230         for (String sbiCode : sbiCodes) {
231             boolean foundSbiCode = false;
232             boolean foundSbiCodeMeta = false;
233             for (int i = 0; i < sbi_activiteit.getRowCount(); i++) {
234                 if (sbiCode.equals(sbi_activiteit.getValue(i, "sbi_code").toString())) {
235                     LOG.debug("SBI code " + sbiCode + " gevonden in sbi_activiteit");
236                     foundSbiCode = true;
237                     break;
238                 }
239             }
240             if (!foundSbiCode) {
241                 fail("SBI code niet gevonden in sbi_activiteit, verwacht te vinden: " + sbiCode);
242             }
243             for (int i = 0; i < herkomst_metadata.getRowCount(); i++) {
244                 if (sbiCode.equals(herkomst_metadata.getValue(i, "waarde").toString())) {
245                     LOG.debug("SBI code " + sbiCode + " gevonden in herkomst_metadata");
246                     foundSbiCodeMeta = true;
247                     break;
248                 }
249             }
250             if (!foundSbiCodeMeta) {
251                 fail("SBI code niet gevonden in herkomst_metadata, verwacht te vinden: " + sbiCode);
252             }
253         }
254 
255         ITable functionaris = rsgb.createDataSet().getTable("functionaris");
256         assertEquals(aantalFunctionarissen, functionaris.getRowCount(),
257                 "Het aantal 'functionaris' records klopt niet");
258         if (kvkNummer == 77513010) {
259             // gebruik een gesorteerde weergaven van functionaris tabel omdat de volgorde van de rijen anders kan zijn terwijl je dan toch dezelfde data hebt geladen
260             // er is een Gevolmachtigde Directeur met beperkte volmacht in de laatste rij
261             functionaris = rsgb.createQueryTable("functionaris", "select * from functionaris order by functie ASC");
262             assertEquals("Vennoot", functionaris.getValue(aantalFunctionarissen - 1, "functie"));
263             assertNull(functionaris.getValue(aantalFunctionarissen - 1, "functionaristypering"));
264             assertNull(functionaris.getValue(aantalFunctionarissen - 1, "volledig_beperkt_volmacht"));
265             assertEquals("10000", functionaris.getValue(aantalFunctionarissen - 1, "beperking_bev_in_euros").toString());
266         }
267     }
