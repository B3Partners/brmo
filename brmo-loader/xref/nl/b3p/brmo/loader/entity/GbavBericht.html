39  public class GbavBericht extends Bericht {
40  
41      private static final Log LOG = LogFactory.getLog(GbavXMLReader.class);
42  
43      private boolean xpathEvaluated = false;
44      private boolean hasAddedBSNHashes = false;
45  
46      private String bsn = null;
47  
48      private List<String> bsnList = new ArrayList<>();
49  
50      public GbavBericht(String brXml) {
51          super(brXml);
52          super.setSoort(BR_GBAV);
53      }
54  
55      private void evaluateXPath() {
56          if (xpathEvaluated) {
57              return;
58          }
59  
60          xpathEvaluated = true;
61  
62          if (datum != null && bsn != null && !bsnList.isEmpty()) {
63              return;
64          }
65          try {
66              DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
67              DocumentBuilder builder = factory.newDocumentBuilder();
68              Document doc = builder.parse(new InputSource(new StringReader(getBrXml())));
69  
70              XPathFactory xPathfactory = XPathFactory.newInstance();
71              XPath xpath = xPathfactory.newXPath();
72  
73              // probeer bestandsdatum te achterhalen
74              XPathExpression expr = xpath.compile("persoon/categorieen/categorie[nummer='01']/rubrieken/rubriek[nummer='8220']/waarde/text()");
75              this.setDatumAsString(expr.evaluate(doc));
76              // kan ook op categorie 4 voorkomen
77              if (this.getDatum() == null) {
78                  expr = xpath.compile("persoon/categorieen/categorie[nummer='04']/rubrieken/rubriek[nummer='8220']/waarde/text()");
79                  this.setDatumAsString(expr.evaluate(doc));
80              }
81  
82              // van de verschillende personen BSN uitlezen en hashen
83              expr = xpath.compile("persoon/categorieen/categorie[nummer='01']/rubrieken/rubriek[nummer='0120']/waarde/text()");
84              this.bsn = expr.evaluate(doc);
85              this.bsnList.add(bsn);
86              // ouder 1
87              expr = xpath.compile("persoon/categorieen/categorie[nummer='02']/rubrieken/rubriek[nummer='0120']/waarde/text()");
88              String _bsn = expr.evaluate(doc);
89              if (_bsn != null && _bsn != "") {
90                  this.bsnList.add(_bsn);
91              }
92              // ouder 2
93              expr = xpath.compile("persoon/categorieen/categorie[nummer='03']/rubrieken/rubriek[nummer='0120']/waarde/text()");
94              _bsn = expr.evaluate(doc);
95              if (_bsn != null && _bsn != "") {
96                  this.bsnList.add(_bsn);
97              }
98              // partner
99              expr = xpath.compile("persoon/categorieen/categorie[nummer='05']/rubrieken/rubriek[nummer='0120']/waarde/text()");
100             _bsn = expr.evaluate(doc);
101             if (_bsn != null && _bsn != "") {
102                 this.bsnList.add(_bsn);
103             }
104             // kind
105             expr = xpath.compile("persoon/categorieen/categorie[nummer='09']/rubrieken/rubriek[nummer='0120']/waarde/text()");
106             _bsn = expr.evaluate(doc);
107             if (_bsn != null && _bsn != "") {
108                 this.bsnList.add(_bsn);
109             }
110         } catch (Exception e) {
111             LOG.error("Fout tijdens parsen van gbav xml", e);
112         }
113     }
114 
115     /**
116      * probeer datum uit bericht te halen via xpath.
117      *
118      * @return datum of {@code null}
119      */
120     public Date parseDatum() {
121         this.evaluateXPath();
122         return this.datum;
123     }
124 
125     public String getBsn() {
126         this.evaluateXPath();
127         return this.bsn;
128     }
129 
130     public List<String> getBsnList() {
131         this.evaluateXPath();
132         return this.bsnList;
133     }
134 
135     public void setBsnMap(Map<String, String> bsnHashes) {
136         if (!hasAddedBSNHashes) {
137             StringBuilder sb = new StringBuilder("<bsnhashes>\n");
138             for (Map.Entry<String, String> entry : bsnHashes.entrySet()) {
139                 if (!entry.getKey().isEmpty() && !entry.getValue().isEmpty()) {
140                     sb.append("<").append(GbavXMLReader.PREFIX).append(entry.getKey()).append('>')
141                             .append(entry.getValue())
142                             .append("</").append(GbavXMLReader.PREFIX).append(entry.getKey()).append(">\n");
143                 }
144             }
145             sb.append("</bsnhashes>\n");
146 
147             this.hasAddedBSNHashes = true;
148             LOG.debug("toevoegen bsn hashes aan br_xml: " + sb);
149             sb.insert(0, "<root>\n");
150 
151             int endOfProlog = this.getBrXml().indexOf("?>");
152             if (endOfProlog > 0) {
153                 endOfProlog = endOfProlog + 2;
154             } else {
155                 endOfProlog = 0;
156             }
157             StringBuilder br = new StringBuilder(this.getBrXml());
158             br.insert(endOfProlog, sb).append("</root>");
159 
160             this.setBrXml(br.toString());
161         }
162     }
163 
164     public void setDatumAsString(String d) {
165         if (d == null || d.isEmpty()) {
166             return;
167         }
168 
169         SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
170         try {
171             this.datum = sdf.parse(d);
172         } catch (ParseException pe) {
173             LOG.error("Error while parsing date: " + datum, pe);
174         }
175     }
