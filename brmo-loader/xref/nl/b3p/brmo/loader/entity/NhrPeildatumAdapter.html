3   import java.text.SimpleDateFormat;
4   import java.util.Date;
5   import javax.xml.bind.annotation.adapters.XmlAdapter;
6   
7   /**
8    *
9    * @author Matthijs Laan
10   */
11  public class NhrPeildatumAdapter extends XmlAdapter<String, Date> {
12      public static final String NHR_PEILDATUM_FORMAT = "yyyyMMddHHmmssSSS";
13      private final SimpleDateFormat dateFormat = new SimpleDateFormat(NHR_PEILDATUM_FORMAT);
14  
15      @Override
16      public String marshal(Date v) throws Exception {
17          synchronized (dateFormat) {
18              return dateFormat.format(v);
19          }
20      }
21  
22      @Override
23      public Date unmarshal(String v) throws Exception {
24          synchronized (dateFormat) {
25              return dateFormat.parse(v);
26          }
27      }
28  }
