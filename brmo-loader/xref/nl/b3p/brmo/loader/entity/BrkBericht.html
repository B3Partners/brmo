37      public void setVervallenInfo(String objectRef, Date datum) {
38          this.objectRef = objectRef;
39          this.datum = datum;
40  
41          xpathEvaluated = true;
42      }
43  
44      private void evaluateXPath() {
45          if (xpathEvaluated) {
46              return;
47          }
48  
49          xpathEvaluated = true;
50          
51          if (objectRef!=null && datum!=null) {
52              //al uit de header gehaald
53              return;
54          }
55  
56          // dit moet dus een standberciht zijn
57          try {
58  
59              DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
60              DocumentBuilder builder = factory.newDocumentBuilder();
61              Document doc = builder.parse(new InputSource(new StringReader(getBrXml())));
62  
63              XPathFactory xPathfactory = XPathFactory.newInstance();
64              XPath xpath = xPathfactory.newXPath();
65              
66              XPathExpression expr = xpath.compile("/KadastraalObjectSnapshot/*[local-name()= 'Perceel' or local-name()='Appartementsrecht']/identificatie/namespace/text()");
67              objectRef = expr.evaluate(doc);
68  
69              expr = xpath.compile("/KadastraalObjectSnapshot/*[local-name()= 'Perceel' or local-name()='Appartementsrecht']/identificatie/lokaalId/text()");
70              objectRef += ":" + expr.evaluate(doc);
71  
72              expr = xpath.compile("/KadastraalObjectSnapshot/*[local-name()= 'toestandsdatum' or local-name()='toestandsdatum']/text()");
73              setDatumAsString(expr.evaluate(doc));
74          } catch (Exception e) {
75              log.error("Error while getting brk referentie", e);
76          }
77      }
78  
79      public void setDatumAsString(String d) {
80          if (d == null || d.isEmpty()) {
81              return;
82          }
83  
84          SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
85          try {
86              datum = sdf.parse(d);
87          } catch (ParseException pe) {
88              log.error("Error while parsing date: " + datum, pe);
89          }
90      }
91  
92      @Override
93      public String getObjectRef() {
94          evaluateXPath();
95  
96          return objectRef;
97      }
98  
99      @Override
100     public Date getDatum() {
101         evaluateXPath();
102 
103         return datum;
104     }

106      public String getRestoredFileName(Date bestanddatum, Integer volgordenummer){
107         try {
108             final SimpleDateFormat output = new SimpleDateFormat("yyyyMMdd");
109             
110             String prefix = "BKE-MUTBX01";
111             String kadGemCode;
112             String perceelnummer;
113             String brkdatum = output.format(bestanddatum);
114             String sectie;
115             String appartementsrechtVolgnummer;
116 
117             String basePath = "/KadastraalObjectSnapshot/*[local-name()= 'Perceel' or local-name()='Appartementsrecht']/kadastraleAanduiding/";
118             XPathFactory xPathfactory = XPathFactory.newInstance();
119             XPath xpath = xPathfactory.newXPath();
120             DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
121             DocumentBuilder builder = factory.newDocumentBuilder();
122             Document doc = builder.parse(new InputSource(new StringReader(this.getBrXml())));
123 
124             // in geval van een verwijderbericht is de "wordt" leeg, db_xml <empty/>
125             if (this.getBrXml().contains("<empty/>") && StringUtils.isNotBlank(this.getBrOrgineelXml())) {
126                 // dan proberen om de db_origineel_xml te gebruiken.
127                 basePath = "/Mutatie/kadastraalObject/AanduidingKadastraalObject/kadastraleAanduiding/";
128                 doc = builder.parse(new InputSource(new StringReader(this.getBrOrgineelXml())));
129             }
130 
131             XPathExpression expr = xpath.compile(basePath + "AKRKadastraleGemeenteCode/waarde/text()");
132             kadGemCode = expr.evaluate(doc);
133             
134             expr = xpath.compile(basePath + "sectie/text()");
135             sectie = expr.evaluate(doc);
136             
137             expr = xpath.compile(basePath + "perceelnummer/text()");
138             perceelnummer = expr.evaluate(doc);
139 
140             expr = xpath.compile(basePath + "appartementsrechtVolgnummer/text()");
141             appartementsrechtVolgnummer = expr.evaluate(doc);
142             if (StringUtils.isNotBlank(appartementsrechtVolgnummer)) {
143                 appartementsrechtVolgnummer = "A" + appartementsrechtVolgnummer;
144             }
145 
146             String aanduiding = kadGemCode + sectie + perceelnummer + appartementsrechtVolgnummer;
147             log.debug("gevonden aanduiding voor herstelde bestandsnaam: " + aanduiding);
148             String filename = "bestandsnaam kon niet worden hersteld";
149             if (StringUtils.isNotBlank(aanduiding)) {
150                 filename = prefix + "-" + aanduiding + "-" + brkdatum + "-" + volgordenummer.toString() + ".zip";
151             }
152             return filename;
153         } catch (ParserConfigurationException | XPathExpressionException | SAXException | IOException  ex) {
154             log.error("Cannot create filename from xml: ", ex);
155             return "";
156         }
157     }
