20  import java.io.StringWriter;
21  import java.util.Date;
22  import java.util.HashMap;
23  import java.util.Map;
24  import javax.xml.stream.XMLInputFactory;
25  import javax.xml.stream.XMLOutputFactory;
26  import javax.xml.stream.XMLStreamException;
27  import javax.xml.stream.XMLStreamReader;
28  import javax.xml.transform.Transformer;
29  import javax.xml.transform.TransformerConfigurationException;
30  import javax.xml.transform.TransformerFactory;
31  import javax.xml.transform.stax.StAXResult;
32  import javax.xml.transform.stax.StAXSource;
33  import nl.b3p.brmo.loader.BrmoFramework;
34  import nl.b3p.brmo.loader.entity.GbavBericht;
35  import org.apache.commons.logging.Log;
36  import org.apache.commons.logging.LogFactory;
37  
38  /**
39   *
40   * @author Mark Prins
41   */
42  public class GbavXMLReader extends BrmoXMLReader {
43  
44      private static final Log log = LogFactory.getLog(GbavXMLReader.class);
45      private static final String PERSOON = "persoon";
46      public static final String PREFIX = "NL.BRP.Persoon.";
47      private final XMLInputFactory factory = XMLInputFactory.newInstance();
48      private final XMLStreamReader streamReader;
49      private final Transformer transformer;
50      private final XMLOutputFactory xmlof;
51      private GbavBericht nextBericht = null;
52      private int volgordeNummer = 0;
53  
54      public GbavXMLReader(InputStream in) throws XMLStreamException, TransformerConfigurationException {
55          streamReader = factory.createXMLStreamReader(in);
56          TransformerFactory tf = TransformerFactory.newInstance();
57          transformer = tf.newTransformer();
58          xmlof = XMLOutputFactory.newInstance();
59          // xmlof.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, Boolean.TRUE);
60  
61          init();
62      }
63  
64      @Override
65      public void init() throws XMLStreamException {
66          soort = BrmoFramework.BR_GBAV;
67          try {
68              while (streamReader.hasNext()) {
69                  if (streamReader.isStartElement()) {
70                      String localName = streamReader.getLocalName();
71                      if (localName.equals(PERSOON)) {
72                          break;
73                      }
74                  }
75                  streamReader.next();
76              }
77          } catch (XMLStreamException ex) {
78              log.error("Error while streaming XML", ex);
79          }
80          if (getBestandsDatum() == null) {
81              setBestandsDatum(new Date());
82          }
83      }
84  
85      @Override
86      public boolean hasNext() throws Exception {
87          if (nextBericht != null) {
88              return true;
89          }
90          try {
91              while (streamReader.hasNext()) {
92                  if (streamReader.isStartElement() && streamReader.getLocalName().equals(PERSOON)) {
93                      StringWriter sw = new StringWriter();
94                      transformer.transform(new StAXSource(streamReader), new StAXResult(xmlof.createXMLStreamWriter(sw)));
95                      nextBericht = new GbavBericht(sw.toString());
96                      nextBericht.setDatum(nextBericht.parseDatum());
97                      if (nextBericht.getDatum() == null) {
98                          nextBericht.setDatum(new Date());
99                      }
100                     nextBericht.setVolgordeNummer(volgordeNummer);
101 
102                     String bsn = nextBericht.getBsn();
103                     String bsnHash = this.getHash(bsn);
104                     nextBericht.setObjectRef(PREFIX + bsnHash);
105 
106                     Map<String, String> bsns = new HashMap<>();
107                     nextBericht.getBsnList().forEach((_bsn) -> {
108                         log.debug("toevoegen bsn en hash: " + _bsn + ":" + this.getHash(_bsn));
109                         bsns.put(_bsn, this.getHash(_bsn));
110                     });
111                     log.debug("aantal BSN in bericht:" + bsns.size());
112                     nextBericht.setBsnMap(bsns);
113                     return true;
114                 } else {
115                     streamReader.next();
116                 }
117             }
118         } catch (XMLStreamException ex) {
119             log.error("Streamfout tijdens parsen GBA-V XML", ex);
120         }
121         return false;
122     }
123 
124     /**
125      * geeft het volgende bericht uit de stream, mits
126      * {@link #hasNext() } {@code true} geeft. Gebruik dus eerst
127      * {@link #hasNext()}.
128      *
129      * @return volgend bericht
130      * @throws Exception soms
131      */
132     @Override
133     public GbavBericht next() throws Exception {
134         GbavBericht b = nextBericht;
135         nextBericht = null;
136         volgordeNummer++;
137         return b;
138     }
139 
140 }
