9   import javax.xml.stream.XMLStreamWriter;

11  import javax.xml.transform.TransformerConfigurationException;
12  import javax.xml.transform.TransformerException;
13  import javax.xml.transform.TransformerFactory;
14  import javax.xml.transform.stax.StAXResult;
15  import javax.xml.transform.stax.StAXSource;
16  import nl.b3p.brmo.loader.BrmoFramework;
17  import nl.b3p.brmo.loader.entity.BrkBericht;
18  import org.apache.commons.logging.Log;

21  /**
22   * reader of BRK stand files
23   *
24   * @author Boy de Wit
25   */
26  public class BrkSnapshotXMLReader extends BrmoXMLReader {
27  
28      private static final Log log = LogFactory.getLog(BrkSnapshotXMLReader.class);
29  
30      private final XMLInputFactory factory = XMLInputFactory.newInstance();
31      private final XMLStreamReader streamReader;
32      private final Transformer transformer;
33  
34      private static final String KAD_OBJ_SNAP = "KadastraalObjectSnapshot";
35      // gebied
36      private static final String PRODUCTSPECIFICATIE = "productSpecificatie";
37      private static final String KENMERKNAAM = "kenmerknaam";
38      private static final String KENMERKWAARDE = "kenmerkwaarde";
39      // burgerlijke gemeente
40      private static final String BRK_DATUM = "BRKDatum";
41      private static final String GEMEENTE = "naamBurgerlijkeGemeente";
42      private static final String MUTATIE = "Mutatie";
43      private static final String NS_MUTATIE = "http://www.kadaster.nl/schemas/brk-levering/product-mutatie/v20120901";
44      private static final String NS_XLINK = "http://www.w3.org/1999/xlink";
45  
46      private static class MutatieGegevens {
47          boolean inMutatieObjectRef = false;
48          String mutatieObjectRef = null;
49          boolean isMutatieZonderWordt = true;
50          Integer volgnummerKadastraalObjectDatum = null;
51      }
52      private MutatieGegevens mutatieGegevens = null;
53  
54      public BrkSnapshotXMLReader(InputStream in)
55              throws XMLStreamException,
56              TransformerConfigurationException {
57  
58          streamReader = factory.createXMLStreamReader(in);
59          TransformerFactory tf = TransformerFactory.newInstance();
60          transformer = tf.newTransformer();
61  
62          init();
63      }
64  
65      @Override
66      public void init() throws XMLStreamException {
67          soort = BrmoFramework.BR_BRK;
68          positionToNext();
69      }
70  
71      /**
72       * Positioneer de XML stream aan het start element voor een
73       * KadastraalObjectSnapshot (als child van een GemeenteGebaseerdeStand:stand
74       * of als child van een Mutatie:wordt) of aan het end element voor een Mutatie:Mutatie
75       * zonder Mutatie:wordt (vervallen perceel).
76       *
77       * @throws XMLStreamException
78       */
79  
80      private void positionToNext() throws XMLStreamException {
81          boolean inMutatie = false;
82          mutatieGegevens = null;
83          boolean inGemeente = false;
84          boolean inWas = false;
85          boolean inProductSpecificatie = false;
86          String inProductSpecificatieNaam = null;
87          while (streamReader.hasNext()) {
88              if (streamReader.isStartElement()) {
89                  if (!inWas && streamReader.getLocalName().equals(KAD_OBJ_SNAP)) {
90                      // KadastraalObjectSnapshot gevonden niet in "was" van mutatie,
91                      // hou streamReader op dit start element voor next()
92                      break;
93                  }
94                  if (streamReader.getLocalName().equals(BRK_DATUM)) {
95                      // GemeenteGebaseerdeStand:BRKDatum:BRKDatum of Mutatie:BRKDatum
96                      setDatumAsString(streamReader.getElementText());
97                  } else if (streamReader.getLocalName().equals(PRODUCTSPECIFICATIE)) {
98                      inProductSpecificatie = true;
99                  } else if (inProductSpecificatie && streamReader.getLocalName().equals(KENMERKNAAM)) {
100                     inProductSpecificatieNaam = streamReader.getElementText();
101                 } else if (inProductSpecificatie && streamReader.getLocalName().equals(KENMERKWAARDE)) {
102                     if ("Peildatum".equals(inProductSpecificatieNaam)) {
103                         setDatumAsString(streamReader.getElementText());
104                     } else if ("Gebiednummer".equals(inProductSpecificatieNaam)) {
105                         setGebied(streamReader.getElementText());
106                     }
107                 } else if (streamReader.getLocalName().equals(GEMEENTE)) {
108                     inGemeente = true;
109                 } else if (inGemeente && streamReader.getLocalName().equals("waarde")) {
110                     setGebied(streamReader.getElementText());
111                 } else if(streamReader.getLocalName().equals(MUTATIE)) {
112                     inMutatie = true;
113                     mutatieGegevens = new MutatieGegevens();
114                 } else if(inMutatie && streamReader.getLocalName().equals("volgnummerKadastraalObjectDatum")) {
115                     mutatieGegevens.volgnummerKadastraalObjectDatum = Integer.parseInt(streamReader.getElementText());
116                 } else if(inMutatie && streamReader.getLocalName().equals("kadastraalObject") && streamReader.getNamespaceURI().equals(NS_MUTATIE)) {
117                     // Mutatie:AanduidingKadastraalObject -> Mutatie:kadastraalObject -> (AppartementsrechtRef | PerceelRef)
118                     mutatieGegevens.inMutatieObjectRef = true;
119                 } else if(inMutatie && mutatieGegevens.inMutatieObjectRef) {
120                     // Haal id op van Appartementsrecht of PerceelRef
121                     mutatieGegevens.mutatieObjectRef = streamReader.getAttributeValue(NS_XLINK, "href");
122                     if(mutatieGegevens.mutatieObjectRef != null) {
123                         int i = mutatieGegevens.mutatieObjectRef.lastIndexOf('.');
124                         if(i != -1) {
125                             mutatieGegevens.mutatieObjectRef = mutatieGegevens.mutatieObjectRef.substring(0,i) + ':' + mutatieGegevens.mutatieObjectRef.substring(i+1);
126                         }
127                     }
128                     mutatieGegevens.inMutatieObjectRef = false;
129                 } else if(inMutatie && streamReader.getLocalName().equals("was")) {
130                     // Bij mutatie "was" skippen
131                     inWas = true;
132                 } else if(inMutatie && streamReader.getLocalName().equals("wordt")) {
133                     mutatieGegevens.isMutatieZonderWordt = false;
134                 }
135             } else if (streamReader.isEndElement()) {
136                 if(streamReader.getLocalName().equals(PRODUCTSPECIFICATIE)) {
137                     inProductSpecificatie = false;
138                     inProductSpecificatieNaam = null;
139                 }
140                 if(streamReader.getLocalName().equals(GEMEENTE)) {
141                     inGemeente = false;
142                 }
143                 if(streamReader.getLocalName().equals("was")) {
144                     inWas = false;
145                 }
146                 if(inMutatie && streamReader.getLocalName().equals(MUTATIE)) {
147                     break;
148                 }
149             }
150 
151             streamReader.next();
152         }
153     }
154 
155     @Override
156     public boolean hasNext() {
157         try {
158             return mutatieGegevens != null && mutatieGegevens.isMutatieZonderWordt || streamReader.hasNext();
159         } catch (XMLStreamException ex) {
160             log.error("Error while streaming XML", ex);
161         }
162         return false;
163     }
164 
165     @Override
166     public BrkBericht next() throws TransformerException, XMLStreamException {
167         BrkBericht b;
168 
169         if(mutatieGegevens != null && mutatieGegevens.isMutatieZonderWordt) {
170             // Vervallen perceel
171             b = new BrkBericht("<empty/>");
172 
173             b.setVervallenInfo(mutatieGegevens.mutatieObjectRef, getBestandsDatum());
174             b.setVolgordeNummer(mutatieGegevens.volgnummerKadastraalObjectDatum);
175         } else {
176             StringWriter sw = new StringWriter();
177 
178             // Vanwege splitsing is repairen van namespaces nodig, anders ontbreekt
179             // de namespace prefix declaratie xmlns:xlink="http://www.w3.org/1999/xlink"
180             // in de output en mislukt de XSL transformatie
181             XMLOutputFactory of = XMLOutputFactory.newInstance();
182             of.setProperty(XMLOutputFactory.IS_REPAIRING_NAMESPACES, Boolean.TRUE);
183             XMLStreamWriter writer = of.createXMLStreamWriter(sw);
184 
185             transformer.transform(new StAXSource(streamReader), new StAXResult(writer));
186             b = new BrkBericht(sw.toString());
187             //volgende gegevens komen uit de header, in bericht worden gegevens verder aangevuld.
188             b.setVolgordeNummer(mutatieGegevens != null ? mutatieGegevens.volgnummerKadastraalObjectDatum : -1);
189             b.setDatum(getBestandsDatum());
190             b.setObjectRef(mutatieGegevens != null ? mutatieGegevens.mutatieObjectRef : null);
191         }
192         positionToNext();
193         return b;
194     }
