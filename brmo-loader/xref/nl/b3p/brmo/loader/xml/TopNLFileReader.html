17  package nl.b3p.brmo.loader.xml;
18  
19  import java.io.File;
20  import java.io.FileInputStream;
21  import java.util.Date;
22  import java.util.StringTokenizer;
23  import java.util.zip.ZipEntry;
24  import java.util.zip.ZipInputStream;
25  import nl.b3p.brmo.loader.BrmoFramework;
26  import nl.b3p.brmo.loader.entity.Bericht;
27  
28  /**
29   *
30   * @author Mark Prins
31   */
32  public class TopNLFileReader extends BrmoXMLReader {
33  
34      private long fileSize = 0;
35  
36      public TopNLFileReader(String fileName, String soort) throws Exception {
37          if (fileName != null) {
38              this.setBestandsNaam(fileName);
39          }
40          this.soort = (soort == null ? BrmoFramework.BR_TOPNL : soort);
41          this.init();
42      }
43  
44      /**
45       *
46       * @throws Exception als het bestand niet bestaat of leeg is
47       */
48      @Override
49      public void init() throws Exception {
50          File input = new File(this.getBestandsNaam());
51          fileSize = input.length();
52          this.setBestandsDatum(new Date());
53      }
54  
55      public long getFileSize() {
56          return fileSize;
57      }
58  
59      /**
60       * probeer uit de naam af te leiden wat gebied en zoomnivo zijn.
61       *
62       * @param naam naam van de zipfile met gml bestanden.
63       */
64      @Override
65      public void setBestandsNaam(String naam) {
66          super.setBestandsNaam(naam);
67          StringTokenizer parts = new StringTokenizer(
68                  naam.substring(naam.lastIndexOf(File.separator) + 1, naam.length()),
69                  "_-.", false);
70  
71          if (parts.countTokens() > 2) {
72              this.setGebied("" + parts.nextToken() + " blad " + parts.nextToken());
73          } else {
74              this.setGebied("Nederland");
75          }
76      }
77  
78      @Override
79      public boolean hasNext() throws Exception {
80          return false;
81      }
82  
83      @Override
84      public Bericht next() throws Exception {
85          return null;
86      }
87  }
