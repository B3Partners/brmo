55    public NhrXMLReader(InputStream in) throws Exception {
56      initTemplates();
57  
58      ByteArrayOutputStream bos = new ByteArrayOutputStream();
59      in = new TeeInputStream(in, bos, true);
60  
61      // Split input naar multiple berichten
62      DOMResult r = new DOMResult();
63      Transformer transformer = splitTemplates.newTransformer();
64      StreamSource source = new StreamSource(in);
65      // Prevent external entity resolution
66      source.setSystemId("");
67      transformer.transform(source, r);
68  
69      JAXBContext jc = JAXBContext.newInstance(NhrBerichten.class, NhrBericht.class, Bericht.class);
70      Unmarshaller unmarshaller = jc.createUnmarshaller();
71  
72      NhrBerichten b = (NhrBerichten) unmarshaller.unmarshal(new DOMSource(r.getNode()));
73  
74      if (b == null || b.berichten == null) {
75        throw new BrmoLeegBestandException("Geen BRMO berichten gevonden in NHR XML");
76      }
77  
78      if (!b.berichten.isEmpty()) {
79        setBestandsDatum(b.berichten.get(0).getDatum());
80      }
81  
82      iterator = b.berichten.iterator();
83  
84      brOrigXML = bos.toString(StandardCharsets.UTF_8);
85      LOG.debug("Originele nHR xml is: \n" + brOrigXML);
86  
87      init();
88    }
89  
90    private synchronized void initTemplates() throws Exception {
91      if (splitTemplates == null) {
92        LOG.info("Initializing NHR split XSL templates...");
93        Source xsl =
94            new StreamSource(NhrXMLReader.class.getResourceAsStream("/xsl/nhr-split-3.0.xsl"));
95        TransformerFactory tf = TransformerFactory.newInstance();
96        tf.setURIResolver(
97            (href, base) -> new StreamSource(NhrXMLReader.class.getResourceAsStream("/xsl/" + href)));
98        NhrXMLReader.splitTemplates = tf.newTemplates(xsl);
99  
100       t = TransformerFactory.newInstance().newTransformer();
101     }
102   }

114   @Override
115   public NhrBericht next() throws Exception {
116     NhrBericht b = iterator.next();
117 
118     StringWriter sw = new StringWriter();
119     t.transform(new DOMSource(b.getNode().getFirstChild()), new StreamResult(sw));
120     // opzoeklijst van bsn en/of volledige naam en hash toevoegen
121     StringBuilder xml = new StringBuilder(sw.toString());
122     String bsns = getXML(extractBSN(xml.toString()));
123     LOG.trace("gevonden bsn en naam sleutels: " + bsns);
124     // insert bsnhashes voor de laatste node
125     xml.insert(xml.lastIndexOf("</"), bsns);
126 
127     b.setBrXml(xml.toString());
128 
129     // we willen alleen bij het eerst geparsede nhr bericht de originele soap hebben
130     if (volgorde == 0) {
131       b.setBrOrgineelXml(brOrigXML);
132     }
133 
134     b.setVolgordeNummer(volgorde++);
135     return b;
136   }

138   /**
139    * maakt een map met bsn,bsnhash.
140    *
141    * @param brXml string
142    * @return hashmap met bsn/naamzonderspaties,bsnhash
143    */
144   public Map<String, String> extractBSN(String brXml) {
145     Map<String, String> bsnHashes = new HashMap<>();
146     //        LOG.trace("deel bericht xml: "+brXml);
147     try {
148       DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
149       factory.setNamespaceAware(true);
150       DocumentBuilder builder = factory.newDocumentBuilder();
151       Document doc = builder.parse(new InputSource(new StringReader(brXml)));
152 
153       NodeList nodeList = doc.getElementsByTagNameNS("*", "bsn");
154       for (int i = 0; i < nodeList.getLength(); i++) {
155         LOG.debug(
156             "BSN: "
157                 + i
158                 + ": "
159                 + nodeList.item(i).getNodeName()
160                 + " - "
161                 + nodeList.item(i).getTextContent());
162         bsnHashes.put(
163             nodeList.item(i).getTextContent(), getHash(nodeList.item(i).getTextContent()));
164       }
165 
166       nodeList = doc.getElementsByTagNameNS("*", "volledigeNaam");
167       String _cleanName;
168       for (int i = 0; i < nodeList.getLength(); i++) {
169         if (nodeList.item(i).getParentNode().getLocalName().equals("natuurlijkPersoon")
170             || nodeList.item(i).getParentNode().getLocalName().equals("buitenlandseVennootschap")) {
171           LOG.debug(
172               "NAAM: "
173                   + i
174                   + ": "
175                   + nodeList.item(i).getNodeName()
176                   + " - "
177                   + nodeList.item(i).getTextContent());
178           _cleanName =
179               nodeList
180                   .item(i)
181                   .getTextContent()
182                   .replace(" ", "")
183                   .replace("'", "")
184                   .replace(".", "")
185                   .replace(",", "")
186                   // komt in bedrijfsnaam voor 'A/S'
187                   .replace("/", "")
188                   // komt in bedrijfsnaam voor '(Nederland)'
189                   .replace("(", "")
190                   .replace(")", "")
191                   // komt in bedrijfsnaam voor '& Co'
192                   .replace("&", "");
193           bsnHashes.put(_cleanName, getHash(_cleanName));
194         }
195       }
196     } catch (ParserConfigurationException | SAXException | IOException e) {
197       LOG.error("Fout tijdens toevoegen bsn hashes", e);
198     }
199     return bsnHashes;
200   }

202   public String getXML(Map<String, String> map) {
203     StringBuilder root = new StringBuilder();
204     if (!map.isEmpty()) {
205       root.append(
206           "<cat:bsnhashes xmlns:cat=\"http://schemas.kvk.nl/schemas/hrip/catalogus/2015/02\">");
207 
208       String type;
209       for (Map.Entry<String, String> entry : map.entrySet()) {
210         if (!entry.getKey().isEmpty() && !entry.getValue().isEmpty()) {
211           root.append("<cat:");
212           // NB BSN's kunnen met een 0 beginnen, dus niet NumberUtils#isCreatable
213           // gebruiken
214           if (NumberUtils.isParsable(entry.getKey())) {
215             type = PREFIX_BSN;
216           } else {
217             type = PREFIX_NAAM;
218           }
219           root.append(type)
220               .append(entry.getKey())
221               .append(">")
222               .append(entry.getValue())
223               .append("</cat:")
224               .append(type)
225               .append(entry.getKey())
226               .append(">");
227         }
228       }
229       root.append("</cat:bsnhashes>");
230     }
231     return root.toString();
232   }
