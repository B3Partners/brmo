147     private String getVersion(DataSource dataSource) throws SQLException {
148         String sql = "SELECT waarde FROM " + BrmoFramework.BRMO_METADATA_TABEL + " WHERE naam = 'brmoversie'";
149         final Connection c = dataSource.getConnection();
150         GeometryJdbcConverter geomToJdbc = GeometryJdbcConverterFactory.getGeometryJdbcConverter(c);
151         Object o = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(c, sql, new ScalarHandler());
152         DbUtils.closeQuietly(c);
153         return o.toString();
154     }

208     public Thread toRsgb(Bericht.STATUS status, ProgressUpdateListener listener) throws BrmoException {
209         RsgbProxy rsgbProxy = new RsgbProxy(dataSourceRsgb, stagingProxy, status, listener);
210         rsgbProxy.setEnablePipeline(enablePipeline);
211         if(pipelineCapacity != null) {
212             rsgbProxy.setPipelineCapacity(pipelineCapacity);
213         }
214         rsgbProxy.setRenewConnectionAfterCommit(renewConnectionAfterCommit);
215         rsgbProxy.setOrderBerichten(orderBerichten);
216         rsgbProxy.setErrorState(errorState);
217         Thread t = new Thread(rsgbProxy);
218         t.start();
219         return t;
220     }

222     /**
223      *
224      * @param mode geeft aan wat ids zijn (laadprocessen of berichten)
225      * @param ids array van ids
226      * @param listener voortgangs listener
227      * @return running transformatie thread
228      * @throws BrmoException if any
229      */
230     public Thread toRsgb(RsgbProxy.BerichtSelectMode mode, long[] ids, ProgressUpdateListener listener) throws BrmoException {
231         String soort = "";
232         if (mode == RsgbProxy.BerichtSelectMode.BY_LAADPROCES) {
233             try {
234                 soort = stagingProxy.getLaadProcesById(ids[0]).getSoort();
235             } catch (SQLException ex) {
236                 throw new BrmoException(ex);
237             }
238         }
239 
240         Runnable worker;
241         if (soort.equalsIgnoreCase(BR_BGTLIGHT)) {
242             // filter soort, als bgt dan als proces verwerken, niet als berichtenset
243             worker = new BGTLightRsgbTransformer(dataSourceRsgbBgt, stagingProxy, ids, listener);
244         } else if(TopNLType.isTopNLType(soort)){
245             try{
246                 worker = new TopNLRsgbTransformer(dataSourceTopNL, stagingProxy, ids, listener);
247             } catch (JAXBException | SQLException ex) {
248                 throw new BrmoException("Probleem met topparser initialiseren: ", ex);
249             }
250         }else{
251             worker = new RsgbProxy(dataSourceRsgb, stagingProxy, mode, ids, listener);
252             ((RsgbProxy) worker).setEnablePipeline(enablePipeline);
253             if (pipelineCapacity != null) {
254                 ((RsgbProxy) worker).setPipelineCapacity(pipelineCapacity);
255             }
256             ((RsgbProxy) worker).setRenewConnectionAfterCommit(renewConnectionAfterCommit);
257             ((RsgbProxy) worker).setOrderBerichten(orderBerichten);
258             ((RsgbProxy) worker).setErrorState(errorState);
259         }
260         Thread t = new Thread(worker);
261         t.start();
262         return t;
263     }

324     /**
325      * laden van BR data uit een bestand. <br>
326      * NB na gebruik zelf de database verbinding sluiten / opruimen met
327      * {@link #closeBrmoFramework()}.
328      *
329      * @param type basis registratie type
330      * @param fileName bestand
331      * @param listener voortgangsmonitor
332      * @throws BrmoException als er een fout optreed tijdens verwerking
333      */
334     public void loadFromFile(String type, String fileName, final ProgressUpdateListener listener) throws BrmoException {
335         try {
336             if (fileName.toLowerCase().endsWith(".zip") && !type.equalsIgnoreCase(BR_BGTLIGHT)) {
337                 log.info("Openen ZIP bestand " + fileName);
338                 ZipInputStream zip = null;
339                 try {
340                     File f = new File(fileName);
341                     if(listener != null) {
342                         listener.total(f.length());
343                     }
344                     CountingInputStream zipCis = new CountingInputStream(new FileInputStream(f)) {
345                         @Override
346                         protected void afterRead(int n) {
347                             super.afterRead(n);
348                             if(listener != null) {
349                                 listener.progress(getByteCount());
350                             }
351                         }
352                     };
353                     zip = new ZipInputStream(zipCis);
354                     ZipEntry entry = zip.getNextEntry();
355                     while (entry != null) {
356                         if (!entry.getName().toLowerCase().endsWith(".xml")) {
357                             log.warn("Overslaan zip entry geen XML: " + entry.getName());
358                         } else {
359                             log.info("Lezen XML bestand uit zip: " + entry.getName());
360                             stagingProxy.loadBr(new CloseShieldInputStream(zip), type, fileName + "/" + entry.getName(), null, null);
361                         }
362                         entry = zip.getNextEntry();
363                     }
364                 } finally {
365                     if(zip != null) {
366                         zip.close();
367                     }
368                 }
369                 log.info("Klaar met ZIP bestand " + fileName);
370             } else {
371                 File f = new File(fileName);
372                 if(listener != null) {
373                     listener.total(f.length());
374                 }
375                 FileInputStream fis = null;
376                 try {
377                     fis = new FileInputStream(fileName);
378                     stagingProxy.loadBr(fis, type, fileName, null, listener);
379                 } finally {
380                     IOUtils.closeQuietly(fis);
381                 }
382             }
383         } catch(Exception e) {
384             if (e instanceof BrmoException) {
385                 throw (BrmoException)e;
386             } else {
387                 throw new BrmoException("Fout bij loaden basisregistratie gegevens", e);
388             }
389         }
390     }

394      *
395      * @param type type registratie, bijv. {@value BrmoFramework#BR_BRK}
396      * @param stream datastream
397      * @param fileName te gebruiken bestandsnaam om laadproces te identificeren
398      * @throws BrmoException als er een algemene fout optreed
399      * @throws BrmoDuplicaatLaadprocesException als het "bestand"
400      * {@code fileName} al geladen is
401      * @throws BrmoLeegBestandException als het "bestand" {@code fileName} leeg
402      * is
403      */
404     public void loadFromStream(String type, InputStream stream, String fileName)
405             throws BrmoException, BrmoDuplicaatLaadprocesException, BrmoLeegBestandException {
406         try {
407             stagingProxy.loadBr(stream, type, fileName, null, null);
408         } catch (Exception e) {
409             if (e instanceof BrmoDuplicaatLaadprocesException) {
410                 throw (BrmoDuplicaatLaadprocesException) e;
411             }
412             if (e instanceof BrmoLeegBestandException) {
413                 throw (BrmoLeegBestandException) e;
414             }
415             throw new BrmoException("Fout bij laden " + type + " berichten uit bestand " + fileName, e);
416         }
417     }
