56      public void check(){
57          afgiftes.forEach((afgifte) -> {
58              check(afgifte);
59          });
60      }
61  
62      /**
63       * afgifte controle.
64       *
65       * @param afgifte de te controleren afgifte
66       */
67      private void check(Afgifte afgifte){
68          try {
69              LaadProces lp = staging.getLaadProcesByRestoredFilename(afgifte.getBestandsnaam());
70              // kijk of afgifte bestaat in staging
71              if (lp != null) {
72                  afgifte.setFoundInStaging(true);
73                  processFoundLaadprocess(afgifte, lp);
74              }
75          } catch (SQLException ex) {
76              log.error("Error querying staging for laadproces for afgifte: " + afgifte.toString(), ex);
77          }
78                  
79      }
80      /**
81       * zoek de status van alle bijbehorende berichten op en leg die vast in de
82       * afgifte.
83       *
84       * @param afgifte de te controleren afgifte
85       * @param lp het laadproces waar de berichten bij zoeken
86       * @throws SQLException if any
87       */
88      private void processFoundLaadprocess(Afgifte afgifte, LaadProces lp) throws SQLException{
89          List<Bericht> berichten = staging.getBerichtByLaadProces(lp);
90          Map<Bericht.STATUS, Integer> counts = afgifte.getStatussen();
91          berichten.stream().map((bericht) -> {
92              if(!counts.containsKey(bericht.getStatus())){
93                  counts.put(bericht.getStatus(), 0);
94              }
95              return bericht;
96          }).forEachOrdered((bericht) -> {
97              counts.put(bericht.getStatus(), counts.get(bericht.getStatus()) + 1);
98          });
99      }
