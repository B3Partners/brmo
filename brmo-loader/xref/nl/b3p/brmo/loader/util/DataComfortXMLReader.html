3   import java.util.ArrayList;
4   import java.util.List;
5   import javax.xml.stream.XMLInputFactory;
6   import javax.xml.stream.XMLStreamReader;
7   import javax.xml.transform.Source;
8   import org.geotools.gml.stream.XmlStreamGeometryReader;
9   import org.javasimon.SimonManager;
10  import org.javasimon.Split;
11  import org.locationtech.jts.geom.Geometry;
12  import org.locationtech.jts.geom.MultiPolygon;
13  import org.locationtech.jts.geom.Polygon;
14  
15  /**
16   * @author Boy de Wit
17   */
18  public class DataComfortXMLReader {
19  
20    private static final int LEVEL_ROOT = 0;
21    private static final int LEVEL_DATA = 1;
22    private static final int LEVEL_COMFORT = 2;
23    private static final int LEVEL_TABLE = 3;
24    private static final int LEVEL_DELETE = 4;
25    private final XMLInputFactory xif = XMLInputFactory.newInstance();
26  
27    public DataComfortXMLReader() {
28      xif.setProperty(XMLInputFactory.IS_COALESCING, Boolean.TRUE); // Coalesce characters
29      xif.setProperty(
30          XMLInputFactory.SUPPORT_DTD,
31          Boolean.FALSE); // No XML entity expansions or external entities
32    }
33  
34    public List<TableData> readDataXML(Source dataXML) throws Exception {
35      Split split = SimonManager.getStopwatch("b3p.util.datacomfortxmlreader").start();
36  
37      XMLStreamReader xer = xif.createXMLStreamReader(dataXML);
38      XmlStreamGeometryReader geometryReader = new XmlStreamGeometryReader(xer);
39  
40      int level = LEVEL_ROOT;
41      List<TableData> list = new ArrayList<>();
42      TableData data = null;
43      TableRow row = null;
44      boolean inComfortData = false;
45      boolean inDeleteData = false;
46  
47      root:
48      while (xer.hasNext()) {
49        xer.nextTag();
50        String tag = xer.getLocalName();
51        switch (level) {
52          case LEVEL_ROOT:
53            if ("data".equals(tag)) {
54              level = LEVEL_DATA;
55            }
56            break;
57  
58          case LEVEL_DATA:
59            if (xer.isEndElement()) {
60              break root;
61            }
62  
63            if ("comfort".equals(tag)) {
64              String comfortSearchTable = xer.getAttributeValue(null, "search-table");
65              String comfortSearchColumn = xer.getAttributeValue(null, "search-column");
66              String comfortSearchValue = xer.getAttributeValue(null, "search-value");
67              String snapshotDate = xer.getAttributeValue(null, "snapshot-date");
68              data =
69                  new TableData(
70                      comfortSearchTable, comfortSearchColumn, comfortSearchValue, snapshotDate);
71              level = LEVEL_COMFORT;
72              inComfortData = true;
73              inDeleteData = false;
74            } else if ("delete".equals(tag)) {
75              data = new TableData();
76              level = LEVEL_DELETE;
77              inComfortData = false;
78              inDeleteData = true;
79            } else {
80              row = new TableRow();
81              row.setTable(tag);
82              row.setIgnoreDuplicates("yes".equals(xer.getAttributeValue(null, "ignore-duplicates")));
83              String beginDatumColumn = xer.getAttributeValue(null, "column-dat-beg-geldh");
84              String eindeDatumColumn = xer.getAttributeValue(null, "column-datum-einde-geldh");
85              row.setColumnDatumBeginGeldigheid(beginDatumColumn);
86              row.setColumnDatumEindeGeldigheid(eindeDatumColumn);
87              data = new TableData(row);
88              level = LEVEL_TABLE;
89              inComfortData = false;
90              inDeleteData = false;
91            }
92            list.add(data);
93            break;
94  
95          case LEVEL_COMFORT:
96            if (xer.isEndElement() && "comfort".equals(tag)) {
97              level = LEVEL_DATA;
98            } else {
99              row = new TableRow();
100             row.setTable(tag);
101 
102             data.addRow(row);
103 
104             level = LEVEL_TABLE;
105           }
106           break;
107         case LEVEL_DELETE:
108           if (xer.isEndElement() && "delete".equals(tag)) {
109             level = LEVEL_DATA;
110           } else {
111             row = new TableRow();
112             row.setTable(tag);
113             data.addRow(row);
114             level = LEVEL_TABLE;
115           }
116           break;
117         case LEVEL_TABLE:
118           if (xer.isEndElement()) {
119             if (inComfortData) {
120               level = LEVEL_COMFORT;
121             } else if (inDeleteData) {
122               level = LEVEL_DELETE;
123             } else {
124               level = LEVEL_DATA;
125             }
126           } else {
127             data = list.get(list.size() - 1);
128             row.getColumns().add(tag);
129             String alleenArchief = xer.getAttributeValue(null, "alleen-archief");
130             if ("true".equals(alleenArchief)) {
131               row.setAlleenArchiefColumn(tag);
132             }
133             // Detecteer XML elementen of text
134             xer.next();
135             // Skip whitespace before a possible element
136             while (xer.isWhiteSpace()) {
137               xer.next();
138             }
139             if (xer.isStartElement()) {
140               Split split2 =
141                   SimonManager.getStopwatch("b3p.util.datacomfortxmlreader.parsegml").start();
142               Geometry geom = geometryReader.readGeometry();
143               // TODO/HACK force polygon to multi-polygon
144               if (geom instanceof Polygon) {
145                 geom = new MultiPolygon(new Polygon[] {(Polygon) geom}, geom.getFactory());
146               }
147               // Note: this linearizes curves, we could use a WKTWriter2 instead!
148               row.getValues().add(geom.toString());
149               split2.stop();
150               // After parsing geometry, move to end of enclosing tag
151               while (!(tag.equals(xer.getLocalName()) && xer.isEndElement())) {
152                 xer.nextTag();
153               }
154             } else if (xer.isCharacters()) {
155               StringBuilder t = new StringBuilder();
156               do {
157                 t.append(xer.getText());
158                 xer.next();
159               } while (xer.isCharacters());
160               row.getValues().add(t.toString());
161             } else {
162               assert (xer.isEndElement());
163               row.getValues().add(null);
164             }
165           }
166           break;
167       }
168     }
169     split.stop();
170     return list;
171   }
