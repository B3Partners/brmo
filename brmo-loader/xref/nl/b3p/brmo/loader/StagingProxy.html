356         Split split = SimonManager.getStopwatch("b3p.rsgb.job").start();
357         final String dateTime = new SimpleDateFormat("yyyyMMdd_HHmmss").format(new Date());
358         Split jobSplit = SimonManager.getStopwatch("b3p.rsgb.job." + dateTime).start();
359         ((RsgbProxy)handler).setSimonNamePrefix("b3p.rsgb.job." + dateTime + ".");
360         final RowProcessor processor = new StagingRowHandler();
361 
362         final ProcessDbXmlPipelineocessDbXmlPipeline">ProcessDbXmlPipeline processDbXmlPipeline = new ProcessDbXmlPipeline(handler, transformPipelineCapacity, "b3p.rsgb.job." + dateTime + ".pipeline");
363         if(enablePipeline) {
364             processDbXmlPipeline.start();
365         }
366         int offset = 0;
367         int batch = batchCapacity;
368         final MutableInt processed = new MutableInt(0);
369         final MutableInt lastJid = new MutableInt(0);
370         final boolean doOrderBerichten = orderBerichten;
371         boolean abort = false;
372         try {
373             do {
374                 log.debug(String.format("Ophalen berichten batch Job tabel, offset %d, limit %d", lastJid.intValue(), batch));
375                 String sql = "select jid, id, datum, volgordenummer, object_ref, br_xml, soort from "
376                         + BrmoFramework.JOB_TABLE + " where jid > " + lastJid.intValue() + " order by jid ";
377                 sql = geomToJdbc.buildPaginationSql(sql, 0, batch);
378                 log.debug("SQL voor ophalen berichten batch: " + sql);
379 
380                 processed.setValue(0);
381                 final Split getBerichten = SimonManager.getStopwatch("b3p.rsgb.job." + dateTime + ".staging.berichten.getbatch").start();
382                 Exception e = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(getConnection(), sql, new ResultSetHandler<Exception>() {
383                     @Override
384                     public Exception handle(ResultSet rs) throws SQLException {
385                         getBerichten.stop();
386                         final Split processResultSet = SimonManager.getStopwatch("b3p.rsgb.job." + dateTime + ".staging.berichten.processresultset").start();
387 
388                         while(rs.next()) {
389                             try {
390                                 Bericht bericht = processor.toBean(rs, Bericht.class);
391                                 if (bericht.getVolgordeNummer() >= 0  && !doOrderBerichten) {
392                                     // mutaties hebben volgnr >=0 dan sortering verplicht
393                                     throw new Exception(String.format("Het sorteren van berichten staat uit, terwijl bericht (id: %d) geen standbericht is (volgnummer >=0)", bericht.getId()));
394                                 }

398

400

403                                         // Exception during transform
404

406                                         continue;
407                                     }

409                                     workUnit.setTypeOfWork(BerichtTypeOfWork.PROCESS_DBXML);
410                                     Split pipelinePut = SimonManager.getStopwatch("b3p.rsgb.job." + dateTime + ".pipeline.processdbxml.put").start();
411                                     processDbXmlPipeline.getQueue().put(workUnit);
412                                     pipelinePut.stop();

417                                 }

419                             } catch(Exception e) {
420                                 return e;
421                             }
422                             processed.increment();
423                         }
424                         processResultSet.stop();
425                         return null;
426                     }
427                 });
428 
429                 offset += processed.intValue();
430 
431                 // If handler threw exception processing row, rethrow it
432                 if(e != null) {
433                     throw e;
434                 }
435 
436             } while(processed.intValue() > 0 && (offset < total));
437             if(offset < total) {
438                 log.warn(String.format("Minder berichten verwerkt (%d) dan verwacht (%d)!", offset, total));
439             }

441             removeJob();
442         } catch(Exception t) {
443             abort = true;
444             throw t;
445         } finally {

448                     // Let threads stop by themselves, don't join

452                     try {
453                         processDbXmlPipeline.join();
454                     } catch(InterruptedException e) {
455                     }

460         split.stop();
