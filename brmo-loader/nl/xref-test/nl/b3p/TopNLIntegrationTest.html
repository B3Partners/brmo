277     @BeforeEach
278     public void setUp() throws Exception {
279         dsStaging = new BasicDataSource();
280         dsStaging.setUrl(params.getProperty("staging.jdbc.url"));
281         dsStaging.setUsername(params.getProperty("staging.user"));
282         dsStaging.setPassword(params.getProperty("staging.passwd"));
283         dsStaging.setAccessToUnderlyingConnectionAllowed(true);
284         staging = new DatabaseDataSourceConnection(dsStaging);
285 
286         dsTopnl = new BasicDataSource();
287         dsTopnl.setUrl(params.getProperty("topnl.jdbc.url"));
288         dsTopnl.setUsername(params.getProperty("topnl.user"));
289         dsTopnl.setPassword(params.getProperty("topnl.passwd"));
290         dsTopnl.setAccessToUnderlyingConnectionAllowed(true);
291         topnl = new DatabaseDataSourceConnection(dsTopnl);
292 
293         brmo = new BrmoFramework(dsStaging, null, null, dsTopnl, null);
294 
295         if (this.isOracle) {
296             staging =
297                     new DatabaseConnection(
298                             OracleConnectionUnwrapper.unwrap(dsStaging.getConnection()),
299                             params.getProperty("staging.user").toUpperCase());
300             staging.getConfig()
301                     .setProperty(
302                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
303                             new Oracle10DataTypeFactory());
304             staging.getConfig()
305                     .setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
306 
307             topnl =
308                     new DatabaseConnection(
309                             OracleConnectionUnwrapper.unwrap(dsTopnl.getConnection()),
310                             params.getProperty("topnl.user").toUpperCase());
311             topnl.getConfig()
312                     .setProperty(
313                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
314                             new Oracle10DataTypeFactory());
315             topnl.getConfig()
316                     .setProperty(DatabaseConfig.FEATURE_SKIP_ORACLE_RECYCLEBIN_TABLES, true);
317         } else if (this.isPostgis) {
318             staging.getConfig()
319                     .setProperty(
320                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
321                             new PostgresqlDataTypeFactory());
322             topnl.getConfig()
323                     .setProperty(
324                             DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
325                             new PostgresqlDataTypeFactory());
326         }
327         FlatXmlDataSetBuilder fxdb = new FlatXmlDataSetBuilder();
328         fxdb.setCaseSensitiveTableNames(false);
329         // lege bericht en laadproces tabellen
330         IDataSet stagingDataSet =
331                 fxdb.build(
332                         new FileInputStream(
333                                 new File(
334                                         TopNLIntegrationTest.class
335                                                 .getResource("/staging-empty-flat.xml")
336                                                 .toURI())));
337 
338         sequential.lock();
339 
340         DatabaseOperation.CLEAN_INSERT.execute(staging, stagingDataSet);
341 
342         assumeTrue(
343                 0L == brmo.getCountBerichten(null, "STAGING_OK"), "Er zijn STAGING_OK berichten");
344         assumeTrue(
345                 0L == brmo.getCountLaadProcessen(null, "STAGING_OK"),
346                 "Er zijn STAGING_OK laadprocessen");
347     }

349     @AfterEach
350     public void cleanup() throws Exception {
351         brmo.closeBrmoFramework();
352         brmo = null;
353         CleanUtil.cleanSTAGING(staging, false);
354         staging.close();
355         dsStaging.close();
356 
357         topnl.close();
358         dsTopnl.close();
359 
360         try {
361             sequential.unlock();
362         } catch (IllegalMonitorStateException e) {
363             // in geval van niet waar gemaakte assumptions
364             LOG.debug(
365                     "unlock van thread is mislukt, mogelijk niet ge-lock-ed of test overgeslagen.");
366         }
367     }

369     @DisplayName("TopNL in Staging")
370     @ParameterizedTest(name = "argumenten #{index}: type: {0}, bestand: {1}, gebied: {4}")
371     @MethodSource("argumentsProvider")
372     public void loadTopNLInStaging(
373             TopNLType bestandType,
374             String bestandNaam,
375             long aantalBerichten,
376             long aantalProcessen,
377             String lpGebied)
378             throws BrmoException {
379         // omdat soms de bestandsnaam wordt aangepast moet dit een harde test fout zijn, dus geen
380         // assumeNotNull
381         assertNotNull(
382                 TopNLIntegrationTest.class.getResource(bestandNaam),
383                 "Het test bestand moet er zijn.");
384 
385         brmo.loadFromFile(
386                 bestandType.getType(),
387                 TopNLIntegrationTest.class.getResource(bestandNaam).getFile(),
388                 null);
389 
390         final List<LaadProces> processen = brmo.listLaadProcessen();
391         final List<Bericht> berichten = brmo.listBerichten();
392 
393         assertNotNull(berichten, "De verzameling berichten bestaat niet.");
394         assertEquals(
395                 aantalBerichten, berichten.size(), "Het aantal berichten is niet als verwacht.");
396         assertNotNull(processen, "De verzameling processen bestaat niet.");
397         assertEquals(
398                 aantalProcessen,
399                 processen.size(),
400                 () -> "Het aantal processen is niet als verwacht voor " + bestandNaam);
401         assertEquals(lpGebied, processen.get(0).getGebied(), "Het gebied moet als verwacht zijn.");
402     }

404     @DisplayName("verwerk TopNL")
405     @ParameterizedTest(name = "argumenten rij {index}: type: {0}, bestand: {1}, gebied: {4}")
406     @MethodSource("argumentsProvider")
407     public void processTopNL(
408             TopNLType bestandType,
409             String bestandNaam,
410             long aantalBerichten,
411             long aantalProcessen,
412             String lpGebied)
413             throws Exception {
414         // omdat soms de bestandsnaam wordt aangepast moet dit een harde test fout zijn, dus geen
415         // assumeNotNull
416         assertNotNull(
417                 TopNLIntegrationTest.class.getResource(bestandNaam),
418                 "Het test bestand moet er zijn.");
419 
420         brmo.loadFromFile(
421                 bestandType.getType(),
422                 TopNLIntegrationTest.class.getResource(bestandNaam).getFile(),
423                 null);
424         final List<LaadProces> processen = brmo.listLaadProcessen();
425         assertEquals(aantalProcessen, processen.size(), "Er is 1 laadproces.");
426 
427         List<Long> ids = new ArrayList<>();
428         for (TopNLType type : TopNLType.values()) {
429             ids.addAll(
430                     Arrays.asList(
431                             brmo.getLaadProcessenIds(
432                                     "bestand_datum", "ASC", type.getType(), "STAGING_OK")));
433         }
434 
435         LOG.debug("Te verwerken LP ids:" + ids);
436         long[] lpIds = ArrayUtils.toPrimitive(ids.toArray(new Long[ids.size()]));
437 
438         Thread t =
439                 brmo.toRsgb(
440                         RsgbProxy.BerichtSelectMode.BY_LAADPROCES,
441                         lpIds,
442                         new ProgressUpdateListener() {
443                             @Override
444                             public void total(long total) {
445                                 LOG.info("Totaal te verwerken: " + total);
446                             }
447 
448                             @Override
449                             public void progress(long progress) {
450                                 LOG.info("Verwerkt: " + progress);
451                             }
452 
453                             @Override
454                             public void exception(Throwable t) {
455                                 LOG.error(t);
456                                 fail("Fout tijdens verwerken. " + t.getLocalizedMessage());
457                             }
458                         });
459         t.join();
460         assertEquals(
461                 LaadProces.STATUS.RSGB_TOPNL_OK.toString(),
462                 staging.createDataSet().getTable("laadproces").getValue(0, "status"),
463                 "Alles OK");
464     }
